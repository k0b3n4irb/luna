1//! MCP server for Luna — wraps `luna-api::Emulator` and exposes its
2//! methods as MCP tools via [`rmcp`].
3//!
4//! Designed so Claude (or any MCP-aware client) can drive the
5//! emulator end-to-end:
6//!
7//! - `load_rom { path }` → loads a cartridge, returns its metadata.
8//! - `reset` → power-on reset.
9//! - `step { count }` → advance the CPU N instructions.
10//! - `step_until_frame { max_steps }` → advance to the next PPU
11//!   frame boundary or hit the cap.
12//! - `state` → JSON snapshot of every observable bit of emulator
13//!   state (CPU regs, PPU regs + occupancy, APU/SPC, scheduler).
14//! - `screenshot { force_display? }` → PNG of the current
15//!   framebuffer, returned base64-encoded so MCP clients can render
16//!   it inline.
17//! - `drain_audio { max }` → consume up to N stereo (i16,i16)
18//!   samples from the APU queue, returned as a flat
19//!   `[l0, r0, l1, r1, …]` array.
20//! - `peek_memory { bank, offset, count }` → read through the bus.
21//! - `peek_aram { offset, count }` → direct SPC700 ARAM read.
22//! - `peek_vram { offset, count }` → direct VRAM read.
23//! - `poke_memory { bank, offset, data }` → inject WRAM bytes.
24//! - `search_memory { pattern }` → find a byte pattern in WRAM.
25//! - `run_until_pc { pc, max_steps }` → step to a target PC.
26//! - `set_cpu_register { reg, val }` → set a CPU register.
27//!
28//! Transport is stdio by default ([`serve_stdio`]); a future commit
29//! will add HTTP-SSE for browser clients.
30

46/// MCP service wrapper around a single-process Luna `Emulator`.
47///
48/// All tools take a shared `Arc<Mutex<Emulator>>` lock for the
49/// duration of the call. That's fine for the stdio transport (one
50/// client at a time); a multi-client HTTP transport would need a
51/// per-session emulator.
52#[derive(Clone)]
53pub struct LunaServer {
54    emulator: Arc<Mutex<Emulator>>,
55    tool_router: ToolRouter<Self>,
56}

75/// `set_joypad` parameters.
76#[derive(Debug, Clone, Deserialize, JsonSchema)]
77pub struct SetJoypadParams {
78    /// Controller index: `0` = Player 1 (`$4218/$4219`),
79    /// `1` = Player 2 (`$421A/$421B`).
80    pub port: u8,
81    /// 16-bit JOY1 bitmask. Bit layout (high → low):
82    /// B, Y, Select, Start, Up, Down, Left, Right, A, X, L, R,
83    /// 0, 0, 0, 0. So `0x1000` = Start, `0x8000` = B,
84    /// `0xF000` = Start + Select + Y + B, etc.
85    pub mask: u16,
86}

88/// `step_until_frame` parameters.
89#[derive(Debug, Clone, Deserialize, JsonSchema)]
90pub struct StepUntilFrameParams {
91    /// Maximum instructions to execute before bailing out (safety
92    /// belt against runaway loops). 1 000 000 is a reasonable
93    /// default for "one game frame".
94    pub max_steps: u64,
95}

97/// `screenshot` parameters.

287    #[rmcp::tool(
288        description = "Load a SNES ROM (.sfc / .smc) from a path on the host filesystem. \
289                                Returns parsed cartridge metadata."
290    )]
291    async fn load_rom(
292        &self,
293        Parameters(params): Parameters<LoadRomParams>,
294    ) -> Result<rmcp::Json<LoadRomResult>, ErrorData> {
295        let info = {
296            let mut em = self.emulator.lock().await;
297            em.load_rom(&PathBuf::from(params.path))
298                .map_err(|e| api_err_to_mcp(&e))?
299        };
300        Ok(rmcp::Json(LoadRomResult { rom: info }))
301    }
302
303    #[rmcp::tool(description = "Reset the loaded emulator to its power-on state. \
304                                Errors if no ROM is currently loaded.")]
305    async fn reset(&self) -> Result<rmcp::Json<EmptyOk>, ErrorData> {
306        {
307            let mut em = self.emulator.lock().await;
308            em.reset().map_err(|e| api_err_to_mcp(&e))?;
309        }
310        Ok(rmcp::Json(EmptyOk { ok: true }))
311    }

313    #[rmcp::tool(
314        description = "Set the joypad button bitmask for controller `port` (0 = P1, \
315                                1 = P2). Bit layout matches the SNES JOY1L/JOY1H pair: \
316                                B(15) Y(14) Select(13) Start(12) Up(11) Down(10) Left(9) \
317                                Right(8) A(7) X(6) L(5) R(4) + 4-bit signature. The press \
318                                is latched on the next VBlank auto-read (one frame later) \
319                                — hold the mask for at least 2 frames before reading back \
320                                game state, then write 0 to release."
321    )]
322    async fn set_joypad(
323        &self,
324        Parameters(params): Parameters<SetJoypadParams>,
325    ) -> Result<rmcp::Json<EmptyOk>, ErrorData> {
326        {
327            let mut em = self.emulator.lock().await;
328            em.set_joypad(params.port, params.mask)
329                .map_err(|e| api_err_to_mcp(&e))?;

366    #[rmcp::tool(
367        description = "Return a JSON snapshot of the emulator's full observable state — \

376    #[rmcp::tool(
377        description = "Render the current PPU framebuffer (256×224, composited \
378                                BG3-over-BG1-over-BG2 + sprites) as a PNG and return it \
379                                base64-encoded."
380    )]
381    async fn screenshot(
382        &self,
383        Parameters(params): Parameters<ScreenshotParams>,
384    ) -> Result<rmcp::Json<ScreenshotResult>, ErrorData> {
385        let png = {
386            let em = self.emulator.lock().await;
387            em.render_frame_png(params.force_display)
388                .map_err(|e| api_err_to_mcp(&e))?
389        };
390        let png_base64 = base64::engine::general_purpose::STANDARD.encode(&png);
391        Ok(rmcp::Json(ScreenshotResult {
392            png_base64,
393            width: 256,
394            height: 224,
395        }))
396    }

398    #[rmcp::tool(
399        description = "Drain up to `max` stereo audio samples from the APU output \
400                                queue. Returns interleaved [l, r, l, r, …] signed-16-bit samples \
401                                at 32 kHz."
402    )]
403    async fn drain_audio(
404        &self,
405        Parameters(params): Parameters<DrainAudioParams>,
406    ) -> Result<rmcp::Json<DrainAudioResult>, ErrorData> {
407        let samples = {
408            let mut em = self.emulator.lock().await;
409            em.drain_audio(params.max).map_err(|e| api_err_to_mcp(&e))?
410        };
411        let frames = samples.len();
412        let mut flat = Vec::with_capacity(frames * 2);
413        for (l, r) in samples {
414            flat.push(l);
415            flat.push(r);
416        }
417        Ok(rmcp::Json(DrainAudioResult {
418            samples: flat,
419            frames,
420        }))
421    }

599/// Trivial `{ ok: true }` payload for tools whose only failure mode
600/// is an explicit error.
601#[derive(Debug, Clone, Serialize, JsonSchema)]
602pub struct EmptyOk {
603    /// Always `true`. Errors come back as an MCP error response.
604    pub ok: bool,
605}

610/// Map [`luna_api::ApiError`] onto an MCP `internal_error` payload.
611fn api_err_to_mcp(e: &ApiError) -> ErrorData {
612    ErrorData::internal_error(e.to_string(), None)
613}

615/// Run the Luna MCP server on stdio until the client disconnects.
616///
617/// Intended entry point for the `luna mcp serve` CLI subcommand and
618/// for `claude_desktop_config.json`-style spawns. Blocks until the
619/// MCP client closes the stream or sends a shutdown.
620pub async fn serve_stdio() -> Result<(), Box<dyn std::error::Error + Send + Sync>> {
621    let (stdin, stdout) = stdio();
622    let server = LunaServer::new().serve((stdin, stdout)).await?;
623    server.waiting().await?;
624    Ok(())
625}

637    /// We can't exercise the full MCP protocol from a unit test
638    /// without setting up an in-memory transport pair (which is its

643    async fn server_state_works_without_rom() {
644        let s = LunaServer::new();

679    /// Smoke-test the full happy path: load a tiny ROM, step it,
680    /// dump state, render a PNG. Uses the same demo cart the
681    /// `luna-api` tests use, just to ensure the MCP wrappers
682    /// faithfully forward.
683    #[tokio::test]
684    async fn server_load_step_state_screenshot_round_trip() {
685        let s = LunaServer::new();
686        // Write demo cart to a tempfile so `load_rom` (which takes
687        // a path) can read it.
688        let path = PathBuf::from("/tmp/luna_mcp_demo.smc");
689        std::fs::write(&path, demo_lorom()).unwrap();
690        let info = s
691            .load_rom(Parameters(LoadRomParams {
692                path: path.to_string_lossy().into(),
693            }))
694            .await
695            .unwrap();
696        assert_eq!(info.0.rom.mapper, "LoRom");
697        let stepped = s.step(Parameters(StepParams { count: 100 })).await.unwrap();
698        assert!(stepped.0.executed > 0);
699        let st = s.state().await;
700        assert!(st.0.state.rom.is_some());
701        let png = s
702            .screenshot(Parameters(ScreenshotParams::default()))
703            .await
704            .unwrap();
705        assert_eq!(png.0.width, 256);
706        assert_eq!(png.0.height, 224);
707        // PNG header check via base64-decode.
708        let bytes = base64::engine::general_purpose::STANDARD
709            .decode(&png.0.png_base64)
710            .unwrap();
711        assert!(bytes.starts_with(b"\x89PNG"));
712        let _ = std::fs::remove_file(&path);
713    }

715    fn demo_lorom() -> Vec<u8> {
716        let mut rom = vec![0u8; 0x8000];
717        rom[0x7FFC] = 0x00;
718        rom[0x7FFD] = 0x80;
719        let title = b"LUNA MCP DEMO        ";
720        rom[0x7FC0..0x7FC0 + title.len()].copy_from_slice(title);
721        rom[0x7FD5] = 0x20;
722        rom[0x7FD7] = 0x07;
723        rom[0x7FD8] = 0x00;
724        let mut sum = 0u32;
725        for (i, b) in rom.iter().enumerate() {
726            if !(0x7FDC..=0x7FDF).contains(&i) {
727                sum += u32::from(*b);
728            }
729        }
730        let checksum = (sum & 0xFFFF) as u16;
731        let complement = !checksum;
732        rom[0x7FDC] = complement as u8;
733        rom[0x7FDD] = (complement >> 8) as u8;
734        rom[0x7FDE] = checksum as u8;
735        rom[0x7FDF] = (checksum >> 8) as u8;
736        rom
737    }
