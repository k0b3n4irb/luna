1//! 65C816 status register (`P`) and the hidden `E` (emulation) flag.
2
3use serde::{Deserialize, Serialize};
4
5/// Status register bit positions.
6///
7/// `M` and `X` only have meaning when `E = 0` (native mode). In emulation
8/// mode (`E = 1`) they are forced to `1` (8-bit accumulator and index
9/// registers, 6502-style).
10pub mod bit {
11    /// Negative.
12    pub const N: u8 = 0b1000_0000;
13    /// Overflow.
14    pub const V: u8 = 0b0100_0000;
15    /// Memory / Accumulator width — `1` = 8-bit, `0` = 16-bit. (Native only.)
16    pub const M: u8 = 0b0010_0000;
17    /// Index width — `1` = 8-bit, `0` = 16-bit. (Native only.)
18    pub const X: u8 = 0b0001_0000;
19    /// Decimal mode (BCD arithmetic on ADC / SBC).
20    pub const D: u8 = 0b0000_1000;
21    /// IRQ disable.
22    pub const I: u8 = 0b0000_0100;
23    /// Zero.
24    pub const Z: u8 = 0b0000_0010;
25    /// Carry.
26    pub const C: u8 = 0b0000_0001;
27}
28
29/// Wrapper around the 8-bit `P` register, with named bit accessors.
30#[derive(Debug, Clone, Copy, PartialEq, Eq, Default, Serialize, Deserialize)]
31pub struct StatusFlags(pub u8);
32
33impl StatusFlags {
34    /// Power-on / reset value of the status register on the 65C816:
35    /// `M = X = I = 1`, all others cleared.
36    pub const RESET: Self = Self(bit::M | bit::X | bit::I);
37
38    /// Raw `P` register byte.
39    #[inline]
40    #[must_use]
41    pub const fn bits(self) -> u8 {
42        self.0
43    }
44
45    /// `true` if **all** the bits in `mask` are currently set.
46    #[inline]
47    #[must_use]
48    pub const fn contains(self, mask: u8) -> bool {
49        (self.0 & mask) == mask
50    }
51
52    /// Set every bit in `mask`.
53    #[inline]
54    pub const fn insert(&mut self, mask: u8) {
55        self.0 |= mask;
56    }
57
58    /// Clear every bit in `mask`.
59    #[inline]
60    pub const fn remove(&mut self, mask: u8) {
61        self.0 &= !mask;
62    }
63
64    /// Set every bit in `mask` to `value`.
65    #[inline]
66    pub const fn set(&mut self, mask: u8, value: bool) {
67        if value {
68            self.insert(mask);
69        } else {
70            self.remove(mask);
71        }
72    }
73
74    /// `true` if the accumulator is 8-bit (M flag set).
75    #[inline]
76    #[must_use]
77    pub const fn acc8(self) -> bool {
78        self.contains(bit::M)
79    }
80
81    /// `true` if the index registers are 8-bit (X flag set).
82    #[inline]
83    #[must_use]
84    pub const fn idx8(self) -> bool {
85        self.contains(bit::X)
86    }
87}

89#[cfg(test)]
90mod tests {
91    use super::*;
92
93    #[test]
94    fn reset_value_has_m_x_i_set() {
95        let p = StatusFlags::RESET;
96        assert!(p.contains(bit::M));
97        assert!(p.contains(bit::X));
98        assert!(p.contains(bit::I));
99        assert!(!p.contains(bit::N));
100        assert!(!p.contains(bit::V));
101        assert!(!p.contains(bit::D));
102        assert!(!p.contains(bit::Z));
103        assert!(!p.contains(bit::C));
104    }
105
106    #[test]
107    fn set_toggles_bits() {
108        let mut p = StatusFlags::default();
109        p.set(bit::C, true);
110        assert!(p.contains(bit::C));
111        p.set(bit::C, false);
112        assert!(!p.contains(bit::C));
113    }
114
115    #[test]
116    fn acc8_idx8_helpers() {
117        let mut p = StatusFlags::default();
118        assert!(!p.acc8());
119        assert!(!p.idx8());
120        p.insert(bit::M);
121        assert!(p.acc8());
122        p.insert(bit::X);
123        assert!(p.idx8());
124    }
