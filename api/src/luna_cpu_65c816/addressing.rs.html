1//! 65C816 addressing-mode helpers.
2//!
3//! Each helper fetches operand bytes from `PB:PC`, applies the relevant
4//! transformation (direct page offset, bank wrap, index addition…) and
5//! returns the **effective 24-bit address** to read from / write to.
6//!
7//! Cycle accounting is the bus's responsibility (via `io_cycle`); these
8//! helpers only deal with address calculation.
9
10use crate::cpu::Cpu;
11use luna_bus::{Addr24, Bus, make_addr};
12
13/// Access kind for indexed addressing modes. Only `abs,X`/`abs,Y` and
14/// `(dp),Y` charge a different internal cycle for reads vs. stores/RMW:
15/// a **read** pays the page-cross cycle only when the page actually
16/// changed (or the index is 16-bit) — ares `idle4()` — whereas a
17/// **write** always pays one fixed internal cycle. (The extra RMW dead
18/// cycle is charged separately in `modify_memory`.)
19#[derive(Debug, Clone, Copy, PartialEq, Eq)]
20pub enum Access {
21    /// Loads / comparisons: pay the page-cross cycle conditionally.
22    Read,
23    /// Stores and read-modify-write: pay one fixed index cycle.
24    Write,
25}
26
27/// Absolute: `LDA $abs` — operand is a 16-bit address in the data bank.
28#[inline]
29pub fn absolute<B: Bus>(cpu: &mut Cpu, bus: &mut B) -> Addr24 {
30    let offset = cpu.fetch_u16(bus);
31    make_addr(cpu.db, offset)
32}
33
34/// Absolute Long: `LDA $long` — operand is a 24-bit address.
35#[inline]
36pub fn absolute_long<B: Bus>(cpu: &mut Cpu, bus: &mut B) -> Addr24 {
37    cpu.fetch_u24(bus)
38}

40/// Direct Page: `LDA $dp` — operand is a u8 offset added to `DP`.
41///
42/// In emulation mode with `DP.low == 0`, the effective address wraps
43/// within the current 256-byte page of the direct page (6502 behavior).
44/// In all other cases the effective address wraps within the 16-bit
45/// page-zero of bank 0.
46#[inline]
47pub fn direct_page<B: Bus>(cpu: &mut Cpu, bus: &mut B) -> Addr24 {
48    let offset = u16::from(cpu.fetch_u8(bus));
49    cpu.idle2(bus);
50    cpu.bank0_wrap = true;
51    make_addr(0, cpu.dp.wrapping_add(offset))
52}
53
54/// Direct Page Indexed X: `LDA $dp,X`.
55///
56/// Wraps with the same emulation-mode caveat as [`direct_page`].
57#[inline]
58pub fn direct_page_indexed_x<B: Bus>(cpu: &mut Cpu, bus: &mut B) -> Addr24 {
59    let base = u16::from(cpu.fetch_u8(bus));
60    cpu.idle2(bus);
61    cpu.io(bus);
62    cpu.bank0_wrap = true;
63    let off = if cpu.e && (cpu.dp & 0xFF) == 0 {
64        // Emulation: wrap within the 256-byte direct page.
65        let dp_high = cpu.dp & 0xFF00;
66        let wrapped = (base as u8).wrapping_add(cpu.x8());
67        dp_high | u16::from(wrapped)
68    } else {
69        cpu.dp.wrapping_add(base).wrapping_add(cpu.x)
70    };
71    make_addr(0, off)
72}

74/// Direct Page Indexed Y: `STX $dp,Y` (the index-Y direct-page family).
75#[inline]
76pub fn direct_page_indexed_y<B: Bus>(cpu: &mut Cpu, bus: &mut B) -> Addr24 {
77    let base = u16::from(cpu.fetch_u8(bus));
78    cpu.idle2(bus);
79    cpu.io(bus);
80    cpu.bank0_wrap = true;
81    let off = if cpu.e && (cpu.dp & 0xFF) == 0 {
82        let dp_high = cpu.dp & 0xFF00;
83        let wrapped = (base as u8).wrapping_add(cpu.y8());
84        dp_high | u16::from(wrapped)
85    } else {
86        cpu.dp.wrapping_add(base).wrapping_add(cpu.y)
87    };
88    make_addr(0, off)
89}

91/// Read a 16-bit pointer at the given bank-0 offset.
92///
93/// Inlined helper used by the various direct-page-indirect modes.
94#[inline]
95fn read_ptr16<B: Bus>(bus: &mut B, ptr_off: u16) -> u16 {
96    let lo = bus.read(make_addr(0, ptr_off));
97    let hi = bus.read(make_addr(0, ptr_off.wrapping_add(1)));
98    u16::from(lo) | (u16::from(hi) << 8)
99}

101/// Direct Page Indirect: `LDA ($dp)` — read a 16-bit pointer at DP+dp,
102/// resolved against the data bank.
103#[inline]
104pub fn direct_page_indirect<B: Bus>(cpu: &mut Cpu, bus: &mut B) -> Addr24 {
105    let dp_off = u16::from(cpu.fetch_u8(bus));
106    cpu.idle2(bus);
107    let ptr_off = cpu.dp.wrapping_add(dp_off);
108    let offset = read_ptr16(bus, ptr_off);
109    make_addr(cpu.db, offset)
110}

112/// Direct Page Indirect Long: `LDA [$dp]` — read a 24-bit pointer.
113#[inline]
114pub fn direct_page_indirect_long<B: Bus>(cpu: &mut Cpu, bus: &mut B) -> Addr24 {
115    let dp_off = u16::from(cpu.fetch_u8(bus));
116    cpu.idle2(bus);
117    let ptr_off = cpu.dp.wrapping_add(dp_off);
118    let lo = bus.read(make_addr(0, ptr_off));
119    let mid = bus.read(make_addr(0, ptr_off.wrapping_add(1)));
120    let hi = bus.read(make_addr(0, ptr_off.wrapping_add(2)));
121    Addr24::from(lo) | (Addr24::from(mid) << 8) | (Addr24::from(hi) << 16)
122}

124/// Direct Page Indirect Y: `LDA ($dp),Y` — read pointer, add Y with
125/// bank carry into the data bank.
126#[inline]
127pub fn direct_page_indirect_y<B: Bus>(cpu: &mut Cpu, bus: &mut B, acc: Access) -> Addr24 {
128    let dp_off = u16::from(cpu.fetch_u8(bus));
129    cpu.idle2(bus);
130    let ptr_off = cpu.dp.wrapping_add(dp_off);
131    let base_off = read_ptr16(bus, ptr_off);
132    let new_off = base_off.wrapping_add(cpu.y);
133    match acc {
134        Access::Read => cpu.idle4(bus, base_off, new_off),
135        Access::Write => cpu.io(bus),
136    }
137    let bank = if new_off < base_off {
138        cpu.db.wrapping_add(1)
139    } else {
140        cpu.db
141    };
142    make_addr(bank, new_off)
143}

145/// Direct Page Indirect Long Y: `LDA [$dp],Y` — read 24-bit pointer,
146/// add Y with full bank carry.
147#[inline]
148pub fn direct_page_indirect_long_y<B: Bus>(cpu: &mut Cpu, bus: &mut B) -> Addr24 {
149    let base = direct_page_indirect_long(cpu, bus);
150    base.wrapping_add(Addr24::from(cpu.y)) & 0x00FF_FFFF
151}

153/// Direct Page Indexed X Indirect: `LDA ($dp,X)`.
154///
155/// The X register is added BEFORE the indirect read (pointer fetch).
156#[inline]
157pub fn direct_page_indexed_indirect<B: Bus>(cpu: &mut Cpu, bus: &mut B) -> Addr24 {
158    let dp_off = u32::from(cpu.fetch_u8(bus));
159    cpu.idle2(bus);
160    cpu.io(bus);
161    // ares: V.l = readDirectX(U.l + X.w, 0); V.h = readDirectX(U.l + X.w, 1).
162    let address = dp_off.wrapping_add(u32::from(cpu.x));
163    let lo = read_direct_x(cpu, bus, address, 0);
164    let hi = read_direct_x(cpu, bus, address, 1);
165    let offset = u16::from(lo) | (u16::from(hi) << 8);
166    make_addr(cpu.db, offset)
167}

169/// Fetch one byte of a `(dp,X)` pointer. `address` is `dp_operand + X`;
170/// `offset` is 0 (low byte) or 1 (high byte).
171///
172/// Resolve one byte of a `(direct,X)` pointer.
173///
174/// In **emulation mode with `D.l == 0`** the index addition wraps within
175/// the 256-byte direct page to form the pointer *base* (`D.w | n8(dp+X)`);
176/// the two pointer bytes are then read **linearly** from `base` and
177/// `base+1` (bank-0 wrap). Every other case (E with `D.l != 0`, or native)
178/// is a plain 16-bit add within bank 0.
179///
180/// This diverges subtly from ares `readDirectX` (`memory.cpp:62-67`), which
181/// re-applies the page-wrap to the *high* byte (`D.w | n8(dp+X+1)`). The
182/// `SingleStepTests/65816` suite — hardware-derived, so the empirical gate —
183/// disagrees only when `n8(dp+X) == 0xFF`:
184/// - case `e1.e 125` (`D=0x7700`, `dp+X=0x197`) needs the base wrap: bytes
185///   at `0x7797`/`0x7798`;
186/// - case `e1.e 8669` (`D=0xF400`, `dp+X=0xFF`) needs the linear high byte:
187///   `0xF4FF`/`0xF500`, not the page-wrapped `0xF400`.
188///
189/// Wrapping the base once and adding `offset` linearly satisfies both.
190fn read_direct_x<B: Bus>(cpu: &Cpu, bus: &mut B, address: u32, offset: u32) -> u8 {
191    let dp = u32::from(cpu.dp);
192    let addr = if cpu.e && (cpu.dp & 0x00FF) == 0 {
193        (dp | (address & 0x00FF)).wrapping_add(offset) & 0xFFFF
194    } else {
195        dp.wrapping_add(address).wrapping_add(offset) & 0xFFFF
196    };
197    bus.read(addr)
198}

200/// Absolute Indexed by X: `LDA $abs,X`.
201#[inline]
202pub fn absolute_indexed_x<B: Bus>(cpu: &mut Cpu, bus: &mut B, acc: Access) -> Addr24 {
203    let base_off = cpu.fetch_u16(bus);
204    let effective = base_off.wrapping_add(cpu.x);
205    match acc {
206        Access::Read => cpu.idle4(bus, base_off, effective),
207        Access::Write => cpu.io(bus),
208    }
209    // Carry into the data bank.
210    if effective < base_off {
211        make_addr(cpu.db.wrapping_add(1), effective)
212    } else {
213        make_addr(cpu.db, effective)
214    }
215}

241/// Stack Relative: `LDA $sr,S` — operand is a u8 offset added to SP.
242#[inline]
243pub fn stack_relative<B: Bus>(cpu: &mut Cpu, bus: &mut B) -> Addr24 {
244    let off = u16::from(cpu.fetch_u8(bus));
245    cpu.io(bus);
246    cpu.bank0_wrap = true;
247    make_addr(0, cpu.sp.wrapping_add(off))
248}
249
250/// Stack Relative Indirect Y: `LDA ($sr,S),Y`.
251#[inline]
252pub fn stack_relative_indirect_y<B: Bus>(cpu: &mut Cpu, bus: &mut B) -> Addr24 {
253    let off = u16::from(cpu.fetch_u8(bus));
254    cpu.io(bus);
255    let ptr_off = cpu.sp.wrapping_add(off);
256    let base_off = read_ptr16(bus, ptr_off);
257    cpu.io(bus);
258    let new_off = base_off.wrapping_add(cpu.y);
259    let bank = if new_off < base_off {
260        cpu.db.wrapping_add(1)
261    } else {
262        cpu.db
263    };
264    make_addr(bank, new_off)
265}

267/// Read an 8- or 16-bit operand from a given effective address.
268///
269/// Width is governed by the M flag for accumulator-targeted reads or by
270/// X for index-register-targeted reads. Callers pass the relevant flag
271/// query as a closure to avoid having two helpers.
272#[inline]
273pub fn read_byte<B: Bus>(bus: &mut B, addr: Addr24) -> u8 {
274    bus.read(addr)
275}
276
277/// Read 16 bits little-endian with the high byte carrying into the next
278/// bank (ares `readBank`: `addr + 1`). This is the bank/long-mode rule.
279///
280/// Direct-page and stack-relative accesses instead wrap the high byte
281/// within bank 0; the opcode handlers use [`crate::cpu::Cpu::read_word16`],
282/// which honors the per-instruction `bank0_wrap` latch. Kept here as the
283/// pure bank-carrying primitive for tests.
284#[inline]
285pub fn read_word<B: Bus>(bus: &mut B, addr: Addr24) -> u16 {
286    let lo = bus.read(addr);
287    let hi = bus.read(addr.wrapping_add(1) & 0x00FF_FFFF);
288    u16::from(lo) | (u16::from(hi) << 8)
289}
