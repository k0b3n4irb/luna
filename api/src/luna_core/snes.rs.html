26/// `serde` helper for a heap-boxed fixed byte array (`Box<[u8; N]>`),
27/// which `serde_bytes` does not cover directly. Used for the 128 KB WRAM.
28pub(crate) mod boxed_byte_array {
29    use serde::{Deserialize, Deserializer, Serializer};
30
31    /// Serialize `Box<[u8; N]>` as raw bytes (the `&Box` from serde's
32    /// `with` call site deref-coerces to this `&[u8; N]`).
33    pub(crate) fn serialize<S, const N: usize>(
34        data: &[u8; N],
35        serializer: S,
36    ) -> Result<S::Ok, S::Error>
37    where
38        S: Serializer,
39    {
40        serializer.serialize_bytes(&data[..])
41    }
42
43    /// Deserialize a byte blob back into `Box<[u8; N]>` (length must match).
44    pub(crate) fn deserialize<'de, D, const N: usize>(
45        deserializer: D,
46    ) -> Result<Box<[u8; N]>, D::Error>
47    where
48        D: Deserializer<'de>,
49    {
50        let bytes = <serde_bytes::ByteBuf>::deserialize(deserializer)?;
51        let arr: [u8; N] = bytes
52            .into_vec()
53            .try_into()
54            .map_err(|_| serde::de::Error::custom("byte array length mismatch"))?;
55        Ok(Box::new(arr))
56    }
57}

99    /// Total master cycles consumed since reset.
100    pub total_mclk: MCycles,

103    /// Real SPC700 + 64 KB ARAM + IPL ROM + mailboxes. Runs in
104    /// parallel with the main CPU at a 21 mclk : 1 spc-cycle ratio.
105    pub apu_real: Apu,
106    /// `true` once the SPC700 has hit an opcode our handler doesn't
107    /// implement (panic-caught). Subsequent reads of `$2140-$2143`
108    /// fall back to the cached state and the dumb mailbox stub takes
109    /// over for any further CPU writes so the game doesn't deadlock.
110    pub apu_panicked: bool,
111    /// Legacy heuristic mailbox stub — used only after the real APU

131    /// Video region as decoded from the cartridge header. Drives

136    /// 17-bit WRAM address counter accessed through the

140    pub wm_addr: u32,
141    /// Manual-mode joypad shift state.
142    ///
143    /// Per ares' `joypad.cpp` + Mesen2's `ControlManager`, when the

155    pub joypad2_shift: u16,

157    /// CPU memory-data register (open-bus latch): the last byte driven on

161    pub mdr: u8,

163    /// Optional CPU↔APU mailbox traffic log (`$2140-$2143`). When

170    /// Optional SA-1 MMIO traffic log (`$2200-$23FF`). When `Some`, every
171    /// CPU read/write of an SA-1 control/status register is appended as a
172    /// [`Sa1LogEvent`] for diagnosing the CPU↔SA-1 handshake (e.g. the

174    #[serde(skip)]
175    pub sa1_log: Option<Vec<Sa1LogEvent>>,
176
177    /// Optional CPU instruction trace. When `Some`, every call to

349/// Master cycles per PPU scanline on NTSC (1364 mclk = 4 dots × 341).
350pub const MCYCLES_PER_SCANLINE: u32 = 1364;
351
352/// Line-relative master-cycle position at which the once-per-scanline DRAM
353/// refresh halts the CPU. ares `cpu/timing.cpp:71` sets
354/// `dramRefreshPosition = 530 + 8 - dmaCounter()` in hcounter (master-clock)
355/// units; with the common `dmaCounter() == 0` this is 538. Cross-checked
356/// against Mesen2 (the refresh fired between line-positions 472 and 580 on the
357/// SMRPG boot, bracketing 538).
358const DRAM_REFRESH_POS: u32 = 538;
359
360/// Master cycles the CPU is halted for DRAM refresh each scanline. ares
361/// `cpu/timing.cpp:24-28` performs 5 refresh accesses of `step(6)+step(2)` =
362/// 8 mclk each → 40 mclk total. luna applies it as one lump stall (the CPU
363/// does no work; the APU/PPU/coproc keep running), shifting the CPU↔APU phase
364/// exactly as hardware does. Omitting this made luna's CPU run ~40 mclk/line
365/// fast vs ares/Mesen, accumulating CPU↔SPC drift (e.g. the SMRPG Akao
366/// timer-poll freeze).
367const DRAM_REFRESH_CYCLES: u32 = 40;

369/// Convert the running master-clock counter into the PPU's current
370/// (H, V) dot coordinate. 1364 master cycles per scanline; the
371/// scanline count depends on region (262 NTSC / 312 PAL). Each
372/// "dot" is 4 master cycles, so H is the line-relative master
373/// clock divided by 4 (range 0..340).
374#[inline]
375fn current_hv(mclk_total: u64, scanlines: u16) -> (u16, u16) {
376    let per_frame = u64::from(MCYCLES_PER_SCANLINE) * u64::from(scanlines);
377    let in_frame = mclk_total % per_frame;
378    let v = (in_frame / u64::from(MCYCLES_PER_SCANLINE)) as u16;
379    let h = ((in_frame % u64::from(MCYCLES_PER_SCANLINE)) / 4) as u16;
380    (h, v)
381}
382
383/// The exact horizontal master-clock position within the current scanline
384/// (0..1363) — Mesen2's `MemoryManager::GetHClock`, used verbatim as the
385/// Event Viewer's `Cycle`. Unlike [`current_hv`]'s H (which divides by 4 to a
386/// PPU dot for IRQ/HTIME comparison), this keeps full master-cycle precision
387/// so the overlay can place an event at the exact column (`x = hclock / 2`).
388#[inline]
389fn current_hclock(mclk_total: u64, scanlines: u16) -> u16 {
390    let per_frame = u64::from(MCYCLES_PER_SCANLINE) * u64::from(scanlines);
391    (mclk_total % per_frame % u64::from(MCYCLES_PER_SCANLINE)) as u16
392}

447impl Snes {
448    /// Build a new machine from a parsed cartridge.
449    ///
450    /// Returns [`UnsupportedMapper`] if the cartridge needs a coprocessor
451    /// luna does not yet emulate (S-DD1, SPC7110). All currently supported
452    /// mappers (`LoROM` / `HiROM` / `ExHiROM` / SA-1 / Super FX) succeed.
453    pub fn try_from_cartridge(cart: Cartridge) -> Result<Self, UnsupportedMapper> {
454        let sram_bytes = (cart.header.sram_size_kb as usize) * 1024;
455        let region = cart.header.region;
456        let mapper: Box<dyn Mapper + Send> = match cart.header.mapper_kind {
457            MapperKind::LoRom => Box::new(LoRomMapper::new(cart.rom, sram_bytes)),
458            kind @ (MapperKind::HiRom | MapperKind::ExHiRom) => {
459                Box::new(HiRomMapper::with_kind(kind, cart.rom, sram_bytes))
460            }
461            // SA-1 — phase-2: ROM banking + I-RAM + BW-RAM + multiplier
462            // MMIO wrapped in a [`Sa1Chip`] that also drives the SA-1's
463            // own 65C816 (released from reset by main-CPU writes to
464            // `$2200 CCNT`).
465            MapperKind::Sa1 => Box::new(Sa1Chip::new(Sa1Mapper::new(cart.rom, sram_bytes))),
466            // Super FX — the GSU is self-contained (no embedded 65C816), so
467            // the whole chip lives in `SuperFxMapper`, driven by the
468            // `step_coproc` hook. The Game Pak **work** RAM (the GSU's plot
469            // target) is NOT the header SRAM byte — that byte is battery
470            // *save* RAM. The work-RAM size comes from the extended-header
471            // expansion-RAM byte `$FFBD` (`1024 << n`); GSU carts that don't
472            // carry it default to 64 KB (Mesen2 `BaseCartridge.cpp`). YI is
473            // 32 KB, Doom / Stunt Race 64 KB, Star Fox 64 KB (default).
474            // Over-allocating wraps GSU RAM/RAMBR addressing wrong, so size
475            // it faithfully rather than to a 128 KB upper bound.
476            MapperKind::SuperFx => {
477                let ram = if cart.header.expansion_ram_kb > 0 {
478                    cart.header.expansion_ram_kb as usize * 1024
479                } else {
480                    0x1_0000
481                };
482                Box::new(SuperFxMapper::new(cart.rom, ram))
483            }
484            // DSP-1 (Super Mario Kart, Pilotwings) — a base ROM/SRAM map
485            // (HiROM or LoROM per the board) plus the NEC uPD7725 chip,
486            // fed the cartridge's `dsp1b.rom` firmware. Without firmware
487            // the chip stays inert (the game still runs).
488            MapperKind::Dsp1 => {
489                let hirom = cart.header.dsp_hirom;
490                let firmware = cart.coprocessor_firmware().map(<[u8]>::to_vec);
491                Box::new(Dsp1Mapper::new(
492                    cart.rom,
493                    sram_bytes,
494                    firmware.as_deref(),
495                    hirom,
496                ))
497            }
498            // S-DD1 — graphics decompression chip (Star Ocean, SF Alpha 2);
499            // LoROM-based MMC, the whole chip in `Sdd1Mapper`.
500            MapperKind::Sdd1 => Box::new(Sdd1Mapper::new(cart.rom, sram_bytes)),
501            other => return Err(UnsupportedMapper(other)),
502        };
503
504        let mut ppu = Ppu::new();
505        // Flip STAT78's region bit (bit 4) for PAL carts. NTSC and
506        // "Unknown" land at the same default (bit clear).
507        if matches!(region, luna_cartridge::Region::Pal) {
508            ppu.stat78 |= 0x10;
509        }
510
511        Ok(Self {
512            cpu: Cpu::new(),
513            ppu,
514            dma: Dma::new(),
515            cpu_regs: CpuRegs::new(),

715    /// Run the CPU reset sequence: read the reset vector at `$00:FFFC`
716    /// via the bus and load `PC`.
717    pub fn reset(&mut self) {
718        // 0. Cartridge coprocessor (Super FX / SA-1 / …): re-power it
719        //    first, BEFORE the main CPU re-reads its reset vector. The
720        //    reset line on real hardware resets the cart chip too, not
721        //    just the main CPU. Doing this first guarantees a mid-run GSU
722        //    isn't still owning the ROM bus when the CPU fetches the
723        //    vector. ROM and battery SRAM persist; the coproc registers,
724        //    internal RAM and its own CPU return to power-on. Leaving the
725        //    GSU mid-execution here froze Doom on `Reset` (it never
726        //    rebooted). No-op for plain LoROM / HiROM carts.
727        self.mapper.reset();
728
729        // 1. CPU: re-read the reset vector through the bus. VRAM / WRAM /
730        //    SRAM persist across a reset (real hardware doesn't clear them).
731        let scanlines = self.region_scanlines();
732        let ppu_line_snapshot = self.ppu_line;
733        let vblank_start_snapshot = vblank_start_line(self.region);
734        let cpu_pc_snapshot = (u32::from(self.cpu.pb) << 16) | u32::from(self.cpu.pc);
735        {
736            let Self {
737                cpu,
738                ppu,
739                dma,
740                cpu_regs,
741                apu_real,
742                apu_stub_fallback,
743                apu_panicked,
744                wram,
745                mapper,
746                fast_rom,
747                nmi_pending,
748                irq_pending,
749                total_mclk,
750                wm_addr,
751                joypad_strobe,
752                joypad1_shift,
753                joypad2_shift,
754                mdr,
755                mailbox_log,
756                sa1_log,
757                mem_trace_log,
758                nocash_log,
759                ..
760            } = self;
761            let mut bus = SnesBus {
762                wram,
763                mapper: mapper.as_mut(),
764                ppu,
765                dma,
766                cpu_regs,
767                apu_real,
768                apu_stub_fallback,
769                apu_panicked,
770                fast_rom: *fast_rom,
771                nmi: nmi_pending,
772                irq: irq_pending,
773                mclk_total: total_mclk,

775                scpu_mar: 0,

793            cpu.reset(&mut bus);
794        }
795
796        // 2. Power-on-style reset of the rest of the system. CRITICAL: the
797        //    APU. A reset re-runs the game's boot sound-driver upload, whose
798        //    IPL-ROM handshake deadlocks unless the SPC700 is back at its
799        //    ready state — leaving the APU running its driver made `Reset`
800        //    appear to do nothing (the main CPU spun on the upload forever).
801        //    CPU registers ($42xx), master clock, frame/scanline counters
802        //    and pending interrupts also return to power-on; VRAM/WRAM/SRAM

811        self.frame_count = 0;
812        self.nmis_serviced = 0;
813        self.nmi_pending = false;
814        self.irq_pending = false;

816        self.joypad_strobe = false;
817        self.joypad1_shift = 0;
818        self.joypad2_shift = 0;

821    /// Execute one CPU instruction. Returns the master-cycle cost of
822    /// that instruction (accumulated through [`Bus::io_cycle`]).
823    pub fn step(&mut self) -> MCycles {
824        let before = self.total_mclk;
825        // Capture the pre-instruction snapshot for the CPU tracer
826        // before the destructure below moves `cpu` out of `self`.
827        if let Some(log) = self.cpu_trace_log.as_mut() {
828            if log.events.len() < log.max_events {
829                log.events.push(CpuTraceEvent {
830                    mclk_total: before,
831                    pc_full: (u32::from(self.cpu.pb) << 16) | u32::from(self.cpu.pc),
832                    a: self.cpu.a,
833                    x: self.cpu.x,
834                    y: self.cpu.y,
835                    sp: self.cpu.sp,
836                    p: self.cpu.p.bits(),
837                    db: self.cpu.db,
838                    dp: self.cpu.dp,
839                    e: self.cpu.e,
840                });
841            }
842        }
843        let scanlines = self.region_scanlines();
844        let ppu_line_snapshot = self.ppu_line;
845        let vblank_start_snapshot = vblank_start_line(self.region);
846        let cpu_pc_snapshot = (u32::from(self.cpu.pb) << 16) | u32::from(self.cpu.pc);
847        let (rb_line, rb_mil, rb_fc, rb_ns);
848        {
849            let Self {
850                cpu,
851                ppu,
852                dma,
853                cpu_regs,
854                apu_real,
855                apu_stub_fallback,
856                apu_panicked,
857                wram,
858                mapper,
859                fast_rom,
860                nmi_pending,
861                irq_pending,
862                total_mclk,
863                wm_addr,
864                joypad_strobe,
865                joypad1_shift,
866                joypad2_shift,
867                mdr,
868                mailbox_log,
869                sa1_log,
870                mem_trace_log,
871                nocash_log,
872                mcycles_in_line,
873                frame_count,
874                nmis_serviced,
875                ..
876            } = self;
877            let mut bus = SnesBus {
878                wram,
879                mapper: mapper.as_mut(),
880                ppu,
881                dma,
882                cpu_regs,
883                apu_real,
884                apu_stub_fallback,
885                apu_panicked,
886                fast_rom: *fast_rom,
887                nmi: nmi_pending,
888                irq: irq_pending,
889                mclk_total: total_mclk,
890                scanlines_per_frame: scanlines,
891                scpu_mar: 0,
892                ppu_line: ppu_line_snapshot,
893                mcycles_in_line: *mcycles_in_line,
894                frame_count: *frame_count,
895                nmis_serviced: *nmis_serviced,
896                sched_enabled: true,
897                vblank_start_line: vblank_start_snapshot,
898                cpu_pc_full: cpu_pc_snapshot,
899                mailbox_log,
900                sa1_log,
901                mem_trace_log,
902                nocash_log,
903                wm_addr,
904                joypad_strobe,
905                joypad1_shift,
906                joypad2_shift,
907                mdr,
908            };
909            // The scanline scheduler now advances per bus access inside
910            // `bus.io_cycle` (per-scanline rendering: each line is drawn
911            // with the register state live at that point, not the end-of-
912            // instruction snapshot). Read the live cursor back out below.
913            cpu.step(&mut bus);
914            rb_line = bus.ppu_line;
915            rb_mil = bus.mcycles_in_line;
916            rb_fc = bus.frame_count;
917            rb_ns = bus.nmis_serviced;
918        }
919        self.ppu_line = rb_line;
920        self.mcycles_in_line = rb_mil;
921        self.frame_count = rb_fc;
922        self.nmis_serviced = rb_ns;
923
924        let consumed = self.total_mclk - before;
925
926        // The cartridge coprocessor (SA-1 / Super FX / DSP-1 / …) now
927        // advances per bus access inside `bus.io_cycle` (Phase 1
928        // cycle-accuracy milestone), in lockstep with the APU and the
929        // PPU scanline scheduler — not as one end-of-instruction lump.
930        // This also removes the old DMA double-charge: the coproc used
931        // to be advanced both per-byte during a transfer (DmaBusView::
932        // tick) *and* again by this lump, running it ~2× too fast
933        // through DMA-heavy code.
934
935        // Sample the coprocessor's level-driven IRQ line into the CPU's
936        // dedicated *level* input (ares `coprocessor/sa1/io.cpp:134-163`,
937        // Super FX `coprocessor/superfx/io.cpp:18-22`): the device holds
938        // the S-CPU IRQ pin until the program acks (Super FX `$3031` read,
939        // SA-1 SIC), so this is a level, not an edge. Sampling it fresh
940        // each instruction (set AND clear) — rather than re-arming the
941        // sticky `pending_irq` — means a single coproc IRQ is serviced
942        // exactly once: re-arming `pending_irq` while `I` masked the
943        // handler used to leave a stale latch that double-serviced after
944        // the `RTI`, corrupting Star Fox's object table.
945        //
946        // The **H/V-timer IRQ is also a level**, not an edge (ares
947        // `cpu/irq.cpp`: `status.irqLine` held until `timeup()`/$4211 read;
948        // Mesen2 `_irqFlag` -> `SetIrqSource(Ppu)` held until `ClearIrqSource`).
949        // `cpu_regs.irq_flag` IS that level — raised at the H/V coincidence in
950        // `poll_hv_irq`, lowered when the program reads $4211. Sampling it into
951        // the CPU's level input (instead of the old one-shot `pending_irq`
952        // edge) means an H/V IRQ that fires while `I` is masked is HELD and
953        // serviced the moment `I` clears, never lost. Doom chains H+V IRQs
954        // (re-arming VTIME) to write INIDISP every frame; the edge model
955        // dropped ~64% of those writes -> its letterbox border flickered.
956        self.cpu
957            .set_irq_line(self.cpu_regs.irq_flag || self.mapper.coproc_main_irq_pending());
958
959        // Apply interrupt edges the scanline scheduler latched during this
960        // instruction's bus accesses. Deferring the CPU poke to the
961        // instruction boundary keeps NMI/IRQ delivery timing identical to
962        // the old end-of-step model — the CPU only services interrupts
963        // between instructions anyway.
964        if self.nmi_pending {
965            self.nmi_pending = false;
966            self.cpu.trigger_nmi();
967        }
968        if self.irq_pending {
969            self.irq_pending = false;
970            self.cpu.trigger_irq();
971        }
972
973        consumed
974    }

1156    /// Side-effect-free, no-clock debug peek of `count` bytes from `bank:offset`
1157    /// (16-bit offset wraps). For memory inspectors: unlike a real bus read it
1158    /// charges **no** `io_cycle` (so it never advances the master clock / APU)
1159    /// and never touches MMIO (so it never toggles the OPHCT/OPVCT or BG-scroll
1160    /// latches, clears the NMI/IRQ flags, or advances VMADD/OAMADD/the WRAM
1161    /// port). WRAM and ROM/SRAM/coproc-work-RAM return their real bytes; the
1162    /// `$2000-$5FFF` register band returns `0`.
1163    pub fn dbg_peek_bytes(&mut self, bank: u8, offset: u16, count: usize) -> Vec<u8> {
1164        let mut out = Vec::with_capacity(count);
1165        for i in 0..count {
1166            let off = offset.wrapping_add(i as u16);
1167            let v = if matches!(bank, 0x00..=0x3F | 0x80..=0xBF) && off < 0x2000 {
1168                // Low-RAM WRAM mirror.
1169                self.wram[usize::from(off)]
1170            } else if matches!(bank, 0x7E..=0x7F) {
1171                // Full WRAM ($7E-$7F).
1172                self.wram[(usize::from(bank - 0x7E) << 16) | usize::from(off)]
1173            } else if matches!(bank, 0x00..=0x3F | 0x80..=0xBF) && (0x3000..=0x37FF).contains(&off)
1174            {
1175                // SA-1 I-RAM ($3000-$37FF) is side-effect-free — read it from
1176                // the mapper. (Lumping it into the register band below made
1177                // every I-RAM peek return 0, which silently broke SA-1 I-RAM
1178                // inspection and cross-emulator I-RAM differentials.)
1179                self.mapper.read(make_addr(bank, off)).unwrap_or(0xFF)
1180            } else if matches!(bank, 0x00..=0x3F | 0x80..=0xBF) && (0x2000..=0x5FFF).contains(&off)
1181            {
1182                // PPU/APU/CPU/coproc register band — read side effects, so 0.
1183                0
1184            } else {
1185                // ROM / SRAM / coproc work-RAM — side-effect-free here.
1186                self.mapper.read(make_addr(bank, off)).unwrap_or(0xFF)
1187            };
1188            out.push(v);
1189        }
1190        out
1191    }
1192
1193    /// Debug poke: write `data` to WRAM (`$7E-$7F` or the `$00-3F`/`$80-BF`
1194    /// low-RAM mirror) directly, bypassing the bus. For injecting a test
1195    /// state without a full save-state. Addresses outside WRAM are ignored
1196    /// (poking ROM/registers is meaningless / unsafe). Returns bytes written.
1197    pub fn dbg_poke_bytes(&mut self, bank: u8, offset: u16, data: &[u8]) -> usize {
1198        let mut written = 0;
1199        for (i, &b) in data.iter().enumerate() {
1200            let off = offset.wrapping_add(i as u16);
1201            let idx = if matches!(bank, 0x7E..=0x7F) {
1202                (usize::from(bank - 0x7E) << 16) | usize::from(off)
1203            } else if matches!(bank, 0x00..=0x3F | 0x80..=0xBF) && off < 0x2000 {
1204                usize::from(off)
1205            } else {
1206                continue;
1207            };
1208            self.wram[idx] = b;
1209            written += 1;
1210        }
1211        written
1212    }

1294    /// Resolve `addr` against the WRAM regions; returns the in-array
1295    /// offset if it maps to WRAM, else `None`.
1296    fn wram_offset(addr: Addr24) -> Option<usize> {
1297        let bank = bank_of(addr);
1298        let offset = offset_of(addr);
1299        // LowRAM mirror: banks $00-$3F and $80-$BF, offsets $0000-$1FFF.

1304        if matches!(bank, 0x7E..=0x7F) {
1305            let high = usize::from(bank - 0x7E) << 16;
1306            return Some(high | usize::from(offset));
1307        }
1308        None
1309    }

1405/// DMA-side view of the system: holds the minimum needed for a sync
1406/// transfer (WRAM + cartridge + PPU) **without** carrying a reference
1407/// to the [`Dma`] controller, so the controller can borrow itself
1408/// mutably while running.
1409struct DmaBusView<'a> {
1410    wram: &'a mut [u8; 0x20000],
1411    mapper: &'a mut dyn Mapper,
1412    ppu: &'a mut Ppu,
1413    /// Shared 17-bit WRAM-port address (`$2181-$2183` WMADD), so a DMA to
1414    /// `$2180` (WMDATA) writes WRAM and auto-increments — the same state
1415    /// the CPU port uses.
1416    wm_addr: &'a mut u32,
1417    /// Optional DMA→VRAM transfer-time trace (moved in from the [`Dma`]
1418    /// controller for the duration of one MDMA burst). `None` = off.
1419    dma_trace: Option<&'a mut DmaTraceLog>,
1420    /// A-bus source address of the most recent `read_a` — paired with
1421    /// the immediately-following `write_b` to record a VRAM byte's
1422    /// source (DMA reads then writes each byte in lockstep, A→B).
1423    last_a_addr: u32,
1424    /// Frame / scanline / vblank snapshot at the start of this DMA burst,
1425    /// stamped onto each `DmaTraceEvent` for per-VBlank bucketing.
1426    trace_frame: u64,
1427    trace_line: u16,
1428    trace_blank: bool,
1429    /// Exact horizontal master-clock (0..1363) at this DMA burst/line,
1430    /// stamped onto each `DmaTraceEvent` so the Event Viewer can plot
1431    /// `(hclock, line)` at full column precision (Mesen2 `GetHClock`).
1432    trace_hclock: u16,
1433    /// DMA channel (0-7) currently driving the transfer — set by the
1434    /// controller via [`DmaBus::set_active_channel`] before each channel's
1435    /// segment, so captured B-bus writes carry their source channel
1436    /// (Mesen2 `dma->GetActiveChannel()`).
1437    dma_channel: u8,
1438}
1439
1440impl DmaBus for DmaBusView<'_> {
1441    fn read_a(&mut self, addr: Addr24) -> u8 {
1442        // Remember this byte's source so the paired write_b (A→B runs
1443        // read-then-write per byte) can record where a VRAM byte came from.
1444        self.last_a_addr = addr;
1445        if let Some(o) = SnesBus::wram_offset(addr) {
1446            return self.wram[o];
1447        }
1448        // A-side ROM / SRAM reads via mapper; everything else is open
1449        // bus until those subsystems land.
1450        self.mapper.read(addr).unwrap_or(0xFF)
1451    }
1452
1453    fn write_a(&mut self, addr: Addr24, value: u8) {
1454        if let Some(o) = SnesBus::wram_offset(addr) {
1455            self.wram[o] = value;
1456            return;
1457        }
1458        // SRAM writes go through the mapper; ROM writes drop.
1459        let _ = self.mapper.write(addr, value);
1460    }
1461
1462    fn read_b(&mut self, b_offset: u8) -> u8 {
1463        // B-bus range $00-$3F = PPU. $80 = WMDATA ($2180): a DMA reading
1464        // WRAM via the port returns WRAM[WMADD] and auto-increments — same
1465        // as the CPU port (`read_inner` $2180). APU $40-$43 is still open
1466        // bus on the DMA path.
1467        if b_offset <= 0x3F {
1468            self.ppu.read(b_offset)
1469        } else if b_offset == 0x80 {
1470            let a = (*self.wm_addr & 0x1FFFF) as usize;
1471            let v = self.wram[a];
1472            *self.wm_addr = (*self.wm_addr + 1) & 0x1FFFF;
1473            v
1474        } else {
1475            0xFF
1476        }
1477    }
1478
1479    fn write_b(&mut self, b_offset: u8, value: u8) {
1480        if b_offset <= 0x3F {
1481            // DMA B-bus trace: capture (source → VMADD → byte) BEFORE the
1482            // write, since the $2119 (high) write auto-increments VMADD.
1483            // Captures EVERY PPU B-bus write ($2100-$213F), not just the
1484            // VRAM ports — the Event Viewer categorises OAM ($2104),
1485            // CGRAM ($2122), etc. DMA writes too. `vram_word` is only
1486            // meaningful for the $2118/$2119 ports; VRAM-only consumers
1487            // (the CLI `--dma-trace` CSV) filter on `b_offset`.
1488            if let Some(log) = self.dma_trace.as_mut() {
1489                if log.events.len() < log.max_events {
1490                    log.events.push(DmaTraceEvent {
1491                        src_full: self.last_a_addr,
1492                        vram_word: self.ppu.vram.address,
1493                        b_offset,
1494                        value,
1495                        channel: self.dma_channel,
1496                        frame: self.trace_frame,
1497                        line: self.trace_line,
1498                        hclock: self.trace_hclock,
1499                        blank: self.trace_blank,
1500                        force_blank: self.ppu.inidisp & 0x80 != 0,
1501                    });
1502                }
1503            }
1504            // CGDATA ($2122) is never dropped during active display (handled
1505            // at the source in Ppu::write — ares io.cpp:55-60); VRAM/OAM still
1506            // drop via their own `active_display` gates.
1507            self.ppu.write(b_offset, value);
1508        } else if matches!(b_offset, 0x80..=0x83) {
1509            // WRAM port ($2180-$2183), mirroring the CPU path: $80 WMDATA
1510            // writes WRAM[WMADD]++ ; $81-$83 set the 17-bit WMADD. Games
1511            // (e.g. Kirby Super Star's boot) DMA ROM→WRAM through $2180 to
1512            // populate low WRAM — dropping it left WRAM $0000+ zero, so the
1513            // boot's `JMP $000E` hit a `$00` (BRK) and crashed.
1514            match b_offset {
1515                0x80 => {
1516                    let a = (*self.wm_addr & 0x1FFFF) as usize;
1517                    self.wram[a] = value;
1518                    *self.wm_addr = (*self.wm_addr + 1) & 0x1FFFF;
1519                }
1520                0x81 => *self.wm_addr = (*self.wm_addr & !0x000FF) | u32::from(value),
1521                0x82 => *self.wm_addr = (*self.wm_addr & !0x0FF00) | (u32::from(value) << 8),
1522                _ => *self.wm_addr = (*self.wm_addr & !0x10000) | (u32::from(value & 1) << 16),
1523            }
1524        }
1525    }
1526
1527    fn tick(&mut self, mcycles: u32) {
1528        // Forward the per-byte DMA tick into the mapper so SA-1 (and
1529        // future coprocs) advance at DMA cadence instead of being
1530        // frozen until the next main-CPU instruction step. Without
1531        // this, ~544-byte OAM DMAs leave the SA-1 paused for ~4 kHz
1532        // mclks and then catch up in one ~700-instruction burst —
1533        // ruining the synchronisation the demo's `$3001 SA1_SYNC`
1534        // handshake depends on.
1535        //
1536        // No S-CPU bus access drives this DMA-side tick (the CPU is halted
1537        // for the transfer), so there is no shared-bus `conflict()` partner —
1538        // pass `mar = 0` (the ares `dma.cpp` DMA-vs-coproc contention is a
1539        // separate, finer refinement).
1540        self.mapper.step_coproc(mcycles, 0);
1541    }
1542
1543    fn set_active_channel(&mut self, channel: u8) {
1544        self.dma_channel = channel;
1545    }
1546}

1591        let force_blank = self.ppu.inidisp & 0x80 != 0;
1592        if let Some(log) = self.mem_trace_log.as_mut() {
1593            if log.events.len() >= log.max_events {
1594                return;
1595            }
1596            let addr_full = if matches!(kind, MemEventKind::NmiSignal) {
1597                0x00_4210
1598            } else {
1599                0x00_4211
1600            };
1601            log.events.push(MemTraceEvent {
1602                mclk_total: mclk,
1603                pc_full: pc,
1604                addr_full,
1605                kind,
1606                value,
1607                line,
1608                hclock: current_hclock(mclk, self.scanlines_per_frame),
1609                blank,
1610                force_blank,
1611            });
1612        }
1613    }

1616impl SnesBus<'_> {
1617    /// Advance the scanline scheduler by `mcycles`, firing per-line events
1618    /// at each boundary. Called from [`Bus::io_cycle`] mid-instruction.
1619    ///
1620    /// Phase 4: the advance is split at scanline boundaries so the H/V-IRQ
1621    /// can be polled over the H-range each chunk covers *within* one line
1622    /// (dot-precise), instead of only at the boundary. Boundary crossing
1623    /// is otherwise identical to before (chunks never overshoot a line, so
1624    /// the per-line events in `sched_one_line` still fire exactly once).
1625    /// Returns the total HDMA master-cycle stall accumulated across the
1626    /// line crossings in this advance (Phase 4); the caller charges it.
1627    fn sched_advance(&mut self, mcycles: u32) -> u32 {
1628        let mut remaining = mcycles;
1629        let mut stall = 0u32;
1630        while remaining > 0 {
1631            let room = MCYCLES_PER_SCANLINE - self.mcycles_in_line;
1632            let chunk = remaining.min(room);
1633            let lo = self.mcycles_in_line;
1634            let hi = lo + chunk;
1635            // Poll the IRQ over [lo, lo+chunk) on the CURRENT line before
1636            // any boundary crossing advances `ppu_line` (the V-counter).
1637            self.poll_hv_irq(lo, hi);
1638            // DRAM refresh: once per scanline the CPU is halted ~40 mclk
1639            // (ares `cpu/timing.cpp`). Fire when this chunk crosses the
1640            // refresh position — like `poll_hv_irq`'s htime, the half-open
1641            // [lo, hi) covers each line position once, so this triggers
1642            // exactly once per line with no persistent flag. Charged as a
1643            // stall the caller re-advances every subsystem by, so the APU/PPU
1644            // run during the halt (the CPU↔APU phase shift hardware produces).
1645            if lo <= DRAM_REFRESH_POS && DRAM_REFRESH_POS < hi {
1646                stall += DRAM_REFRESH_CYCLES;
1647            }
1648            self.mcycles_in_line += chunk;
1649            remaining -= chunk;
1650            if self.mcycles_in_line >= MCYCLES_PER_SCANLINE {
1651                self.mcycles_in_line -= MCYCLES_PER_SCANLINE;
1652                stall += self.sched_one_line();
1653            }
1654        }
1655        stall
1656    }
1657
1658    /// Dot-precise H/V-counter IRQ poll over the half-open master-cycle
1659    /// range `[mclk_lo, mclk_hi)` of the current scanline (`ppu_line` =
1660    /// V-counter). Mirrors ares `cpu/irq.cpp:18-31` and Mesen2
1661    /// `InternalRegisters::UpdateIrqLevel`: the trigger is a level
1662    /// `(!hirq || h == htime) && (!virq || v == vtime)` whose **rising
1663    /// edge** sets the flag. Within a line H is monotonic, so the H-IRQ
1664    /// level is a one-dot pulse → exactly one edge per line at
1665    /// `h == htime`; V-only is a whole-line block → one edge at the
1666    /// matching line's start (`mclk_lo == 0`). The half-open interval
1667    /// crosses each dot once, so this needs no persistent level state and
1668    /// never double-fires. NMITIMEN bit 4 = H-IRQ enable, bit 5 = V-IRQ
1669    /// enable; HTIME/VTIME are 9-bit (dots / lines).
1670    ///
1671    /// Deferred (gaps, not regressions): the ares "no trigger on the last
1672    /// dot of the field" guard, the htime==0 / detect→assert delay, and
1673    /// the $4211 TIMEUP hold window (mirror of the RDNMI hold).
1674    fn poll_hv_irq(&mut self, mclk_lo: u32, mclk_hi: u32) {
1675        let hirq = self.cpu_regs.nmitimen & 0x10 != 0;
1676        let virq = self.cpu_regs.nmitimen & 0x20 != 0;
1677        if !hirq && !virq {
1678            return;
1679        }
1680        // V gate: with V-IRQ enabled the line must match, else the level
1681        // can never rise on this scanline.
1682        if virq && self.ppu_line != self.cpu_regs.vtime {
1683            return;
1684        }
1685        let fire = if hirq {
1686            // H-IRQ: fire when this chunk crosses the `htime` dot.
1687            let trig = u32::from(self.cpu_regs.htime) * 4;
1688            mclk_lo <= trig && trig < mclk_hi
1689        } else {
1690            // V-only: H is irrelevant; fire at the matching line's start.
1691            mclk_lo == 0
1692        };
1693        if fire {
1694            // Raise the held level only (ares `status.irqLine`, Mesen `_irqFlag`).
1695            // `irq_flag` stays set until the program reads $4211; the CPU samples
1696            // it as a *level* via `set_irq_line` at each instruction boundary, so
1697            // the IRQ is never lost to `I`-masking the way the old one-shot
1698            // `*self.irq` edge was (it coalesced/dropped ~64% of Doom's chained
1699            // H+V writes). No edge latch is set here any more.
1700            self.cpu_regs.irq_flag = true;
1701            self.trace_irq_signal(MemEventKind::IrqSignal, self.cpu_regs.nmitimen);
1702        }
1703    }
1704
1705    /// Cross one scanline boundary, applying the per-line PPU events.
1706    /// Mirrors the former `Snes::advance_one_scanline`, but raises NMI/IRQ
1707    /// via the bus `nmi`/`irq` latches (the CPU is borrowed here; `step`
1708    /// applies the edge at the instruction boundary).
1709    /// Returns the master-cycle cost of any HDMA performed on this line
1710    /// crossing (frame-start setup + per-line transfer), so the caller can
1711    /// charge the CPU the stall (Phase 4).
1712    fn sched_one_line(&mut self) -> u32 {
1713        let vblank_start = self.vblank_start_line;
1714        let scanlines = self.scanlines_per_frame;
1715        let mut hdma_stall = 0u32;
1716
1717        // Super Scope light-gun: when the CRT beam reaches the aimed scanline,
1718        // the gun strobes IOBit, latching the PPU H/V counters (OPHCT/OPVCT)
1719        // the game reads back for the beam position.
1720        if let Some((h, v)) = self.cpu_regs.scope_latch_target(self.ppu_line) {
1721            self.ppu.latch_counters(h, v);
1722        }
1723
1724        // Render the visible line that just finished, with its end-of-line
1725        // register state (HDMA for the next line fires after the increment).
1726        if self.ppu_line < vblank_start {
1727            self.ppu
1728                .render_current_scanline(self.ppu_line, luna_ppu::RenderOptions::default());
1729            // Latch that this frame showed visible content if the line was
1730            // scanned out un-blanked. Front-ends use the per-frame snapshot
1731            // (not the instantaneous INIDISP bit, which a Super FX title
1732            // re-asserts every VBlank to prep its next buffer) to decide
1733            // whether to publish the frame.
1734            if self.ppu.inidisp & 0x80 == 0 {
1735                self.ppu.frame_visible_content_accum = true;
1736            }
1737        }
1738
1739        self.ppu_line += 1;
1740        if self.ppu_line == vblank_start {
1741            // Entering VBlank: latch the $4210 NMI flag + HVBJOY.7.
1742            self.cpu_regs.nmi_flag = true;
1743            self.cpu_regs.hvbjoy |= 0x80;
1744            if self.cpu_regs.nmitimen & 0x80 != 0 {
1745                *self.nmi = true;
1746                self.nmis_serviced = self.nmis_serviced.saturating_add(1);
1747                self.trace_irq_signal(MemEventKind::NmiSignal, self.cpu_regs.nmitimen);
1748            }
1749            // OAM address auto-reset (ares `object.cpp:31-32`), unless
1750            // forced-blank.
1751            if self.ppu.inidisp & 0x80 == 0 {
1752                self.ppu.oam.reload_address_from_latch();
1753            }
1754            // Joypad auto-read latch + manual-mode shift reload.
1755            self.cpu_regs.latch_joypad_auto_read();
1756            if self.cpu_regs.nmitimen & 0x01 != 0 {
1757                // The hardware auto-read strobes and clocks BOTH controllers
1758                // 16 times, leaving their shift registers EXHAUSTED. A
1759                // subsequent *un-strobed* manual read of $4016/$4017 then
1760                // returns 1 on the data line (ares `controller/gamepad`
1761                // `data()` returns 1 past 16 clocks; ares `cpu/io.cpp:16,20`
1762                // `data.bit(0,1) = controllerPort.data()`), NOT the button
1763                // bits. Reloading the latched value here made idle reads
1764                // return B=0, so any game that polls $4016/$4017.d0 for the
1765                // idle-high data line took the wrong branch — e.g. Donkey
1766                // Kong Country's controller/autofire routine at $80:C13D /
1767                // $80:C16C, which then corrupted its debounce counters and
1768                // looped the attract/game-start sequence.
1769                *self.joypad1_shift = 0xFFFF;
1770                *self.joypad2_shift = 0xFFFF;
1771            }
1772        } else if self.ppu_line == vblank_start + 3 {
1773            self.cpu_regs.clear_joypad_busy();
1774        }
1775
1776        // H/V-counter IRQ is now polled dot-precisely per bus access in
1777        // `poll_hv_irq` (Phase 4), not latched at the scanline boundary.
1778
1779        if self.ppu_line >= scanlines {
1780            // Frame wrap.
1781            self.ppu_line = 0;
1782            self.cpu_regs.hvbjoy &= !0x80;
1783            // P1 — faithful `nmiLine`: ares clears it when `nmiValid` falls
1784            // (vcounter < vdisp), i.e. at VBlank end, NOT only on a $4210 read
1785            // (irq.cpp `nmiLine = nmiValid`). Without this the flag stays
1786            // stale-true into the next frame — the latent bug a faithful
1787            // late-NMI-enable (P2) would turn into a spurious NMI (the SMRPG
1788            // black-screen tripwire). The actual NMI is still fired at VBlank
1789            // entry via `*self.nmi`; this only fixes what `$4210` reads outside
1790            // VBlank (now 0, matching hardware).
1791            self.cpu_regs.nmi_flag = false;
1792            self.frame_count = self.frame_count.saturating_add(1);
1793            // Snapshot whether the frame that just completed showed any
1794            // visible content, paired with the frame counter bump so a
1795            // front-end polling at this boundary reads a consistent value.
1796            self.ppu.latch_frame_content();
1797            // Interlace field parity flips every frame at the V-counter wrap
1798            // (ares counter/inline.hpp:32), exposed at STAT78 bit 7. Phase A:
1799            // flag only — no vertical doubling yet.
1800            self.ppu.field = !self.ppu.field;
1801            let mut view = DmaBusView {
1802                wram: &mut *self.wram,
1803                mapper: &mut *self.mapper,
1804                ppu: &mut *self.ppu,
1805                wm_addr: &mut *self.wm_addr,
1806                // hdma_init only reads table headers/pointers (A-bus); it makes
1807                // no B-bus register writes, so there is nothing to trace here.
1808                // The per-scanline transfers below are what the Event Viewer
1809                // captures.
1810                dma_trace: None,
1811                last_a_addr: 0,
1812                trace_frame: self.frame_count,
1813                trace_line: self.ppu_line,
1814                trace_blank: self.ppu_line >= self.vblank_start_line,
1815                trace_hclock: current_hclock(*self.mclk_total, self.scanlines_per_frame),
1816                dma_channel: 0,
1817            };
1818            hdma_stall += self.dma.hdma_init(&mut view);
1819        }
1820
1821        // HDMA on every visible scanline (end-of-HBlank ordering).
1822        if self.ppu_line < vblank_start {
1823            // Trace HDMA register writes for the Event Viewer, exactly like
1824            // MDMA: hdma_run_line stamps each channel via set_active_channel
1825            // (HdmaChannelFlag | ch, Mesen2 SnesDmaController.cpp:264), and the
1826            // shared DmaBusView::write_b records the DmaTraceEvent. The trace
1827            // log is moved into the view for the line and returned after.
1828            let mut trace = self.dma.dma_trace.take();
1829            let mut view = DmaBusView {
1830                wram: &mut *self.wram,
1831                mapper: &mut *self.mapper,
1832                ppu: &mut *self.ppu,
1833                wm_addr: &mut *self.wm_addr,
1834                dma_trace: trace.as_mut(),
1835                last_a_addr: 0,
1836                trace_frame: self.frame_count,
1837                trace_line: self.ppu_line,
1838                trace_blank: self.ppu_line >= self.vblank_start_line,
1839                trace_hclock: current_hclock(*self.mclk_total, self.scanlines_per_frame),
1840                dma_channel: 0,
1841            };
1842            hdma_stall += self.dma.hdma_run_line(&mut view);
1843            self.dma.dma_trace = trace;
1844        }
1845        hdma_stall
1846    }
1847}

1850    fn read(&mut self, addr: Addr24) -> u8 {
1851        let value = self.read_inner(addr);
1852        // Latch the byte on the CPU data bus (ares' `r.mdr`): open-bus
1853        // reads below return this. An open-bus read returns the prior
1854        // MDR (read_inner hands back `*self.mdr`), so this is idempotent
1855        // for those — it only changes on a real fetch.
1856        *self.mdr = value;
1857        self.trace_mem_access(addr, MemEventKind::Read, value);
1858        value
1859    }
1860    fn write(&mut self, addr: Addr24, value: u8) {
1861        // A write drives the data bus too, so it updates the MDR.
1862        *self.mdr = value;
1863        self.trace_mem_access(addr, MemEventKind::Write, value);
1864        self.write_inner(addr, value);
1865    }
1866    fn io_cycle(&mut self, mcycles: MCycles) {
1867        self.advance_time(mcycles, true);
1868    }
1869    fn nmi_pending(&self) -> bool {
1870        *self.nmi
1871    }
1872    fn irq_pending(&self) -> bool {
1873        // H/V-timer IRQ is the held level `cpu_regs.irq_flag` (no longer the
1874        // one-shot `*self.irq` edge); coproc holds its own level line.
1875        *self.irq || self.cpu_regs.irq_flag || self.mapper.coproc_main_irq_pending()
1876    }

1879impl SnesBus<'_> {
1880    /// Advance the master clock and the time-driven subsystems by
1881    /// `mcycles` (Phase 1 cycle-accuracy: per-bus-access synchronisation
1882    /// instead of one end-of-instruction lump).
1883    ///
1884    /// `advance_coproc` is `false` only on the DMA accounting path: the
1885    /// coprocessor already advanced per transferred byte inside
1886    /// [`DmaBusView::tick`], so charging it again with the lumped DMA
1887    /// cost here would double-count it.
1888    fn advance_time(&mut self, mcycles: MCycles, advance_coproc: bool) {
1889        // HDMA + DRAM refresh steal master cycles. `sched_advance` returns
1890        // the stall cost of anything crossed (per-line HDMA and the
1891        // once-per-line DRAM refresh); the CPU is halted for that long, so we
1892        // re-advance everything (master clock, APU, PPU, coproc, IRQ poll) by
1893        // the stall in a follow-up pass. The loop converges — each scanline's
1894        // HDMA + refresh run once and a stall rarely spans a full 1364-mclk
1895        // line.
1896        let mut step = mcycles;
1897        loop {
1898            *self.mclk_total = self.mclk_total.saturating_add(step);
1899            // APU in lockstep with the CPU at bus-access granularity.
1900            // `Apu::step` carries the sub-84-mclk remainder in
1901            // `mclk_deficit`, so per-access stepping composes exactly with
1902            // the old lump (same SPC instruction count) — only the CPU↔APU
1903            // port interleaving is finer.
1904            if !*self.apu_panicked {
1905                self.apu_real.step(step as u32);
1906                if self.apu_real.cpu.stopped {
1907                    *self.apu_panicked = true;
1908                }
1909            }
1910            // PPU scanline scheduler + cartridge coprocessor. Gated by
1911            // `sched_enabled` so debug peeks / mapping tests don't tick
1912            // emulation forward.
1913            if !self.sched_enabled {
1914                return;
1915            }
1916            let stall = self.sched_advance(step as u32);
1917            if advance_coproc {
1918                self.mapper.step_coproc(step as u32, self.scpu_mar);
1919            }
1920            // Charge the HDMA + refresh stall only on the CPU-instruction
1921            // path. The DMA-lump path (`advance_coproc == false`) lets them
1922            // fire but accounts its own lumped time; proper DMA↔HDMA
1923            // interleaving is Phase 5, so don't re-loop there.
1924            if stall == 0 || !advance_coproc {
1925                return;
1926            }
1927            step = MCycles::from(stall);
1928        }
1929    }
1930
1931    fn read_inner(&mut self, addr: Addr24) -> u8 {
1932        // Hold this access as the S-CPU memory-address register (ares
1933        // `cpu.r.mar`) so the per-access coproc step can model SA-1
1934        // `conflict()` bus contention against it.
1935        self.scpu_mar = addr;
1936        let speed = address_speed(addr, self.fast_rom);
1937        self.io_cycle(speed.mcycles());
1938
1939        if let Some(o) = Self::wram_offset(addr) {
1940            return self.wram[o];
1941        }
1942        if let Some(off) = Self::ppu_offset(addr) {
1943            // $2137 SLHV — reading also latches the H/V counters
1944            // into OPHCT / OPVCT. The actual returned byte is open
1945            // bus (we hand back the PPU's open-bus latch).
1946            if off == luna_ppu::register::SLHV {
1947                let (h, v) = current_hv(*self.mclk_total, self.scanlines_per_frame);
1948                self.ppu.latch_counters(h, v);
1949            }
1950            return self.ppu.read(off);
1951        }
1952        if let Some(port) = Self::apu_port(addr) {
1953            // Mailbox reads: prefer the real SPC (now timer-driven,
1954            // so its driver actually loops). Fall back to the
1955            // heuristic stub only if the SPC has stopped on an
1956            // unimplemented opcode.
1957            let value = if *self.apu_panicked {
1958                self.apu_stub_fallback.read(port)
1959            } else {
1960                self.apu_real.cpu_read_port(port)
1961            };
1962            if let Some(log) = self.mailbox_log.as_mut() {
1963                log.push(MailboxEvent {
1964                    mclk_total: *self.mclk_total,
1965                    pc_full: self.cpu_pc_full,
1966                    kind: MailboxEventKind::Read,
1967                    port: port as u8,
1968                    value,
1969                });
1970            }
1971            return value;
1972        }
1973        if let Some(off) = Self::wram_port_offset(addr) {
1974            // $2180 WMDATA — read byte at the 17-bit counter, advance.
1975            // $2181-$2183 are write-only; reads return open bus.
1976            if off == 0x80 {
1977                let a = (*self.wm_addr & 0x1FFFF) as usize;
1978                let v = self.wram[a];
1979                *self.wm_addr = (*self.wm_addr + 1) & 0x1FFFF;
1980                return v;
1981            }
1982            return *self.mdr;
1983        }
1984        if let Some(offset) = Self::is_joypad_serial(addr) {
1985            // $4016/$4017 manual serial read.
1986            //
1987            // Per Mesen2 (`BaseControlDevice.cpp::StrobeProcessRead`
1988            // + `SnesController.cpp::ReadRam`) and ares
1989            // (`controller/gamepad/gamepad.cpp::data`): while strobe
1990            // is high, the shift register is continuously re-latched
1991            // from the live controller state — so every read while
1992            // strobe is high returns bit B (the MSB in luna's MSB-
1993            // first layout). Once strobe falls, the buffer freezes
1994            // and subsequent reads shift one MSB-first bit out per
1995            // call; reads past slot 16 return 1 (pulled-high serial
1996            // line — the "device signature" 4 zeros at slots 12-15
1997            // already live in the upper bits, then the shift fills
1998            // 1s from the LSB).
1999            //
2000            // luna used to ALWAYS shift, regardless of strobe, so
2001            // games that polled "write 1; read 16x" (Bomberman's
2002            // title-menu pattern) drained the buffer into all-1s
2003            // after the first sweep and saw a phantom "every button
2004            // pressed" state forever — instant menu auto-advance.
2005            // A Mouse on this port answers the manual serial read with its own
2006            // 32-bit stream (device signature + signed dx/dy) instead of the
2007            // pad shift register. The $4016 strobe drives its latch (below).
2008            let port = usize::from(offset != 0x4016);
2009            if let Some(bit) = self.cpu_regs.port_serial_bit(port) {
2010                return bit;
2011            }
2012            let shift = if offset == 0x4016 {
2013                &mut *self.joypad1_shift
2014            } else {
2015                &mut *self.joypad2_shift
2016            };
2017            let live = if offset == 0x4016 {
2018                self.cpu_regs.joypad1
2019            } else {
2020                self.cpu_regs.joypad2
2021            };
2022            if *self.joypad_strobe {
2023                *shift = live;
2024            }
2025            let bit = (*shift >> 15) & 1;
2026            *shift = shift.wrapping_shl(1) | 1;
2027            return bit as u8;
2028        }
2029        if let Some(offset) = Self::dma_offset(addr) {
2030            return self.dma.read_register(offset).unwrap_or(*self.mdr);
2031        }
2032        if Self::is_mdmaen(addr) || Self::is_hdmaen(addr) {
2033            // MDMAEN / HDMAEN are write-only; reads return open bus.
2034            return *self.mdr;
2035        }
2036        if let Some(reg_off) = Self::cpu_reg_offset(addr) {
2037            // $4212 HVBJOY: bit 7 = vblank (latched in `cpu_regs.hvbjoy`),
2038            // bit 6 = hblank (live H-counter), bit 0 = auto-read busy
2039            // (latched in `cpu_regs.hvbjoy`).
2040            //
2041            // Per ares' `cpu/io.cpp`:
2042            //   data.bit(6) = hcounter() <= 2 || hcounter() >= 1096;
2043            // The `hcounter()` is in master cycles (0..1364); our
2044            // `current_hv` returns H in *dots* (`mclk / 4`, 0..341),
2045            // so the equivalent threshold is `h == 0 || h >= 274`.
2046            //
2047            // Without the live hblank bit, games that do `BIT $4212;
2048            // BVC -5` (SMW, many others) hang in an infinite busy-wait.
2049            if reg_off == 0x4212 {
2050                let (h, _) = current_hv(*self.mclk_total, self.scanlines_per_frame);
2051                let in_hblank = h == 0 || h >= 274;
2052                let hblank_bit = if in_hblank { 0x40 } else { 0x00 };
2053                return (self.cpu_regs.hvbjoy & !0x40) | hblank_bit;
2054            }
2055            if reg_off == 0x4210 {
2056                // RDNMI 4-cycle hold window. Mesen2 keeps the flag
2057                // set when HClock < 6 on the NMI scanline, so a
2058                // mainline `BPL $4210` after the NMI handler ACKed
2059                // the same flag still sees the bit. luna's H is in
2060                // dots (4 mclk each), so < 6 mclks ≈ < 2 dots.
2061                let (h, _) = current_hv(*self.mclk_total, self.scanlines_per_frame);
2062                let in_hold = self.ppu_line == self.vblank_start_line && h < 2;
2063                return self.cpu_regs.read_rdnmi(in_hold);
2064            }
2065            if let Some(v) = self.cpu_regs.read(reg_off) {
2066                return v;
2067            }
2068            // Write-only registers fall through to open bus.
2069            return *self.mdr;
2070        }
2071        if let Some(v) = self.mapper.read(addr) {
2072            if let Some(reg) = Self::sa1_reg(addr) {
2073                if let Some(log) = self.sa1_log.as_mut() {

2103    fn write_inner(&mut self, addr: Addr24, value: u8) {
2104        // S-CPU memory-address register (ares `cpu.r.mar`) — see `read_inner`.
2105        self.scpu_mar = addr;
2106        let speed = address_speed(addr, self.fast_rom);
2107        self.io_cycle(speed.mcycles());
2108
2109        // Nocash debug TTY: capture `$21FC` writes (no$/Mesen console port —
2110        // the SDK's `SNES_NOCASH`/`SNES_ASSERT`) when enabled, bounded so a
2111        // runaway never grows unbounded. Otherwise `$21FC` is open-bus.
2112        if let Some(buf) = self.nocash_log.as_mut() {
2113            let bank = (addr >> 16) as u8;
2114            if matches!(bank, 0x00..=0x3F | 0x80..=0xBF) && addr as u16 == 0x21FC {
2115                if buf.len() < 1 << 20 {
2116                    buf.push(value);
2117                }
2118                return;
2119            }
2120        }
2121
2122        if let Some(o) = Self::wram_offset(addr) {
2123            self.wram[o] = value;
2124            return;
2125        }
2126        if let Some(off) = Self::ppu_offset(addr) {
2127            // Gap G7: refresh the PPU's "active display" flag before
2128            // every register write so VMDATA / OAMDATA / CGDATA writes
2129            // that land during the visible portion of a non-blanked
2130            // frame silently drop the data (the address/latch state
2131            // still advances). ares `ppu_io.cpp:19-45` / Mesen2
2132            // `SnesPpu.cpp:2046-2057`.
2133            self.ppu.active_display =
2134                self.ppu_line < self.vblank_start_line && (self.ppu.inidisp & 0x80) == 0;
2135
2136            // Phase 2 of gap G6 — intra-line partial flush. If the
2137            // CPU is writing a render-affecting PPU register ($2100..$2133)
2138            // mid-scanline, commit the in-progress dots with the OLD
2139            // state BEFORE applying the write so the partial line gets
2140            // the pre-write pixels. (Mesen2 SnesPpu.cpp:1884-1886
2141            // RenderScanline-before-write pattern.)
2142            if off < 0x34 && self.ppu_line < self.vblank_start_line {
2143                let (h, _) = current_hv(*self.mclk_total, self.scanlines_per_frame);
2144                let dot = h.min(luna_ppu::FRAME_W as u16);
2145                self.ppu.flush_partial_scanline(
2146                    self.ppu_line,
2147                    dot,
2148                    luna_ppu::RenderOptions::default(),
2149                );
2150            }
2151            // $2100 INIDISP — a write that exits forced-blank exactly
2152            // at the vblank-entry scanline triggers the OAM address
2153            // auto-reset, same as the per-line vblank hook. ares
2154            // `ppu_io.cpp:194`, Mesen2 `SnesPpu.cpp:1889-1896`.
2155            if off == 0x00 {
2156                let was_force_blank = self.ppu.inidisp & 0x80 != 0;
2157                let will_force_blank = value & 0x80 != 0;
2158                if was_force_blank && !will_force_blank && self.ppu_line == self.vblank_start_line {
2159                    self.ppu.oam.reload_address_from_latch();
2160                }
2161            }
2162            self.ppu.write(off, value);
2163            if off == 0x00 {
2164                // INIDISP just changed forced-blank: recompute `active_display`
2165                // LIVE so a MID-SCANLINE forced-blank gates VRAM/OAM writes
2166                // correctly. `active_display` is otherwise only refreshed per
2167                // scanline (advance_scheduler), so without this a game that
2168                // force-blanks mid-line then DMAs (Tales' attack-frame OBJ-tile
2169                // upload at line ~209) has its writes wrongly dropped — luna's
2170                // stale cache said active-display while forced-blank was set.
2171                // ares checks `displayDisable` live at the write (ppu io.cpp).
2172                self.ppu.active_display =
2173                    self.ppu_line < self.vblank_start_line && (self.ppu.inidisp & 0x80) == 0;
2174            }
2175            return;
2176        }
2177        if let Some(port) = Self::apu_port(addr) {
2178            // CPU writes the byte to BOTH the real APU's to_spc port
2179            // (so the SPC700 reads it at $F4-$F7) and the fallback
2180            // stub (in case the SPC has panicked and we need it
2181            // later). Cheap, no consistency issues since the stub
2182            // is only consulted when the real APU is dead.
2183            self.apu_real.cpu_write_port(port, value);
2184            self.apu_stub_fallback.write(port, value);
2185            if let Some(log) = self.mailbox_log.as_mut() {
2186                log.push(MailboxEvent {
2187                    mclk_total: *self.mclk_total,
2188                    pc_full: self.cpu_pc_full,
2189                    kind: MailboxEventKind::Write,
2190                    port: port as u8,
2191                    value,
2192                });
2193            }
2194            return;
2195        }
2196        if let Some(off) = Self::wram_port_offset(addr) {
2197            match off {
2198                // $2180 WMDATA — write byte at the 17-bit counter,
2199                // auto-advance.
2200                0x80 => {
2201                    let a = (*self.wm_addr & 0x1FFFF) as usize;
2202                    self.wram[a] = value;
2203                    *self.wm_addr = (*self.wm_addr + 1) & 0x1FFFF;
2204                }
2205                // $2181 WMADDL — counter bits 0..7.
2206                0x81 => *self.wm_addr = (*self.wm_addr & !0x000FF) | u32::from(value),
2207                // $2182 WMADDM — counter bits 8..15.
2208                0x82 => {
2209                    *self.wm_addr = (*self.wm_addr & !0x0FF00) | (u32::from(value) << 8);
2210                }
2211                // $2183 WMADDH — counter bit 16 (only bit 0 of the
2212                // value is used; upper bits ignored).
2213                0x83 => {
2214                    *self.wm_addr = (*self.wm_addr & !0x10000) | (u32::from(value & 0x01) << 16);
2215                }
2216                _ => {}
2217            }
2218            return;
2219        }
2220        if let Some(offset) = Self::is_joypad_serial(addr) {
2221            // $4016 strobe write.
2222            //
2223            // Per Mesen2 (`BaseControlDevice.cpp::StrobeProcessWrite`)
2224            // and ares (`controller/gamepad/gamepad.cpp` latch path):
2225            // the buffer parallel-loads on the **falling** edge
2226            // (strobe 1→0), NOT on the rising edge. Rising-edge and
2227            // held-high writes leave the buffer alone — the live
2228            // refresh during reads (handled in the read path above)
2229            // is what keeps the strobe-high reads in sync.
2230            //
2231            // luna used to reload on rising edge and on hold, which
2232            // wasn't observable on its own but combined with the
2233            // missing read-side refresh produced the Bomberman menu
2234            // glitch.
2235            if offset == 0x4016 {
2236                let next_strobe = (value & 0x01) != 0;
2237                if *self.joypad_strobe && !next_strobe {
2238                    // Falling edge — latch live state.
2239                    *self.joypad1_shift = self.cpu_regs.joypad1;
2240                    *self.joypad2_shift = self.cpu_regs.joypad2;
2241                }
2242                *self.joypad_strobe = next_strobe;
2243                // Every connected peripheral latches off the $4016 strobe.
2244                self.cpu_regs.latch_devices(next_strobe);
2245            }
2246            // $4017 writes drive the expansion-port output pins —
2247            // ignored by an emulator that doesn't model the expansion.
2248            return;
2249        }
2250        if let Some(offset) = Self::dma_offset(addr) {
2251            self.dma.write_register(offset, value);
2252            // Let a DMA-observing coprocessor (S-DD1) capture the channel's
2253            // source address / length so it can set up on-the-fly graphics
2254            // decompression. Non-observing mappers ignore `$43xx` writes.
2255            let _ = self.mapper.write(addr, value);
2256            return;
2257        }
2258        if Self::is_mdmaen(addr) {
2259            // Trigger sync DMA on every channel selected in `value`.
2260            // We splat the SnesBus borrows: `dma` is mutated by the DMA
2261            // call, the other refs flow into DmaBusView. This borrow-split
2262            // lets the DMA run without re-entering the Bus impl. The
2263            // DMA→VRAM trace is moved into the view for the burst (so its
2264            // $2118/9 writes are captured) and restored after.
2265            //
2266            // 8 mclk one-shot start overhead, then 8 mclk per byte
2267            // (fullsnes §"SNES DMA Timing"). `advance_coproc = false`:
2268            // coprocs already advanced per byte via `DmaBusView::tick`, so
2269            // the master-clock charge here must not re-charge them.
2270            if self.dma.hdmaen == 0 {
2271                // Fast path — no HDMA armed, so nothing the DMA could
2272                // cross matters: run the whole burst as one lump (the
2273                // legacy behaviour, byte-identical). Covers virtually all
2274                // forced-blank / vblank uploads.
2275                let mut trace = self.dma.dma_trace.take();
2276                let bytes = {
2277                    let mut view = DmaBusView {
2278                        wram: self.wram,
2279                        mapper: self.mapper,
2280                        ppu: self.ppu,
2281                        wm_addr: self.wm_addr,
2282                        dma_trace: trace.as_mut(),
2283                        last_a_addr: 0,
2284                        trace_frame: self.frame_count,
2285                        trace_line: self.ppu_line,
2286                        trace_blank: self.ppu_line >= self.vblank_start_line,
2287                        trace_hclock: current_hclock(*self.mclk_total, self.scanlines_per_frame),
2288                        dma_channel: 0,
2289                    };
2290                    self.dma.run_mdma(&mut view, value)
2291                };
2292                self.dma.dma_trace = trace;
2293                self.advance_time(u64::from(8 + bytes.saturating_mul(8)), false);
2294                return;
2295            }
2296
2297            // Segmented path (Phase 5) — HDMA is armed, so a long burst
2298            // must yield to HDMA at scanline boundaries instead of landing
2299            // all at once. Drive the DMA in segments bounded by the next
2300            // line crossing; `advance_time` then runs that line's HDMA via
2301            // `sched_one_line`. A DMA yields to HDMA at most once per line,
2302            // so line-granular segmentation reproduces ares' per-byte
2303            // `dmaEdge` ordering for this case.
2304            self.advance_time(8, false); // one-shot start overhead
2305            let mut trace = self.dma.dma_trace.take();
2306            loop {
2307                let room_mclk = MCYCLES_PER_SCANLINE.saturating_sub(self.mcycles_in_line);
2308                // Bytes that fit before the next boundary (≥1 to progress).
2309                // 8 mclk/byte, matching `DmaChannel::run_segment`.
2310                let seg_bytes = (room_mclk / 8).max(1);
2311                let done = {
2312                    let mut view = DmaBusView {
2313                        wram: self.wram,
2314                        mapper: self.mapper,
2315                        ppu: self.ppu,
2316                        wm_addr: self.wm_addr,
2317                        dma_trace: trace.as_mut(),
2318                        last_a_addr: 0,
2319                        trace_frame: self.frame_count,
2320                        trace_line: self.ppu_line,
2321                        trace_blank: self.ppu_line >= self.vblank_start_line,
2322                        trace_hclock: current_hclock(*self.mclk_total, self.scanlines_per_frame),
2323                        dma_channel: 0,
2324                    };
2325                    self.dma.run_mdma_segment(&mut view, value, seg_bytes)
2326                };
2327                // Charge this segment's master clock; crosses the line
2328                // boundary and fires HDMA for any visible line crossed.
2329                self.advance_time(u64::from(done) * 8, false);
2330                if self.dma.mdma_cursor.is_none() {
2331                    break; // burst complete
2332                }
2333            }
2334            self.dma.dma_trace = trace;
2335            return;
2336        }
2337        if Self::is_hdmaen(addr) {
2338            self.dma.hdmaen = value;
2339            return;
2340        }
2341        if let Some(reg_off) = Self::cpu_reg_offset(addr) {
2342            // WRIO ($4201) bit 7 0→1 transition latches the PPU H/V
2343            // counters. Check BEFORE handing off to CpuRegs::write so
2344            // we can see the previous value.
2345            if reg_off == 0x4201 {
2346                let prev = self.cpu_regs.wrio;
2347                if prev & 0x80 == 0 && value & 0x80 != 0 {
2348                    let (h, v) = current_hv(*self.mclk_total, self.scanlines_per_frame);
2349                    self.ppu.latch_counters(h, v);
2350                }
2351            }
2352            // P2 — late NMI enable: raising NMITIMEN.7 (0→1) while the NMI line
2353            // is still asserted ($4210 flag set, i.e. mid-VBlank, un-read) fires
2354            // the NMI now (ares irq.cpp `nmitimenUpdate`: `if nmiEnable.raise(7)
2355            // && nmiLine → nmiTransition`). Checked BEFORE the CpuRegs::write so
2356            // we see the previous NMITIMEN.7. P1's VBlank-end clear of `nmi_flag`
2357            // is what makes this safe: the line is no longer stale-true outside
2358            // VBlank, so this can't fire the spurious NMI that black-screened
2359            // SMRPG when the naive port was attempted.
2360            if reg_off == 0x4200
2361                && self.cpu_regs.nmitimen & 0x80 == 0
2362                && value & 0x80 != 0
2363                && self.cpu_regs.nmi_flag
2364            {
2365                *self.nmi = true;
2366                self.trace_irq_signal(MemEventKind::NmiSignal, value);
2367            }
2368            if self.cpu_regs.write(reg_off, value) {
2369                return;
2370            }
2371            // CpuRegs returned false → maybe a register that lives
2372            // elsewhere (e.g. $420D MEMSEL → fast_rom). Handle here.
2373            if reg_off == 0x420D {
2374                self.fast_rom = value & 0x01 != 0;
2375            }
2376            return;
2377        }
2378        if let Some(reg) = Self::sa1_reg(addr) {
2379            if let Some(log) = self.sa1_log.as_mut() {
2380                log.push(Sa1LogEvent {
2381                    mclk_total: *self.mclk_total,
2382                    pc_full: self.cpu_pc_full,
2383                    kind: MailboxEventKind::Write,
2384                    reg,
2385                    value,
2386                });
2387            }
2388        }
2389        // Mapper claims SRAM writes; anything not yet routed drops.
2390        let _ = self.mapper.write(addr, value);
2391    }

3591    ///   7. NOP-pauses then `STP`s.
3592    fn demo_sa1_irq_cart() -> Cartridge {
3593        let mut rom = vec![0xEA; 32 * 1024];
3594        rom[0x7FFC] = 0x00;
3595        rom[0x7FFD] = 0x80;
3596        let off = 0x7FC0;
3597        for (i, b) in b"LUNA SA1 IRQ DEMO    ".iter().enumerate() {
3598            rom[off + i] = *b;
3599        }
3600        rom[off + 0x15] = 0x23;
3601        rom[off + 0x17] = 0x05;
3602        rom[off + 0x18] = 0x00;
3603        rom[off + 0x19] = 0x01;
3604        rom[off + 0x1C] = 0x34;
3605        rom[off + 0x1D] = 0x12;
3606        rom[off + 0x1E] = 0xCB;
3607        rom[off + 0x1F] = 0xED;
3608
3609        let mut p = 0usize;
3610        let emit = |bytes: &[u8], rom: &mut [u8], p: &mut usize| {
3611            for b in bytes {
3612                rom[*p] = *b;
3613                *p += 1;
3614            }
3615        };
3616        // SEI, native mode, 16-bit A/X/Y.
3617        emit(&[0x78], &mut rom, &mut p);
3618        emit(&[0x18, 0xFB], &mut rom, &mut p);
3619        emit(&[0xC2, 0x30], &mut rom, &mut p);
3620
3621        // CRV = $3000:  LDA #$3000 ; STA $002203
3622        emit(&[0xA9, 0x00, 0x30], &mut rom, &mut p);
3623        emit(&[0x8F, 0x03, 0x22, 0x00], &mut rom, &mut p);
3624        // CIV = $3010:  LDA #$3010 ; STA $002207
3625        emit(&[0xA9, 0x10, 0x30], &mut rom, &mut p);
3626        emit(&[0x8F, 0x07, 0x22, 0x00], &mut rom, &mut p);
3627
3628        // Back to 8-bit accumulator for byte writes.
3629        emit(&[0xE2, 0x20], &mut rom, &mut p);
3630
3631        // CIE = $80 — enable S-CPU → SA-1 IRQ.
3632        emit(&[0xA9, 0x80], &mut rom, &mut p);
3633        emit(&[0x8F, 0x0A, 0x22, 0x00], &mut rom, &mut p);
3634
3635        // Seed SA-1 program at I-RAM $3000:
3636        //   $3000: CLI                          58
3637        //   $3001..$300F: NOP loop              EA…
3638        //   $3010 (IRQ handler):
3639        //         LDA #$AA                       A9 AA
3640        //         STA $3500                      8D 00 35
3641        //         STP                            DB
3642        // We store byte-by-byte with STA absolute long ($8F).
3643        let writes: &[(u32, u8)] = &[
3644            (0x00_3000, 0x58), // CLI
3645            (0x00_3001, 0xEA),
3646            (0x00_3002, 0xEA),
3647            (0x00_3003, 0xEA),
3648            (0x00_3004, 0xEA),
3649            (0x00_3005, 0xEA),
3650            (0x00_3006, 0xEA),
3651            (0x00_3007, 0xEA),
3652            (0x00_3008, 0xEA),
3653            (0x00_3009, 0xEA),
3654            (0x00_300A, 0xEA),
3655            (0x00_300B, 0xEA),
3656            (0x00_300C, 0xEA),
3657            (0x00_300D, 0xEA),
3658            (0x00_300E, 0xEA),
3659            (0x00_300F, 0xEA),
3660            (0x00_3010, 0xA9), // LDA #
3661            (0x00_3011, 0xAA),
3662            (0x00_3012, 0x8D), // STA abs
3663            (0x00_3013, 0x00),
3664            (0x00_3014, 0x35),
3665            (0x00_3015, 0xDB), // STP
3666        ];
3667        for (addr, byte) in writes {
3668            // LDA #imm
3669            emit(&[0xA9, *byte], &mut rom, &mut p);
3670            // STA $aabbcc (absolute long: 8F lo mid hi)
3671            emit(
3672                &[
3673                    0x8F,
3674                    (*addr & 0xFF) as u8,
3675                    ((*addr >> 8) & 0xFF) as u8,
3676                    ((*addr >> 16) & 0xFF) as u8,
3677                ],
3678                &mut rom,
3679                &mut p,
3680            );
3681        }
3682
3683        // Release SA-1: default CCNT is $20 (bit 5 = reset). A write
3684        // of $00 clears bit 5, producing the 1→0 release edge.
3685        emit(&[0xA9, 0x00], &mut rom, &mut p);
3686        emit(&[0x8F, 0x00, 0x22, 0x00], &mut rom, &mut p);
3687
3688        // Run-up: 40 NOPs so the SA-1 can reach its NOP loop.
3689        for _ in 0..40 {
3690            emit(&[0xEA], &mut rom, &mut p);
3691        }
3692
3693        // Trigger SA-1 IRQ: CCNT bit 7 0→1 edge.
3694        emit(&[0xA9, 0x80], &mut rom, &mut p);
3695        emit(&[0x8F, 0x00, 0x22, 0x00], &mut rom, &mut p);
3696
3697        // More NOPs to give the SA-1 time to service the IRQ.
3698        for _ in 0..60 {
3699            emit(&[0xEA], &mut rom, &mut p);
3700        }
3701
3702        // STP — main CPU done.
3703        emit(&[0xDB], &mut rom, &mut p);
3704        let _ = p;
3705        Cartridge::from_bytes(rom).unwrap()
3706    }
