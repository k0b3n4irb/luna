1//! CPU-system registers at `$4200-$420F` and `$4210-$421F`.
2//!
3//! These live "between" the 65C816 and the rest of the chip: the
4//! multiplier / divider are functional units, NMITIMEN and the
5//! interrupt-status registers gate the CPU's interrupt service, and
6//! HVBJOY exposes the PPU's H/V-blank state to game code.
7//!
8//! Reference: <https://problemkaputt.de/fullsnes.htm> §"S-CPU Registers".
9
10use crate::controller::{Mouse, PortDevice, SuperScope};
11
12/// CPU-side registers.
13///
14/// Reads from write-only registers return open-bus (we return the last
15/// value seen on the CPU bus, which the calling [`crate::Snes`]
16/// `SnesBus` tracks). Writes to read-only registers are silently dropped.
17#[derive(Debug, Default, serde::Serialize, serde::Deserialize)]
18pub struct CpuRegs {
19    /// `$4200` — NMITIMEN: NMI enable (bit 7), V-IRQ (bit 5),
20    /// H-IRQ (bit 4), joypad auto-read (bit 0).
21    pub nmitimen: u8,
22    /// `$4201` — WRIO: programmable I/O pins (write side). Bit 7
23    /// connects to the PPU latch (writing 0 latches H/V counters).
24    pub wrio: u8,
25    /// `$4202` — WRMPYA: multiplicand.
26    pub wrmpya: u8,
27    /// `$4203` — WRMPYB: multiplier. Writing this triggers the
28    /// 8x8 = 16-bit multiplication; the result lands in `rdmpy`.
29    pub wrmpyb: u8,
30    /// `$4204/$4205` — WRDIVL/H: 16-bit dividend.
31    pub wrdiv: u16,
32    /// `$4206` — WRDVDD: 8-bit divisor. Writing this triggers the
33    /// 16/8 = 16-bit division; quotient in `rddiv`, remainder in `rdmpy`.
34    pub wrdvdd: u8,
35    /// `$4207/$4208` — HTIMEL/H: 9-bit H-counter trigger (low + bit 0
36    /// of high).
37    pub htime: u16,
38    /// `$4209/$420A` — VTIMEL/H: 9-bit V-counter trigger.
39    pub vtime: u16,
40
41    /// `$4210` read side — bit 7 = "NMI flag" (set on vblank when
42    /// NMI is enabled, OR on any read in some hardware). Cleared on
43    /// read. Bits 0-3 = CPU version (we return 2, the most common rev).
44    pub nmi_flag: bool,
45    /// `$4211` read side — bit 7 = "IRQ flag" (set when the IRQ
46    /// condition fires). Cleared on read.
47    pub irq_flag: bool,
48    /// `$4212` read side — HVBJOY: bit 7 = V-blank, bit 6 = H-blank,
49    /// bit 0 = joypad auto-read busy.
50    pub hvbjoy: u8,
51
52    /// Result of the multiplication ($4216/$4217) or the division
53    /// remainder. Hardware shares the same 16-bit "RDMPY" register.
54    pub rdmpy: u16,
55    /// Result of the division ($4214/$4215).
56    pub rddiv: u16,
57
58    // ---------- Joypad auto-read ----------
59    /// Current bitmask for player 1 (16-bit). Bit layout per SNES
60    /// hardware: B Y SEL START Up Down Left Right A X L R 0 0 0 0
61    /// (MSB → LSB). The front-end pushes this via [`Self::set_joypad`].
62    pub joypad1: u16,
63    /// Same for player 2.
64    pub joypad2: u16,
65    /// `$4218/$4219` — latched player-1 state copied here at the
66    /// start of every `VBlank` when `NMITIMEN.0` is set. Stays put
67    /// between latches, so games that read these registers see a
68    /// stable per-frame snapshot.
69    pub joypad1_latched: u16,
70    /// `$421A/$421B` — latched player-2 state.
71    pub joypad2_latched: u16,
72
73    /// Device on controller port 1 (the auto-read + `$4016` serial path).
74    pub port1: PortDevice,
75    /// Device on controller port 2 (the auto-read + `$4017` serial path).
76    pub port2: PortDevice,
77    /// SNES Mouse state — shared by whichever port is set to
78    /// [`PortDevice::Mouse`] (a host has at most one mouse).
79    pub mouse: Mouse,
80    /// Super Scope state — used when a port is [`PortDevice::SuperScope`].
81    pub super_scope: SuperScope,
82}
83
84impl CpuRegs {
85    /// Build a powered-on register file (all zero).
86    #[must_use]
87    pub fn new() -> Self {
88        Self::default()
89    }

91    /// One serial bit for `port`'s peripheral (Mouse / Super Scope) on a
92    /// manual `$4016` (port 0) / `$4017` (port 1) read; `None` for a pad.
93    pub fn port_serial_bit(&mut self, port: usize) -> Option<u8> {
94        match if port == 0 { self.port1 } else { self.port2 } {
95            PortDevice::Mouse => Some(self.mouse.data() & 1),
96            PortDevice::SuperScope => Some(self.super_scope.data() & 1),
97            PortDevice::Pad => None,
98        }
99    }
100
101    /// If a Super Scope is connected and aimed onscreen, and the beam has
102    /// reached its aimed scanline `line`, the `(h, v)` counter values its
103    /// light-gun latches into OPHCT/OPVCT (which the game reads for the beam
104    /// position). `None` otherwise — no scope, offscreen, or wrong scanline.
105    pub fn scope_latch_target(&self, line: u16) -> Option<(u16, u16)> {
106        let active = self.port1 == PortDevice::SuperScope || self.port2 == PortDevice::SuperScope;
107        if active && !self.super_scope.offscreen() && u32::from(line) == self.super_scope.cy as u32
108        {
109            Some(self.super_scope.latch_hv())
110        } else {
111            None
112        }
113    }
114
115    /// Drive the shared `$4016` strobe line into every connected peripheral's
116    /// serial latch.
117    pub fn latch_devices(&mut self, strobe: bool) {
118        if self.port1 == PortDevice::Mouse || self.port2 == PortDevice::Mouse {
119            self.mouse.latch(strobe);
120        }
121        if self.port1 == PortDevice::SuperScope || self.port2 == PortDevice::SuperScope {
122            self.super_scope.latch(strobe);
123        }
124    }

126    /// Read a register at the 16-bit CPU bank-0 offset (`$4200-$421F`).
127    /// Returns `None` if `offset` is outside that range.
128    pub const fn read(&mut self, offset: u16) -> Option<u8> {
129        let v = match offset {
130            // Most $4200-$420A registers are write-only; reads return
131            // open-bus on real hardware. We return 0 so tests can
132            // distinguish "definitely write-only" from "stub MMIO" at
133            // the SnesBus level.
134            0x4210 => {
135                // RDNMI: bit 7 = nmi flag (cleared by this read).
136                // For the cleared-on-read semantics with the 4-cycle
137                // hold window at the start of the NMI scanline, use
138                // [`Self::read_rdnmi`] from a bus-level caller that
139                // knows the current hclock. Plain `read()` always
140                // clears — fine for direct test callers, not for games
141                // that race a `BPL $4210` mainline against an NMI
142                // handler that ACKs the same flag (CT, Terranigma).
143                let v = if self.nmi_flag { 0x80 } else { 0x00 } | 0x02; // CPU rev 2
144                self.nmi_flag = false;
145                v
146            }
147            0x4211 => {
148                // TIMEUP: bit 7 = irq flag (cleared by this read).
149                let v = if self.irq_flag { 0x80 } else { 0x00 };
150                self.irq_flag = false;
151                v
152            }
153            0x4212 => self.hvbjoy,
154            0x4214 => self.rddiv as u8,
155            0x4215 => (self.rddiv >> 8) as u8,
156            0x4216 => self.rdmpy as u8,
157            0x4217 => (self.rdmpy >> 8) as u8,
158            // Joypad auto-read latches at $4218-$421F. Standard SNES
159            // only has 2 controller ports, so $421C-$421F (joypads 3
160            // and 4 via multitap) return 0.
161            0x4218 => self.joypad1_latched as u8,
162            0x4219 => (self.joypad1_latched >> 8) as u8,
163            0x421A => self.joypad2_latched as u8,
164            0x421B => (self.joypad2_latched >> 8) as u8,
165            0x421C..=0x421F => 0x00,
166            0x4200..=0x420F | 0x4213 => return None,
167            _ => return None,
168        };
169        Some(v)
170    }
171
172    /// `$4210` RDNMI read with the hardware 4-cycle hold window honoured.
173    ///
174    /// Per Mesen2 (`InternalRegisters.cpp:229-241`) and ares
175    /// (`cpu_irq.cpp::rdnmi`): reading `$4210` clears the NMI flag on
176    /// every cycle EXCEPT the first ~6 master cycles of the NMI
177    /// scanline. During that brief window the hardware forces the flag
178    /// to stay set, so a CPU mainline that polls `BPL $4210` is not
179    /// starved by its own NMI handler `ACKing` the same flag a few cycles
180    /// earlier. Mesen2's comment explicitly names Terranigma; Chrono
181    /// Trigger uses the same Square/Quintet idiom and hangs in luna
182    /// without the hold window.
183    ///
184    /// `in_hold` should be `true` only when the bus knows the current
185    /// scanline is the NMI scanline AND the H-clock is < 6 master
186    /// cycles (= < 2 dots at luna's H resolution).
187    pub const fn read_rdnmi(&mut self, in_hold: bool) -> u8 {
188        let v = if self.nmi_flag { 0x80 } else { 0x00 } | 0x02; // CPU rev 2
189        if !in_hold {
190            self.nmi_flag = false;
191        }
192        v
193    }
194
195    /// Push the current per-frame joypad state into the auto-read
196    /// latches. Called by [`crate::Snes::advance_one_scanline`] when
197    /// the scheduler crosses the `VBlank` entry line, IF `NMITIMEN.0`
198    /// is set. Also raises `HVBJOY.0` (auto-read busy) for the few
199    /// scanlines the hardware spends in the auto-read sequence — see
200    /// [`Self::clear_joypad_busy`].
201    ///
202    /// D-pad opposing-direction quirk: per ares' `gamepad.cpp`
203    /// `latch()`, real hardware physically prevents `up + down` and
204    /// `left + right` from being asserted simultaneously. We resolve
205    /// the case by clearing both opposing bits — keyboard players
206    /// who hit conflicting keys see a "no-direction" outcome rather
207    /// than a glitched latch.
208    pub fn latch_joypad_auto_read(&mut self) {
209        if self.nmitimen & 0x01 == 0 {
210            return;
211        }
212        // A Mouse on a port feeds the auto-read its first 16 protocol bits
213        // (signature + buttons + speed) instead of the pad's button mask —
214        // that is how the SDK's `mouseInit` detects it.
215        self.joypad1_latched = match self.port1 {
216            PortDevice::Mouse => self.mouse.auto_read_16(),
217            PortDevice::SuperScope => self.super_scope.auto_read_16(),
218            PortDevice::Pad => Self::clean_dpad(self.joypad1),
219        };
220        self.joypad2_latched = match self.port2 {
221            PortDevice::Mouse => self.mouse.auto_read_16(),
222            PortDevice::SuperScope => self.super_scope.auto_read_16(),
223            PortDevice::Pad => Self::clean_dpad(self.joypad2),
224        };
225        self.hvbjoy |= 0x01;
226    }
227
228    /// Clear opposing D-pad bits (Up vs Down, Left vs Right) to model
229    /// the physical lockout on real-HW controllers. Bit layout per
230    /// the SNES JOY1L/JOY1H pair: bit 11 = Up, bit 10 = Down, bit 9
231    /// = Left, bit 8 = Right.
232    const fn clean_dpad(mask: u16) -> u16 {
233        let mut m = mask;
234        if m & 0x0C00 == 0x0C00 {
235            m &= !0x0C00; // up + down → drop both
236        }
237        if m & 0x0300 == 0x0300 {
238            m &= !0x0300; // left + right → drop both
239        }
240        m
241    }
242
243    /// Drop the auto-read busy bit. Called a few scanlines after
244    /// [`Self::latch_joypad_auto_read`] so games polling
245    /// `$4212 & 0x01` see it clear before they continue.
246    pub const fn clear_joypad_busy(&mut self) {
247        self.hvbjoy &= !0x01;
248    }
249
250    /// Set the current button bitmask for controller `idx` (`0` or
251    /// `1`). The bitmask isn't visible to game code until the next
252    /// auto-read latch (typically the next `VBlank`).
253    ///
254    /// Bit layout (high → low): B Y SEL START Up Down Left Right
255    /// A X L R 0 0 0 0.
256    pub const fn set_joypad(&mut self, idx: usize, mask: u16) {
257        match idx {
258            0 => self.joypad1 = mask,
259            1 => self.joypad2 = mask,
260            _ => {}
261        }
262    }
263
264    /// Write a register at the 16-bit CPU bank-0 offset.
265    pub fn write(&mut self, offset: u16, value: u8) -> bool {
266        match offset {
267            0x4200 => self.nmitimen = value,
268            0x4201 => self.wrio = value,
269            0x4202 => self.wrmpya = value,
270            0x4203 => {
271                self.wrmpyb = value;
272                // 8x8 = 16-bit unsigned multiplication. Result writes
273                // RDMPY ($4216/$4217). Real hardware takes 8 mclk.
274                self.rdmpy = u16::from(self.wrmpya) * u16::from(self.wrmpyb);
275            }
276            0x4204 => self.wrdiv = (self.wrdiv & 0xFF00) | u16::from(value),
277            0x4205 => self.wrdiv = (self.wrdiv & 0x00FF) | (u16::from(value) << 8),
278            0x4206 => {
279                self.wrdvdd = value;
280                // 16/8 = 16-bit unsigned division. Quotient → RDDIV
281                // ($4214/$4215). Remainder → RDMPY ($4216/$4217).
282                // Division by zero on the SNES gives:
283                //   quotient = $FFFF
284                //   remainder = original dividend
285                if value == 0 {
286                    self.rddiv = 0xFFFF;
287                    self.rdmpy = self.wrdiv;
288                } else {
289                    let dividend = u32::from(self.wrdiv);
290                    let divisor = u32::from(value);
291                    self.rddiv = (dividend / divisor) as u16;
292                    self.rdmpy = (dividend % divisor) as u16;
293                }
294            }
295            0x4207 => self.htime = (self.htime & 0xFF00) | u16::from(value),
296            0x4208 => self.htime = (self.htime & 0x00FF) | (u16::from(value & 0x01) << 8),
297            0x4209 => self.vtime = (self.vtime & 0xFF00) | u16::from(value),
298            0x420A => self.vtime = (self.vtime & 0x00FF) | (u16::from(value & 0x01) << 8),
299            // $420B (MDMAEN) and $420C (HDMAEN) are routed to the DMA
300            // controller, not this struct.
301            0x420B | 0x420C => return false,
302            0x420D => {
303                // MEMSEL — bit 0 = FastROM enable. Stored externally
304                // (luna-core::Snes.fast_rom), so we return "not us".
305                return false;
306            }
307            // Read-only registers — writes ignored.
308            0x4210..=0x421F => {}
309            _ => return false,
310        }
311        true
312    }

501    #[test]
502    fn dpad_lockout_clears_opposing_directions() {
503        // Bit layout per SNES JOY1L/JOY1H pair (high → low):
504        //   B Y SEL START Up Down Left Right A X L R 0 0 0 0
505        let mut r = CpuRegs::new();
506        r.write(0x4200, 0x01); // auto-read enable
507        // Up + Down + B held → keep B, drop both Up + Down.
508        r.set_joypad(0, 0x8000 | 0x0800 | 0x0400);
509        r.latch_joypad_auto_read();
510        assert_eq!(r.joypad1_latched & 0x0C00, 0, "up + down lockout");
511        assert_eq!(r.joypad1_latched & 0x8000, 0x8000, "B preserved");
512
513        // Left + Right + Start held → keep Start, drop L + R.
514        r.set_joypad(0, 0x1000 | 0x0200 | 0x0100);
515        r.latch_joypad_auto_read();
516        assert_eq!(r.joypad1_latched & 0x0300, 0, "left + right lockout");
517        assert_eq!(r.joypad1_latched & 0x1000, 0x1000, "Start preserved");
518
519        // Only Up held (no opposing) → passes through.
520        r.set_joypad(0, 0x0800);
521        r.latch_joypad_auto_read();
522        assert_eq!(r.joypad1_latched, 0x0800);
523    }

527        // Compile-time sanity check that we agree with ares'
528        // gamepad.cpp shift order. Test by name so a future
529        // bit-flip will fire here loudly.
530        let b = 0x8000u16;
531        let y = 0x4000;
532        let sel = 0x2000;
533        let start = 0x1000;
534        let up = 0x0800;
535        let down = 0x0400;
536        let left = 0x0200;
537        let right = 0x0100;
538        let a = 0x0080;
539        let x = 0x0040;
540        let l = 0x0020;
541        let r_ = 0x0010;
542        let all = b | y | sel | start | up | down | left | right | a | x | l | r_;
543        // Bits 3..0 are the device signature — always zero for a
544        // standard gamepad. So a "press everything" mask is
545        // 0xFFF0.
546        assert_eq!(all, 0xFFF0);
547    }
