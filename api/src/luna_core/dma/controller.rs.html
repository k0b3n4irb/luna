1//! [`Dma`] — the 8-channel SNES DMA controller.
2//!
3//! Owns the array of channels and the `MDMAEN` / `HDMAEN` register
4//! semantics. `Dma::run_mdma` is the entry point invoked by the bus
5//! when software writes `$420B`.

10/// One DMA byte landing in VRAM (`$2118`/`$2119`), captured at transfer
11/// time. The byte value is the one the DMA actually read from the
12/// source this instant — immune to a coprocessor (e.g. Super FX)
13/// overwriting its source buffer between the transfer and a later VRAM
14/// dump (the double-buffer confound).
15#[derive(Debug, Clone, Copy)]
16pub struct DmaTraceEvent {
17    /// 24-bit A-bus source address of this byte.
18    pub src_full: u32,
19    /// PPU VRAM word address (`$2116/7` VMADD) the byte targets.
20    pub vram_word: u16,
21    /// B-bus register offset: the byte targets `$2100 + b_offset` (e.g.
22    /// `0x18`/`0x19` = `$2118`/`$2119` VRAM; `0x04` = `$2104` OAM).
23    pub b_offset: u8,
24    /// The transferred byte.
25    pub value: u8,
26    /// DMA channel (0-7) that performed this transfer — Mesen2's
27    /// `DebugEventInfo::DmaChannel` (read from `dma->GetActiveChannel()`),
28    /// driving the Event Viewer's per-channel filter.
29    pub channel: u8,
30    /// Completed-frame counter at the start of the owning DMA burst — lets a
31    /// consumer bucket DMA→VRAM bytes by frame (the per-VBlank budget check).
32    pub frame: u64,
33    /// PPU scanline at the start of the owning burst.
34    pub line: u16,
35    /// Exact horizontal master-clock (0..1363) at the transfer — Mesen2's
36    /// `GetHClock` (the Event Viewer plots events at `(hclock, line)`).
37    pub hclock: u16,
38    /// `true` if the burst started in the vertical-blank window
39    /// (`line >= vblank_start`).
40    pub blank: bool,
41    /// `true` if INIDISP (`$2100`) forced-blank (bit 7) was set at the write.
42    /// A VRAM write is safe iff `blank || force_blank` (V-blank *or* forced
43    /// blank); otherwise it races active display and the PPU drops it.
44    pub force_blank: bool,
45}

56/// The SNES DMA controller — 8 channels + a pair of global registers.
57#[derive(Default, serde::Serialize, serde::Deserialize)]
58pub struct Dma {
59    /// The eight DMA channels (indexed 0-7).
60    pub channels: [DmaChannel; 8],
61    /// `$420B MDMAEN` — last written value. Reading it returns 0 (write
62    /// only). The fast path is to call [`Dma::run_mdma`] directly with
63    /// the value being written.
64    pub mdmaen: u8,
65    /// `$420C HDMAEN` — HDMA enable mask. Stored but not yet acted upon
66    /// (HDMA is in a later phase).
67    pub hdmaen: u8,
68    /// Optional DMA→VRAM transfer-time trace. `None` = disabled. The
69    /// bus moves this into the per-transfer [`DmaBus`] view so the
70    /// view's `$2118/9` writes can record (source → VMADD → byte).
71    ///
72    /// Diagnostic only — not part of the save-state (`serde(skip)` →
73    /// defaults to `None` on restore).
74    #[serde(skip)]
75    pub dma_trace: Option<DmaTraceLog>,
76    /// In-progress sync-DMA cursor (Phase 5): the masked channel set and
77    /// the channel currently mid-transfer, so a burst can be driven in
78    /// scanline-bounded segments. `None` = no sync DMA in flight.
79    /// Transient run-state — never saved (a state is only taken at a
80    /// frame boundary, never mid-DMA).
81    #[serde(skip)]
82    pub mdma_cursor: Option<MdmaCursor>,
83}

85/// Resume state for a segmented sync DMA (Phase 5). The per-channel
86/// progress lives on each [`DmaChannel`]; this only tracks which channels
87/// were triggered and which one the driver is currently servicing.
88#[derive(Debug, Clone, Copy)]
89pub struct MdmaCursor {
90    /// The `$420B` channel mask for this burst.
91    pub mask: u8,
92    /// Next channel index (0..=8) to service; `8` = burst complete.
93    pub current_ch: u8,
94}

120    /// Read a channel register at the 24-bit B-bus(-style) offset
121    /// inside `$4300-$437F`. Returns `None` if the address is outside
122    /// the DMA register window.
123    #[must_use]
124    pub fn read_register(&self, offset: u16) -> Option<u8> {
125        if !(0x4300..=0x437F).contains(&offset) {
126            return None;
127        }
128        let channel = ((offset >> 4) & 0x07) as usize;
129        let reg = (offset & 0x0F) as u8;
130        Some(self.channels[channel].read(reg))
131    }
132
133    /// Write a channel register.
134    pub fn write_register(&mut self, offset: u16, value: u8) -> bool {
135        if !(0x4300..=0x437F).contains(&offset) {
136            return false;
137        }
138        let channel = ((offset >> 4) & 0x07) as usize;
139        let reg = (offset & 0x0F) as u8;
140        self.channels[channel].write(reg, value);
141        true
142    }
143
144    /// Execute the sync-DMA pass triggered by a write of `mask` to
145    /// `$420B MDMAEN`. Channels with their bit set in `mask` run in
146    /// ascending order; each transfers `das` bytes (or 64 KB if
147    /// `das == 0`). Returns the total number of bytes transferred
148    /// across all triggered channels (useful for cycle counting in
149    /// later phases).
150    pub fn run_mdma<B: DmaBus>(&mut self, bus: &mut B, mask: u8) -> u32 {
151        // One-shot wrapper: an unbounded budget runs the whole burst in a
152        // single call and clears the cursor (the legacy lump behaviour).
153        self.run_mdma_segment(bus, mask, u32::MAX)
154    }
155
156    /// Transfer at most `max_bytes` of the sync DMA triggered by `mask`,
157    /// resuming across calls (Phase 5). The first call on an idle
158    /// controller latches the cursor from `mask`; subsequent calls
159    /// continue the same burst (the passed `mask` is ignored while a
160    /// cursor is live). Channels run in ascending order; a channel is
161    /// serviced to completion before advancing. Returns the bytes
162    /// transferred *this* call; when the burst finishes,
163    /// [`Self::mdma_cursor`] is cleared back to `None`.
164    pub fn run_mdma_segment<B: DmaBus>(&mut self, bus: &mut B, mask: u8, max_bytes: u32) -> u32 {
165        if self.mdma_cursor.is_none() {
166            self.mdmaen = mask;
167            self.mdma_cursor = Some(MdmaCursor {
168                mask,
169                current_ch: 0,
170            });
171        }
172        let mut budget = max_bytes;
173        let mut total = 0u32;
174        // Work the masked channels in order, within this call's budget.
175        loop {
176            let cur = self.mdma_cursor.expect("cursor set above");
177            if cur.current_ch >= 8 || budget == 0 {
178                break;
179            }
180            let ch = cur.current_ch as usize;
181            if cur.mask & (1 << ch) != 0 {
182                // Tag the bus view with the active channel so its B-bus
183                // writes carry the channel (Mesen2 `dma->GetActiveChannel()`).
184                bus.set_active_channel(ch as u8);
185                let done = self.channels[ch].run_segment(bus, budget);
186                total += done;
187                budget -= done;
188                if self.channels[ch].seg_running {
189                    // Hit the budget mid-channel — resume this channel on
190                    // the next call (don't advance the cursor).
191                    break;
192                }
193            }
194            // Channel finished (or not masked): advance to the next.
195            self.mdma_cursor.as_mut().expect("cursor").current_ch += 1;
196        }
197        if self.mdma_cursor.is_some_and(|c| c.current_ch >= 8) {
198            self.mdma_cursor = None;
199        }
200        total
201    }
202
203    /// Frame-start HDMA initialisation. Called once per frame
204    /// (typically at the entry of the pre-render scanline). For each
205    /// channel whose bit is set in `$420C HDMAEN`, copies the table
206    /// start pointer, reads the first header byte, and in indirect
207    /// mode the first data pointer. Channels not enabled in
208    /// [`Self::hdmaen`] are left untouched.
209    /// Returns the master-cycle cost of the frame-start setup so the CPU
210    /// can be charged the stall (Phase 4). Per ares `cpu/dma.cpp`
211    /// `hdmaSetup`: `step(8)` overhead + one header read per enabled
212    /// channel; folded here into the canonical `18 + 8·channels` figure.
213    pub fn hdma_init<B: DmaBus>(&mut self, bus: &mut B) -> u32 {
214        // ares `hdmaReset` + Mesen2 `InitHdmaChannels`: every frame, reset the
215        // per-channel run flags for ALL 8 channels — enabled or not. Mesen
216        // notes NOT resetting `DoTransfer` glitches Aladdin / Super Ghouls'n
217        // Ghosts. `hdma_active` is luna's `!HdmaFinished`.
218        for ch in 0..8 {
219            self.channels[ch].hdma_active = true;
220            self.channels[ch].hdma_do_transfer = false;
221        }
222        if self.hdmaen == 0 {
223            return 0;
224        }
225        let mut enabled = 0u32;
226        for ch in 0..8 {
227            // "Set DoTransfer to true for ALL channels if any HDMA channel is
228            // enabled" (Mesen2 SnesDmaController.cpp:131, ares dma.cpp:143).
229            // This is what lets a channel enabled MID-frame transfer from its
230            // (stale) running table pointer — the references do NOT re-copy
231            // the source address on a mid-frame enable.
232            self.channels[ch].hdma_do_transfer = true;
233            if self.hdmaen & (1 << ch) != 0 {
234                // Enabled at V=0: copy source→pointer and read the first entry
235                // (may terminate the channel if the header is 0).
236                self.channels[ch].hdma_start_frame(bus);
237                enabled += 1;
238            }
239        }
240        HDMA_OVERHEAD_MCLK + 8 * enabled
241    }
242
243    /// Per-scanline HDMA step. Called once per visible scanline
244    /// (lines 0..=224 NTSC). Each enabled, still-active channel fires up
245    /// to one mode-pattern's worth of bytes through its configured B-bus
246    /// offset. Returns the **master-cycle cost** of the line's HDMA so the
247    /// CPU can be charged the stall (Phase 4): `18 + 8·bytes` when any
248    /// channel was active, else 0 (ares `cpu/dma.cpp` `hdmaRun`: `step(8)`
249    /// overhead + 8 mclk per transferred byte, folded into the canonical
250    /// 18-mclk per-scanline overhead).
251    pub fn hdma_run_line<B: DmaBus>(&mut self, bus: &mut B) -> u32 {
252        let mut bytes = 0u32;
253        let mut any_active = false;
254        for ch in 0..8 {
255            if self.hdmaen & (1 << ch) != 0 {
256                // Live HDMAEN gate (ares `hdmaActive()` / Mesen2 per-line
257                // `HdmaChannels & (1<<i)`): a channel enabled mid-frame runs
258                // from here using the `do_transfer`/pointer state left by
259                // `hdma_init` — the references keep that (stale) state; they do
260                // NOT re-copy the source on a mid-frame enable.
261                // Tag B-bus writes this channel makes with its channel id so
262                // the Event Viewer can plot them — faithful to Mesen2's
263                // `_activeChannel = HdmaChannelFlag | i` (SnesDmaController.cpp
264                // :264). The HDMA flag (bit 6) marks the event as HDMA-sourced;
265                // consumers mask `& 7` for the channel number.
266                bus.set_active_channel(HDMA_CHANNEL_FLAG | ch as u8);
267                // A channel active at line start does work this line.
268                any_active |= self.channels[ch].hdma_active;
269                // ares `hdmaFinished()`: this is the last active HDMA channel
270                // iff no higher-indexed enabled channel is still active. Snapshot
271                // it before the step (later channels haven't advanced this line —
272                // matching ares' transfer-all-then-advance-all ordering). Drives
273                // the indirect-terminator 1-byte reload quirk (`hdma_step_line`).
274                let last_active = !((ch + 1..8)
275                    .any(|j| self.hdmaen & (1 << j) != 0 && self.channels[j].hdma_active));
276                bytes += self.channels[ch].hdma_step_line(bus, last_active);
277            }
278        }
279        if any_active {
280            HDMA_OVERHEAD_MCLK + 8 * bytes
281        } else {
282            0
283        }
284    }
285}

287/// Fixed per-scanline HDMA overhead in master cycles when ≥1 channel is
288/// active — the canonical hardware figure (anomie / bsnes / higan),
289/// folding ares' `step(8)` + DMA-clock alignment + reload reads.
290const HDMA_OVERHEAD_MCLK: u32 = 18;

387    #[test]
388    fn run_mdma_segment_chunked_matches_one_shot() {
389        // Increment 0 invariant: splitting a sync DMA into arbitrary
390        // byte-bounded segments produces byte-identical B-bus output and
391        // identical channel end-state vs. running it in one shot.
392        // Two masked channels with odd sizes + different modes, so chunk
393        // boundaries fall mid-pattern and mid-channel.
394        fn configure(dma: &mut Dma, bus: &mut MockBus) {
395            for i in 0..256u32 {
396                bus.a[(0x7E_0000 + i) as usize] = i as u8;
397                bus.a[(0x7F_0000 + i) as usize] = 0x80u8.wrapping_add(i as u8);
398            }
399            // Ch0: mode 1 (2 regs, pattern [0,1]), BBAD $18, 70 bytes.
400            dma.channels[0].params = DmaParams::from_byte(0x01);
401            dma.channels[0].bbad = 0x18;
402            dma.channels[0].a_addr = 0x0000;
403            dma.channels[0].a_bank = 0x7E;
404            dma.channels[0].das = 70;
405            // Ch2: mode 0 (1 reg), BBAD $22, 37 bytes.
406            dma.channels[2].params = DmaParams::from_byte(0x00);
407            dma.channels[2].bbad = 0x22;
408            dma.channels[2].a_addr = 0x0000;
409            dma.channels[2].a_bank = 0x7F;
410            dma.channels[2].das = 37;
411        }
412        let mask = 0b0000_0101; // channels 0 and 2
413
414        // (a) one shot
415        let mut bus_ref = MockBus::new();
416        let mut dma_ref = Dma::new();
417        configure(&mut dma_ref, &mut bus_ref);
418        let n_ref = dma_ref.run_mdma(&mut bus_ref, mask);
419
420        // (b) segmented in cycling chunk sizes until the cursor clears
421        let mut bus_seg = MockBus::new();
422        let mut dma_seg = Dma::new();
423        configure(&mut dma_seg, &mut bus_seg);
424        let chunks = [1u32, 7, 8, 9, 170, 65535];
425        let mut n_seg = 0u32;
426        let mut i = 0;
427        while dma_seg.mdma_cursor.is_some() || n_seg == 0 {
428            n_seg += dma_seg.run_mdma_segment(&mut bus_seg, mask, chunks[i % chunks.len()]);
429            i += 1;
430            assert!(i < 1000, "segmented run did not terminate");
431        }
432
433        assert_eq!(n_ref, 70 + 37);
434        assert_eq!(n_seg, n_ref, "same total bytes");
435        assert_eq!(bus_seg.b, bus_ref.b, "byte-identical B-bus output");
436        assert!(
437            dma_seg.mdma_cursor.is_none(),
438            "cursor cleared at completion"
439        );
440        for ch in [0usize, 2] {
441            assert_eq!(dma_seg.channels[ch].a_addr, dma_ref.channels[ch].a_addr);
442            assert_eq!(dma_seg.channels[ch].das, dma_ref.channels[ch].das);
443            assert!(!dma_seg.channels[ch].seg_running);
444        }
445    }

448    fn hdma_charges_overhead_plus_per_byte_stall() {
449        // Phase 4 HDMA time cost: 18 mclk/scanline overhead when any
450        // channel is active, + 8 mclk per transferred byte. Table
451        // `02 11 00` = non-repeat 2-line entry (transfer line 1, gap
452        // line 2), then terminator. Mode 0 (1 byte/line) → BBAD $22.
453        let mut bus = MockBus::new();
454        bus.a[0x00_2000] = 0x02; // line count
455        bus.a[0x00_2001] = 0x11; // data byte
456        bus.a[0x00_2002] = 0x00; // terminator
457        let mut dma = Dma::new();
458        dma.channels[0].params = DmaParams::from_byte(0); // mode 0, direct
459        dma.channels[0].bbad = 0x22;
460        dma.channels[0].a_addr = 0x2000;
461        dma.channels[0].a_bank = 0x00;
462        dma.hdmaen = 0b0000_0001; // channel 0 HDMA enabled
463
464        // Frame setup: 18 overhead + 8 per enabled channel (1).
465        assert_eq!(dma.hdma_init(&mut bus), HDMA_OVERHEAD_MCLK + 8);

468        assert_eq!(dma.hdma_run_line(&mut bus), HDMA_OVERHEAD_MCLK + 8);
469        assert_eq!(bus.b[0x22], 0x11);
470
471        // Line 2 is an active gap (still active at line start, 0 bytes,
472        // reads the terminator) → overhead only.
473        assert_eq!(dma.hdma_run_line(&mut bus), HDMA_OVERHEAD_MCLK);

479    #[test]
480    fn hdma_mid_frame_enable_uses_stale_pointer_not_source() {
481        // Faithful port (ares hdmaSetup / Mesen2 InitHdmaChannels, audit #9): a
482        // channel enabled MID-frame does NOT re-copy its source address. When
483        // any HDMA is enabled at V=0, hdma_init arms `DoTransfer=true` for ALL
484        // channels, so a later-enabled channel runs from its (stale) running
485        // table pointer. luna previously lazily re-init'd from source — an
486        // invention absent from BOTH references (Yoshi's Island still renders
487        // correctly under the faithful model; validated by screenshot).
488        let mut bus = MockBus::new();
489        // ch7 is enabled at V=0 → makes hdma_init run its full loop.
490        bus.a[0x00_1000] = 0x01; // 1-line entry
491        bus.a[0x00_1001] = 0x11;
492        bus.a[0x00_1002] = 0x00; // terminator
493        // ch0's STALE pointer points at $00:3000 ($DD); its SOURCE is a
494        // different place ($00:2000 = $AA) that must NOT be read mid-frame.
495        bus.a[0x00_3000] = 0xDD;
496        bus.a[0x00_2000] = 0xAA;
497
498        let mut dma = Dma::new();
499        dma.channels[7].params = DmaParams::from_byte(0);
500        dma.channels[7].bbad = 0x30;
501        dma.channels[7].a_addr = 0x1000;
502        dma.channels[0].params = DmaParams::from_byte(0);
503        dma.channels[0].bbad = 0x22;
504        dma.channels[0].a_addr = 0x2000; // source — must stay untouched
505        dma.channels[0].a2a = 0x3000; // stale running pointer
506        dma.channels[0].ntlr = 0x02; // stale line counter (non-repeat gap next)
507
508        dma.hdmaen = 0x80; // only ch7 at V=0
509        dma.hdma_init(&mut bus);
510        assert!(
511            dma.channels[0].hdma_do_transfer,
512            "init armed DoTransfer for ALL channels"
513        );
514        assert_eq!(
515            dma.channels[0].a2a, 0x3000,
516            "source NOT copied into pointer"
517        );
518
519        // Mid-frame enable of ch0 → transfers from the stale pointer ($DD).
520        dma.hdmaen = 0x81;
521        dma.hdma_run_line(&mut bus);

555    #[test]
556    fn hdma_transfer_tags_writes_with_the_hdma_channel_flag() {
557        // Regression: HDMA per-scanline register writes must be tagged with
558        // their channel so the Event Viewer can plot them — faithful to
559        // Mesen2's `_activeChannel = HdmaChannelFlag | i`. The tag is
560        // `0x40 | ch`; consumers mask `& 7` for the channel number. (Without
561        // this, raster/gradient HDMA effects were invisible in the overlay.)
562        let mut bus = MockBus::new();
563        // Channel 3, mode 0, one transferred line of 0x5A → $2122.
564        bus.a[0x00_2000] = 0x01; // 1-line, non-repeat
565        bus.a[0x00_2001] = 0x5A; // data byte
566        bus.a[0x00_2002] = 0x00; // terminator
567        let mut dma = Dma::new();
568        dma.channels[3].params = DmaParams::from_byte(0); // mode 0, direct
569        dma.channels[3].bbad = 0x22;
570        dma.channels[3].a_addr = 0x2000;
571        dma.channels[3].a_bank = 0x00;
572        dma.hdmaen = 0b0000_1000; // channel 3
573
574        dma.hdma_init(&mut bus); // header reads only — no B-bus writes
575        assert!(
576            bus.tagged_writes.is_empty(),
577            "hdma_init must not write B-bus registers"
578        );
579
580        dma.hdma_run_line(&mut bus);
581        assert_eq!(
582            bus.tagged_writes,
583            vec![(HDMA_CHANNEL_FLAG | 3, 0x22, 0x5A)],
584            "the transfer is tagged with HdmaChannelFlag | channel 3"
585        );
586        // Consumer view: the channel number is the low 3 bits.
587        assert_eq!(bus.tagged_writes[0].0 & 7, 3);
588    }

590    #[test]
591    fn hdma_indirect_terminator_1byte_quirk_tracks_the_last_active_channel() {
592        // Row #10 integration: the "last active channel" that drives the
593        // 1-byte indirect-terminator quirk (ares `hdmaFinished()`) is computed
594        // across channels. Two indirect channels: ch0 terminates on line 1
595        // while ch1 (higher index) is still active → ch0 is NOT last, reads 2
596        // pointer bytes. ch1 terminates on line 2 as the sole survivor → it IS
597        // last, reads only 1.
598        let mut bus = MockBus::new();
599        // ch0 table $00:1000 — 1-line entry → $3456, then 0 terminator whose
600        // pointer bytes ($EE,$FF) sit at offsets 4/5.
601        for (i, b) in [0x01u8, 0x56, 0x34, 0x00, 0xEE, 0xFF].iter().enumerate() {
602            bus.a[0x00_1000 + i] = *b;
603        }
604        bus.a[0x7E_3456] = 0xAB;
605        // ch1 table $00:2000 — 2-line entry → $5678, then 0 terminator ($AA,$BB).
606        for (i, b) in [0x02u8, 0x78, 0x56, 0x00, 0xAA, 0xBB].iter().enumerate() {
607            bus.a[0x00_2000 + i] = *b;
608        }
609        bus.a[0x7E_5678] = 0xCD;
610        bus.a[0x7E_5679] = 0xEF;
611
612        let mut dma = Dma::new();
613        for (ch, addr) in [(0usize, 0x1000u16), (1, 0x2000)] {
614            dma.channels[ch].params = DmaParams::from_byte(0x40); // indirect, mode 0
615            dma.channels[ch].bbad = 0x22;
616            dma.channels[ch].a_addr = addr;
617            dma.channels[ch].a_bank = 0x00;
618            dma.channels[ch].dasb = 0x7E;
619        }
620        dma.hdmaen = 0b0000_0011;
621        dma.hdma_init(&mut bus);
622
623        // Line 1: ch0 terminates but ch1 is still active → NOT last → 2 bytes.
624        dma.hdma_run_line(&mut bus);
625        assert!(!dma.channels[0].hdma_active);
626        assert_eq!(
627            dma.channels[0].a2a, 0x1006,
628            "ch0 read header + 2 indirect bytes (a later channel is active)"
629        );
630        assert_eq!(dma.channels[0].das, 0xFFEE, "full 2-byte indirect pointer");
631        assert!(
632            dma.channels[1].hdma_active,
633            "ch1 still inside its 2-line entry"
634        );
635
636        // Line 2: ch1 terminates as the sole remaining channel → last → 1 byte.
637        dma.hdma_run_line(&mut bus);
638        assert!(!dma.channels[1].hdma_active);
639        assert_eq!(
640            dma.channels[1].a2a, 0x2005,
641            "ch1 read header + 1 indirect byte (last active channel quirk)"
642        );
643        assert_eq!(
644            dma.channels[1].das, 0xAA00,
645            "ares: firstByte << 8, one short"
646        );
647    }
