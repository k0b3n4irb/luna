6/// Can the DMA A-bus reach `addr`? It cannot touch the B-bus or CPU
7/// I/O — invalid reads return open bus and invalid writes are dropped
8/// (ares `dma.cpp:54-60`). Blocked in banks `00-3f`/`80-bf`:
9/// `2100-21ff` (B-bus), `4000-41ff`, `4200-421f`, `4300-437f`.
10const fn valid_a(addr: Addr24) -> bool {
11    if addr & 0x40_FF00 == 0x2100 {
12        return false;
13    }
14    if addr & 0x40_FE00 == 0x4000 {
15        return false;
16    }
17    if addr & 0x40_FFE0 == 0x4200 {
18        return false;
19    }
20    if addr & 0x40_FF80 == 0x4300 {
21        return false;
22    }
23    true
24}
25
26/// Is `addr` in WRAM — `$7E/$7F:xxxx` or the `$00-3f/$80-bf:0000-1fff`
27/// mirror? A transfer from WRAM to B-bus `$2180` (WMDATA) is invalid
28/// (ares `dma.cpp:94`).
29const fn is_wram_a(addr: Addr24) -> bool {
30    addr & 0xFE_0000 == 0x7E_0000 || addr & 0x40_E000 == 0x0000
31}
32
33/// A-bus read with the `validA` gate (open bus = 0 when blocked).
34fn read_a_valid<B: DmaBus>(bus: &mut B, addr: Addr24) -> u8 {
35    if valid_a(addr) { bus.read_a(addr) } else { 0 }
36}
37
38/// A-bus write with the `validA` gate (dropped when blocked).
39fn write_a_valid<B: DmaBus>(bus: &mut B, addr: Addr24, value: u8) {
40    if valid_a(addr) {
41        bus.write_a(addr, value);
42    }
43}

49/// Direction of the DMA transfer.
50#[derive(Debug, Clone, Copy, PartialEq, Eq, serde::Serialize, serde::Deserialize)]
51pub enum Direction {
52    /// A-bus → B-bus (CPU → PPU). The common case: uploading tiles,
53    /// palettes, OAM, etc.
54    AToB,
55    /// B-bus → A-bus (PPU → CPU). Rare in practice but legal.
56    BToA,
57}
58
59/// A-bus address increment behaviour.
60#[derive(Debug, Clone, Copy, PartialEq, Eq, serde::Serialize, serde::Deserialize)]
61pub enum Increment {
62    /// `+1` per byte (the canonical "uploading a linear buffer" case).
63    Up,
64    /// `-1` per byte.
65    Down,
66    /// Fixed (no increment / decrement). Used by some games that DMA
67    /// from a register-mapped source.
68    Fixed,
69}
70
71/// DMA transfer pattern (`$43x0` bits 0-2). Each mode specifies how
72/// many bytes per "cycle" and which B-bus offset each goes to.
73///
74/// Notation below: `b` = `BBADx`. So mode 1 writes alternating to
75/// `$2100 + b` and `$2100 + b + 1`.
76#[derive(Debug, Clone, Copy, PartialEq, Eq, serde::Serialize, serde::Deserialize)]
77pub enum TransferMode {
78    /// Mode 0 — 1 byte to `b`. (e.g. palette stream into `$2122`.)
79    OneByteOneReg,
80    /// Mode 1 — 2 bytes alternating to `b`, `b+1`. (e.g. VRAM via
81    /// `$2118`/`$2119`.)
82    TwoBytesTwoRegs,
83    /// Mode 2 — 2 bytes to the same register `b`. (e.g. OAM stream
84    /// into `$2104`.)
85    TwoBytesOneReg,
86    /// Mode 3 — 4 bytes: `b`, `b`, `b+1`, `b+1`. (e.g. color math
87    /// pair-of-pairs.)
88    FourBytesTwoPairs,
89    /// Mode 4 — 4 bytes: `b`, `b+1`, `b+2`, `b+3`. (BG mode 7 has
90    /// 4 sequential MMIO addresses for various blob uploads.)
91    FourBytesFourRegs,
92    /// Mode 5 — 4 bytes alternating to `b`, `b+1`, `b`, `b+1`. (Mirror
93    /// of mode 1 doubled — rare.)
94    FourBytesTwoRegsAlt,
95    /// Mode 6 — alias of mode 2.
96    TwoBytesOneRegAlt,
97    /// Mode 7 — alias of mode 3.
98    FourBytesTwoPairsAlt,
99}
100
101impl TransferMode {
102    /// Per-byte B-bus offset increments within one "cycle" of the
103    /// pattern. The pattern length is the slice's len.
104    #[must_use]
105    pub const fn pattern(self) -> &'static [u8] {
106        match self {
107            Self::OneByteOneReg => &[0],
108            Self::TwoBytesTwoRegs => &[0, 1],
109            Self::TwoBytesOneReg | Self::TwoBytesOneRegAlt => &[0, 0],
110            Self::FourBytesTwoPairs | Self::FourBytesTwoPairsAlt => &[0, 0, 1, 1],
111            Self::FourBytesFourRegs => &[0, 1, 2, 3],
112            Self::FourBytesTwoRegsAlt => &[0, 1, 0, 1],
113        }
114    }
115
116    /// Decode the low 3 bits of `$43x0`.
117    #[must_use]
118    pub const fn from_bits(bits: u8) -> Self {
119        match bits & 0x07 {
120            0 => Self::OneByteOneReg,
121            1 => Self::TwoBytesTwoRegs,
122            2 => Self::TwoBytesOneReg,
123            3 => Self::FourBytesTwoPairs,
124            4 => Self::FourBytesFourRegs,
125            5 => Self::FourBytesTwoRegsAlt,
126            6 => Self::TwoBytesOneRegAlt,
127            _ => Self::FourBytesTwoPairsAlt,
128        }
129    }
130}

132/// Fully-decoded `$43x0 DMAPx` register.
133#[derive(Debug, Clone, Copy, serde::Serialize, serde::Deserialize)]
134pub struct DmaParams {
135    /// A→B (CPU → PPU) or B→A (PPU → CPU).
136    pub direction: Direction,
137    /// How the A-bus address moves per byte (`+1`, `-1`, or fixed).
138    pub a_increment: Increment,
139    /// The mode pattern (1/2/4 bytes per cycle, register layout).
140    pub mode: TransferMode,
141    /// Bit 6 of `$43x0` — HDMA indirect-mode flag. Only honoured by
142    /// HDMA (not P1.2 scope).
143    pub hdma_indirect: bool,
144}
145
146impl DmaParams {
147    /// Decode `$43x0` into a `DmaParams`.
148    #[must_use]
149    pub const fn from_byte(byte: u8) -> Self {
150        let direction = if byte & 0x80 != 0 {
151            Direction::BToA
152        } else {
153            Direction::AToB
154        };
155        // Bits 3-4: 0b00 = +1 (Up), 0b10 = -1 (Down), anything with
156        // bit 3 set = Fixed.
157        let a_increment = match (byte >> 3) & 0x03 {
158            0b00 => Increment::Up,
159            0b10 => Increment::Down,
160            _ => Increment::Fixed,
161        };
162        Self {
163            direction,
164            a_increment,
165            mode: TransferMode::from_bits(byte),
166            hdma_indirect: byte & 0x40 != 0,
167        }
168    }
169
170    /// Encode back to `$43x0`. Useful for read-back semantics.
171    #[must_use]
172    pub fn to_byte(self) -> u8 {
173        let mut b = 0;
174        if self.direction == Direction::BToA {
175            b |= 0x80;
176        }
177        if self.hdma_indirect {
178            b |= 0x40;
179        }
180        b |= match self.a_increment {
181            Increment::Up => 0b00 << 3,
182            Increment::Down => 0b10 << 3,
183            Increment::Fixed => 0b01 << 3, // canonical "fixed" encoding
184        };
185        b |= match self.mode {
186            TransferMode::OneByteOneReg => 0,
187            TransferMode::TwoBytesTwoRegs => 1,
188            TransferMode::TwoBytesOneReg => 2,
189            TransferMode::FourBytesTwoPairs => 3,
190            TransferMode::FourBytesFourRegs => 4,
191            TransferMode::FourBytesTwoRegsAlt => 5,
192            TransferMode::TwoBytesOneRegAlt => 6,
193            TransferMode::FourBytesTwoPairsAlt => 7,
194        };
195        b
196    }
197}

203/// One of the eight DMA channels.
204#[derive(Debug, Clone, Copy, Default, serde::Serialize, serde::Deserialize)]
205pub struct DmaChannel {
206    /// `$43x0` — decoded parameters (drive the transfer logic).
207    pub params: DmaParams,
208    /// `$43x0` — the raw byte exactly as last written. On hardware the
209    /// register is a plain 8-bit R/W latch: reads return every bit
210    /// verbatim, including bit 5 (unused) and the independent
211    /// fixed/reverse increment bits, none of which survive a round-trip
212    /// through [`DmaParams`]. Games that park their direct page on the
213    /// `$43xx` block and use these registers as fast zero-page scratch
214    /// (e.g. Ms. Pac-Man's graphics decompressor stores its source
215    /// pointer's low byte in `$4340`) rely on lossless read-back; a
216    /// re-encode via [`DmaParams::to_byte`] silently corrupts a quarter
217    /// of all values (any with bit 5 set). ares `cpu/io.cpp` readIO
218    /// reconstructs the byte from independently-stored bits — same
219    /// observable result.
220    pub dmap: u8,
221    /// `$43x1` — B-bus base offset (`$2100 + bbad`).
222    pub bbad: u8,
223    /// `$43x2/$43x3` — A-bus address (low / high). 16-bit; the bank
224    /// comes from `a_bank`. The pair forms the 24-bit source/dest of
225    /// the A-bus side. Increments per byte per `params.a_increment`.
226    pub a_addr: u16,
227    /// `$43x4` — A-bus bank (does NOT auto-increment on bank crossing).
228    pub a_bank: u8,
229    /// `$43x5/$43x6` — DMA byte count. Counts DOWN to zero; `0x0000`
230    /// initially means **64 KB** (special wrap, not zero).
231    pub das: u16,
232    /// `$43x7` — HDMA indirect bank (HDMA only).
233    pub dasb: u8,
234    /// `$43x8/$43x9` — HDMA table pointer (HDMA only).
235    pub a2a: u16,
236    /// `$43xA` — HDMA line counter (HDMA only). Bit 7 = repeat flag,
237    /// bits 6:0 = lines remaining in the current entry. The whole
238    /// byte is the most-recently-fetched header; bits 6:0 are
239    /// decremented each scanline while the channel is active.
240    pub ntlr: u8,
241    /// `$43xB` / `$43xF` — unused mirror byte exposed in real hardware
242    /// (some games rely on the open-bus value).
243    pub unused: u8,
244    /// HDMA: whether the channel can still fire this frame — luna's name for
245    /// ares/Mesen2's `!HdmaFinished`. `hdma_init` resets it true for ALL
246    /// channels every frame; `hdma_start_frame` clears it when the table
247    /// starts with a terminator, and the per-line reload clears it when the
248    /// table runs out.
249    pub hdma_active: bool,
250    /// HDMA: whether **this** scanline fires a transfer (ares/Mesen2
251    /// `DoTransfer`). `hdma_init` sets it true for ALL channels when any HDMA
252    /// is enabled; reload-on-entry re-sets it; in between it is the repeat
253    /// flag's value.
254    pub hdma_do_transfer: bool,
255
256    // --- transient sync-DMA segment cursor (Phase 5) ---
257    // These hold a sync DMA's progress *between* [`Self::run_segment`]
258    // calls so a long burst can be split at scanline boundaries (letting
259    // HDMA preempt). They are run-state only — never part of a save-state
260    // (a state is only ever taken at a frame boundary, never mid-DMA).
261    /// Bytes still to transfer in the in-progress sync DMA (`0` = idle).
262    #[serde(skip)]
263    pub seg_remaining: u32,
264    /// Pattern position (`mode.pattern()` index) carried across segments.
265    #[serde(skip)]
266    pub seg_byte_idx: u8,
267    /// `true` while a sync DMA is mid-burst (resume vs. fresh start).
268    #[serde(skip)]
269    pub seg_running: bool,
270}

272impl Default for DmaParams {
273    fn default() -> Self {
274        Self {
275            direction: Direction::AToB,
276            a_increment: Increment::Up,
277            mode: TransferMode::OneByteOneReg,
278            hdma_indirect: false,
279        }
280    }
281}

283impl DmaChannel {
284    /// Build a channel in its post-reset state (all zeros).
285    #[must_use]
286    pub fn new() -> Self {
287        Self::default()
288    }
289
290    /// Read a per-channel register at offset `0x0..=0xF` (i.e. the
291    /// low nibble of `$43xN`).
292    #[must_use]
293    pub const fn read(&self, offset: u8) -> u8 {
294        match offset & 0x0F {
295            0x0 => self.dmap,
296            0x1 => self.bbad,
297            0x2 => self.a_addr as u8,
298            0x3 => (self.a_addr >> 8) as u8,
299            0x4 => self.a_bank,
300            0x5 => self.das as u8,
301            0x6 => (self.das >> 8) as u8,
302            0x7 => self.dasb,
303            0x8 => self.a2a as u8,
304            0x9 => (self.a2a >> 8) as u8,
305            0xA => self.ntlr,
306            0xB | 0xF => self.unused,
307            _ => 0xFF, // $43xC-$43xE: truly unused
308        }
309    }
310
311    /// Write a per-channel register at offset `0x0..=0xF`.
312    pub fn write(&mut self, offset: u8, value: u8) {
313        match offset & 0x0F {
314            0x0 => {
315                // Keep the raw latch for lossless read-back AND the
316                // decoded params that drive the transfer logic.
317                self.dmap = value;
318                self.params = DmaParams::from_byte(value);
319            }
320            0x1 => self.bbad = value,
321            0x2 => self.a_addr = (self.a_addr & 0xFF00) | u16::from(value),
322            0x3 => self.a_addr = (self.a_addr & 0x00FF) | (u16::from(value) << 8),
323            0x4 => self.a_bank = value,
324            0x5 => self.das = (self.das & 0xFF00) | u16::from(value),
325            0x6 => self.das = (self.das & 0x00FF) | (u16::from(value) << 8),
326            0x7 => self.dasb = value,
327            0x8 => self.a2a = (self.a2a & 0xFF00) | u16::from(value),
328            0x9 => self.a2a = (self.a2a & 0x00FF) | (u16::from(value) << 8),
329            0xA => self.ntlr = value,
330            0xB | 0xF => self.unused = value,
331            _ => {} // $43xC-$43xE: writes dropped
332        }
333    }
334
335    /// Execute the channel's DMA against the given bus, to completion.
336    /// Runs until `das` reaches zero (a starting `das = 0` transfers
337    /// 64 KB). Updates `a_addr` (with `params.a_increment`) and `das` as
338    /// it goes; the B-bus offset cycles through `params.mode.pattern()`.
339    /// Returns the number of bytes transferred.
340    ///
341    /// Thin wrapper over [`Self::run_segment`] with an unbounded budget.
342    pub fn run<B: DmaBus>(&mut self, bus: &mut B) -> u32 {
343        self.run_segment(bus, u32::MAX)
344    }
345
346    /// Transfer at most `max_bytes` of this channel's sync DMA, resuming
347    /// across calls (Phase 5: the driver splits a burst at scanline
348    /// boundaries so HDMA can preempt). The first call on an idle channel
349    /// latches the byte count (`das`, or 64 KB if `das == 0`) and pattern
350    /// position; subsequent calls continue from where the last left off.
351    /// Returns the bytes transferred *this* call; check
352    /// [`Self::seg_running`] to see whether the burst is finished.
353    pub fn run_segment<B: DmaBus>(&mut self, bus: &mut B, max_bytes: u32) -> u32 {
354        // Each byte transfer is ~8 master cycles on real hardware. We
355        // feed that to the bus's per-byte tick so coprocessors (SA-1,
356        // etc.) run interleaved with the DMA instead of waking up after a
357        // multi-kilocycle freeze — matches ares + Mesen2. See
358        // `DmaBus::tick`. (The master clock for APU/PPU/HDMA is advanced
359        // by the SnesBus driver per segment, not here.)
360        const MCLK_PER_DMA_BYTE: u32 = 8;
361
362        if !self.seg_running {
363            // 0x0000 means 64 KB (ares' `do { } while(--transferSize)`
364            // wraps a zero start to 65536 iterations).
365            self.seg_remaining = if self.das == 0 {
366                0x1_0000
367            } else {
368                u32::from(self.das)
369            };
370            self.seg_byte_idx = 0;
371            self.seg_running = true;
372        }
373
374        let pattern = self.params.mode.pattern();
375        let mut done = 0u32;
376        while done < max_bytes && self.seg_remaining > 0 {
377            let b_offset = self.bbad.wrapping_add(pattern[self.seg_byte_idx as usize]);
378            let a_addr: Addr24 = make_addr(self.a_bank, self.a_addr);
379            // A WRAM→WMDATA ($2180) transfer is invalid: the B-bus side
380            // is suppressed (ares dma.cpp:94).
381            let b_valid = !(b_offset == 0x80 && is_wram_a(a_addr));
382            match self.params.direction {
383                Direction::AToB => {
384                    let v = read_a_valid(bus, a_addr);
385                    if b_valid {
386                        bus.write_b(b_offset, v);
387                    }
388                }
389                Direction::BToA => {
390                    let v = if b_valid { bus.read_b(b_offset) } else { 0 };
391                    write_a_valid(bus, a_addr, v);
392                }
393            }
394            // Advance A-bus address per params.
395            self.a_addr = match self.params.a_increment {
396                Increment::Up => self.a_addr.wrapping_add(1),
397                Increment::Down => self.a_addr.wrapping_sub(1),
398                Increment::Fixed => self.a_addr,
399            };
400            self.seg_byte_idx = ((self.seg_byte_idx as usize + 1) % pattern.len()) as u8;
401            self.seg_remaining -= 1;
402            done += 1;
403            // Cooperative coprocessor tick after each byte.
404            bus.tick(MCLK_PER_DMA_BYTE);
405        }
406        if self.seg_remaining == 0 {
407            // Hardware leaves `das = 0` at the end of a sync DMA.
408            self.das = 0;
409            self.seg_running = false;
410        }
411        done
412    }
413
414    // ---------------------------------------------------------------
415    // HDMA — per-scanline transfers
416    // ---------------------------------------------------------------
417
418    /// Frame-start HDMA setup for this channel. Resets the running
419    /// table pointer (`a2a`) to `a_addr`, reads the first header byte
420    /// into `ntlr`, and in indirect mode also reads the 16-bit data
421    /// pointer into `das`.
422    ///
423    /// Sets `hdma_active = true` and `hdma_do_transfer = true` if the
424    /// channel has at least one entry; sets them false (channel
425    /// disabled for the frame) if the table starts with a 0 byte.
426    pub fn hdma_start_frame<B: DmaBus>(&mut self, bus: &mut B) {
427        self.a2a = self.a_addr;
428        let header = read_a_valid(bus, make_addr(self.a_bank, self.a2a));
429        self.a2a = self.a2a.wrapping_add(1);
430        self.ntlr = header;
431        if header == 0 {
432            self.hdma_active = false;
433            self.hdma_do_transfer = false;
434            return;
435        }
436        if self.params.hdma_indirect {
437            let lo = read_a_valid(bus, make_addr(self.a_bank, self.a2a));
438            self.a2a = self.a2a.wrapping_add(1);
439            let hi = read_a_valid(bus, make_addr(self.a_bank, self.a2a));
440            self.a2a = self.a2a.wrapping_add(1);
441            self.das = u16::from(lo) | (u16::from(hi) << 8);
442        }
443        self.hdma_active = true;
444        self.hdma_do_transfer = true;
445    }
446
447    /// One HDMA scanline step. If `hdma_do_transfer` is set, transfers
448    /// one "unit" (1, 2, or 4 bytes per [`TransferMode`]) from the
449    /// table/data pointer to the B-bus. Decrements the line counter
450    /// (`ntlr` bits 6:0); when it reaches 0, reads the next header
451    /// byte and either reloads from a new entry, or terminates the
452    /// channel for the rest of the frame.
453    ///
454    /// `last_active` is true when this is the last still-active HDMA
455    /// channel (ares `hdmaFinished()`: no higher-indexed channel is
456    /// active). It only matters for the indirect-reload quirk below.
457    ///
458    /// Returns the number of bytes transferred on this line (0 if
459    /// the channel is done or this line was a non-repeat gap).
460    pub fn hdma_step_line<B: DmaBus>(&mut self, bus: &mut B, last_active: bool) -> u32 {
461        if !self.hdma_active {
462            return 0;
463        }
464        let mut transferred: u32 = 0;
465        if self.hdma_do_transfer {
466            // Emit one full mode "unit". Pattern length = unit size.
467            let pattern = self.params.mode.pattern();
468            for &b_off in pattern {
469                let b_offset = self.bbad.wrapping_add(b_off);
470                let src = if self.params.hdma_indirect {
471                    make_addr(self.dasb, self.das)
472                } else {
473                    make_addr(self.a_bank, self.a2a)
474                };
475                let value = read_a_valid(bus, src);
476                if self.params.hdma_indirect {
477                    self.das = self.das.wrapping_add(1);
478                } else {
479                    self.a2a = self.a2a.wrapping_add(1);
480                }
481                // Suppress the WRAM→WMDATA ($2180) B-bus write.
482                if !(b_offset == 0x80 && is_wram_a(src)) {
483                    bus.write_b(b_offset, value);
484                }
485                transferred += 1;
486            }
487        }
488        // Decrement the FULL 8-bit line counter (repeat bit included) and
489        // reload a new header when the low 7 bits reach 0 — the ares/Mesen2
490        // model (`Gsu`-adjacent `SnesDmaController`: `LineCounterAndRepeat--`
491        // then reload if `& 0x7F == 0`). This is why a header whose low 7
492        // bits are 0 (e.g. `$80`) is a **128-line** entry, not a 1-line one:
493        // the counter walks $80→$7F→…→$00 before the low 7 bits hit 0.
494        // Contra III's title HDMA opens with `$80` for the 128-line logo
495        // band; the old `(ntlr & 0x7F).saturating_sub(1)` treated it as a
496        // single line, so the logo vanished after one scanline.
497        self.ntlr = self.ntlr.wrapping_sub(1);
498        if self.ntlr & 0x7F == 0 {
499            // Entry exhausted: read next header byte (ares `hdmaReload`,
500            // dma.cpp:152-171). A 0 header terminates the channel; a
501            // non-zero one arms a fresh entry that transfers next line.
502            let next = read_a_valid(bus, make_addr(self.a_bank, self.a2a));
503            self.a2a = self.a2a.wrapping_add(1);
504            self.ntlr = next;
505            let completed = next == 0;
506            self.hdma_active = !completed;
507            self.hdma_do_transfer = !completed;
508            if self.params.hdma_indirect {
509                // ares reads the indirect data pointer on EVERY reload,
510                // including a terminating (0) header (dma.cpp:162-169). The
511                // first byte lands in the high half (`data << 8`); the second
512                // shifts it down and takes the high half. But if this reload
513                // both terminates the channel AND it is the last active HDMA
514                // channel (`hdmaCompleted && hdmaFinished()`), ares reads only
515                // that FIRST byte — the address ends one short and one A-bus
516                // cycle is saved (the row #10 last-active-channel quirk). For
517                // a live entry (or a non-last terminator) the normal 2-byte
518                // load runs, giving `das = lo | hi << 8` exactly as before.
519                let first = read_a_valid(bus, make_addr(self.a_bank, self.a2a));
520                self.a2a = self.a2a.wrapping_add(1);
521                self.das = u16::from(first) << 8;
522                if !(completed && last_active) {
523                    let second = read_a_valid(bus, make_addr(self.a_bank, self.a2a));
524                    self.a2a = self.a2a.wrapping_add(1);
525                    self.das = (u16::from(second) << 8) | (self.das >> 8);
526                }
527            }
528        } else {
529            // Continuation line: transfer iff the (decremented) repeat bit
530            // is set — repeat entries hold bit 7 high until they reload.
531            self.hdma_do_transfer = self.ntlr & 0x80 != 0;
532        }

1064    #[test]
1065    fn write_then_read_round_trip_on_every_offset() {
1066        let mut ch = DmaChannel::new();
1067        // Walk every offset, write a known byte, read it back.
1068        for (off, val) in (0..=0x0Fu8).map(|o| (o, o.wrapping_mul(7) ^ 0xAA)) {
1069            ch.write(off, val);
1070            // Offsets $C..$E are write-discarded.
1071            if matches!(off, 0xC..=0xE) {
1072                continue;
1073            }
1074            // Every other offset — including $0 (DMAPx) — is a plain
1075            // 8-bit latch and must round-trip every bit verbatim.
1076            assert_eq!(ch.read(off), val, "offset ${off:X} must round-trip");
1077        }
1078    }

1080    #[test]
1081    fn dmap_reads_back_raw_byte_not_reencoded() {
1082        // Regression: $43x0 is a full 8-bit R/W latch. Re-encoding the
1083        // read through DmaParams::to_byte() drops bit 5 (and folds the
1084        // independent fixed/reverse increment bits), corrupting a
1085        // quarter of all values. Ms. Pac-Man parks DP on $4300 and
1086        // stores its decompressor's source-pointer low byte in $4340
1087        // (DMAP4); a lossy read-back desyncs the LZ stream so the
1088        // end-marker is never seen → infinite boot loop.
1089        let mut ch = DmaChannel::new();
1090        for v in 0u8..=0xFF {
1091            ch.write(0x0, v);
1092            assert_eq!(ch.read(0x0), v, "DMAPx must read back ${v:02X} verbatim");
1093        }
1094    }
