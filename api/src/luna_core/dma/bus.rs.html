1//! The [`DmaBus`] trait — the minimum view of the system DMA needs.
2//!
3//! Splitting A-bus and B-bus access this way matches the hardware
4//! topology: DMA shuffles bytes between two physical buses, the CPU's
5//! 24-bit one (A) and the PPU's 8-bit `$2100 + offset` one (B). It also
6//! lets the production [`luna_core::Snes`] do direct WRAM ↔ PPU memory
7//! access without recursive [`luna_bus::Bus`] traffic (which would
8//! otherwise re-enter the dispatch and re-tick `io_cycle`).
9
10use luna_bus::Addr24;
11
12/// View of the system exposed to a DMA channel during a transfer.
13pub trait DmaBus {
14    /// Read one byte from the CPU's 24-bit A-bus.
15    fn read_a(&mut self, addr: Addr24) -> u8;
16
17    /// Write one byte to the CPU's 24-bit A-bus.
18    fn write_a(&mut self, addr: Addr24, value: u8);
19
20    /// Read one byte from the PPU's B-bus at `$2100 + b_offset`.
21    fn read_b(&mut self, b_offset: u8) -> u8;
22
23    /// Write one byte to the PPU's B-bus at `$2100 + b_offset`.
24    fn write_b(&mut self, b_offset: u8, value: u8);
25
26    /// Per-byte cooperative tick — called by the DMA channel after
27    /// every transferred byte so cartridge coprocessors (SA-1, Super
28    /// FX, DSP-N, …) get a chance to run *during* the burst instead
29    /// of being frozen until the main CPU's instruction step
30    /// completes ~4 kHz of mclks later.
31    ///
32    /// Matches ares (`coprocessor/sa1/sa1.cpp:63-94` — `Thread::step(2) + Thread::synchronize(cpu)` interleaved by the cooperative scheduler) and Mesen2 (`SnesDmaController::CopyDmaByte` → `IncMasterClock4` → `_memoryManager->Exec()` → `cart->SyncCoprocessors()` → `Sa1::Run()` every 2 mclks during DMA).
33    ///
34    /// Default `no-op` for the test mock buses; production buses
35    /// override to forward to the coprocessor's `step_coproc`.
36    fn tick(&mut self, _mcycles: u32) {}
37
38    /// Set the DMA channel (0-7) currently driving the bus. The controller
39    /// calls this before each channel's transfer so the bus view can tag
40    /// captured B-bus writes with their source channel — mirrors Mesen2's
41    /// `dma->GetActiveChannel()` (`SnesEventManager.cpp:40-42`). Default
42    /// no-op for the test mock buses.
43    fn set_active_channel(&mut self, _channel: u8) {}
44}
