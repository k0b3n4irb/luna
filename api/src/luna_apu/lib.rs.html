1//! SNES APU — orchestrates the `luna-cpu-spc700` core, 64 KB of ARAM,
2//! the IPL ROM mapped over the top page, and the four mailbox ports
3//! facing the main CPU.
4//!
5//! The audio DSP (sample generation, voice mixing) is the
6//! cycle-accurate ares port in [`dsp`]; `$F2`/`$F3` route through it.
7//!
8//! # Mailbox direction model
9//!
10//! The four SNES mailbox ports are actually **two** registers per
11//! port (one per direction):
12//!
13//! ```text
14//!                     ┌─────────────────────────────┐
15//!   CPU side          │       inside the APU         │       SPC700
16//! ─────────────       │                             │     ───────────
17//!   $2140-$2143 ──W──▶│ to_spc_ports[0..3] (RAM)    │◀──R─ $00F4-$00F7
18//!                     │                             │
19//!   $2140-$2143 ◀──R──│ to_cpu_ports[0..3] (RAM)    │──W──▶ $00F4-$00F7
20//!                     └─────────────────────────────┘
21//! ```
22//!
23//! A read at the same port from either side returns what **the other
24//! side last wrote**, not your own writes. This matches real hardware
25//! and is what the IPL ROM relies on for the boot handshake.
26//!
27//! # Clock ratio
28//!
29//! SNES master = 21.477 MHz, SPC = 1.024 MHz, ratio ≈ 21. We run one
30//! SPC instruction per 84 master cycles, which assumes ~4-cycle
31//! average SPC instructions — good enough until we wire timer-driven
32//! cycle accounting in.

36/// Read one ARAM byte as the **SPC700** sees it: `$FFC0-$FFFF` returns
37/// the 64-byte IPL ROM while `$F1` bit 7 is set, otherwise the physical
38/// RAM underneath. The DSP reads physical ARAM directly and bypasses
39/// this overlay (ares: the IPL mapping is SMP-side only).
40const fn aram_with_ipl(aram: &[u8; 0x10000], control: u8, addr: u16) -> u8 {
41    if addr >= IPL_ROM_BASE && control & 0x80 != 0 {
42        IPL_ROM[(addr - IPL_ROM_BASE) as usize]
43    } else {
44        aram[addr as usize]
45    }
46}

48/// Whether an SPC700 bus access uses the **internal** wait-state field
49/// (`$F0` bits 6-7) rather than the external one (bits 4-5). Per ares
50/// `smp/timing.cpp::wait`: IO registers (`$00F0-$00FF`) and the IPLROM
51/// (`$FFC0-$FFFF`, while `$F1` bit 7 exposes it). Idle cycles are also
52/// internal (handled at the call site — they have no address).
53const fn spc_ws_internal(addr: u16, control: u8) -> bool {
54    (addr & 0xFFF0) == 0x00F0 || (addr >= IPL_ROM_BASE && control & 0x80 != 0)
55}

61/// `serde` helper for a heap-boxed fixed byte array (`Box<[u8; N]>`),
62/// which `serde_bytes` does not cover directly. Used by the save-state
63/// machinery for the 64 KB ARAM.
64pub(crate) mod boxed_byte_array {
65    use serde::{Deserialize, Deserializer, Serializer};
66
67    /// Serialize `Box<[u8; N]>` as raw bytes (the `&Box` from serde's
68    /// `with` call site deref-coerces to this `&[u8; N]`).
69    pub(crate) fn serialize<S, const N: usize>(
70        data: &[u8; N],
71        serializer: S,
72    ) -> Result<S::Ok, S::Error>
73    where
74        S: Serializer,
75    {
76        serializer.serialize_bytes(&data[..])
77    }
78
79    /// Deserialize a byte blob back into `Box<[u8; N]>` (length must match).
80    pub(crate) fn deserialize<'de, D, const N: usize>(
81        deserializer: D,
82    ) -> Result<Box<[u8; N]>, D::Error>
83    where
84        D: Deserializer<'de>,
85    {
86        let bytes = <serde_bytes::ByteBuf>::deserialize(deserializer)?;
87        let arr: [u8; N] = bytes
88            .into_vec()
89            .try_into()
90            .map_err(|_| serde::de::Error::custom("byte array length mismatch"))?;
91        Ok(Box::new(arr))
92    }
93}

95/// Nominal master cycles per SPC instruction (~4 SPC cycles × the
96/// 20.97 master/SPC ratio). Since Phase 2 this is **no longer** the
97/// scheduler quantum — `Apu::step` charges each instruction its real
98/// per-opcode cost — but it is kept as a convenience multiplier for
99/// tests that want "≈N instructions of headroom".
100pub const MASTER_CYCLES_PER_SPC_STEP: u32 = 84;

102/// NTSC SNES master clock (Hz) — the CPU/PPU timebase.
103pub const MASTER_CLOCK_HZ: u64 = 21_477_272;

105/// SPC700 / S-DSP clock (Hz): the APU crystal ÷ 24. The crystal is
106/// nominally 24.576 MHz (→ 1.024 MHz) but real hardware measures
107/// ~24.607 MHz; ares (`apuFrequency = 32040·768`) and Mesen2 both use the
108/// measured value → `24_606_720` ÷ 24 = **`1_025_280` Hz** (and a `32_040` Hz
109/// DSP output). luna's textbook 1.024 MHz ran the SPC ~0.125 % slow, which
110/// shifts the CPU↔SPC clock alignment during the boot/upload handshake
111/// (differential vs ares: this value moves luna's IPL-upload-loop exit
112/// measurably closer to ares — necessary, though not alone sufficient,
113/// for the Tales of Phantasia OP).
114pub const SPC_CLOCK_HZ: u64 = 1_025_280;
115
116/// SPC clock in Mesen2's **2× domain** (`_state.Cycle` units): each SPC
117/// cycle is 2 of these, so the master→SPC target `master_clock × clockRatio`
118/// has half-SPC-cycle resolution. `clockRatio = spcSampleRate·64 /
119/// masterClock`, and `spcSampleRate·64 = 32_040 · 64 = 2_050_560 = 2 ×
120/// SPC_CLOCK_HZ` — the integer numerator luna uses for the zero-drift
121/// fractional accumulator (`cpu_target_2x`).
122pub const SPC_2X_HZ: u64 = 2 * SPC_CLOCK_HZ;
123
124/// SPC cycles per audio sample: 32 SPC cycles → one DSP sample. At
125/// [`SPC_CLOCK_HZ`] this yields the `32_040` Hz output rate ares and Mesen2
126/// produce (the host audio backend resamples to the device rate).
127pub const SPC_CYCLES_PER_SAMPLE: u32 = 32;

129/// SPC700 wait-state dividers (ares `smp/timing.cpp`), indexed by the 2-bit
130/// wait-state field from `$F0` (external bits 4-5 / internal bits 6-7).
131/// Expressed in luna's calibrated ws=0 baseline (where one access advances
132/// `spc_pos_2x` by 2 and the timer/sample subdividers by 1), so `[0]` is a
133/// no-op vs the old fixed model:
134/// - clock = ares `cycleWaitStates {2,4,10,20}` (the 2× domain advance);
135/// - timer = ares `timerWaitStates {2,4,8,16}` ÷ 2 → `{1,2,4,8}` — the
136///   hardware glitch where clock dividers 8/16 run as 10/20 but the timers
137///   do **not**, so clock and timer diverge at ws=2/3;
138/// - sample = clock ÷ 2 → `{1,2,5,10}` — the DSP follows the SMP clock, not
139///   the timer rate.
140const SPC_CLOCK_WAIT_2X: [u32; 4] = [2, 4, 10, 20];
141const SPC_TIMER_WAIT: [u32; 4] = [1, 2, 4, 8];
142const SPC_SAMPLE_WAIT: [u32; 4] = [1, 2, 5, 10];

178/// All APU state owned by [`Apu`]: SPC700 core + 64 KB ARAM + the two
179/// 4-byte mailbox arrays.
180#[derive(serde::Serialize, serde::Deserialize)]
181pub struct Apu {
182    /// The SPC700 CPU.
183    pub cpu: Spc700,
184    /// Cycle-accurate ares-port S-DSP. Owns its own register file
185    /// (mirrored into `dsp_regs` for legacy introspection), Voice[8],
186    /// Echo, Noise, BRR, Latch, Clock, `MainVol` state. `tick_voices`
187    /// drives one `dsp.main()` per 32 SPC cycles → one stereo sample.
188    pub dsp: dsp::Dsp,
189    /// 64 KB of physical audio RAM. The 64-byte IPL ROM is *not* baked
190    /// in — it's a read overlay over `$FFC0..=$FFFF` on the SPC side,
191    /// gated on `$F1` bit 7 (see [`aram_with_ipl`] / [`Apu::peek`]). The
192    /// DSP reads this array directly, bypassing the overlay.
193    #[serde(with = "boxed_byte_array")]
194    pub aram: Box<[u8; 0x10000]>,
195    /// CPU → SPC mailbox (CPU writes `$2140-$2143`, SPC reads `$F4-$F7`).
196    pub to_spc_ports: [u8; 4],
197    /// SPC → CPU mailbox (SPC writes `$F4-$F7`, CPU reads `$2140-$2143`).
198    /// This is Mesen2's `OutputReg`; the CPU reads it as-of the SPC run to
199    /// `target-1` (see [`Self::run_to_target`]) — the echo half-cycle.
200    pub to_cpu_ports: [u8; 4],
201    /// Shadow of the CPU→SPC mailbox (Mesen2 `NewCpuRegs`): the value the
202    /// CPU most recently wrote. Copied into [`Self::to_spc_ports`]
203    /// (`CpuRegs`, what the SPC actually reads) either immediately or one
204    /// SPC cycle later — the write-visibility delay (see
205    /// [`Self::cpu_write_port`] / [`Self::pending_cpu_reg_update`]).
206    #[serde(default)]
207    new_to_spc_ports: [u8; 4],
208    /// Mesen2 `_pendingCpuRegUpdate`: when a CPU mailbox write lands while
209    /// the SPC is more than 1 (2×) unit behind, the SPC sees the new value
210    /// only after one more SPC cycle. Applied in [`Self::run_one_cycle`]
211    /// (Mesen `ProcessCycle`). Lets e.g. Kishin Douji Zenki boot.
212    #[serde(default)]
213    pending_cpu_reg_update: bool,
214    /// SPC executed position in the **2× SPC clock domain** (Mesen2
215    /// `_state.Cycle`): `SPC_2X_HZ` units, advanced `+2` per SPC cycle. The
216    /// driver runs the SPC up to `cpu_target_2x - 1` so it sits half a 1×
217    /// SPC cycle behind the CPU at every mailbox access.
218    #[serde(default)]
219    spc_pos_2x: u64,
220    /// CPU position in the same 2× domain (`master_clock × clockRatio`),
221    /// accumulated with zero drift. The SPC chases `cpu_target_2x - 1`.
222    #[serde(default)]
223    cpu_target_2x: u64,
224    /// Fractional remainder of [`Self::cpu_target_2x`] (`< MASTER_CLOCK_HZ`):
225    /// the sub-unit phase of `master_clock × clockRatio`, needed to decide a
226    /// mailbox write's immediate-vs-pending visibility exactly like Mesen.
227    #[serde(default)]
228    cpu_clock_frac: u64,
229    /// `$F1` SPC control register — bit 7 (use IPL ROM) is the only
230    /// bit we honour for now; the rest are stored verbatim for round-
231    /// trip diagnostics.
232    pub control: u8,
233    /// `$F0` TEST register. Bit 0 = `timersDisable`, bit 3 =
234    /// `timersEnable` (ares io.cpp:81-94) — together they gate timer
235    /// advance. Reset default `0x0A` (timersEnable + ramWritable set)
236    /// keeps timers running, matching the ares power-on state. The
237    /// RAM-writable/disable and wait-state bits are stored, not modelled.
238    pub test: u8,
239    /// `true` once the SPC has executed at least one instruction past
240    /// the IPL ROM region (i.e. it `JMP`'d into user code). When that
241    /// happens we expect uploaded music driver code; until our
242    /// opcode coverage catches up to the most popular drivers, this
243    /// is mostly informational.
244    pub past_iplrom: bool,
245
246    // ------------- DSP (audio synth) — owned by `dsp` -------------
247    /// Last value written to `$F2` — the DSP register-index port.
248    /// `$F3` reads/writes `dsp.registers[dsp_index & 0x7F]` via
249    /// [`dsp::Dsp::read`] / [`dsp::Dsp::write`], which fans out the
250    /// data to the relevant Voice/Echo/Noise/etc field.
251    pub dsp_index: u8,
252    /// Mixed L audio output for the most recent sample produced by
253    /// [`dsp::Dsp::main`]. Consumers (audio backend) prefer
254    /// [`Self::audio_queue`] for proper sample-rate output, but a few
255    /// debug paths read this last-sample snapshot.
256    pub audio_left: i16,
257    /// Mixed R audio output — companion to [`Self::audio_left`].
258    pub audio_right: i16,
259    /// FIFO of stereo PCM samples ready for the host audio backend
260    /// to consume. Sized to a few frames at 32 kHz so brief audio
261    /// underruns don't cause sustained drift; the host drains it on
262    /// every UI frame.
263    ///
264    /// Transient playback buffer — not part of the save-state. It
265    /// `serde(skip)`-defaults to an empty queue on restore.
266    #[serde(skip)]
267    pub audio_queue: std::collections::VecDeque<(i16, i16)>,
268    /// SPC cycles owed to the audio-sample tick (32-cycle base
269    /// clock). Drives both ADSR rate progression and the per-voice
270    /// position counter.
271    sample_tick_deficit: u32,
272
273    // ------------- Timers -------------
274    /// Reload values at `$FA` (T0), `$FB` (T1), `$FC` (T2). A value
275    /// of `0` is treated as `256` per the SPC700 spec.
276    pub timer_reload: [u8; 3],
277    /// 4-bit output counters at `$FD` (T0), `$FE` (T1), `$FF` (T2).
278    /// Increment when the timer crosses its reload; reset to 0 when
279    /// the CPU reads the register.
280    pub timer_output: [u8; 3],
281    /// Internal divider counter for each timer.
282    pub timer_internal: [u16; 3],
283    /// `true` if `$F1.bits[0..2]` enabled the corresponding timer.
284    /// A 0→1 transition resets the internal counter on real HW; we
285    /// honour that.
286    pub timer_enabled: [bool; 3],
287    /// Sub-tick counter for the timer base clocks. T0/T1 tick once
288    /// every 128 SPC cycles; T2 ticks once every 16 SPC cycles. We
289    /// derive both from the same counter rather than running them
290    /// separately.
291    pub timer_subdivider: u32,
292
293    /// Optional full SPC700 instruction trace: `(events, max_events)`. A
294    /// pre-opcode register snapshot per SPC700 instruction, capped at
295    /// `max_events` (ring buffer: drops the oldest half when full).
296    /// Transient — `serde(skip)` so it never enters a save-state.
297    #[serde(skip)]
298    spc_trace: Option<(Vec<Spc700TraceEvent>, usize)>,
299}

308    /// Build a freshly-reset APU. ARAM is zeroed, the IPL ROM is
309    /// mapped at `$FFC0..=$FFFF`, the SPC700 is reset (which reads
310    /// its reset vector from the IPL ROM and lands at `$FFC0`).
311    #[must_use]

314        // read overlay over $FFC0-$FFFF (gated on $F1 bit 7, applied in
315        // the bus read path) rather than baked in — so a game that
316        // clears bit 7 reclaims the underlying RAM, and the DSP always
317        // reads physical ARAM.
318        let aram: Box<[u8; 0x10000]> = vec![0u8; 0x10000]
319            .into_boxed_slice()
320            .try_into()
321            .expect("64 KB slice into fixed array");

323            cpu: Spc700::new(),
324            dsp: dsp::Dsp::new(),

326            to_spc_ports: [0; 4],

329            pending_cpu_reg_update: false,
330            spc_pos_2x: 0,
331            cpu_target_2x: 0,
332            cpu_clock_frac: 0,
333            control: 0x80, // bit 7: IPL ROM exposed
334            test: 0x0A,    // timersEnable + ramWritable (ares power-on)
335            past_iplrom: false,
336            dsp_index: 0,
337            audio_left: 0,
338            audio_right: 0,
339            audio_queue: std::collections::VecDeque::with_capacity(8192),
340            sample_tick_deficit: 0,
341            timer_reload: [0; 3],
342            timer_output: [0; 3],
343            timer_internal: [0; 3],

375    /// Tick the three SPC timers by `spc_cycles` of headroom.
376    ///
377    /// T0 / T1 base clock: 8 kHz = one tick every 128 SPC cycles.
378    /// T2 base clock: 64 kHz = one tick every 16 SPC cycles.
379    ///
380    /// On a tick, if the timer is enabled, the internal counter
381    /// increments; when it reaches the reload value (0 = 256), the
382    /// 4-bit output counter wraps-increments and the internal counter
383    /// resets.
384    fn tick_timers(&mut self, spc_cycles: u32) {
385        let before = self.timer_subdivider;
386        self.timer_subdivider = self.timer_subdivider.wrapping_add(spc_cycles);
387        let after = self.timer_subdivider;
388
389        // $F0 TEST master gate (ares timing.cpp:45-49): when timersEnable
390        // (bit 3) is clear or timersDisable (bit 0) is set, the stage→
391        // output propagation is suppressed (timers freeze). The clock
392        // divider (`timer_subdivider`) keeps running above, so phase
393        // resumes when re-enabled.
394        if self.test & 0x08 == 0 || self.test & 0x01 != 0 {
395            return;
396        }
397
398        // T2 ticks at the 16-cycle boundary.
399        let t2_ticks = (after / 16) - (before / 16);
400        for _ in 0..t2_ticks {
401            self.tick_one_timer(2);
402        }
403        // T0 and T1 tick at the 128-cycle boundary.
404        let slow_ticks = (after / 128) - (before / 128);
405        for _ in 0..slow_ticks {
406            self.tick_one_timer(0);
407            self.tick_one_timer(1);
408        }
409    }

411    /// Tick the per-voice envelope state machines. We accumulate SPC
412    /// cycles in `sample_tick_deficit`; each time it crosses a
413    /// 32-cycle boundary the DSP advances by one 32 kHz audio sample
414    /// — which is when the ADSR phase machine actually moves and
415    /// the per-voice position counter increments.
416    fn tick_voices(&mut self, spc_cycles: u32) {
417        self.sample_tick_deficit = self.sample_tick_deficit.saturating_add(spc_cycles);
418        while self.sample_tick_deficit >= SPC_CYCLES_PER_SAMPLE {
419            self.sample_tick_deficit -= SPC_CYCLES_PER_SAMPLE;
420            // One full 32-step DSP macro pipeline → one stereo sample.
421            // `dsp.main` drains internal latches (KON, KOFF, ENDX, BRR
422            // step, envelope, gaussian, echo FIR) and writes a single
423            // (i16,i16) into `dsp.last_sample`.
424            let (l, r) = self.dsp.main(&mut self.aram);
425            if self.audio_queue.len() < AUDIO_QUEUE_CAPACITY {
426                self.audio_queue.push_back((l, r));
427            }
428            self.audio_left = l;
429            self.audio_right = r;
430        }
431    }

433    /// Advance every voice's ADSR state by one 32 kHz audio sample.
434    /// Advance one timer (0, 1, or 2) by one base-clock tick.
435    fn tick_one_timer(&mut self, idx: usize) {
436        if !self.timer_enabled[idx] {
437            return;
438        }
439        self.timer_internal[idx] = self.timer_internal[idx].wrapping_add(1);
440        let target = if self.timer_reload[idx] == 0 {
441            256
442        } else {
443            u16::from(self.timer_reload[idx])
444        };
445        if self.timer_internal[idx] >= target {
446            self.timer_internal[idx] = 0;
447            self.timer_output[idx] = (self.timer_output[idx] + 1) & 0x0F;
448        }
449    }

472    /// Main CPU writes `value` to mailbox port (0..=3). Mesen2
473    /// `CpuWriteRegister`: the SPC was already `Run` to `target-1` (by
474    /// [`Self::step`] before this call). Shadow the value into `NewCpuRegs`
475    /// ([`Self::new_to_spc_ports`]); make it visible to the SPC (`CpuRegs` =
476    /// [`Self::to_spc_ports`]) **immediately** when the CPU is within one 2×
477    /// unit of the SPC (`master_clock·ratio − Cycle ≤ 1`), otherwise **one
478    /// SPC cycle later** via [`Self::pending_cpu_reg_update`] — the
479    /// write-visibility delay Mesen needs for e.g. Kishin Douji Zenki.
480    pub const fn cpu_write_port(&mut self, port: usize, value: u8) {
481        if self.new_to_spc_ports[port] == value {
482            return;
483        }
484        self.new_to_spc_ports[port] = value;
485        // `master_clock·ratio − Cycle = (cpu_target_2x − spc_pos_2x) + frac`,
486        // where `frac = cpu_clock_frac / MASTER_CLOCK_HZ ∈ [0,1)`. After
487        // `run_to_target`, `cpu_target_2x − spc_pos_2x ∈ {0, 1}`. So the
488        // write is immediate unless the SPC stopped exactly one unit short
489        // *and* there is a non-zero fractional phase.
490        let behind = self.cpu_target_2x.saturating_sub(self.spc_pos_2x);
491        if behind == 0 || (behind == 1 && self.cpu_clock_frac == 0) {
492            self.to_spc_ports[port] = value;
493        } else {
494            self.pending_cpu_reg_update = true;
495        }
496    }

520    /// Advance the SPC700 by `mclk` master cycles. Converts master
521    /// cycles into an SPC-cycle budget at the exact 21.477 MHz : 1.024
522    /// MHz ratio (with fractional carry), then runs SPC instructions —
523    /// each charged its **actual** per-opcode cost (incl. the taken-
524    /// branch penalty) — until the budget is spent. Timers and the DSP
525    /// voice clock advance by the same real per-instruction cycles.
526    pub fn step(&mut self, mclk: u32) {
527        // Advance the CPU's position in the 2× SPC clock domain (Mesen2
528        // `_state.Cycle` units = `master_clock × clockRatio`), carrying the
529        // fractional remainder so the long-run rate is exactly
530        // `SPC_2X_HZ / MASTER_CLOCK_HZ` with zero drift.
531        self.cpu_clock_frac += u64::from(mclk) * SPC_2X_HZ;
532        self.cpu_target_2x += self.cpu_clock_frac / MASTER_CLOCK_HZ;
533        self.cpu_clock_frac %= MASTER_CLOCK_HZ;
534        self.run_to_target();
535    }

537    /// Mesen2 `Spc::Run`: step the SPC cycle-by-cycle (one bus access each,
538    /// `+2` in the 2× domain) until it reaches `cpu_target_2x - 1` — i.e.
539    /// **half a 1× SPC cycle behind** the CPU. The `-1` is load-bearing: it
540    /// holds the SPC at `target-1` so a CPU mailbox access reads `OutputReg`
541    /// (and writes `CpuRegs`) as-of that half-short position — the faithful
542    /// echo / write visibility that the whole-cycle approximations couldn't
543    /// reproduce. Stops mid-instruction (the SPC core is resumable).
544    fn run_to_target(&mut self) {
545        // `-1` matches Mesen (`Spc.Instructions.cpp`): each step advances 2.
546        let target = self.cpu_target_2x.saturating_sub(1);
547        while self.spc_pos_2x < target {
548            if self.cpu.stopped {
549                // STOP/SLEEP: execution is frozen forever (no resume path);
550                // keep the counter synced so we don't busy-loop.
551                self.spc_pos_2x = self.cpu_target_2x;
552                break;
553            }
554            // Advance by the access's wait-state cost (ares `cycleWaitStates`;
555            // 2 at the default ws=0, i.e. the old fixed `+= 2`).
556            self.spc_pos_2x += u64::from(self.run_one_cycle());
557        }
558    }

583    fn run_one_spc(&mut self) -> u32 {
584        // Atomic whole-instruction path — used only by the trajectory harness
585        // (`trace_step_one`), which runs with a frozen mailbox, so there is no
586        // pending CPU→SPC write to reconcile here (the cycle-exact driver
587        // `run_one_cycle` owns `pending_cpu_reg_update`).
588        // SPC700 instruction trace (`--spc-trace`): pre-opcode register
589        // snapshot. Copy the registers out first so the trace borrow does
590        // not alias `self.cpu`.
591        if self.spc_trace.is_some() {
592            let ev = Spc700TraceEvent {
593                pc: self.cpu.pc,
594                a: self.cpu.a,
595                x: self.cpu.x,
596                y: self.cpu.y,
597                sp: self.cpu.sp,
598                psw: self.cpu.psw.0,
599                spc_cycle: self.timer_subdivider,
600                t2_int: self.timer_internal[2],
601                t2_out: self.timer_output[2],
602            };
603            if let Some((events, max)) = self.spc_trace.as_mut() {
604                if *max > 0 {
605                    if events.len() >= *max {
606                        events.drain(0..*max / 2);
607                    }
608                    events.push(ev);
609                }
610            }
611        }
612        let (cycles, clocked) = {
613            let mut bus = ApuBusView {
614                aram: &mut self.aram,
615                to_spc_ports: &mut self.to_spc_ports,
616                new_to_spc_ports: &mut self.new_to_spc_ports,
617                to_cpu_ports: &mut self.to_cpu_ports,
618                control: &mut self.control,
619                test: &mut self.test,
620                dsp_index: &mut self.dsp_index,
621                dsp: &mut self.dsp,
622                timer_reload: &mut self.timer_reload,
623                timer_output: &mut self.timer_output,
624                timer_internal: &mut self.timer_internal,
625                timer_enabled: &mut self.timer_enabled,
626                timer_subdivider: &mut self.timer_subdivider,
627                sample_tick_deficit: &mut self.sample_tick_deficit,
628                audio_queue: &mut self.audio_queue,
629                audio_left: &mut self.audio_left,
630                audio_right: &mut self.audio_right,
631                clocked: 0,
632                cost_2x: 2,
633            };
634            let cycles = u32::from(self.cpu.step(&mut bus));
635            (cycles, bus.clocked)
636        };
637        let unclocked = cycles.saturating_sub(clocked);
638        if unclocked > 0 {
639            self.tick_timers(unclocked);
640            self.tick_voices(unclocked);
641        }
642        if self.cpu.pc < IPL_ROM_BASE {
643            self.past_iplrom = true;
644        }
645        cycles
646    }

648    /// Advance the SPC700 by exactly one cycle (one bus access = `+2` in the
649    /// 2× domain), clocking timers + DSP per cycle (Mesen2 `ProcessCycle`).
650    fn run_one_cycle(&mut self) -> u32 {
651        if !self.cpu.in_instruction && self.spc_trace.is_some() {
652            let ev = Spc700TraceEvent {
653                pc: self.cpu.pc,
654                a: self.cpu.a,
655                x: self.cpu.x,
656                y: self.cpu.y,
657                sp: self.cpu.sp,
658                psw: self.cpu.psw.0,
659                spc_cycle: self.timer_subdivider,
660                t2_int: self.timer_internal[2],
661                t2_out: self.timer_output[2],
662            };
663            if let Some((events, max)) = self.spc_trace.as_mut() {
664                if *max > 0 {
665                    if events.len() >= *max {
666                        events.drain(0..*max / 2);
667                    }
668                    events.push(ev);
669                }
670            }
671        }
672        let mut bus = ApuBusView {
673            aram: &mut self.aram,
674            to_spc_ports: &mut self.to_spc_ports,
675            new_to_spc_ports: &mut self.new_to_spc_ports,
676            to_cpu_ports: &mut self.to_cpu_ports,
677            control: &mut self.control,
678            test: &mut self.test,
679            dsp_index: &mut self.dsp_index,
680            dsp: &mut self.dsp,
681            timer_reload: &mut self.timer_reload,
682            timer_output: &mut self.timer_output,
683            timer_internal: &mut self.timer_internal,
684            timer_enabled: &mut self.timer_enabled,
685            timer_subdivider: &mut self.timer_subdivider,
686            sample_tick_deficit: &mut self.sample_tick_deficit,
687            audio_queue: &mut self.audio_queue,
688            audio_left: &mut self.audio_left,
689            audio_right: &mut self.audio_right,
690            clocked: 0,
691            cost_2x: 2,
692        };
693        self.cpu.step_cycle(&mut bus);
694        // The access's 2×-domain wait-state cost (ares `cycleWaitStates`),
695        // captured before `bus` drops so the borrow ends here. `run_to_target`
696        // advances `spc_pos_2x` by this instead of a fixed 2.
697        let cost_2x = bus.cost_2x;
698        // Mesen2 `ProcessCycle`: a CPU→SPC mailbox write that was deferred
699        // (the SPC was >1 unit behind when the CPU wrote it) becomes visible
700        // to the SPC one cycle after the write — apply it now, after this
701        // cycle's bus access.
702        if self.pending_cpu_reg_update {
703            self.to_spc_ports = self.new_to_spc_ports;
704            self.pending_cpu_reg_update = false;
705        }
706        if self.cpu.pc < IPL_ROM_BASE {
707            self.past_iplrom = true;
708        }
709        cost_2x
710    }

714    /// free-run exactly one SPC700 instruction (full timer/DSP clocking, a
715    /// frozen mailbox), and return the snapshot. Not used in normal

723            self.cpu.y,

732/// Bus view of the APU created on each SPC700 step. Splits the APU's
733/// fields the way the borrow checker needs them — `aram` and
734/// `to_cpu_ports` are mutable (the SPC writes those), `to_spc_ports`
735/// is read-only from the SPC's side (the CPU writes those).
736struct ApuBusView<'a> {
737    aram: &'a mut [u8; 0x10000],
738    to_spc_ports: &'a mut [u8; 4],
739    /// `NewCpuRegs` shadow — cleared alongside `to_spc_ports` on a `$F1`
740    /// bit-4/5 input-port clear so the pending copy-all can't resurrect it.
741    new_to_spc_ports: &'a mut [u8; 4],
742    to_cpu_ports: &'a mut [u8; 4],
743    control: &'a mut u8,
744    test: &'a mut u8,
745    dsp_index: &'a mut u8,
746    dsp: &'a mut dsp::Dsp,
747    timer_reload: &'a mut [u8; 3],
748    timer_output: &'a mut [u8; 3],
749    timer_internal: &'a mut [u16; 3],
750    timer_enabled: &'a mut [bool; 3],
751    /// SPC-cycle accumulator feeding the timer dividers (128 → T0/T1,
752    /// 16 → T2). Advanced one per bus cycle by [`Self::clock_cycle`].
753    timer_subdivider: &'a mut u32,
754    /// SPC cycles since the last DSP sample; one 32 kHz sample is
755    /// produced every [`SPC_CYCLES_PER_SAMPLE`].
756    sample_tick_deficit: &'a mut u32,
757    audio_queue: &'a mut std::collections::VecDeque<(i16, i16)>,
758    audio_left: &'a mut i16,
759    audio_right: &'a mut i16,
760    /// Count of bus cycles clocked during the current instruction, so
761    /// the caller can reconcile any cycles with no bus activity (the
762    /// SLEEP/STOP halt window emits fewer bus ops than its cycle cost).
763    clocked: u32,
764    /// 2×-domain clock cost of the most recent bus access, set by
765    /// `clock_cycle` from the SPC700 wait-state dividers (`$F0`). `2` at the
766    /// default (ws=0) → identical to the old fixed `spc_pos_2x += 2`;
767    /// `{2,4,10,20}` once a game sets non-zero wait-states (ares
768    /// `smp/timing.cpp` `cycleWaitStates`). `run_to_target` reads it back.
769    cost_2x: u32,
770}

772/// Advance one SPC timer (0/1/2) by one base-clock tick: bump the
773/// internal counter, and on reaching the reload target (0 ⇒ 256) wrap-
774/// increment the 4-bit output and reset.
775fn tick_one_timer(
776    reload: [u8; 3],
777    output: &mut [u8; 3],
778    internal: &mut [u16; 3],
779    enabled: [bool; 3],
780    idx: usize,
781) {
782    if !enabled[idx] {
783        return;
784    }
785    internal[idx] = internal[idx].wrapping_add(1);
786    let target = if reload[idx] == 0 {
787        256
788    } else {
789        u16::from(reload[idx])
790    };
791    if internal[idx] >= target {
792        internal[idx] = 0;
793        output[idx] = (output[idx] + 1) & 0x0F;
794    }
795}

797impl ApuBusView<'_> {
798    /// Clock the timers **and** the S-DSP by exactly one SPC cycle, in
799    /// position — the faithful ares grammar (`wait()` → `stepTimers` +
800    /// `synchronize(dsp)`; Mesen2 `IncCycleCount` → `Timer.Run` +
801    /// `dsp->Exec`). Called at the start of every read / write / idle so
802    /// a `$FD-$FF` timer read or a `$F3` DSP write lands on the correct
803    /// cycle. This is the whole point of the per-cycle SPC700 port: the
804    /// number of these calls per opcode equals its true cycle count.
805    fn clock_cycle(&mut self, ws_internal: bool) {
806        // SPC700 wait-state divider (ares `smp/timing.cpp::wait`): pick the
807        // 2-bit wait-state field — internal (`$F0` bits 6-7) for idle cycles,
808        // IO registers (`$00Fx`) and IPLROM (`$FFC0+`); external (bits 4-5) for
809        // normal RAM. At the power-on default (ws=0) every divider is the
810        // baseline, so this is byte-identical to the old fixed model.
811        let ws = (if ws_internal {
812            *self.test >> 6
813        } else {
814            *self.test >> 4
815        } & 0x03) as usize;
816        self.cost_2x = SPC_CLOCK_WAIT_2X[ws]; // 2× clock advance; read by run_to_target
817
818        self.clocked = self.clocked.wrapping_add(1);
819
820        // --- timers --- advance by the timer divider; a stage-0 tick fires for
821        // each 16- (T2) / 128- (T0/T1) boundary crossed. The step is ≤ 8 < 16,
822        // so at most one crossing per access — equivalent to the old
823        // `after % N == 0` at ws=0 (step 1).
824        let before = *self.timer_subdivider;
825        *self.timer_subdivider = before.wrapping_add(SPC_TIMER_WAIT[ws]);
826        let after = *self.timer_subdivider;
827        // $F0 TEST gate (ares timing.cpp:45-49): output propagation is
828        // suppressed when timersEnable (bit 3) is clear or timersDisable
829        // (bit 0) is set. The divider keeps running so phase resumes on
830        // re-enable.
831        if *self.test & 0x08 != 0 && *self.test & 0x01 == 0 {
832            if before / 16 != after / 16 {
833                tick_one_timer(
834                    *self.timer_reload,
835                    self.timer_output,
836                    self.timer_internal,
837                    *self.timer_enabled,
838                    2,
839                );
840            }
841            if before / 128 != after / 128 {
842                tick_one_timer(
843                    *self.timer_reload,
844                    self.timer_output,
845                    self.timer_internal,
846                    *self.timer_enabled,
847                    0,
848                );
849                tick_one_timer(
850                    *self.timer_reload,
851                    self.timer_output,
852                    self.timer_internal,
853                    *self.timer_enabled,
854                    1,
855                );
856            }
857        }
858
859        // --- S-DSP: one 32 kHz sample every 32 SPC clocks. The DSP follows the
860        // SMP *clock* (ares `step()` syncs the dsp thread), so it scales with
861        // the clock divider, not the timer one.
862        *self.sample_tick_deficit += SPC_SAMPLE_WAIT[ws];
863        if *self.sample_tick_deficit >= SPC_CYCLES_PER_SAMPLE {
864            *self.sample_tick_deficit -= SPC_CYCLES_PER_SAMPLE;
865            let (l, r) = self.dsp.main(self.aram);
866            if self.audio_queue.len() < AUDIO_QUEUE_CAPACITY {
867                self.audio_queue.push_back((l, r));
868            }
869            *self.audio_left = l;
870            *self.audio_right = r;
871        }
872    }
873}

875impl SpcBus for ApuBusView<'_> {
876    fn read(&mut self, addr: u16) -> u8 {
877        self.clock_cycle(spc_ws_internal(addr, *self.control));
878        match addr {
879            // $F0 — testing register, write-only on real HW. Return 0.
880            0x00F0 => 0,
881            // $F1 — control register. Reads return... 0 on real HW
882            // (it's effectively write-only). Match that.
883            0x00F1 => 0,
884            // $F2 — DSP register-index port. Real HW returns the
885            // last value written; we model that.
886            0x00F2 => *self.dsp_index,
887            // $F3 — DSP register-data port. Now routed through the
888            // cycle-accurate ares-port S-DSP (`crates/luna-apu/dsp.rs`).
889            0x00F3 => {
890                // DSP reads have NO side effects (ares dsp/memory.cpp:1-3).
891                // ENDX ($7C) is cleared only by a write to $7C or by KON —
892                // never on read. Clearing it here drops end-of-sample bits
893                // a driver may read more than once.
894                self.dsp.read(*self.dsp_index & 0x7F)
895            }
896            // $F4-$F7 — mailbox FROM the main CPU.
897            0x00F4..=0x00F7 => self.to_spc_ports[(addr - 0x00F4) as usize],
898            // $F8-$F9 — AUXIO4/5 scratch registers. ares io.cpp:49-53
899            // returns the last value written; we keep it in ARAM (the
900            // write path stores it there) and read it straight back.
901            0x00F8..=0x00F9 => self.aram[addr as usize],
902            // $FA-$FC — timer reload values. Write-only on real HW,
903            // reads return 0.
904            0x00FA..=0x00FC => 0,
905            // $FD-$FF — timer outputs. 4-bit counters that **clear on
906            // read** — the SPC driver typically does
907            // `MOV A,$FD / BNE somewhere` once per its main loop.
908            0x00FD..=0x00FF => {
909                let idx = (addr - 0x00FD) as usize;
910                let v = self.timer_output[idx];
911                self.timer_output[idx] = 0;
912                v
913            }
914            // Everything else — ARAM, with the IPL ROM overlaid over
915            // $FFC0-$FFFF while $F1 bit 7 is set.
916            _ => aram_with_ipl(self.aram, *self.control, addr),
917        }
918    }
919
920    fn write(&mut self, addr: u16, value: u8) {
921        self.clock_cycle(spc_ws_internal(addr, *self.control));
922        match addr {
923            // $F0 — TEST register. Bit 0 = timersDisable, bit 3 =
924            // timersEnable gate the timers (ares io.cpp:81-94); the
925            // other bits (RAM writable/disable, wait states) are stored
926            // but not yet modelled. The P-flag write gate is omitted
927            // (writes with PSW.P set are pathological for $F0).
928            0x00F0 => *self.test = value,
929            // $F1 — control register. Bit 7 controls IPL ROM
930            // visibility (we don't yet model un-mapping). Bits 0-2
931            // enable timers T0/T1/T2; a 0→1 transition resets the
932            // corresponding internal counter on real HW.
933            0x00F1 => {
934                for i in 0..3 {
935                    let bit = 1u8 << i;
936                    let was_enabled = self.timer_enabled[i];
937                    let now_enabled = value & bit != 0;
938                    if now_enabled && !was_enabled {
939                        self.timer_internal[i] = 0;
940                        self.timer_output[i] = 0;
941                    }
942                    self.timer_enabled[i] = now_enabled;
943                }
944                // Bits 4/5 clear the CPU→SMP input mailbox ports the SPC
945                // reads at $F4-$F7 (ares io.cpp:113-123): bit 4 → ports
946                // 0/1, bit 5 → ports 2/3.
947                // Clear the `NewCpuRegs` shadow too (Mesen Spc::Write
948                // $F1) so a deferred pending copy-all can't resurrect a
949                // cleared port.
950                if value & 0x10 != 0 {
951                    self.to_spc_ports[0] = 0;
952                    self.to_spc_ports[1] = 0;
953                    self.new_to_spc_ports[0] = 0;
954                    self.new_to_spc_ports[1] = 0;
955                }
956                if value & 0x20 != 0 {
957                    self.to_spc_ports[2] = 0;
958                    self.to_spc_ports[3] = 0;
959                    self.new_to_spc_ports[2] = 0;
960                    self.new_to_spc_ports[3] = 0;
961                }
962                *self.control = value;
963            }
964            // $F2 — DSP register-index port.
965            0x00F2 => *self.dsp_index = value,
966            // $F3 — DSP register-data port. Routed through the
967            // cycle-accurate ares-port S-DSP. `dsp.write` handles all
968            // register side effects (KON/KOFF latching, FLG bit-flip,
969            // ESA/DIR/EON/PMON fanout, per-voice volume/pitch/SRCN/
970            // ADSR/GAIN demuxing, FIR taps) matching ares' memory.cpp.
971            0x00F3 => {
972                let idx = *self.dsp_index & 0x7F;
973                self.dsp.write(idx, value);
974            }
975            // $F4-$F7 — mailbox TO the main CPU.
976            0x00F4..=0x00F7 => self.to_cpu_ports[(addr - 0x00F4) as usize] = value,
977            // $F8-$F9 — auxiliary RAM-mapped regs. Store in ARAM so
978            // reads come back consistent.
979            0x00F8..=0x00F9 => self.aram[addr as usize] = value,
980            // $FA-$FC — timer reload (target) values.
981            0x00FA..=0x00FC => {
982                let idx = (addr - 0x00FA) as usize;
983                self.timer_reload[idx] = value;
984            }
985            // $FD-$FF — timer outputs are read-only; writes drop.
986            0x00FD..=0x00FF => {}
987            // Everything else — ARAM.
988            _ => self.aram[addr as usize] = value,
989        }
990    }
991
992    fn idle(&mut self) {
993        // Idle cycles have no address → ares uses the internal wait-states.
994        self.clock_cycle(true);
995    }

1202    fn wait_states_divide_the_spc_clock() {
1203        // ares smp/timing.cpp: non-zero `$F0` wait-states slow the SPC clock
1204        // ({2,4,10,20} per access). A tight `BRA self` loop over a fixed
1205        // master-clock budget executes proportionally fewer instructions as
1206        // the divider grows. Count instructions via the SPC trace.
1207        fn instr_count(ws: u8) -> usize {
1208            let mut apu = Apu::new();
1209            // BRA -2 (2F FE) at $0200 — a one-instruction self-loop, all
1210            // external RAM accesses + the 2-cycle taken-branch idle.
1211            apu.aram[0x0200] = 0x2F;
1212            apu.aram[0x0201] = 0xFE;
1213            apu.cpu.pc = 0x0200;
1214            // timersEnable (bit 3) + both internal (6-7) and external (4-5)
1215            // wait-state fields = `ws`, so every access (reads + idles) uses it.
1216            apu.test = 0x08 | (ws << 4) | (ws << 6);
1217            apu.enable_spc_trace(1_000_000);
1218            apu.step(5_000 * MASTER_CYCLES_PER_SPC_STEP);
1219            apu.take_spc_trace().len()
1220        }
1221        let n0 = instr_count(0);
1222        let n1 = instr_count(1); // clock divider 2× → ~half the instructions
1223        let n3 = instr_count(3); // clock divider 10× → ~a tenth
1224        assert!(n0 > 1_000, "ws=0 baseline ran: {n0}");
1225        let r1 = n1 as f64 / n0 as f64;
1226        assert!(
1227            (0.4..=0.6).contains(&r1),
1228            "ws=1 ≈ half: n0={n0} n1={n1} ratio={r1:.3}"
1229        );
1230        let r3 = n3 as f64 / n0 as f64;
1231        assert!(
1232            (0.05..=0.15).contains(&r3),
1233            "ws=3 ≈ tenth: n0={n0} n3={n3} ratio={r3:.3}"
