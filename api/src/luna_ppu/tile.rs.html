1//! Tile decode primitives.
2//!
3//! SNES tiles are stored in VRAM in a planar layout:
4//! - **2bpp**: 16 bytes / tile. Each pair of bytes encodes one row of
5//!   8 pixels — byte 0 holds the LSB of each pixel, byte 1 the MSB.
6//! - **4bpp**: 32 bytes / tile. Same encoding as two stacked 2bpp tiles
7//!   (planes 0/1 contiguous, planes 2/3 right after).
8//! - **8bpp**: 64 bytes / tile. Four stacked 2bpp planes.
9//!
10//! See `ARCHITECTURE.md` §6.2 and fullsnes §"BG Modes".
11
12/// Decode one row of an 8×8 2bpp tile.
13///
14/// `lo` is the byte whose bits hold each pixel's bit 0; `hi` holds
15/// each pixel's bit 1. Returned indices are in `0..=3`. Pixel 0 is the
16/// **leftmost** pixel — i.e. bit 7 of each byte.
17#[must_use]
18pub fn decode_2bpp_row(lo: u8, hi: u8) -> [u8; 8] {
19    let mut out = [0u8; 8];
20    for (i, slot) in out.iter_mut().enumerate() {
21        let shift = 7 - i;
22        let bit0 = (lo >> shift) & 1;
23        let bit1 = (hi >> shift) & 1;
24        *slot = (bit1 << 1) | bit0;
25    }
26    out
27}
28
29/// Decode one row of an 8×8 4bpp tile.
30///
31/// `p0..=p3` are the four plane bytes for the row. Pixel n is built
32/// as `(p3[n]<<3) | (p2[n]<<2) | (p1[n]<<1) | p0[n]`. Returned indices
33/// are in `0..=15`.
34#[must_use]
35pub fn decode_4bpp_row(p0: u8, p1: u8, p2: u8, p3: u8) -> [u8; 8] {
36    let mut out = [0u8; 8];
37    for (i, slot) in out.iter_mut().enumerate() {
38        let shift = 7 - i;
39        let b0 = (p0 >> shift) & 1;
40        let b1 = (p1 >> shift) & 1;
41        let b2 = (p2 >> shift) & 1;
42        let b3 = (p3 >> shift) & 1;
43        *slot = (b3 << 3) | (b2 << 2) | (b1 << 1) | b0;
44    }
45    out
46}

48/// Convert a 15-bit BGR555 color to 24-bit RGB888.
49///
50/// Hardware packs each channel as 5 bits in the order BGR (blue is
51/// bit 10-14, green bit 5-9, red bit 0-4). The bit-15 high bit is
52/// ignored by the renderer.
53#[must_use]
54pub const fn bgr555_to_rgb888(color: u16) -> [u8; 3] {
55    let r5 = (color & 0x001F) as u8;
56    let g5 = ((color >> 5) & 0x001F) as u8;
57    let b5 = ((color >> 10) & 0x001F) as u8;
58    [scale_5_to_8(r5), scale_5_to_8(g5), scale_5_to_8(b5)]
59}
60
61/// Expand a 5-bit channel to 8 bits using the canonical SNES
62/// "replicate-top-3-bits" formula — gives 0 → 0 and 31 → 255 exactly,
63/// with smooth quantization in between.
64#[inline]
65#[must_use]
66pub const fn scale_5_to_8(c5: u8) -> u8 {
67    let c5 = c5 & 0x1F;
68    (c5 << 3) | (c5 >> 2)
69}

71/// Apply SNES master brightness (`$2100` bits 0-3, 0..15) to an
72/// already-converted RGB888 triple.
73///
74/// Per ares `color.cpp`, `L = (1 + l)/16 * (l ? 1.0 : 0.25)`:
75/// brightness 1..15 scale linearly as `in * (brightness + 1) / 16`
76/// (full at 15), and brightness **0** is an extra ÷4 darker — i.e.
77/// `in / 64`, not pure black. ares notes this is hardware-measured
78/// ("luma=0 is not 100% black; but much darker than linear scaling");
79/// Mesen2 models it as pure black (`in * b / 15`), but ares is the
80/// hardware reference here.
81#[inline]
82#[must_use]
83pub fn apply_brightness(rgb: [u8; 3], brightness: u8) -> [u8; 3] {
84    let b = u16::from(brightness & 0x0F);
85    let scale = b + 1;
86    // Brightness 0 gets an additional right-shift of 2 (the ×0.25).
87    let extra = if b == 0 { 2 } else { 0 };
88    [
89        (((u16::from(rgb[0]) * scale) >> 4) >> extra) as u8,
90        (((u16::from(rgb[1]) * scale) >> 4) >> extra) as u8,
91        (((u16::from(rgb[2]) * scale) >> 4) >> extra) as u8,
92    ]
93}

104    fn row_2bpp_all_zero_is_all_color_0() {
105        assert_eq!(decode_2bpp_row(0x00, 0x00), [0, 0, 0, 0, 0, 0, 0, 0]);

108    #[test]
109    fn row_2bpp_leftmost_pixel_is_msb() {
110        // lo = 0x80 (bit 7 set), hi = 0x00 → only pixel 0 has color 1.
111        assert_eq!(decode_2bpp_row(0x80, 0x00), [1, 0, 0, 0, 0, 0, 0, 0]);
112    }
113
114    #[test]
115    fn row_2bpp_combines_planes_into_2bit_index() {
116        // lo = $A5 (bits 7,5,2,0), hi = $5A (bits 6,4,3,1).
117        // Pixel n (left→right) gets (hi.bit(7-n)<<1 | lo.bit(7-n)):
118        //   0: (0<<1)|1 = 1
119        //   1: (1<<1)|0 = 2
120        //   2: (0<<1)|1 = 1
121        //   3: (1<<1)|0 = 2
122        //   4: (1<<1)|0 = 2
123        //   5: (0<<1)|1 = 1
124        //   6: (1<<1)|0 = 2
125        //   7: (0<<1)|1 = 1
126        assert_eq!(decode_2bpp_row(0xA5, 0x5A), [1, 2, 1, 2, 2, 1, 2, 1]);
127    }
128
129    #[test]
130    fn row_2bpp_full_palette_3_when_both_planes_full() {
131        assert_eq!(decode_2bpp_row(0xFF, 0xFF), [3, 3, 3, 3, 3, 3, 3, 3]);
132    }
