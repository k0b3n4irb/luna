34/// Render one scanline of BG1 in Mode 0 (2bpp).
35///
36/// Pixels are produced by:
37/// 1. adding the BG1 H/V scroll to `(x, y)`
38/// 2. looking up the tilemap entry at the resulting (tile-row, tile-col)
39/// 3. decoding the 2bpp tile row addressed by the entry
40/// 4. mapping the 0..=3 palette index through the entry's palette
41///    offset into CGRAM and converting BGR555 → RGB888
42///
43/// Color index 0 in any BG palette is **transparent**; this rendering
44/// pass replaces it with CGRAM index 0 (the backdrop / "color 0"
45/// global), which matches the behaviour of a single-BG composite.
46// Legacy RGB-output renderers, kept only as convenience helpers for the
47// unit tests below — the production pipeline uses the `*_indexed_*`
48// renderers + the compositor in `ppu.rs`. `#[cfg(test)]` so they neither
49// ship in the library nor pollute its public API.
50#[cfg(test)]
51#[must_use]
52fn render_bg1_scanline(ppu: &Ppu, y: u16) -> Scanline {
53    render_bg1_scanline_with(ppu, y, RenderOptions::default())
54}

56/// BG1's bit-depth derived from `BGMODE` (the low 3 bits select the
57/// PPU mode). Returns 2, 4 or 8 — matching the SNES per-mode mapping
58/// from <https://problemkaputt.de/fullsnes.htm> §"PPU Background
59/// Modes":
60///
61/// | Mode | BG1 | BG2 | BG3 | BG4 |
62/// |:----:|----:|----:|----:|----:|
63/// |  0   | 2   | 2   | 2   | 2   |
64/// |  1   | 4   | 4   | 2   | —   |
65/// |  2   | 4   | 4   | —   | —   |
66/// |  3   | 8   | 4   | —   | —   |
67/// |  4   | 8   | 2   | —   | —   |
68/// |  5   | 4   | 2   | —   | —   |
69/// |  6   | 4   | —   | —   | —   |
70/// |  7   | 8   | —   | —   | —   |
71///
72/// Modes 5/6 are high-res (512px); Mode 7 is affine. We render either
73/// the way Mode 1/2/3 would (planar tiles + tilemap) for now.
74#[cfg(test)]
75#[must_use]
76fn bg1_bpp(bgmode: u8) -> u8 {
77    match bgmode & 0x07 {
78        0 => 2,
79        1 | 2 | 5 | 6 => 4,
80        3 | 4 | 7 => 8,
81        _ => unreachable!(),
82    }
83}

85/// Same as [`render_bg1_scanline`] but with debug options.
86///
87/// Honours `BGMODE` for BG1 bit-depth:
88///
89/// - **2bpp** (Mode 0): 16 bytes/tile, 4-color sub-palettes at
90///   `CGRAM[palette_off * 4 + idx]`.
91/// - **4bpp** (Mode 1/2/5/6): 32 bytes/tile, 16-color sub-palettes
92///   at `CGRAM[palette_off * 16 + idx]`.
93/// - **8bpp** (Mode 3/4/7): 64 bytes/tile, full 256-colour palette
94///   indexed straight by `idx` (tilemap palette offset ignored).
95///
96/// Mode 7 affine is still handled the planar way for now — close
97/// enough to show *something* until the real Mode-7 path lands.
98#[cfg(test)]
99#[must_use]
100fn render_bg1_scanline_with(ppu: &Ppu, y: u16, opts: RenderOptions) -> Scanline {
101    let mut out = [[0u8; 3]; 256];
102
103    // INIDISP bit 7: forced blank — entire scanline is black, ignoring
104    // master brightness (which is 0 in forced blank anyway per spec).
105    if ppu.inidisp & 0x80 != 0 && !opts.bypass_forced_blank {
106        return out;
107    }
108    let brightness = if opts.bypass_forced_blank {
109        0x0F
110    } else {
111        ppu.inidisp & 0x0F
112    };
113
114    // Backdrop colour (CGRAM index 0).
115    let backdrop = decode_palette(ppu, 0, brightness);
116
117    // Route through the runtime indexed renderer (tilemap sizes, 16×16
118    // tiles, mosaic, Mode-0 palette region) and decode to RGB, so this
119    // debug/API view no longer diverges from the real compositor.
120    let indexed = render_bg_scanline_indexed_with(ppu, 0, y, opts);
121    for (x, px) in indexed.iter().enumerate() {
122        out[x] = match *px {
123            Some((cgram_idx, _)) => decode_palette(ppu, cgram_idx, brightness),
124            None => backdrop,
125        };
126    }
127    out
128}

130/// Decode one full 8-pixel tile row of palette indices out of a planar
131/// tile in VRAM (left-to-right, pre-flip). The BG sampler caches the
132/// result so 8 consecutive same-tile pixels decode the row once (PERF-1).
133///
134/// SNES tile layout is *planar bitplane-pair* in VRAM:
135///
136/// - 2bpp: rows 0..7 each have planes (0, 1) packed as two bytes →
137///   16 bytes per tile, addressed `tile_base + row*2`.
138/// - 4bpp: planes (0, 1) for rows 0..7 fill the first 16 bytes; planes
139///   (2, 3) for rows 0..7 fill the next 16 bytes. So a row's four
140///   plane bytes live at `tile_base + row*2`, `+1`, `+16+row*2`, `+17`.
141/// - 8bpp: same as 4bpp but with planes (4, 5, 6, 7) in the next 32
142///   bytes.
143fn decode_tile_row(ppu: &Ppu, tile_base: usize, row_in_tile: usize, bpp: u8) -> [u8; 8] {
144    let row_off = tile_base + row_in_tile * 2;
145    let p0 = ppu.vram.peek(row_off as u16);
146    let p1 = ppu.vram.peek(row_off.wrapping_add(1) as u16);
147    if bpp == 2 {
148        return decode_2bpp_row(p0, p1);
149    }
150    let p2 = ppu.vram.peek(row_off.wrapping_add(16) as u16);
151    let p3 = ppu.vram.peek(row_off.wrapping_add(17) as u16);
152    let mut row = decode_4bpp_row(p0, p1, p2, p3);
153    if bpp == 4 {
154        return row;
155    }
156    // 8bpp: fold in the upper four planes too.
157    let p4 = ppu.vram.peek(row_off.wrapping_add(32) as u16);
158    let p5 = ppu.vram.peek(row_off.wrapping_add(33) as u16);
159    let p6 = ppu.vram.peek(row_off.wrapping_add(48) as u16);
160    let p7 = ppu.vram.peek(row_off.wrapping_add(49) as u16);
161    for (col, px) in row.iter_mut().enumerate() {
162        let bit = 7 - col;
163        let mask = 1u8 << bit;
164        *px |= ((p4 & mask) >> bit) << 4
165            | ((p5 & mask) >> bit) << 5
166            | ((p6 & mask) >> bit) << 6
167            | ((p7 & mask) >> bit) << 7;
168    }
169    row
170}

172/// Decode a single tile pixel: `decode_tile_row(...)[col_in_tile]`.
173/// Used by the sprite path, which samples scattered pixels with no row

175fn decode_tile_pixel(

177    tile_base: usize,
178    row_in_tile: usize,

209/// Byte offset into VRAM for an SNES sprite tile.
210///
211/// Sprite CHR data lives in two 8 KB pages of 256 tiles × 32 bytes
212/// each, selected by **bit 8 of the 9-bit tile number**:
213///
214/// * Page 0 base: `OBSEL[0..2] << 13` bytes.
215/// * Page 1 base: `page0 + ((nameselect + 1) << 12)` bytes, where
216///   `nameselect = OBSEL[3..4]`.
217///
218/// The page-1 offset can be `0x1000`, `0x2000`, `0x3000` or
219/// `0x4000`. The `0x2000` (nameselect = 1) case places page 1
220/// immediately after page 0; the others either overlap page 0 or
221/// leave a gap. Hardware-faithful per ares' `oam.cpp` and
222/// Mesen2's `_state.OamBaseAddress` + `_state.OamAddressOffset`.
223///
224/// luna previously addressed all 512 tiles linearly from page 0
225/// (equivalent to nameselect = 1, regardless of the register
226/// value) — broken for games that configure the second page
227/// elsewhere.
228#[inline]
229#[must_use]
230pub(crate) fn sprite_tile_byte_offset(obsel: u8, tile: u16) -> usize {
231    // OBSEL bits 0..2 give the OBJ tile base address in 8K-**word**
232    // units (= 16K-byte units, since VRAM is word-addressed by the
233    // PPU). Same interpretation as ares' `tiledataAddress = (n & 7)
234    // << 13` followed by word-indexed `vram[address]`, and as
235    // Mesen2's `OamBaseAddress = (value & 7) << 13` indexed against
236    // `_vram[FetchAddress]` (a `uint16_t*` array). luna's `vram.peek`
237    // is byte-addressed so we double the shift here (`<< 14`) to
238    // convert from word-base to byte-base in one step. Same applies
239    // to the name-select offset (`<< 13` = 8K-byte step per +1).
240    let page0 = ((usize::from(obsel) & 0x07) << 14) & 0xFFFF;
241    let nameselect = usize::from(obsel >> 3) & 0x03;
242    let page1 = page0.wrapping_add((nameselect + 1) << 13) & 0xFFFF;
243    let tile_in_page = usize::from(tile & 0xFF) * 32;
244    if (tile & 0x100) == 0 {
245        (page0 + tile_in_page) & 0xFFFF
246    } else {
247        (page1 + tile_in_page) & 0xFFFF
248    }
249}

251/// Sprite size pair selected by `OBSEL.bits[7:5]` — small vs large.
252///
253/// Per ares' `oam.cpp` `width()`/`height()` + Mesen2's `oamWidth[]`/
254/// `oamHeight[]` tables (`SnesPpuTypes.h`). The two emulators agree
255/// exactly, including the non-square non-square pairs for codes 6
256/// and 7 — and **code 7's large size is `32×32` (not `32×64`)**, an
257/// easy-to-miss hardware quirk.
258#[must_use]
259pub fn sprite_size_pair(obsel: u8) -> ((u16, u16), (u16, u16)) {
260    match (obsel >> 5) & 0x07 {
261        0 => ((8, 8), (16, 16)),
262        1 => ((8, 8), (32, 32)),
263        2 => ((8, 8), (64, 64)),
264        3 => ((16, 16), (32, 32)),
265        4 => ((16, 16), (64, 64)),
266        5 => ((32, 32), (64, 64)),
267        6 => ((16, 32), (32, 64)),
268        7 => ((16, 32), (32, 32)),
269        _ => unreachable!(),
270    }
271}

273/// Decode all 128 OAM entries into [`SpriteEntry`].
274#[must_use]
275pub fn decode_all_sprites(ppu: &Ppu) -> [SpriteEntry; 128] {
276    let (small, large) = sprite_size_pair(ppu.obsel);
277    let mut out = [SpriteEntry {
278        x: 0,
279        y: 0,
280        tile: 0,
281        palette: 0,
282        priority: 0,
283        h_flip: false,
284        v_flip: false,
285        w: 8,
286        h: 8,
287    }; 128];
288    for (idx, slot) in out.iter_mut().enumerate() {
289        let base = (idx * 4) as u16;
290        let x_low = ppu.oam.peek(base);
291        // The live OAM Y is the truth: `$F0` (240) just parks the sprite
292        // off-screen (visible Y is 0..223), the standard hide. We render
293        // sprites per scanline against live OAM (Phase 1), so the game's
294        // VBlank hide-then-repopulate has already settled by the time the
295        // visible lines draw — no resurrection needed. (Removed the P1.19
296        // `$F0 → shadow_y` fallback `6c06f64`, a band-aid for the old
297        // free-running snapshot renderer; it wrongly resurrected genuinely
298        // hidden sprites, e.g. DKC's parked sprites leaking 1-px slivers at
299        // x=255.)
300        let y_pos = ppu.oam.peek(base + 1);
301        let tile_low = ppu.oam.peek(base + 2);
302        let attrs = ppu.oam.peek(base + 3);
303        // High table: 32 bytes at OAM[$200..$220], 2 bits per sprite.
304        let high_byte_idx = 0x200 + (idx / 4);
305        let high_bit_off = (idx % 4) * 2;
306        let high = (ppu.oam.peek(high_byte_idx as u16) >> high_bit_off) & 0x03;
307        let x_high_bit = (high & 0x01) as i16;
308        let is_large = (high & 0x02) != 0;
309        // 9-bit X with sign extension: bit 8 set => x is negative.
310        let mut x = (x_high_bit << 8) | (x_low as i16);
311        if x >= 256 {
312            x -= 512;
313        }
314        let (w, h) = if is_large { large } else { small };
315        // Interlace small-sprite height quirk (ares oam.cpp:67): in OBJ
316        // interlace a *small* sprite (size bit 0) with baseSize >= 6 is
317        // forced to height 16 (the table would give 32). OBJ interlace is
318        // SETINI **bit 1** (`obj.io.interlace = data.bit(1)`, ares io.cpp) —
319        // SEPARATE from the BG/screen interlace in bit 0.
320        let h = if ppu.setini & 0x02 != 0 && !is_large && (ppu.obsel >> 5) & 0x07 >= 6 {
321            16
322        } else {
323            h
324        };
325        *slot = SpriteEntry {
326            x,
327            y: y_pos,
328            tile: (tile_low as u16) | ((attrs as u16 & 0x01) << 8),
329            palette: (attrs >> 1) & 0x07,
330            priority: (attrs >> 4) & 0x03,
331            h_flip: attrs & 0x40 != 0,
332            v_flip: attrs & 0x80 != 0,
333            w,
334            h,
335        };
336    }
337    out
338}

340/// Render one scanline of sprite output.
341///
342/// Returns, for each of the 256 visible columns, either `Some(rgb)`
343/// when a non-transparent sprite covers that pixel, or `None` when
344/// no sprite contributes. The caller composites this on top of the
345/// BG layers.
346///
347/// Per-sprite priority bits *are* decoded but not yet used to slot
348/// sprites between BG layers — that lands once the per-pixel priority
349/// engine is in place. For now the simple "sprites on top of all
350/// BGs" rule is enough for title-screen Mario/Yoshi visibility.
351#[cfg(test)]
352#[must_use]
353fn render_sprites_scanline(ppu: &Ppu, y: u16, opts: RenderOptions) -> [Option<[u8; 3]>; 256] {
354    let mut out: [Option<[u8; 3]>; 256] = [None; 256];
355    if ppu.inidisp & 0x80 != 0 && !opts.bypass_forced_blank {
356        return out;
357    }
358    let brightness = if opts.bypass_forced_blank {
359        0x0F
360    } else {
361        ppu.inidisp & 0x0F
362    };
363    // Delegate to the runtime indexed sprite renderer and decode to RGB,
364    // so this debug/API view stays in lockstep with the real compositor
365    // (tile wrap, Y-wrap, palette, priority) instead of re-implementing.
366    let indexed = render_sprites_scanline_indexed_with(ppu, y, opts);
367    for (x, px) in indexed.iter().enumerate() {
368        if let Some((cgram_idx, _)) = *px {
369            out[x] = Some(decode_palette(ppu, cgram_idx, brightness));
370        }
371    }
372    out
373}

391/// Render the full visible frame using the SNES per-pixel priority
392/// engine. For each pixel we walk a per-BGMODE priority table from
393/// top to bottom; the first layer with a non-transparent pixel at
394/// that x wins. If no layer contributes, the CGRAM backdrop (index 0)
395/// shows through.
396///
397/// The implementation differs from the previous "back-to-front layer
398/// overlay" by:
399///   * tracking transparency vs backdrop explicitly through
400///     [`IndexedScanline`] (CGRAM index + priority bit, `None` =
401///     transparent — not "backdrop colour");
402///   * routing sprites BETWEEN BG layers based on their per-sprite
403///     priority (0-3), not always on top;
404///   * honouring the BG-priority bit from each tilemap entry so a
405///     tile marked "high" sits above a tile marked "low" of the same
406///     BG layer (e.g. SMW status-bar text in front of clouds);
407///   * routing BG3 to the top in Mode 1 when BGMODE bit 3 is set.
408///
409/// Modes 0-4 are fully wired through this engine; modes 5, 6 and 7
410/// fall back to the Mode-1 table (close enough for the games in our
411/// test corpus until a dedicated path lands).
412#[must_use]
413pub fn render_frame_with(ppu: &Ppu, opts: RenderOptions) -> Vec<[u8; 3]> {
414    let mut buf = vec![[0u8; 3]; FRAME_W * FRAME_H];
415    for y in 0..FRAME_H as u16 {
416        let off = y as usize * FRAME_W;
417        render_scanline_into(ppu, y, opts, &mut buf[off..off + FRAME_W]);
418    }
419    buf
420}

455/// Core of [`render_scanline_partial_into`]. `precomp` optionally
456/// supplies a pre-decoded sprite set + line evaluation so the scheduler
457/// decodes sprites once per scanline and shares them across the
458/// overflow-flag read and (in interlace) both field renders (PERF-2).
459/// `None` decodes sprites internally, as the full-frame paths do.
460pub(crate) fn render_scanline_partial_into_from(
461    ppu: &Ppu,
462    y: u16,
463    start_x: u16,
464    end_x: u16,
465    opts: RenderOptions,
466    out: &mut [[u8; 3]],
467    precomp: Option<(&[SpriteEntry; 128], &SpriteEval)>,
468) {
469    debug_assert_eq!(
470        out.len(),
471        FRAME_W,
472        "render_scanline_partial_into requires a slice of FRAME_W ({FRAME_W}) pixels"
473    );
474    let start = usize::from(start_x).min(FRAME_W);
475    let end = usize::from(end_x).min(FRAME_W);
476    if start >= end {
477        return;
478    }
479
480    // INIDISP bit 7: forced blank → range is all black.
481    if ppu.inidisp & 0x80 != 0 && !opts.bypass_forced_blank {
482        for px in &mut out[start..end] {
483            *px = [0, 0, 0];
484        }
485        return;
486    }
487    let brightness = if opts.bypass_forced_blank {
488        0x0F
489    } else {
490        ppu.inidisp & 0x0F
491    };
492
493    // Mode 7 EXTBG ($2133 bit 6): the affine plane is *also* exposed as
494    // BG2, with bit 7 of each pixel selecting a priority and bits 0-6 as
495    // the colour (ares mode7.cpp:44-52 + io.cpp:733-738). It gets its own
496    // priority table.
497    let extbg = ppu.bgmode & 0x07 == 0x07 && ppu.setini & 0x40 != 0;
498    let table = if extbg {
499        MODE7_EXTBG_TABLE
500    } else {
501        priority_table(ppu.bgmode)
502    };
503    let backdrop_bgr5 = cgram_to_bgr5(ppu, 0);
504
505    // Fixed COLDATA — the sub operand when CGWSEL bit 1 = 0, or when
506    // bit 1 = 1 but the sub had no real winner (math.transparent
507    // fallback per ares dac.cpp:124-130 / Mesen2 SnesPpu.cpp:1354-1364).
508    let coldata_bgr5 = (ppu.coldata_r, ppu.coldata_g, ppu.coldata_b);
509
510    // Per-layer window masks for THIS scanline. Once mid-frame window-
511    // register writes are honoured, this must be per-line not per-frame.
512    let masks = compute_window_masks(ppu);
513
514    // Mode 7 uses a dedicated affine renderer for BG1 instead of the
515    // planar tilemap path. BG2-4 are unused in Mode 7.
516    let is_mode7 = ppu.bgmode & 0x07 == 0x07;
517    // Modes 5/6 are hi-res: each dot has two BG subpixels (left = sub
518    // screen, right = main screen). We render 512-wide content as two
519    // 256-wide sample sets and downsample by averaging the pair (the
520    // chosen "Option A" — keeps the framebuffer 256 wide).
521    let is_hires = matches!(ppu.bgmode & 0x07, 5 | 6);
522    // Pseudo-hires ($2133 bit 3) interleaves the sub and main screens
523    // per dot in *non*-5/6 modes — the transparency trick (Kirby
524    // waterfalls, etc.). BG content stays lores (256); only the
525    // main/sub blend kicks in (ares dac.cpp:34). It reuses the exact
526    // same averaging output as true hi-res.
527    let is_pseudo_hires = ppu.setini & 0x08 != 0;
528    let blend_hires = is_hires || is_pseudo_hires;
529
530    // Indexed scanlines for the 4 BG layers (transparent = `None`).
531    // `bgs_above` feeds the main-screen winner, `bgs_below` the sub-
532    // screen winner. Outside hi-res the two are identical — only TM/TS
533    // + TMW/TSW gating differs.
534    let (bgs_above, bgs_below): ([IndexedScanline; 4], [IndexedScanline; 4]) = if is_mode7 {
535        let bg1 = render_mode7_scanline_indexed(ppu, y, opts);
536        // EXTBG: derive BG2 from the same plane — colour = low 7 bits,
537        // priority = bit 7, transparent when the low 7 bits are zero.
538        let bg2 = if extbg {
539            let mut b2: IndexedScanline = [None; 256];
540            for (x, px) in bg1.iter().enumerate() {
541                if let Some((full, _)) = px {
542                    let low7 = full & 0x7F;
543                    if low7 != 0 {
544                        b2[x] = Some((low7, full >> 7));
545                    }
546                }
547            }
548            b2
549        } else {
550            [None; 256]
551        };
552        let m7 = [bg1, bg2, [None; 256], [None; 256]];
553        (m7, m7)
554    } else if is_hires {
555        let mut above = [[None; 256]; 4];
556        let mut below = [[None; 256]; 4];
557        for i in 0..4 {
558            let (a, b) = render_bg_scanline_indexed_hires(ppu, i, y, opts);
559            above[i] = a;
560            below[i] = b;
561        }
562        (above, below)
563    } else {
564        let s = [
565            render_bg_scanline_indexed_with(ppu, 0, y, opts),
566            render_bg_scanline_indexed_with(ppu, 1, y, opts),
567            render_bg_scanline_indexed_with(ppu, 2, y, opts),
568            render_bg_scanline_indexed_with(ppu, 3, y, opts),
569        ];
570        (s, s)
571    };
572    let sprites = match precomp {
573        Some((s, e)) => render_sprites_scanline_indexed_from(ppu, y, opts, s, e),
574        None => render_sprites_scanline_indexed_with(ppu, y, opts),
575    };
576
577    for (x, out_pixel) in out[start..end]
578        .iter_mut()
579        .enumerate()
580        .map(|(i, p)| (i + start, p))
581    {
582        // Per-screen layer enable. TM/TMW gate the main screen, TS/TSW
583        // gate the sub screen. If TM.layer = 1 AND TMW.layer = 1 the
584        // layer is hidden inside its window; if TM.layer = 0 the layer
585        // never appears on main. (Same for TS/TSW.)
586        let screen_layer_enabled = |layer_idx: usize, en_mask: u8, win_mask: u8| -> bool {
587            let layer_bit = 1u8 << layer_idx;
588            if en_mask & layer_bit == 0 {
589                return false;
590            }
591            if win_mask & layer_bit != 0 && masks.combined[layer_idx][x] {
592                return false;
593            }
594            true
595        };
596
597        let main = pick_pixel_winner(ppu, table, &bgs_above, &sprites, x, backdrop_bgr5, |li| {
598            screen_layer_enabled(li, ppu.tm, ppu.tmw)
599        });
600        let sub = pick_pixel_winner(ppu, table, &bgs_below, &sprites, x, backdrop_bgr5, |li| {
601            screen_layer_enabled(li, ppu.ts, ppu.tsw)
602        });
603
604        // Colour math gates:
605        //   * CGADSUB bits 0..4 = per-layer enable; bit 5 = backdrop.
606        //   * CGWSEL bits 5:4 = per-region enable using the colour-math
607        //     window mask (layer index 5).
608        //   * OBJ math also requires the winner palette ≥ 4 (CGRAM index
609        //     ≥ 192) — see ares dac.cpp:103-107.
610        let in_math_window = masks.combined[5][x];
611        let math_globally_enabled = match (ppu.cgwsel >> 4) & 0x03 {
612            0 => true,
613            1 => in_math_window,
614            2 => !in_math_window,
615            _ => false,
616        };
617        let layer_enabled_for_math = if math_globally_enabled {
618            match main.layer {
619                0..=3 => ppu.cgadsub & (1 << main.layer) != 0,
620                4 => (ppu.cgadsub & 0x10) != 0 && main.cgram_idx >= 192,
621                _ => ppu.cgadsub & 0x20 != 0, // backdrop
622            }
623        } else {
624            false
625        };
626
627        // CGWSEL bit 1 selects the math operand source:
628        //   0 → fixed COLDATA always.
629        //   1 → sub-screen winner pixel; if the sub had no real layer
630        //       winner (just the sub backdrop), fall back to COLDATA AND
631        //       disable halve for this dot. ares dac.cpp:124-130 / Mesen2
632        //       SnesPpu.cpp:1354-1364.
633        let subtract = ppu.cgadsub & 0x80 != 0;
634        let mut half = ppu.cgadsub & 0x40 != 0;
635        let math_operand = if ppu.cgwsel & 0x02 != 0 {
636            if sub.real_layer {
637                sub.bgr5
638            } else {
639                half = false;
640                coldata_bgr5
641            }
642        } else {
643            coldata_bgr5
644        };
645
646        let rgb5 = if layer_enabled_for_math {
647            color_math(main.bgr5, math_operand, subtract, half)
648        } else {
649            main.bgr5
650        };
651        // CGWSEL bits 7:6 — force-main-black region. Per ares
652        // (window.cpp:36-38 + dac.cpp:120-122) and Mesen2
653        // (SnesPpuTypes.h:13-19 + SnesPpu.cpp:1307-1326), value 1 =
654        // OutsideWindow, value 2 = InsideWindow.
655        let force_black = match (ppu.cgwsel >> 6) & 0x03 {
656            0 => false,
657            1 => !in_math_window,
658            2 => in_math_window,
659            _ => true,
660        };
661        let main_bgr5 = if force_black { (0, 0, 0) } else { rgb5 };
662
663        // Hi-res / pseudo-hires: the dot's left subpixel is the sub-
664        // screen winner (`sub`), the right is the main-screen result.
665        // Option A collapses the 512-wide line to 256 by averaging the
666        // pair — matching a CRT's horizontal blend (ares dac.cpp:39-40
667        // emits below then above). Otherwise the main result stands alone.
668        let final_bgr5 = if blend_hires {
669            average_bgr5(sub.bgr5, main_bgr5)
670        } else {
671            main_bgr5
672        };
673
674        let rgb888 = [
675            scale_5_to_8(final_bgr5.0),
676            scale_5_to_8(final_bgr5.1),
677            scale_5_to_8(final_bgr5.2),
678        ];
679        *out_pixel = apply_brightness(rgb888, brightness);
680    }
681}

687/// Per-layer combined window masks for one frame.
688///
689/// `combined[layer][x]` is `true` when pixel `x` is "inside" the
690/// effective window region for that layer (W1 ∪/∩/⊕/⊕̄ W2, modulated
691/// by per-window invert bits). It's used by:
692///
693///   * `TMW` / `TSW` to gate which pixels of a layer reach the
694///     main / sub screen.
695///   * `CGWSEL` to decide where colour math applies and where the
696///     main screen is forced to black.
697///
698/// Layers are indexed `0..=3` for BG1..BG4, `4` for OBJ, `5` for
699/// the dedicated colour-math window.
700struct WindowMasks {
701    combined: [[bool; FRAME_W]; 6],
702}
703
704fn compute_window_masks(ppu: &Ppu) -> WindowMasks {
705    let mut out = WindowMasks {
706        combined: [[false; FRAME_W]; 6],
707    };
708    let in_w1 = make_window(ppu.wh0, ppu.wh1);
709    let in_w2 = make_window(ppu.wh2, ppu.wh3);
710    // Per-layer (W?SEL nibble, log bits): {W1 invert, W1 enable,
711    // W2 invert, W2 enable} packed in `sel`'s low 4 bits, plus a
712    // 2-bit logic op selecting OR / AND / XOR / XNOR.
713    let layer_cfg = [
714        (ppu.w12sel & 0x0F, ppu.wbglog & 0x03),
715        (ppu.w12sel >> 4, (ppu.wbglog >> 2) & 0x03),
716        (ppu.w34sel & 0x0F, (ppu.wbglog >> 4) & 0x03),
717        (ppu.w34sel >> 4, (ppu.wbglog >> 6) & 0x03),
718        (ppu.wobjsel & 0x0F, ppu.wobjlog & 0x03),
719        (ppu.wobjsel >> 4, (ppu.wobjlog >> 2) & 0x03),
720    ];
721    for (layer, &(sel, logic_bits)) in layer_cfg.iter().enumerate() {
722        let w1_invert = sel & 0x01 != 0;
723        let w1_enable = sel & 0x02 != 0;
724        let w2_invert = sel & 0x04 != 0;
725        let w2_enable = sel & 0x08 != 0;
726        if !w1_enable && !w2_enable {
727            continue; // no window for this layer → mask stays all-false
728        }
729        for x in 0..FRAME_W {
730            let r1 = if w1_enable {
731                in_w1[x] ^ w1_invert
732            } else {
733                false
734            };
735            let r2 = if w2_enable {
736                in_w2[x] ^ w2_invert
737            } else {
738                false
739            };
740            out.combined[layer][x] = match (w1_enable, w2_enable) {
741                (true, false) => r1,
742                (false, true) => r2,
743                (true, true) => match logic_bits {
744                    0 => r1 || r2,
745                    1 => r1 && r2,
746                    2 => r1 ^ r2,
747                    _ => !(r1 ^ r2),
748                },
749                _ => unreachable!(),
750            };
751        }
752    }
753    out
754}

760/// Render one Mode-7 scanline into an [`IndexedScanline`].
761///
762/// Mode 7 has just one BG layer (BG1) which is the entire VRAM
763/// interpreted as a 128×128-tile field with 8×8 8bpp tiles. This is a
764/// direct port of ares `mode7.cpp`. The per-scanline affine origin is:
765///
766/// ```text
767///   origin_x = M7A·clip(M7HOFS - M7X) + M7B·clip(M7VOFS - M7Y)
768///              + M7B·y + (M7X << 8)
769///   origin_y = M7C·clip(M7HOFS - M7X) + M7D·clip(M7VOFS - M7Y)
770///              + M7D·y + (M7Y << 8)
771///   pixel_x  = (origin_x + M7A·x) >> 8
772///   pixel_y  = (origin_y + M7C·x) >> 8
773/// ```
774///
775/// where matrix values are signed 8.8 fixed point, `clip()` keeps the
776/// 10-bit magnitude + bit-13 sign of the scroll/centre difference, and
777/// each partial product drops its low 6 bits (`& !63`) to match the
778/// hardware multiplier. The integer part is a VRAM byte address into
779/// the tilemap (low byte) and pixel data (high byte) of the interleaved
780/// 64 KB tilemap / tileset.
781///
782/// `M7SEL` bits 1:0 control the wrap / transparent behaviour outside
783/// the 128×128 field:
784///
785///   * `00` / `01` — coordinates wrap (the 7-bit tile mask repeats)
786///   * `10` — return transparent (force palette 0)
787///   * `11` — use tile 0 outside the field
788///
789/// Horizontal and vertical flips (`M7SEL` bits 6, 7) negate the
790/// screen-space coordinate before the matrix multiply.
791#[must_use]
792pub fn render_mode7_scanline_indexed(ppu: &Ppu, y: u16, opts: RenderOptions) -> IndexedScanline {
793    let mut out: IndexedScanline = [None; 256];
794    if ppu.inidisp & 0x80 != 0 && !opts.bypass_forced_blank {
795        return out;
796    }
797    let m7a = i32::from(ppu.m7a);
798    let m7b = i32::from(ppu.m7b);
799    let m7c = i32::from(ppu.m7c);
800    let m7d = i32::from(ppu.m7d);
801    // Centre and scroll are signed 13-bit (already sign-extended in i16).
802    let hcenter = i32::from(ppu.m7x);
803    let vcenter = i32::from(ppu.m7y);
804    let hoffset = i32::from(ppu.m7_hofs);
805    let voffset = i32::from(ppu.m7_vofs);
806    // M7SEL bit layout (ares io.cpp:411-414): bit 0 = H-flip,
807    // bit 1 = V-flip, bits 7:6 = screen-over. NOT bits 7/6/1:0.
808    let h_flip = ppu.m7sel & 0x01 != 0;
809    let v_flip = ppu.m7sel & 0x02 != 0;
810    let screen_over = (ppu.m7sel >> 6) & 0x03;
811
812    // 13-bit sign extend with a 10-bit magnitude (ares mode7.cpp:26-27):
813    // `--s---nnnnnnnnnn` -> `ssssssnnnnnnnnnn`. The Mode-7 multiplier
814    // only keeps a 10-bit magnitude plus the bit-13 sign of the
815    // scroll/centre difference.
816    let clip = |n: i32| -> i32 { if n & 0x2000 != 0 { n | !1023 } else { n & 1023 } };
817
818    // Mosaic (ares mode7.cpp:12-21): Mode 7 uses BG1's mosaic enable
819    // (MOSAIC bit 0). The screen x/y are snapped to the block's top-left
820    // before the affine transform, replicating that pixel across the
821    // block.
822    let mosaic_size = if ppu.mosaic & 0x01 != 0 {
823        i32::from((ppu.mosaic >> 4) & 0x0F) + 1
824    } else {
825        1
826    };
827    let mosaic_y = i32::from(y) / mosaic_size * mosaic_size;
828
829    // V-flip mirrors the screen-space scanline (ares mode7.cpp:24).
830    let yy = if v_flip { 255 - mosaic_y } else { mosaic_y };
831
832    // Per-scanline affine origin. Each partial product drops its low 6
833    // bits (`& !63`) to match the hardware multiplier (ares
834    // mode7.cpp:28-29).
835    let origin_x = ((m7a * clip(hoffset - hcenter)) & !63)
836        + ((m7b * clip(voffset - vcenter)) & !63)
837        + ((m7b * yy) & !63)
838        + (hcenter << 8);
839    let origin_y = ((m7c * clip(hoffset - hcenter)) & !63)
840        + ((m7d * clip(voffset - vcenter)) & !63)
841        + ((m7d * yy) & !63)
842        + (vcenter << 8);
843
844    for x in 0..256i32 {
845        let mosaic_x = x / mosaic_size * mosaic_size;
846        // H-flip mirrors the screen-space dot (ares mode7.cpp:23).
847        let xx = if h_flip { 255 - mosaic_x } else { mosaic_x };
848
849        // 16.8 fixed point -> integer pixel grid (ares mode7.cpp:31-32).
850        let pix_x = (origin_x + m7a * xx) >> 8;
851        let pix_y = (origin_y + m7c * xx) >> 8;
852
853        // OOB if either coordinate leaves the 1024×1024 field (ares
854        // mode7.cpp:39) — the sign bits of a negative value also trip
855        // the `& !1023` test.
856        let out_of_bounds = ((pix_x | pix_y) & !1023) != 0;
857
858        // Tile coordinates wrap into the 128×128 tilemap via the 7-bit
859        // mask; in-tile coordinates take the low 3 bits (ares
860        // mode7.cpp:33-37).
861        let tile_x = ((pix_x >> 3) & 0x7F) as u16;
862        let tile_y = ((pix_y >> 3) & 0x7F) as u16;
863        let palette_address = (((pix_y & 7) << 3) | (pix_x & 7)) as u16;
864
865        // VRAM layout: byte 2n+0 = tilemap entry n, byte 2n+1 = char
866        // data. Screen-over 3 forces tile 0 outside the field; screen-
867        // over 2 forces a transparent pixel (ares mode7.cpp:41-42).
868        let tilemap_entry_addr = (tile_y << 7 | tile_x) * 2;
869        let tile_id = if screen_over == 3 && out_of_bounds {
870            0
871        } else {
872            ppu.vram.peek(tilemap_entry_addr)
873        };
874        let pixel_byte_addr = (u16::from(tile_id) << 7) // tile * 64 words = * 128 bytes
875            .wrapping_add(palette_address * 2)
876            .wrapping_add(1); // odd byte = char data
877        let palette_idx = if screen_over == 2 && out_of_bounds {
878            0
879        } else {
880            ppu.vram.peek(pixel_byte_addr)
881        };
882        if palette_idx == 0 {
883            continue; // transparent
884        }
885        out[x as usize] = Some((palette_idx, 0));
886    }
887    out
888}

890/// Mark every pixel between `left` and `right` (inclusive) as
891/// "inside" the window. If `left > right` the window is empty,
892/// matching the documented hardware behaviour.
893fn make_window(left: u8, right: u8) -> [bool; FRAME_W] {
894    let mut out = [false; FRAME_W];
895    if left <= right {
896        let l = left as usize;
897        let r = (right as usize).min(FRAME_W - 1);
898        out[l..=r].fill(true);
899    }
900    out
901}

903/// Look up a CGRAM index and return the raw 5-bit BGR triple, with
904/// no brightness scaling. Used by the colour-math path so math
905/// happens in 5-bit space (matching real hardware).
906fn cgram_to_bgr5(ppu: &Ppu, cgram_index: u8) -> (u8, u8, u8) {
907    let color = ppu.cgram.color(cgram_index);
908    let r5 = (color & 0x001F) as u8;
909    let g5 = ((color >> 5) & 0x001F) as u8;
910    let b5 = ((color >> 10) & 0x001F) as u8;
911    (r5, g5, b5)
912}

914/// Direct-color-mode decode (CGWSEL bit 0). When set, 8bpp BG /
915/// Mode-7 pixels skip the CGRAM lookup — the pixel byte
916/// (`BBGGGRRR`) plus the tilemap palette group (`bgr`) form the colour
917/// directly, giving more than 256 distinct RGB values without burning
918/// CGRAM entries.
919///
920/// Per ares `dac.cpp:163-170` the output is `0BBb00GG Gg0RRRr0`: the
921/// pixel supplies the high bits of each channel and the 3-bit palette
922/// group supplies one extra low bit each (R←g0, G←g1, B←g2). With
923/// group 0 this reduces to the plain `BBGGGRRR` decode.
924const fn direct_color_to_bgr5(palette_index: u8, group: u8) -> (u8, u8, u8) {
925    let r = ((palette_index & 0x07) << 2) | ((group & 0x01) << 1);
926    let g = (((palette_index >> 3) & 0x07) << 2) | (((group >> 1) & 0x01) << 1);
927    let b = (((palette_index >> 6) & 0x03) << 3) | (((group >> 2) & 0x01) << 2);
928    (r, g, b)
929}

931/// SNES colour-math arithmetic — add or subtract a 5-bit sub-screen
932/// Winner record for a single pixel on one screen (main or sub).
933#[derive(Debug, Clone, Copy)]
934struct PixelWinner {
935    /// 5-bit BGR triplet for this pixel.
936    bgr5: (u8, u8, u8),
937    /// Which layer drew it. `0..=3` = BG1..BG4, `4` = OBJ, `5` = backdrop.
938    layer: u8,
939    /// CGRAM index of the winning pixel — needed for the OBJ palette
940    /// ≥ 4 gate (only sprites with palette ≥ 4, i.e. CGRAM index ≥ 192,
941    /// participate in color math; ares dac.cpp:103-107).
942    cgram_idx: u8,
943    /// `true` when a BG or OBJ layer drew, `false` when only the
944    /// backdrop is the result. Used by the math.transparent fallback
945    /// for the sub screen (ares dac.cpp:69, 124-130 / Mesen2
946    /// SnesPpu.cpp:1354-1364).
947    real_layer: bool,
948}

950/// Walk the priority table for one screen and return the winning pixel.
951/// The screen-specific enable+window gating is supplied via the
952/// `is_enabled` predicate (TM/TMW for main, TS/TSW for sub). Direct
953/// color (CGWSEL bit 0) is applied here for 8bpp BG layers.
954fn pick_pixel_winner(
955    ppu: &Ppu,
956    table: &[LayerSlot],
957    bgs: &[IndexedScanline; 4],
958    sprites: &IndexedScanline,
959    x: usize,
960    backdrop_bgr5: (u8, u8, u8),
961    is_enabled: impl Fn(usize) -> bool,
962) -> PixelWinner {
963    for slot in table {
964        let layer_idx = match slot.kind {
965            LayerKind::Bg => slot.idx as usize,
966            LayerKind::Obj => 4,
967        };
968        if !is_enabled(layer_idx) {
969            continue;
970        }
971        // BG pixels pack the 3-bit tilemap palette group into prio bits
972        // 1-3 (bit 0 = the tile priority bit) so direct colour can read
973        // it; sprites carry their 0-3 priority unpacked with no group.
974        let candidate: Option<(u8, u8)> = match slot.kind {
975            LayerKind::Bg => bgs[slot.idx as usize][x].and_then(|(cg, prio)| {
976                ((prio & 1) == slot.bg_prio).then_some((cg, (prio >> 1) & 0x07))
977            }),
978            LayerKind::Obj => {
979                sprites[x].and_then(|(cg, prio)| (prio == slot.idx).then_some((cg, 0)))
980            }
981        };
982        if let Some((cgram_idx, palette_group)) = candidate {
983            let direct = ppu.cgwsel & 0x01 != 0
984                && matches!(slot.kind, LayerKind::Bg)
985                && bg_bpp(ppu.bgmode, slot.idx as usize) == 8;
986            let bgr5 = if direct {
987                direct_color_to_bgr5(cgram_idx, palette_group)
988            } else {
989                cgram_to_bgr5(ppu, cgram_idx)
990            };
991            return PixelWinner {
992                bgr5,
993                layer: layer_idx as u8,
994                cgram_idx,
995                real_layer: true,
996            };
997        }
998    }
999    PixelWinner {
1000        bgr5: backdrop_bgr5,
1001        layer: 5,
1002        cgram_idx: 0,
1003        real_layer: false,
1004    }
1005}

1007/// colour from a 5-bit main-screen colour, optionally halving the
1008/// result. `subtract` and `half` are passed explicitly so the caller
1009/// can override `half` (the math.transparent empty-sub fallback wants
1010/// to disable halve even though CGADSUB bit 6 is set — see ares
1011/// dac.cpp:124-130 / Mesen2 SnesPpu.cpp:1354-1364).
1012/// Average two 5-bit BGR triples — the hi-res 512→256 horizontal
1013/// downsample (left/sub + right/main subpixels collapsed into one dot).
1014const fn average_bgr5(a: (u8, u8, u8), b: (u8, u8, u8)) -> (u8, u8, u8) {
1015    ((a.0 + b.0) >> 1, (a.1 + b.1) >> 1, (a.2 + b.2) >> 1)
1016}

1018fn color_math(main: (u8, u8, u8), sub: (u8, u8, u8), subtract: bool, half: bool) -> (u8, u8, u8) {
1019    let combine = |m: u8, s: u8| -> u8 {
1020        let m = i32::from(m);
1021        let s = i32::from(s);
1022        let mut r = if subtract { m - s } else { m + s };
1023        if half {
1024            r >>= 1;
1025        }
1026        r.clamp(0, 31) as u8
1027    };
1028    (
1029        combine(main.0, sub.0),
1030        combine(main.1, sub.1),
1031        combine(main.2, sub.2),
1032    )
1033}

1082// Mode 0: BG1, BG2, BG3, BG4 all 2bpp. Real-hardware order:
1083//   OBJ3, BG1H, BG2H, OBJ2, BG1L, BG2L, OBJ1, BG3H, BG4H, OBJ0, BG3L, BG4L

1086    bg(0, 1),
1087    bg(1, 1),
1088    obj(2),
1089    bg(0, 0),
1090    bg(1, 0),
1091    obj(1),
1092    bg(2, 1),
1093    bg(3, 1),
1094    obj(0),
1095    bg(2, 0),
1096    bg(3, 0),

1099// Mode 1, BG3 priority bit 0:
1100//   OBJ3, BG1H, BG2H, OBJ2, BG1L, BG2L, OBJ1, BG3H, OBJ0, BG3L
1101const MODE1_BG3LO_TABLE: &[LayerSlot] = &[
1102    obj(3),
1103    bg(0, 1),
1104    bg(1, 1),
1105    obj(2),
1106    bg(0, 0),
1107    bg(1, 0),
1108    obj(1),
1109    bg(2, 1),

1111    bg(2, 0),

1115//   BG3H, OBJ3, BG1H, BG2H, OBJ2, BG1L, BG2L, OBJ1, OBJ0, BG3L
1116const MODE1_BG3HI_TABLE: &[LayerSlot] = &[
1117    bg(2, 1),
1118    obj(3),
1119    bg(0, 1),
1120    bg(1, 1),
1121    obj(2),
1122    bg(0, 0),
1123    bg(1, 0),
1124    obj(1),
1125    obj(0),
1126    bg(2, 0),

1129// Modes 2 and 3 (BG1+BG2 only):
1130//   OBJ3, BG1H, OBJ2, BG2H, OBJ1, BG1L, OBJ0, BG2L
1131const MODE2OR3_TABLE: &[LayerSlot] = &[
1132    obj(3),
1133    bg(0, 1),
1134    obj(2),
1135    bg(1, 1),
1136    obj(1),
1137    bg(0, 0),
1138    obj(0),
1139    bg(1, 0),
1140];

1142// Modes 5/6 (hi-res, BG1 + BG2 only) — same layer order as Mode 1
1143// minus BG3:
1144//   OBJ3, BG1H, BG2H, OBJ2, BG1L, BG2L, OBJ1, OBJ0
1145const MODE56_TABLE: &[LayerSlot] = &[
1146    obj(3),
1147    bg(0, 1),
1148    bg(1, 1),
1149    obj(2),
1150    bg(0, 0),
1151    bg(1, 0),
1152    obj(1),
1153    obj(0),
1154];

1156// Mode 7: just BG1 (affine) and OBJ. Standard hardware convention:
1157//   OBJ3, OBJ2, OBJ1, BG1, OBJ0
1158// — i.e. only sprite priority 0 falls below the Mode-7 plane.
1159const MODE7_TABLE: &[LayerSlot] = &[obj(3), obj(2), obj(1), bg(0, 0), obj(0)];
1160
1161// Mode 7 EXTBG ($2133.6): the plane is also exposed as BG2, split by
1162// pixel bit 7 into two priorities (ares io.cpp:733-738, numeric
1163// priorities OBJ3=7, OBJ2=6, BG2hi=5, OBJ1=4, BG1=3, OBJ0=2, BG2lo=1):
1164//   OBJ3, OBJ2, BG2H, OBJ1, BG1, OBJ0, BG2L
1165const MODE7_EXTBG_TABLE: &[LayerSlot] =
1166    &[obj(3), obj(2), bg(1, 1), obj(1), bg(0, 0), obj(0), bg(1, 0)];

1168/// Pick the priority table for the current BGMODE.
1169const fn priority_table(bgmode: u8) -> &'static [LayerSlot] {
1170    match bgmode & 0x07 {
1171        0 => MODE0_TABLE,
1172        1 => {
1173            // BGMODE bit 3 = "BG3 priority": when set, BG3 high
1174            // slot moves to the top of the table.
1175            if bgmode & 0x08 != 0 {
1176                MODE1_BG3HI_TABLE
1177            } else {
1178                MODE1_BG3LO_TABLE
1179            }
1180        }
1181        2..=4 => MODE2OR3_TABLE,
1182        5 | 6 => MODE56_TABLE,
1183        7 => MODE7_TABLE,
1184        _ => MODE1_BG3LO_TABLE, // unreachable (bgmode & 7 ∈ 0..=7)
1185    }
1186}

1212/// Resolve a BG layer's geometry, or `None` if the layer is disabled
1213/// in the current mode (bpp 0).
1214const fn bg_geom(ppu: &Ppu, bg_idx: usize) -> Option<BgGeom> {

1218    }

1220        2 => 16,

1223        _ => 16,
1224    };

1226    let (cols, rows) = match bg.tilemap_size & 0x03 {
1227        1 => (64u16, 32u16),

1230        _ => (32u16, 32u16),
1231    };

1234        bpp,
1235        bytes_per_tile,

1238        cols,
1239        rows,
1240        tilemap_size: bg.tilemap_size & 0x03,
1241        tile_pixels: if big_tiles { 16 } else { 8 },
1242        tile_shift: if big_tiles { 4 } else { 3 },
1243        force_wide: matches!(ppu.bgmode & 0x07, 5 | 6),
1244        mode0_palette_base: if ppu.bgmode & 0x07 == 0 {
1245            (bg_idx as u8) << 5
1246        } else {
1247            0
1248        },
1249    })

1252/// Sample one BG pixel at a source position (in BG pixels). Returns the
1253/// CGRAM index + tilemap priority bit, or `None` if transparent. Shared
1254/// by the lores (256) and hires (512) renderers — for hires the caller
1255/// passes a doubled-scroll, 2×-density source coordinate; tiles stay
1256/// 8 source-pixels wide so no other change is needed.
1257///
1258/// Mode 0 gives each BG its own 32-colour CGRAM region (BG1→0, BG2→32,
1259/// BG3→64, BG4→96; ares `background.cpp:111`). Only Mode 0 has that
1260/// offset and Mode 0 is 2bpp on every BG, so it only touches the 2bpp
1261/// arm — max index 96 + 7*4 + 3 = 127.
1262/// Per-scanline memo for [`sample_bg_pixel`]: the most-recent tilemap
1263/// entry (keyed by its VRAM byte offset) and decoded tile row (keyed by
1264/// tile base + row + bpp). Consecutive same-tile pixels reuse both
1265/// instead of re-fetching the entry and re-decoding the row 8× (PERF-1).
1266#[derive(Default)]
1267struct BgSampleCache {
1268    entry: Option<(usize, u16)>,
1269    row: Option<(usize, usize, u8, [u8; 8])>,
1270}
1271
1272fn sample_bg_pixel(
1273    ppu: &Ppu,
1274    g: &BgGeom,
1275    src_x: u16,
1276    src_y: u16,
1277    cache: &mut BgSampleCache,
1278) -> IndexedPixel {
1279    // Horizontal tile span: hi-res (Mode 5/6) forces 16-px-wide columns
1280    // (ares `background.cpp:79` `htiles = 4`); vertical follows the size bit.
1281    let (h_shift, h_pixels) = if g.force_wide {
1282        (4u16, 16u16)
1283    } else {
1284        (g.tile_shift, g.tile_pixels)
1285    };
1286    // Tile coordinates in TILE units (8 or 16 pixels per side).
1287    let tile_col_full = (src_x >> h_shift) & (g.cols - 1);
1288    let tile_row_full = (src_y >> g.tile_shift) & (g.rows - 1);
1289    let sub_x = (tile_col_full >> 5) as usize;
1290    let sub_y = (tile_row_full >> 5) as usize;
1291    let sub_index = match g.tilemap_size {
1292        1 => sub_x,
1293        2 => sub_y,
1294        3 => sub_y * 2 + sub_x,
1295        _ => 0,
1296    };
1297    let tile_col = tile_col_full & 0x1F;
1298    let tile_row = tile_row_full & 0x1F;
1299    let entry_off = g.tilemap_base + sub_index * 0x0800 + ((tile_row * 32 + tile_col) as usize) * 2;
1300    let entry = match cache.entry {
1301        Some((k, w)) if k == entry_off => w,
1302        _ => {
1303            let entry_lo = ppu.vram.peek(entry_off as u16);
1304            let entry_hi = ppu.vram.peek(entry_off.wrapping_add(1) as u16);
1305            let w = u16::from(entry_lo) | (u16::from(entry_hi) << 8);
1306            cache.entry = Some((entry_off, w));
1307            w
1308        }
1309    };
1310    let tile_num = entry & 0x03FF;
1311    let palette_off = ((entry >> 10) & 0x07) as u8;
1312    let prio_bit = ((entry >> 13) & 0x01) as u8;
1313    let h_flip = entry & 0x4000 != 0;
1314    let v_flip = entry & 0x8000 != 0;
1315    // Pixel position within the block, masked per-axis (horizontal may be
1316    // 16 even when vertical is 8, in hi-res).
1317    let mut col_in_block = (src_x & (h_pixels - 1)) as usize;
1318    let mut row_in_block = (src_y & (g.tile_pixels - 1)) as usize;
1319    if h_flip {
1320        col_in_block = (h_pixels as usize - 1) - col_in_block;
1321    }
1322    if v_flip {
1323        row_in_block = (g.tile_pixels as usize - 1) - row_in_block;
1324    }
1325    // A 16-wide/tall block's halves share `tile_num` as the top-left, with
1326    // the canonical sprite-plane offset of +1 right, +16 down. The
1327    // conditions are naturally false for an 8-px axis (col/row never ≥ 8).
1328    let quadrant_offset: u16 =
1329        u16::from(col_in_block >= 8) + if row_in_block >= 8 { 16 } else { 0 };
1330    let final_tile = tile_num.wrapping_add(quadrant_offset) & 0x03FF;
1331    let row_in_tile = row_in_block & 7;
1332    let col_in_tile = col_in_block & 7;
1333    let tile_base = g.char_base + (final_tile as usize) * g.bytes_per_tile;
1334    let row = match cache.row {
1335        Some((b, r, bp, data)) if b == tile_base && r == row_in_tile && bp == g.bpp => data,
1336        _ => {
1337            let data = decode_tile_row(ppu, tile_base, row_in_tile, g.bpp);
1338            cache.row = Some((tile_base, row_in_tile, g.bpp, data));
1339            data
1340        }
1341    };
1342    let idx = row[col_in_tile];
1343    if idx == 0 {
1344        return None;
1345    }
1346    let cgram_idx = match g.bpp {
1347        2 => g.mode0_palette_base + palette_off * 4 + idx,
1348        4 => palette_off.wrapping_mul(16).wrapping_add(idx),
1349        _ => idx,
1350    };
1351    // Pack the 3-bit palette group into prio bits 1-3 (bit 0 = tile
1352    // priority) so 8bpp direct colour can recover it; for 2/4bpp the
1353    // group is already folded into cgram_idx and these bits are ignored.
1354    Some((cgram_idx, prio_bit | (palette_off << 1)))
1355}

1357/// Read one raw 16-bit tilemap word from BG3, used as offset-per-tile
1358/// (OPT) data in Modes 2/4/6. `hpixel`/`vpixel` are BG3 source pixels;
1359/// addressing mirrors [`sample_bg_pixel`] (ares-style 32×32 sub-screen
1360/// quadrants), which the two references prefer over Mesen2's linear
1361/// 64-wide form.
1362fn bg3_tilemap_word(ppu: &Ppu, bg3: &crate::ppu::BgState, hpixel: u16, vpixel: u16) -> u16 {
1363    let (cols, rows) = match bg3.tilemap_size & 0x03 {
1364        1 => (64u16, 32u16),
1365        2 => (32u16, 64u16),
1366        3 => (64u16, 64u16),
1367        _ => (32u16, 32u16),
1368    };
1369    let htile = (hpixel >> 3) & (cols - 1);
1370    let vtile = (vpixel >> 3) & (rows - 1);
1371    let sub_x = (htile >> 5) as usize;
1372    let sub_y = (vtile >> 5) as usize;
1373    let sub_index = match bg3.tilemap_size & 0x03 {
1374        1 => sub_x,
1375        2 => sub_y,
1376        3 => sub_y * 2 + sub_x,
1377        _ => 0,
1378    };
1379    let tile_col = htile & 0x1F;
1380    let tile_row = vtile & 0x1F;
1381    let base = (bg3.tilemap_addr_words as usize) << 1;
1382    let off = base + sub_index * 0x0800 + ((tile_row * 32 + tile_col) as usize) * 2;
1383    u16::from(ppu.vram.peek(off as u16))
1384        | (u16::from(ppu.vram.peek(off.wrapping_add(1) as u16)) << 8)
1385}

1387/// Offset-per-tile (Modes 2/4/6): replace BG1/BG2's scroll for one
1388/// screen tile column with values read from BG3's tilemap. Returns the
1389/// effective `(h_scroll, v_scroll)` for column `screen_col`.
1390///
1391/// The leftmost column (0) always uses the plain scroll; column D≥1
1392/// uses BG3 data one tile to the left (the hardware fetch-pipeline lag
1393/// — ares `background.cpp:52-69` + `fetchOffset`, Mesen2
1394/// `SnesPpu.cpp:153-169,257-276`). H replaces scroll bits 3-9 keeping
1395/// the fine `&7`; V replaces fully. Mode 4 packs both into one word
1396/// (bit 15 picks H vs V); Modes 2/6 use two words. Enable bit =
1397/// `0x2000` for BG1, `0x4000` for BG2.
1398/// Returns `(eff_h, eff_v, h_from_opt)` — the effective scroll for a
1399/// screen column, and whether the H component was overridden by OPT (the
1400/// hi-res caller needs that: the base scroll doubles in hi-res but the OPT
1401/// override does not — ares `background.cpp:66`).
1402fn opt_scroll(
1403    ppu: &Ppu,
1404    bg_idx: usize,
1405    screen_col: u16,
1406    base_h: u16,
1407    base_v: u16,
1408) -> (u16, u16, bool) {
1409    if screen_col == 0 {
1410        return (base_h, base_v, false);
1411    }
1412    let bg3 = bg_state(ppu, 2);
1413    let h_lookup = ((screen_col - 1) << 3).wrapping_add(bg3.h_scroll & !7);
1414    let h_word = bg3_tilemap_word(ppu, &bg3, h_lookup, bg3.v_scroll);
1415    let v_word = bg3_tilemap_word(ppu, &bg3, h_lookup, bg3.v_scroll.wrapping_add(8));
1416    let enable = if bg_idx == 0 { 0x2000u16 } else { 0x4000u16 };
1417    let mut eff_h = base_h;
1418    let mut eff_v = base_v;
1419    let mut h_from_opt = false;
1420    if ppu.bgmode & 0x07 == 4 {
1421        if h_word & enable != 0 {
1422            if h_word & 0x8000 == 0 {
1423                eff_h = (base_h & 0x07) | (h_word & 0x3F8);
1424                h_from_opt = true;
1425            } else {
1426                eff_v = h_word & 0x3FF;
1427            }
1428        }
1429    } else {
1430        if h_word & enable != 0 {
1431            eff_h = (base_h & 0x07) | (h_word & 0x3F8);
1432            h_from_opt = true;
1433        }
1434        if v_word & enable != 0 {
1435            eff_v = v_word & 0x3FF;
1436        }
1437    }
1438    (eff_h, eff_v, h_from_opt)
1439}

1441/// Hi-res interlace vertical source ($2133 bit 0, modes 5/6): the logical
1442/// line a screen scanline `y` samples, with mosaic in ares' order
1443/// (`background.cpp:38-44`). Mosaic snaps the *screen* line to its block
1444/// first; hi-res interlace then doubles (`block<<1`), adding the field bit
1445/// only when mosaic is off (`vpixel<<1 | (field && !mosaic.enable)`) — the
1446/// mosaic offset doubling (`voffset << (hires&&interlace)`) falls out of
1447/// snapping in screen space before the `<<1`. Progressive / non-interlace
1448/// reduces to the plain block. Lores interlace does **not** double (line 40
1449/// lives inside `if(hires())`), so the lores renderer skips this entirely.
1450#[inline]
1451pub(crate) fn hires_interlace_src(ppu: &Ppu, y: u16, mosaic_size: u16) -> u16 {
1452    let block = (y / mosaic_size) * mosaic_size;
1453    if ppu.setini & 0x01 != 0 {
1454        let field_bit = if mosaic_size > 1 {
1455            0
1456        } else {
1457            u16::from(ppu.field)
1458        };
1459        (block << 1) | field_bit
1460    } else {
1461        block
1462    }
1463}

1465/// Same as [`render_bg_scanline_with`] but returns CGRAM indices
1466/// (instead of decoded RGB) and tags each pixel with its tilemap
1467/// priority bit. Transparent pixels (colour 0 in the relevant
1468/// sub-palette) return `None` so the compositor can route them to
1469/// a lower-priority layer instead of the backdrop.
1470#[must_use]
1471pub fn render_bg_scanline_indexed_with(
1472    ppu: &Ppu,
1473    bg_idx: usize,
1474    y: u16,
1475    opts: RenderOptions,
1476) -> IndexedScanline {
1477    let mut out: IndexedScanline = [None; 256];
1478    if ppu.inidisp & 0x80 != 0 && !opts.bypass_forced_blank {
1479        return out;
1480    }
1481    let Some(g) = bg_geom(ppu, bg_idx) else {
1482        return out;
1483    };
1484    let bg = bg_state(ppu, bg_idx);
1485    // No interlace doubling on the lores path: ares `background.cpp:40`
1486    // (`vpixel <<= 1`) is inside `if(hires())`, so lores interlace shows the
1487    // same 224 content on both fields. `y` stays the screen line.
1488    // MOSAIC ($2106): high nibble = block size N (0..15 means 1..16
1489    // pixels per side), low nibble bit `bg_idx` enables mosaic for that
1490    // BG. Within a block we replicate the top-left pixel — implemented
1491    // by snapping both X and Y to the block boundary before sampling.
1492    let mosaic_size = if ppu.mosaic & (1 << bg_idx) != 0 {
1493        u16::from((ppu.mosaic >> 4) & 0x0F) + 1
1494    } else {
1495        1
1496    };
1497    let mosaic_y = (y / mosaic_size) * mosaic_size;
1498    // Offset-per-tile applies to BG1/BG2 in Modes 2/4 (Mode 6 is hi-res,
1499    // handled elsewhere). Effective scroll is recomputed per 8-pixel
1500    // screen column.
1501    let is_opt = matches!(ppu.bgmode & 0x07, 2 | 4) && bg_idx < 2;
1502    let mut opt_col = u16::MAX;
1503    let mut eff = (bg.h_scroll, bg.v_scroll);
1504    let mut cache = BgSampleCache::default();
1505    for x in 0..256u16 {
1506        let mosaic_x = (x / mosaic_size) * mosaic_size;
1507        if is_opt {
1508            let col = mosaic_x >> 3;
1509            if col != opt_col {
1510                opt_col = col;
1511                // Lores (Modes 2/4): the OPT high bits and the scroll low
1512                // bits combine directly — `h_from_opt` is irrelevant.
1513                let (h, v, _) = opt_scroll(ppu, bg_idx, col, bg.h_scroll, bg.v_scroll);
1514                eff = (h, v);
1515            }
1516        }
1517        let src_x = mosaic_x.wrapping_add(eff.0);
1518        let src_y = mosaic_y.wrapping_add(eff.1);
1519        out[x as usize] = sample_bg_pixel(ppu, &g, src_x, src_y, &mut cache);
1520    }
1521    out
1522}

1524/// Render a hi-res (Mode 5/6) BG layer, returning `(above, below)`
1525/// 256-wide sample sets. `above[x]` is the main-screen subpixel sampled
1526/// at hires column `2x+1`; `below[x]` is the sub-screen subpixel at
1527/// `2x`. Horizontal scroll is doubled (ares `background.cpp:39`, Mesen2
1528/// `SnesPpu.cpp:984`); tiles are 16 hires pixels wide (ares
1529/// `background.cpp:79` `htiles = 4`). Mosaic snaps the dot/scanline to
1530/// the block before doubling (ares `background.cpp:42-43`). Interlace
1531/// vertical doubling (ares `background.cpp:40,43`) is **not** applied —
1532/// luna has no interlaced output mode (224-line progressive only).
1533fn render_bg_scanline_indexed_hires(
1534    ppu: &Ppu,
1535    bg_idx: usize,
1536    y: u16,
1537    opts: RenderOptions,
1538) -> (IndexedScanline, IndexedScanline) {
1539    let mut above: IndexedScanline = [None; 256];
1540    let mut below: IndexedScanline = [None; 256];
1541    if ppu.inidisp & 0x80 != 0 && !opts.bypass_forced_blank {
1542        return (above, below);
1543    }
1544    let Some(g) = bg_geom(ppu, bg_idx) else {
1545        return (above, below);
1546    };
1547    let bg = bg_state(ppu, bg_idx);
1548    // Mosaic snaps the screen dot/scanline to the block.
1549    let mosaic_size = if ppu.mosaic & (1 << bg_idx) != 0 {
1550        u16::from((ppu.mosaic >> 4) & 0x0F) + 1
1551    } else {
1552        1
1553    };
1554    // Interlace (mode 5/6 + $2133 bit 0): mosaic-snap in screen space, THEN
1555    // double (field bit dropped when mosaic is on) — ares background.cpp:38-44.
1556    let mosaic_y = hires_interlace_src(ppu, y, mosaic_size);
1557    // Mode 6 is hi-res *and* offset-per-tile (BG1 only; Mode 5 has no
1558    // OPT, Modes 2/4 use the lores path). Effective scroll is per column.
1559    let is_opt = ppu.bgmode & 0x07 == 6 && bg_idx < 2;
1560    let mut opt_col = u16::MAX;
1561    let mut eff = (bg.h_scroll, bg.v_scroll);
1562    let mut h_from_opt = false;
1563    let mut cache = BgSampleCache::default();
1564    for x in 0..256u16 {
1565        let mosaic_x = x / mosaic_size * mosaic_size;
1566        if is_opt {
1567            let col = mosaic_x >> 3;
1568            if col != opt_col {
1569                opt_col = col;
1570                let (h, v, ho) = opt_scroll(ppu, bg_idx, col, bg.h_scroll, bg.v_scroll);
1571                eff = (h, v);
1572                h_from_opt = ho;
1573            }
1574        }
1575        // Hi-res H offset (ares `background.cpp:49,66`): the base scroll
1576        // doubles, but an OPT override does NOT. Non-OPT → `hscroll << 1`;
1577        // OPT-active → `(opt & ~7) + ((hscroll << 1) & 7)`.
1578        let h2 = if h_from_opt {
1579            (eff.0 & 0x3F8).wrapping_add((bg.h_scroll << 1) & 0x07)
1580        } else {
1581            eff.0 << 1
1582        };
1583        let src_y = mosaic_y.wrapping_add(eff.1);
1584        let cx = (mosaic_x << 1).wrapping_add(h2);
1585        below[x as usize] = sample_bg_pixel(ppu, &g, cx, src_y, &mut cache);
1586        above[x as usize] = sample_bg_pixel(ppu, &g, cx.wrapping_add(1), src_y, &mut cache);
1587    }
1588    (above, below)
1589}

1591/// 9-bit X of a sprite (luna's signed X folded into the hardware
1592/// 0..511 range).
1593const fn sprite_x9(sp: &SpriteEntry) -> i32 {
1594    sp.x as i32 & 0x1FF
1595}

1597/// Is the sprite on scanline `y`? The row-in-sprite is taken mod 256
1598/// (vertical wrap) and the sprite must not sit fully in the right
1599/// off-screen dead zone (ares `object.cpp:51-55`). In interlace the
1600/// sprite's *screen* extent is halved (`height >> io.interlace`,
1601/// `object.cpp:53`) — each screen line spans two logical sprite rows.
1602fn sprite_on_line(sp: &SpriteEntry, y: u16, interlace: bool) -> bool {
1603    let x9 = sprite_x9(sp);
1604    if x9 > 256 && x9 + i32::from(sp.w) - 1 < 512 {
1605        return false;
1606    }
1607    let row = y.wrapping_sub(sp.y as u16) & 0xFF;
1608    let screen_h = if interlace { sp.h >> 1 } else { sp.h };
1609    usize::from(row) < screen_h as usize
1610}

1612/// Count of the sprite's 8-px tiles that are not in the off-screen dead
1613/// zone — the unit charged against the 34-tile/line budget (ares
1614/// `object.cpp:133-136`).
1615fn sprite_onscreen_tiles(sp: &SpriteEntry) -> usize {
1616    let tiles = i32::from(sp.w) / 8;
1617    (0..tiles)
1618        .filter(|&tx| {
1619            let sx9 = (i32::from(sp.x) + tx * 8) & 0x1FF;
1620            !(sx9 >= 256 && sx9 + 7 < 512)
1621        })
1622        .count()
1623}

1625/// Result of per-line sprite evaluation. Shared (pre-computed once per
1626/// scanline) between the overflow-flag read and the flush render so the
1627/// 128-entry decode + evaluation isn't repeated (PERF-2).
1628pub(crate) struct SpriteEval {
1629    /// Surviving sprite indices in *fetch* order (back-most first), so
1630    /// the renderer draws them back-to-front and the front sprite wins.
1631    survivors: [u8; 32],
1632    survivor_count: usize,
1633    /// More than 32 sprites on this line ($213E.6).
1634    pub(crate) range_over: bool,
1635    /// More than 34 sprite tiles on this line ($213E.7).
1636    pub(crate) time_over: bool,
1637}
1638
1639/// Evaluate one scanline's sprites: the 32-sprite range limit and the
1640/// 34-tile time limit, starting from `firstSprite` (OAM priority
1641/// rotation). ares `object.cpp:35-49,91-161`, Mesen2
1642/// `SnesPpu.cpp:595-625,693-741`.
1643pub(crate) fn evaluate_sprite_line(ppu: &Ppu, sprites: &[SpriteEntry; 128], y: u16) -> SpriteEval {
1644    let first = ppu.oam.first_sprite() as usize;
1645    // OBJ interlace = SETINI bit 1 (separate from BG interlace, bit 0).
1646    let interlace = ppu.setini & 0x02 != 0;
1647    // Pass 1: first 32 on-line sprites, in evaluation order from
1648    // firstSprite (= priority order, items[0] front-most).
1649    let mut items = [0u8; 32];
1650    let mut count = 0usize;
1651    let mut range_over = false;
1652    for k in 0..128usize {
1653        let i = (first + k) & 0x7F;
1654        if !sprite_on_line(&sprites[i], y, interlace) {
1655            continue;
1656        }
1657        if count < 32 {
1658            items[count] = i as u8;
1659            count += 1;
1660        } else {
1661            range_over = true;
1662            break;
1663        }
1664    }
1665    // Time over: total on-screen tiles of the kept sprites > 34.
1666    let total_tiles: usize = items[..count]
1667        .iter()
1668        .map(|&it| sprite_onscreen_tiles(&sprites[it as usize]))
1669        .sum();
1670    let time_over = total_tiles > 34;
1671    // Pass 2: fetch in reverse (back-most first); keep sprites while the
1672    // cumulative tile count stays within 34. The front sprites that
1673    // overflow the budget are dropped — the SNES "tile over" quirk.
1674    let mut survivors = [0u8; 32];
1675    let mut sc = 0usize;
1676    let mut budget = 0usize;
1677    for idx in (0..count).rev() {
1678        let it = items[idx];
1679        let tiles = sprite_onscreen_tiles(&sprites[it as usize]);
1680        if budget + tiles > 34 {
1681            break;
1682        }
1683        budget += tiles;
1684        survivors[sc] = it;
1685        sc += 1;
1686    }
1687    SpriteEval {
1688        survivors,
1689        survivor_count: sc,
1690        range_over,
1691        time_over,
1692    }
1693}

1726/// Core of [`render_sprites_scanline_indexed_with`] over a pre-decoded
1727/// sprite set + line evaluation, so the per-scanline decode is shared
1728/// across the overflow-flag read and the (possibly doubled, in
1729/// interlace) flush render instead of repeated 2-4× (PERF-2).
1730fn render_sprites_scanline_indexed_from(
1731    ppu: &Ppu,
1732    y: u16,
1733    opts: RenderOptions,
1734    sprites: &[SpriteEntry; 128],
1735    eval: &SpriteEval,
1736) -> IndexedScanline {
1737    let mut out: IndexedScanline = [None; 256];
1738    if ppu.inidisp & 0x80 != 0 && !opts.bypass_forced_blank {
1739        return out;
1740    }
1741    // OBJ interlace (Phase D) = SETINI **bit 1** (not bit 0, which is BG
1742    // interlace). When on, the sprite occupies half the screen lines and a
1743    // screen row samples logical sprite row `screen_row*2 + field` (ares
1744    // object.cpp:109,121-122). Each field is rendered in its own blend pass
1745    // (Phase C), so `Ppu::field` selects the even/odd logical row here.
1746    let interlace = ppu.setini & 0x02 != 0;
1747    let field = usize::from(ppu.field);
1748    // Draw survivors in fetch order: back-most first, front-most last,
1749    // so the front sprite overwrites and wins the pixel.
1750    for s in 0..eval.survivor_count {
1751        let sp = &sprites[eval.survivors[s] as usize];
1752        let row_in_sprite = y.wrapping_sub(sp.y as u16) & 0xFF;
1753        for col in 0..sp.w {
1754            let screen_x = sp.x + col as i16;
1755            if !(0..256).contains(&screen_x) {
1756                continue;
1757            }
1758            let mut sc = col as usize;
1759            let mut sr = row_in_sprite as usize;
1760            if interlace {
1761                sr <<= 1; // logical (even) sprite row — ares object.cpp:109
1762            }
1763            if sp.h_flip {
1764                sc = (sp.w - 1) as usize - sc;
1765            }
1766            if sp.v_flip {
1767                // Vertical flip. For square sprites this is the plain
1768                // `h-1-row`. For the rectangular (taller-than-wide)
1769                // sizes it's the *buggy* hardware flip: the top and
1770                // bottom width-sized halves mirror separately and don't
1771                // swap (ares object.cpp:111-119, Mesen2
1772                // SnesPpu.cpp:716-732 — `pos = 3*w-1-row`).
1773                let w = sp.w as usize;
1774                let h = sp.h as usize;
1775                sr = if w == h {
1776                    h - 1 - sr
1777                } else if sr < w {
1778                    w - 1 - sr
1779                } else {
1780                    3 * w - 1 - sr
1781                };
1782            }
1783            if interlace {
1784                // Pick the field's interleaved logical row: +field when not
1785                // flipped, -field when flipped (ares object.cpp:121-122),
1786                // masked to 8 bits like ares' `y &= 255`.
1787                sr = if sp.v_flip {
1788                    sr.wrapping_sub(field)
1789                } else {
1790                    sr + field
1791                } & 0xFF;
1792            }
1793            let tile_x = (sc / 8) as u16;
1794            let tile_y = (sr / 8) as u16;
1795            let pix_x = sc % 8;
1796            let pix_y = sr % 8;
1797            // Component tile within the 16×16 name page: the low and
1798            // high nibbles of the OAM character byte wrap independently
1799            // and the page (nameselect, bit 8) is held fixed — ares
1800            // object.cpp:130-146, Mesen2 SnesPpu.cpp:735-741. A linear
1801            // add would carry across rows / flip the page at page edges.
1802            let char_byte = sp.tile & 0xFF;
1803            let name_select = sp.tile & 0x100;
1804            let col_nib = ((char_byte & 0x0F) + tile_x) & 0x0F;
1805            let row_nib = ((char_byte >> 4) + tile_y) & 0x0F;
1806            let tile_id = name_select | (row_nib << 4) | col_nib;
1807            let tile_off = sprite_tile_byte_offset(ppu.obsel, tile_id);
1808            let idx = decode_tile_pixel(ppu, tile_off, pix_y, pix_x, 4);
1809            if idx == 0 {
1810                continue;
1811            }
1812            let cgram_idx = 128u16 + (sp.palette as u16) * 16 + (idx as u16);
1813            out[screen_x as usize] = Some((cgram_idx as u8, sp.priority));
1814        }
1815    }
1816    out
1817}

1834/// Bits-per-pixel for any BG in any mode (cf. [`bg1_bpp`]).
1835#[must_use]
1836pub const fn bg_bpp(bgmode: u8, bg_idx: usize) -> u8 {
1837    let m = bgmode & 0x07;
1838    match (m, bg_idx) {
1839        (0, _) => 2,
1840        (1, 0 | 1) => 4,
1841        (1, 2) => 2,
1842        (2, 0 | 1) => 4,
1843        (3, 0) => 8,
1844        (3, 1) => 4,
1845        (4, 0) => 8,
1846        (4, 1) => 2,
1847        (5, 0) => 4,
1848        (5, 1) => 2,
1849        (6, 0) => 4,
1850        (7, 0) => 8,
1851        _ => 0, // BG disabled in this mode
1852    }
1853}

1855/// Render one scanline for the requested BG layer.
1856#[must_use]
1857pub fn render_bg_scanline_with(ppu: &Ppu, bg_idx: usize, y: u16, opts: RenderOptions) -> Scanline {
1858    let mut out = [[0u8; 3]; 256];
1859    if ppu.inidisp & 0x80 != 0 && !opts.bypass_forced_blank {
1860        return out;
1861    }
1862    let brightness = if opts.bypass_forced_blank {
1863        0x0F
1864    } else {
1865        ppu.inidisp & 0x0F
1866    };
1867    let bpp = bg_bpp(ppu.bgmode, bg_idx);
1868    if bpp == 0 {
1869        // BG disabled in this mode → fill with backdrop.
1870        let backdrop = decode_palette(ppu, 0, brightness);
1871        out.fill(backdrop);
1872        return out;
1873    }
1874    let bytes_per_tile = match bpp {
1875        2 => 16,
1876        4 => 32,
1877        8 => 64,
1878        _ => 16,
1879    };
1880    let bg = bg_state(ppu, bg_idx);
1881    let tilemap_base = (bg.tilemap_addr_words as usize) << 1;
1882    let char_base = (bg.char_addr_words as usize) << 1;
1883    let backdrop = decode_palette(ppu, 0, brightness);
1884
1885    // SC bits 0-1 from BG*SC: 0 = 32×32, 1 = 64×32 (extra screen to
1886    // the right), 2 = 32×64 (extra screen below), 3 = 64×64 (extra
1887    // screens right + below + diagonal). Each "extra screen" is a
1888    // full 32×32 sub-tilemap stored at `base + N*0x800` bytes.
1889    let (cols, rows) = match bg.tilemap_size & 0x03 {
1890        0 => (32u16, 32u16),
1891        1 => (64u16, 32u16),
1892        2 => (32u16, 64u16),
1893        3 => (64u16, 64u16),
1894        _ => (32u16, 32u16),
1895    };
1896
1897    for x in 0..256u16 {
1898        let src_x = x.wrapping_add(bg.h_scroll);
1899        let src_y = y.wrapping_add(bg.v_scroll);
1900        let tile_col_full = (src_x / 8) & (cols - 1);
1901        let tile_row_full = (src_y / 8) & (rows - 1);
1902        // Within which 32×32 sub-screen does this (col, row) live?
1903        let sub_x = (tile_col_full >> 5) as usize; // 0 or 1
1904        let sub_y = (tile_row_full >> 5) as usize; // 0 or 1
1905        let sub_index = match bg.tilemap_size & 0x03 {
1906            0 => 0,
1907            1 => sub_x,             // 64x32: right sub-screen offset
1908            2 => sub_y,             // 32x64: bottom sub-screen offset
1909            3 => sub_y * 2 + sub_x, // 64x64: TL/TR/BL/BR
1910            _ => 0,
1911        };
1912        let tile_col = tile_col_full & 0x1F;
1913        let tile_row = tile_row_full & 0x1F;
1914        let entry_off =
1915            tilemap_base + sub_index * 0x0800 + ((tile_row * 32 + tile_col) as usize) * 2;
1916        let entry_lo = ppu.vram.peek(entry_off as u16);
1917        let entry_hi = ppu.vram.peek(entry_off.wrapping_add(1) as u16);
1918        let entry = u16::from(entry_lo) | (u16::from(entry_hi) << 8);
1919        let tile_num = entry & 0x03FF;
1920        let palette_off = ((entry >> 10) & 0x07) as u8;
1921        let h_flip = entry & 0x4000 != 0;
1922        let v_flip = entry & 0x8000 != 0;
1923        let mut row_in_tile = (src_y & 7) as usize;
1924        let mut col_in_tile = (src_x & 7) as usize;
1925        if v_flip {
1926            row_in_tile = 7 - row_in_tile;
1927        }
1928        if h_flip {
1929            col_in_tile = 7 - col_in_tile;
1930        }
1931        let tile_base = char_base + (tile_num as usize) * bytes_per_tile;
1932        let idx = decode_tile_pixel(ppu, tile_base, row_in_tile, col_in_tile, bpp);
1933
1934        out[x as usize] = if idx == 0 {
1935            backdrop
1936        } else {
1937            // Mode 0: per-BG palette region (id << 5). See the
1938            // matching note in render_bg_scanline_indexed_with.
1939            let mode0_palette_base = if ppu.bgmode & 0x07 == 0 {
1940                (bg_idx as u8) << 5
1941            } else {
1942                0
1943            };
1944            let cgram_idx = match bpp {
1945                2 => mode0_palette_base + palette_off * 4 + idx,
1946                4 => palette_off.wrapping_mul(16).wrapping_add(idx),
1947                _ => idx,
1948            };
1949            decode_palette(ppu, cgram_idx, brightness)
1950        };
1951    }
1952    out
1953}

2012/// Render BG `bg_idx`'s full tilemap as an RGBA image (debug view). In
2013/// Mode 7 the `bg_idx` is ignored and the 128×128 8bpp field is rendered.
2014/// A disabled layer (bpp 0 in the current mode) yields a 1×1 black image.
2015///
2016/// NOTE: 8×8 tiles only — 16×16 big-tile mode (BGMODE bits 4-7) is
2017/// rendered as its top-left 8×8 quadrants for now.
2018#[must_use]
2019pub fn render_bg_tilemap(ppu: &Ppu, bg_idx: usize) -> TilemapImage {
2020    if ppu.bgmode & 0x07 == 7 {
2021        return render_mode7_tilemap(ppu);
2022    }
2023    let Some(g) = bg_geom(ppu, bg_idx) else {
2024        return TilemapImage::empty();
2025    };

2027    let height = u32::from(g.rows) * 8;
2028    let mut rgba = vec![0u8; (width * height * 4) as usize];
2029    for tile_row in 0..g.rows {
2030        for tile_col in 0..g.cols {

2233/// Render all 128 OAM sprites as a sheet: one 64×64 cell per sprite (16
2234/// columns × 8 rows), each sprite decoded at its native size from the
2235/// OBJ CHR area using its OBJ palette (`128 + palette*16`). Pixels with
2236/// colour index 0 — and unused cell area — are transparent (alpha 0), so
2237/// the PNG carries true sprite cut-outs. Flips are applied.
2238#[must_use]
2239pub fn render_obj_sheet(ppu: &Ppu) -> TilemapImage {
2240    const CELL: u32 = 64;
2241    const COLS: u32 = 16;
2242    let rows = 128 / COLS;
2243    let width = COLS * CELL;
2244    let height = rows * CELL;

2250        let tiles_h = u32::from(s.h) / 8;
2251        for ty in 0..tiles_h {

2253            for tx in 0..tiles_w {
2254                let src_tx = if s.h_flip { tiles_w - 1 - tx } else { tx };
2255                // OBJ tiles wrap within the 16×16-tile (256-entry) page;
2256                // bit 8 (the page select) is preserved.
2257                let lo = (u32::from(s.tile) & 0xFF)
2258                    .wrapping_add(src_tx)
2259                    .wrapping_add(src_ty * 16)
2260                    & 0xFF;

2262                let tile_base = sprite_tile_byte_offset(ppu.obsel, tile_id);
2263                for py in 0..8u32 {
2264                    let row_in_tile = if s.v_flip { 7 - py } else { py } as usize;

2268                        let idx = decoded[col];
2269                        if idx == 0 {
2270                            continue; // transparent
2271                        }
2272                        let cgram_idx = 128u8
2273                            .wrapping_add(s.palette.wrapping_mul(16))
2274                            .wrapping_add(idx);
2275                        let [r, g, b] = bgr555_to_rgb888(ppu.cgram.color(cgram_idx));
2276                        let x = cx + tx * 8 + px;
2277                        let y = cy + ty * 8 + py;
2278                        let o = ((y * width + x) * 4) as usize;
2279                        rgba[o] = r;
2280                        rgba[o + 1] = g;
2281                        rgba[o + 2] = b;
2282                        rgba[o + 3] = 255;
2283                    }
2284                }
2285            }
2286        }
2287    }
2288    TilemapImage {
2289        width,
2290        height,
2291        rgba,
2292        cols: COLS as u16,
2293        rows: rows as u16,
2294        bpp: 4,
2295        is_mode7: false,
2296        view_x: 0,
2297        view_y: 0,

2299        view_h: 0,
