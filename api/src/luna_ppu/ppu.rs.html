1//! [`Ppu`] register dispatch.
2//!
3//! The PPU is memory-mapped at `$00-$3F:$2100-$213F` (and the mirror in
4//! `$80-$BF`). This module exposes [`Ppu::write`] / [`Ppu::read`]
5//! taking the **low 6 bits of the offset** (`0x00`-`0x3F`) — the bus
6//! is responsible for the bank/region routing.

183/// The SNES Picture Processing Unit.
184///
185/// P1.1 scope: the data-flow plumbing — VRAM, CGRAM, OAM and the
186/// minimum register subset to upload data to them. The rendering side
187/// (modes, scroll, sprites on screen) lands in P1.4+.
188#[derive(serde::Serialize, serde::Deserialize)]
189pub struct Ppu {
190    /// 64 KB tile and tilemap memory.
191    pub vram: Vram,
192    /// 512 B palette memory.
193    pub cgram: Cgram,
194    /// 544 B object attribute memory.
195    pub oam: Oam,
196
197    // ---------- Visual registers (stored but not yet rendered) ----------
198    /// `$2100` INIDISP — bit 7 = forced blank, bits 0-3 = brightness.
199    pub inidisp: u8,
200    /// `$2101` OBSEL — sprite OAM size & character base.
201    pub obsel: u8,
202    /// `$2105` BGMODE — mode (bits 0-2) + BG3 priority + tile sizes.
203    pub bgmode: u8,
204    /// `$2106` MOSAIC.
205    pub mosaic: u8,
206    /// `$2130` CGWSEL — bit 7:6 force-main-black region, bit 5:4
207    /// math-enable region (both reference the colour-math window),
208    /// bit 1 sub-BG/OBJ enable, bit 0 direct-colour mode for 8bpp.
209    pub cgwsel: u8,
210    /// `$2131` CGADSUB — bit 7 add/subtract, bit 6 half-color, bit 5
211    /// backdrop, bit 4 OBJ palettes 4-7, bits 3:0 BG4..BG1 enables.
212    pub cgadsub: u8,
213    /// `$2132` COLDATA: red channel of the fixed sub-screen colour
214    /// (5 bits, 0-31). Updated by writes whose bit 5 is set.
215    pub coldata_r: u8,
216    /// `$2132` COLDATA: green channel (5 bits). Bit 6 selects.
217    pub coldata_g: u8,
218    /// `$2132` COLDATA: blue channel (5 bits). Bit 7 selects.
219    pub coldata_b: u8,
220    /// `$2123-$2125` — per-region window enable / invert bits.
221    /// Layout, per layer (BG1, BG2, BG3, BG4, OBJ, math):
222    ///   bit 0 = window-1 invert (1 = use "outside" semantics)
223    ///   bit 1 = window-1 enable
224    ///   bit 2 = window-2 invert
225    ///   bit 3 = window-2 enable
226    /// `w12sel` packs BG1 (low nibble) + BG2 (high nibble),
227    /// `w34sel` packs BG3 + BG4, `wobjsel` packs OBJ + math.
228    pub w12sel: u8,
229    /// `$2124` W34SEL — same control word for BG3 (low nibble) and
230    /// BG4 (high nibble). See [`Self::w12sel`] for bit semantics.
231    pub w34sel: u8,
232    /// `$2125` WOBJSEL — OBJ (low nibble) and the dedicated colour-
233    /// math window (high nibble). Same per-nibble layout.
234    pub wobjsel: u8,
235    /// `$2126` WH0 — window 1 left X (inclusive).
236    pub wh0: u8,
237    /// `$2127` WH1 — window 1 right X (inclusive).
238    pub wh1: u8,
239    /// `$2128` WH2 — window 2 left X.
240    pub wh2: u8,
241    /// `$2129` WH3 — window 2 right X.
242    pub wh3: u8,
243    /// `$212A` WBGLOG — 2-bit window-combine logic per BG.
244    /// Layout: bits 0-1 = BG1, 2-3 = BG2, 4-5 = BG3, 6-7 = BG4.
245    /// Values: 0 = OR, 1 = AND, 2 = XOR, 3 = XNOR.
246    pub wbglog: u8,
247    /// `$212B` WOBJLOG — bits 0-1 = OBJ logic, bits 2-3 = math.
248    pub wobjlog: u8,
249    /// `$212C` TM — main-screen layer enable mask
250    /// (bit 0..4 = BG1..BG4, OBJ).
251    pub tm: u8,
252    /// `$212D` TS — sub-screen layer enable mask.
253    pub ts: u8,
254    /// `$212E` TMW — main-screen window mask
255    /// (1 bit = disable that layer inside the combined window).
256    pub tmw: u8,
257    /// `$212F` TSW — sub-screen window mask.
258    pub tsw: u8,
259    /// BG1-4 derived state ($2107-$2114).
260    pub bg: [BgState; 4],
261    /// `$211A` M7SEL: bit 0 = H-flip, bit 1 = V-flip, bits 7:6 =
262    /// screen-over mode (00/01 wrap, 10 transparent, 11 use-tile-0).
263    /// (ares io.cpp:411-414 — NOT the bit 7/6/1:0 layout some docs imply.)
264    pub m7sel: u8,
265    /// Mode-7 matrix element A — signed 8.8, scales the horizontal
266    /// component of the projected X axis.
267    pub m7a: i16,
268    /// Mode-7 matrix element B — signed 8.8, scales the vertical
269    /// component of the projected X axis. Writes also retrigger the
270    /// hardware multiplier (see [`Self::mpy_result`]).
271    pub m7b: i16,
272    /// Mode-7 matrix element C — horizontal component of Y axis.
273    pub m7c: i16,
274    /// Mode-7 matrix element D — vertical component of Y axis.
275    pub m7d: i16,
276    /// Mode-7 centre X (signed 13-bit, sign-extended to i16).
277    pub m7x: i16,
278    /// Mode-7 centre Y.
279    pub m7y: i16,
280    /// Mode-7 horizontal scroll `M7HOFS` (`$210D`), signed 13-bit,
281    /// sign-extended to i16. Distinct from `bg[0].h_scroll`: `$210D`
282    /// feeds *both* the 10-bit BG1 scroll and this 13-bit Mode-7
283    /// scroll, which uses the shared Mode-7 latch (ares io.cpp:308).
284    pub m7_hofs: i16,
285    /// Mode-7 vertical scroll `M7VOFS` (`$210E`), signed 13-bit.
286    pub m7_vofs: i16,
287    /// `$2133` SETINI — bit 7: external sync, bit 6: EXTBG (Mode-7
288    /// BG2 overlay), bit 5: hi-res 512×448, bit 4: overscan,
289    /// bit 3: pseudo-512, bit 2: V-mosaic disable, bit 1: interlace.
290    pub setini: u8,
291    /// `$213E` STAT77 — read-side latch for the PPU1 status byte.
292    /// Bit 7 toggles per field in interlace mode; bits 0-3 = chip ID
293    /// (we return 1 — model 5C77).
294    pub stat77: u8,
295    /// OBJ range-over: more than 32 sprites evaluated on some scanline
296    /// this frame. Exposed at `$213E` bit 6; cleared at frame start
297    /// (ares `object.cpp:11-14`).
298    pub obj_range_over: bool,
299    /// OBJ time-over: more than 34 sprite tiles on some scanline this
300    /// frame. Exposed at `$213E` bit 7.
301    pub obj_time_over: bool,
302    /// `$213F` STAT78 — read-side latch for the PPU2 status byte.
303    /// Bit 7 = interlace odd-field flag, bit 4 = region (1 = PAL,
304    /// 0 = NTSC), bits 0-3 = chip revision (= 2).
305    pub stat78: u8,
306    /// Latched H counter (9-bit) from the last $2137/$4201-bit7
307    /// trigger. Exposed at OPHCT.
308    pub ophct: u16,
309    /// Latched V counter (9-bit). Exposed at OPVCT.
310    pub opvct: u16,
311    /// "High-byte pending" flag for OPHCT — first read returns the
312    /// low byte (and sets this), second returns the high byte (and
313    /// clears it).
314    pub ophct_hi_pending: bool,
315    /// Same for OPVCT.
316    pub opvct_hi_pending: bool,
317    /// "Latch hit" bit: set when the H/V counters were latched
318    /// since the last STAT78 read. Exposed at STAT78 bit 6.
319    pub external_latch_hit: bool,
320    /// Interlace field parity, exposed at STAT78 ($213F) bit 7. Toggles
321    /// every frame at the V-counter wrap (ares `counter/inline.hpp:32`
322    /// `time.field ^= 1`) — unconditionally, even in progressive mode.
323    /// Interlace Phase A: state + flag only; vertical doubling is later.
324    pub field: bool,
325    /// `$2134-$2136 MPYL/M/H` — 24-bit hardware multiplier result.
326    /// Updated whenever M7A or M7B's high byte is written:
327    /// `M7A (signed 16) × M7B_high (signed 8) → 24-bit signed`.
328    pub mpy_result: i32,
329    /// Shared latch for the Mode-7 write-twice registers
330    /// (\$211B-\$2120). Each write `value` shifts the latch and
331    /// stores the 16-bit pair `(latch_low, value_high)`.
332    pub m7_latch: u8,
333    /// Latch for the BG H/V scroll write-twice protocol.
334    /// `$210D-$2114` are written low byte first, then high byte (bits
335    /// 0-1 land as the top 2 bits of the 10-bit scroll). The PPU keeps
336    /// TWO shared latches across all eight BG-scroll registers (ares
337    /// `latch.bgofsPPU1`/`bgofsPPU2`): every H write updates both, every V
338    /// write updates only PPU1. The H formula reads bits 3-9 from PPU1 and
339    /// bits 0-2 from PPU2 — the genuine write-twice cross-latch quirk.
340    bgofs_ppu1: u8,
341    bgofs_ppu2: u8,
342
343    // ---------- Open-bus tracking ----------
344    /// Last value seen on the PPU data bus — returned for reads of
345    /// write-only registers.
346    pub open_bus: u8,
347
348    // ---------- Diagnostic counters (debug-only, not on hot path) ----------
349    /// How many times `$2100` (INIDISP) has been written since reset.
350    /// Used by the GUI's Stubs panel to detect "the game never touched
351    /// INIDISP again after init" (= NMI handler not running).
352    pub inidisp_write_count: u64,
353
354    /// Persistent framebuffer, written one scanline at a time by the
355    /// scheduler via [`Ppu::render_current_scanline`]. Consumers
356    /// (`luna-api::Emulator::render_frame_png`, the GUI, etc.) read
357    /// this directly for zero-cost frame access — they no longer need
358    /// to re-render the whole frame on every request.
359    ///
360    /// Tests that hand-build a `Ppu` and call
361    /// [`render_frame_with`](crate::render_frame_with) bypass this
362    /// buffer entirely and get a freshly-computed `Vec`.
363    ///
364    /// Serialized in the save-state: it is the last fully-rendered frame,
365    /// part of observable state, so restoring rewinds the displayed image
366    /// to exactly the save point (front-ends hold the last non-blank frame
367    /// between renders, so this must round-trip).
368    pub framebuffer: Vec<[u8; 3]>,
369
370    /// How far the current scanline has been rendered into the
371    /// framebuffer. Phase 2 of gap G6: a `$21xx` write that lands
372    /// mid-scanline calls [`Ppu::flush_partial_scanline`] to commit
373    /// pixels `last_flushed_dot..current_dot` with the OLD state
374    /// BEFORE the write takes effect. Reset to 0 by
375    /// [`Ppu::scanline_reset`] at every scanline boundary.
376    pub last_flushed_dot: u16,
377
378    /// Per-line OBJ evaluation cache `(y, (sprites, eval))`, decoded once
379    /// per scanline and reused across all of that line's flush segments
380    /// (PERF-3). Without it, every mid-line `$21xx` write that triggers a
381    /// partial flush re-decodes all 128 OAM entries. Caching once per line
382    /// also matches hardware (OBJ is evaluated once, during the prior
383    /// line's `HBlank`) rather than re-evaluating per segment. Transient —
384    /// cleared by [`Ppu::scanline_reset`]; not part of the save-state.
385    #[serde(skip)]
386    line_sprites: Option<LineObjCache>,
387
388    /// `true` when the screen is currently being scanned out — i.e.
389    /// not in forced blank (`INIDISP.7 == 0`) AND the current scanline
390    /// is visible (`< vblank_start_line`). Updated by the bus layer
391    /// before every PPU register write. Gap G7: writes to VRAM
392    /// (`$2118/$2119`), CGRAM (`$2122`), and OAM (`$2104`) during
393    /// active display silently DROP the data byte on real hardware —
394    /// the address/latch counter still advances. ares ppu_io.cpp:19-45,
395    /// Mesen2 SnesPpu.cpp:2046-2057 (`CanAccessVram`).
396    pub active_display: bool,
397
398    /// Accumulator: set `true` whenever a *visible* scanline is rendered
399    /// with forced-blank OFF during the current frame. Snapshotted into
400    /// [`Ppu::frame_visible_content`] (and cleared) at each frame wrap by
401    /// [`Ppu::latch_frame_content`].
402    pub frame_visible_content_accum: bool,
403
404    /// Whether the just-completed frame showed any visible (non-forced-
405    /// blank) content. Distinct from the *instantaneous* INIDISP bit 7:
406    /// a game that re-blanks during `VBlank` every frame (e.g. Super FX
407    /// titles preparing the next double-buffer) reads forced-blank at the
408    /// frame boundary yet *did* display content during active scanout.
409    /// Front-ends gate "publish this frame vs hold the last good one" on
410    /// this, not on the instantaneous flag, so such frames aren't dropped.
411    pub frame_visible_content: bool,
412}

507    /// Render the current visible scanline `y` into the persistent
508    /// framebuffer, using the live PPU register state. Called by the
509    /// scheduler at the end of every visible scanline (gap G6 Phase 1).
510    ///
511    /// Equivalent to a full-line flush: renders pixels
512    /// `last_flushed_dot..FRAME_W` and resets the partial-flush cursor.
513    /// Out-of-range `y` (≥ `FRAME_H`) is a no-op.
514    pub fn render_current_scanline(&mut self, y: u16, opts: RenderOptions) {
515        // OBJ range/time-over flags accumulate over the frame and clear
516        // at its start (ares object.cpp:11-14). Evaluate per line and OR
517        // in; the renderer applies the matching 32/34 drop.
518        if y == 0 {
519            self.obj_range_over = false;
520            self.obj_time_over = false;
521        }
522        // Evaluate this line's sprites ONCE (cached + reused by every
523        // segment, PERF-3) and OR in the overflow flags. Blank lines
524        // contribute no flags (matching the old guard) and need no decode.
525        if self.inidisp & 0x80 == 0 {
526            self.ensure_line_sprites(y);
527            let (range_over, time_over) = self
528                .line_sprites
529                .as_ref()
530                .map_or((false, false), |(_, b)| (b.1.range_over, b.1.time_over));
531            self.obj_range_over |= range_over;
532            self.obj_time_over |= time_over;
533        }
534        self.flush_partial_scanline_inner(y, FRAME_W as u16, opts);
535        self.scanline_reset();
536    }

538    /// Decode + evaluate this scanline's OBJ once and cache it on
539    /// `line_sprites`, keyed by `y`. Re-decodes only when the cached line
540    /// differs (or the cache was cleared at the last boundary). The
541    /// overflow flags live on the cached [`SpriteEval`]; this does NOT
542    /// touch `obj_range_over` / `obj_time_over` (the caller ORs those once
543    /// per line).
544    fn ensure_line_sprites(&mut self, y: u16) {
545        if self.line_sprites.as_ref().is_some_and(|(cy, _)| *cy == y) {
546            return;
547        }
548        let sprites = decode_all_sprites(self);
549        let eval = evaluate_sprite_line(self, &sprites, y);
550        self.line_sprites = Some((y, Box::new((sprites, eval))));
551    }

553    /// Render pixels `last_flushed_dot..end_x` of scanline `y` into the
554    /// persistent framebuffer using the current PPU register state, and
555    /// advance `last_flushed_dot` to `end_x`. Phase 2 of gap G6 — the
556    /// bus layer calls this BEFORE applying a `$21xx` write so the
557    /// in-progress scanline gets the pre-write pixels committed.
558    ///
559    /// Out-of-range `y` (≥ `FRAME_H`) or `end_x <= last_flushed_dot`
560    /// is a no-op.
561    pub fn flush_partial_scanline(&mut self, y: u16, end_x: u16, opts: RenderOptions) {
562        self.flush_partial_scanline_inner(y, end_x, opts);
563    }

565    /// Core of [`flush_partial_scanline`]. Reuses the once-per-line OBJ
566    /// decode + evaluation cached on [`Self::line_sprites`] (PERF-3) so the
567    /// (interlace-doubled) render never re-decodes all 128 OAM entries per
568    /// segment. The cache is `take`n into a local for the render so the
569    /// interlace path can still mutate `self.field`, then restored.
570    fn flush_partial_scanline_inner(&mut self, y: u16, end_x: u16, opts: RenderOptions) {
571        let yi = usize::from(y);
572        if yi >= FRAME_H {
573            return;
574        }
575        let start = self.last_flushed_dot.min(FRAME_W as u16);
576        let end = end_x.min(FRAME_W as u16);
577        if start >= end {
578            return;
579        }
580        // Per-line OBJ eval, decoded once and reused (blank lines need
581        // none). `take` the cache so the borrow doesn't pin `self` during
582        // the interlace `self.field` mutations below; restored at the end.
583        let cached = if self.inidisp & 0x80 == 0 {
584            self.ensure_line_sprites(y);
585            self.line_sprites.take()
586        } else {
587            None
588        };
589        let precomp = cached.as_ref().map(|(_, b)| (&b.0, &b.1));
590        // Stack scratch row → partial render → copy into framebuffer.
591        let mut row = [[0u8; 3]; FRAME_W];
592        let si = usize::from(start);
593        let ei = usize::from(end);
594        if self.setini & 0x01 != 0 {
595            // Interlace (Phase C): the 448-line image is collapsed to 224 by
596            // averaging the two fields (logical lines y*2 and y*2+1) — the
597            // vertical analog of the hi-res 512→256 "Option A" downsample.
598            // Render the range once per field parity and blend → stable,
599            // flicker-free output. (`field` still toggles for STAT78 reads;
600            // the rendered frame uses both parities regardless.)
601            let saved_field = self.field;
602            let mut row_odd = [[0u8; 3]; FRAME_W];
603            self.field = false;
604            render_scanline_partial_into_from(self, y, start, end, opts, &mut row, precomp);
605            self.field = true;
606            render_scanline_partial_into_from(self, y, start, end, opts, &mut row_odd, precomp);
607            self.field = saved_field;
608            for x in si..ei {
609                for c in 0..3 {
610                    row[x][c] = u8::midpoint(row[x][c], row_odd[x][c]);
611                }
612            }
613        } else {
614            render_scanline_partial_into_from(self, y, start, end, opts, &mut row, precomp);
615        }
616        let off = yi * FRAME_W;
617        self.framebuffer[off + si..off + ei].copy_from_slice(&row[si..ei]);
618        self.last_flushed_dot = end;
619        // Restore the per-line OBJ cache for the next segment of this line.
620        self.line_sprites = cached;
621    }

625    /// starts from dot 0 and re-evaluates its sprites.
626    pub fn scanline_reset(&mut self) {
627        self.last_flushed_dot = 0;
628        self.line_sprites = None;
629    }

638    /// Read a PPU register. `offset` is the byte offset from `$2100`
639    /// (`0x00..=0x3F`).
640    ///
641    /// Write-only registers return the open-bus value; we model open
642    /// bus minimally by returning the last byte seen on the PPU bus.
643    pub fn read(&mut self, offset: u8) -> u8 {
644        let value = match offset {
645            register::OAMDATAREAD => self.oam.read(),
646            register::VMDATALREAD => self.vram.read_lo(),
647            register::VMDATAHREAD => self.vram.read_hi(),
648            register::CGDATAREAD => self.cgram.read(),
649            // Mode-7 hardware-multiplier result (also driven by
650            // M7A / M7B writes outside Mode 7 — used as a fast
651            // multiplier).
652            register::MPYL => self.mpy_result as u8,
653            register::MPYM => (self.mpy_result >> 8) as u8,
654            register::MPYH => (self.mpy_result >> 16) as u8,
655            register::STAT77 => {
656                self.stat77
657                    | (u8::from(self.obj_time_over) << 7)
658                    | (u8::from(self.obj_range_over) << 6)
659            }
660            register::STAT78 => {
661                // Reading $213F is documented to clear the shared
662                // BG-scroll write-twice latch AND the "latch hit"
663                // status bit as side effects.
664                let v = self.stat78
665                    | (u8::from(self.field) << 7)
666                    | if self.external_latch_hit { 0x40 } else { 0 };
667                self.bgofs_ppu1 = 0;
668                self.bgofs_ppu2 = 0;
669                self.external_latch_hit = false;
670                // Reading $213F also resets the OPHCT/OPVCT byte-read
671                // flip-flop (ares `io.cpp` $213f: `latch.hcounter = 0;
672                // latch.vcounter = 0;`), so the next OPHCT/OPVCT read
673                // returns the LOW byte. Without this the toggle drifts and
674                // every other V-counter read returns the high byte (0 for
675                // lines < 256) — which made Doom's raster IRQ handler read
676                // V≈0, mis-dispatch to its no-ack branch, and re-fire the
677                // H/V IRQ ~200×/frame, freezing the CPU at I=1 and
678                // flickering the letterbox border.
679                self.ophct_hi_pending = false;
680                self.opvct_hi_pending = false;
681                v
682            }
683            register::OPHCT => {
684                // Read low byte first, then high byte (only bit 0
685                // of the high byte is meaningful — H is 9-bit).
686                if self.ophct_hi_pending {
687                    self.ophct_hi_pending = false;
688                    ((self.ophct >> 8) & 0x01) as u8
689                } else {
690                    self.ophct_hi_pending = true;
691                    self.ophct as u8
692                }
693            }
694            register::OPVCT => {
695                if self.opvct_hi_pending {
696                    self.opvct_hi_pending = false;
697                    ((self.opvct >> 8) & 0x01) as u8
698                } else {
699                    self.opvct_hi_pending = true;
700                    self.opvct as u8
701                }
702            }
703            // Everything else: open bus. The renderer's status registers
704            // ($2134-$213F apart from those above) will be implemented
705            // alongside the renderer.
706            _ => self.open_bus,
707        };
708        self.open_bus = value;
709        value
710    }

712    /// Decode `$2107-$210A BGxSC` (tilemap address + size bits).
713    fn set_bg_tilemap(&mut self, bg_idx: usize, value: u8) {
714        // bits 2-7: tilemap base in 1 KB units (i.e. 0x400 byte
715        // increments = 0x200 word increments).
716        let base_words = u16::from(value & 0xFC) << 8;
717        self.bg[bg_idx].tilemap_addr_words = base_words;
718        self.bg[bg_idx].tilemap_size = value & 0x03;
719    }

721    /// `$210D/$210F/$2111/$2113` H-scroll write-twice protocol.
722    fn write_bg_h_scroll(&mut self, bg_idx: usize, value: u8) {
723        // ares ppu/io.cpp:312-314 (BG1HOFS et al.). The H offset uses BOTH
724        // shared scroll latches: bits 3-9 from the *previous* write
725        // (`bgofs_ppu1 & ~7`), bits 0-2 from the write *two before*
726        // (`bgofs_ppu2 & 7`), bits 8-9 from the new high byte. Both latches
727        // then take the new byte. The two-latch cross is the real hardware
728        // write-twice quirk (it only shows when scroll registers interleave;
729        // a single latch mis-scrolled the sub-tile H offset in many games).
730        self.bg[bg_idx].h_scroll = ((u16::from(value) << 8)
731            | (u16::from(self.bgofs_ppu1) & !7)
732            | (u16::from(self.bgofs_ppu2) & 7))
733            & 0x03FF;
734        self.bgofs_ppu1 = value;
735        self.bgofs_ppu2 = value;
736    }
737
738    /// `$210E/$2110/$2112/$2114` V-scroll write-twice protocol.
739    fn write_bg_v_scroll(&mut self, bg_idx: usize, value: u8) {
740        // ares ppu/io.cpp:323-324 (BG1VOFS): V uses the FULL previous-write
741        // latch (no PPU2 cross) and updates only `bgofs_ppu1`.
742        self.bg[bg_idx].v_scroll = ((u16::from(value) << 8) | u16::from(self.bgofs_ppu1)) & 0x03FF;
743        self.bgofs_ppu1 = value;
744    }

746    /// Mode-7 write-twice helper. Returns the new 16-bit value
747    /// composed of `(latch_low, value_high)` and advances the
748    /// shared latch. The Mode-7 latch is *separate* from the BG
749    /// scroll latch — per fullsnes, every M7A-M7D / M7X / M7Y
750    /// write feeds (and reads from) this single 8-bit register.
751    fn write_m7_pair(&mut self, value: u8) -> u16 {
752        let composed = (u16::from(value) << 8) | u16::from(self.m7_latch);
753        self.m7_latch = value;
754        composed
755    }

757    /// `$210D` also writes the 13-bit Mode-7 H scroll `M7HOFS`, which
758    /// uses the *shared Mode-7 latch* — NOT the BG scroll latch
759    /// (ares io.cpp:308-310). The value is `data << 8 | latch`, masked
760    /// to 13 bits and sign-extended.
761    fn write_m7_hofs(&mut self, value: u8) {
762        let raw = self.write_m7_pair(value) & 0x1FFF;
763        self.m7_hofs = ((raw as i16) << 3) >> 3;
764    }

766    /// `$210E` Mode-7 V scroll `M7VOFS` — see [`Self::write_m7_hofs`].
767    fn write_m7_vofs(&mut self, value: u8) {
768        let raw = self.write_m7_pair(value) & 0x1FFF;
769        self.m7_vofs = ((raw as i16) << 3) >> 3;
770    }

772    /// Latch the current PPU H/V counters into OPHCT/OPVCT. Called
773    /// by the `SnesBus` on:
774    ///   * a WRIO (\$4201) write whose bit 7 transitions from 0 to 1
775    ///   * a read of SLHV (\$2137) — also returns open bus
776    ///
777    /// Both paths feed the SAME pair of latched values; the read
778    /// protocol on OPHCT/OPVCT is separately tracked (low-then-high).
779    pub const fn latch_counters(&mut self, h: u16, v: u16) {
780        self.ophct = h & 0x1FF;
781        self.opvct = v & 0x1FF;
782        self.external_latch_hit = true;
783    }

785    /// Re-run the Mode-7 hardware multiplier. Triggered by writes
786    /// to M7A or the high byte of M7B (i.e. the second of the two
787    /// M7B writes). Computes `signed(M7A) × signed(M7B_high) →
788    /// 24-bit signed result`; reads at MPYL/M/H expose it.
789    fn update_mpy(&mut self) {
790        // `M7B`'s upper byte is what the hardware uses; that's
791        // bits 8..15 of our stored i16, i.e. (self.m7b >> 8) as i8.
792        let a = i32::from(self.m7a);
793        let b = i32::from((self.m7b >> 8) as i8);
794        self.mpy_result = a * b;
795    }

797    /// Write a PPU register. `offset` is the byte offset from `$2100`.
798    pub fn write(&mut self, offset: u8, value: u8) {
799        self.open_bus = value;
800        match offset {
801            register::INIDISP => {
802                self.inidisp = value;
803                self.inidisp_write_count = self.inidisp_write_count.saturating_add(1);
804            }
805            register::OBSEL => self.obsel = value,
806            register::OAMADDL => self.oam.set_address_low(value),
807            register::OAMADDH => self.oam.set_address_high(value),
808            register::OAMDATA => self.oam.write_gated(value, !self.active_display),
809            register::BGMODE => self.bgmode = value,
810            register::MOSAIC => self.mosaic = value,
811            register::BG1SC => self.set_bg_tilemap(0, value),
812            register::BG2SC => self.set_bg_tilemap(1, value),
813            register::BG3SC => self.set_bg_tilemap(2, value),
814            register::BG4SC => self.set_bg_tilemap(3, value),
815            register::BG12NBA => {
816                self.bg[0].char_addr_words = u16::from(value & 0x0F) << 12;
817                self.bg[1].char_addr_words = u16::from((value >> 4) & 0x0F) << 12;
818            }
819            register::BG34NBA => {
820                self.bg[2].char_addr_words = u16::from(value & 0x0F) << 12;
821                self.bg[3].char_addr_words = u16::from((value >> 4) & 0x0F) << 12;
822            }
823            register::BG1HOFS => {
824                // $210D feeds both the BG1 (10-bit) and Mode-7 (13-bit)
825                // scrolls, each with its own latch.
826                self.write_bg_h_scroll(0, value);
827                self.write_m7_hofs(value);
828            }
829            register::BG1VOFS => {
830                self.write_bg_v_scroll(0, value);
831                self.write_m7_vofs(value);
832            }
833            register::BG2HOFS => self.write_bg_h_scroll(1, value),
834            register::BG2VOFS => self.write_bg_v_scroll(1, value),
835            register::BG3HOFS => self.write_bg_h_scroll(2, value),
836            register::BG3VOFS => self.write_bg_v_scroll(2, value),
837            register::BG4HOFS => self.write_bg_h_scroll(3, value),
838            register::BG4VOFS => self.write_bg_v_scroll(3, value),
839            register::M7SEL => self.m7sel = value,
840            register::M7A => {
841                self.m7a = self.write_m7_pair(value) as i16;
842                self.update_mpy();
843            }
844            register::M7B => {
845                self.m7b = self.write_m7_pair(value) as i16;
846                self.update_mpy();
847            }
848            register::M7C => self.m7c = self.write_m7_pair(value) as i16,
849            register::M7D => self.m7d = self.write_m7_pair(value) as i16,
850            register::M7X => {
851                // 13-bit signed value, sign-extended for arithmetic.
852                let raw = self.write_m7_pair(value) & 0x1FFF;
853                self.m7x = ((raw as i16) << 3) >> 3;
854            }
855            register::M7Y => {
856                let raw = self.write_m7_pair(value) & 0x1FFF;
857                self.m7y = ((raw as i16) << 3) >> 3;
858            }
859            register::SETINI => self.setini = value,
860            register::VMAIN => self.vram.vmain = VmainSettings::from_byte(value),
861            register::VMADDL => {
862                let hi = (self.vram.address >> 8) as u8;
863                self.vram.set_address(value, hi);
864            }
865            register::VMADDH => {
866                let lo = self.vram.address as u8;
867                self.vram.set_address(lo, value);
868            }
869            register::VMDATAL => self.vram.write_lo_gated(value, !self.active_display),
870            register::VMDATAH => self.vram.write_hi_gated(value, !self.active_display),
871            register::CGADD => self.cgram.set_address(value),
872            // CGRAM is never dropped on hardware — unlike VRAM/OAM, a CGDATA
873            // write during active display always commits (ares io.cpp:55-60;
874            // only the address is latched, which luna doesn't model). Gating
875            // it broke ROMs that set the backdrop mid-frame (ControllerLatency)
876            // and the HiColor CPU-IRQ palette streams.
877            register::CGDATA => self.cgram.write(value),
878            register::W12SEL => self.w12sel = value,
879            register::W34SEL => self.w34sel = value,
880            register::WOBJSEL => self.wobjsel = value,
881            register::WH0 => self.wh0 = value,
882            register::WH1 => self.wh1 = value,
883            register::WH2 => self.wh2 = value,
884            register::WH3 => self.wh3 = value,
885            register::WBGLOG => self.wbglog = value,
886            register::WOBJLOG => self.wobjlog = value,
887            register::TM => self.tm = value,
888            register::TS => self.ts = value,
889            register::TMW => self.tmw = value,
890            register::TSW => self.tsw = value,
891            register::CGWSEL => self.cgwsel = value,
892            register::CGADSUB => self.cgadsub = value,
893            register::COLDATA => {
894                // Multiple channels can be selected per write (the
895                // high 3 bits act as a per-channel write-enable mask),
896                // and writes ACCUMULATE — drivers set R then G then B
897                // with three sequential writes, and reads of any
898                // unaddressed channel are preserved.
899                let intensity = value & 0x1F;
900                if value & 0x20 != 0 {
901                    self.coldata_r = intensity;
902                }
903                if value & 0x40 != 0 {
904                    self.coldata_g = intensity;
905                }
906                if value & 0x80 != 0 {
907                    self.coldata_b = intensity;
908                }
909            }
910            // FALLTHROUGH for unmodelled registers.
911            // Other registers are stored as raw bytes for now (BG
912            // scroll, window state, etc. — wired here in P1.4+).
913            _ => {
914                // Drop silently; we'll wire each register as the
915                // renderer needs it.
916            }
917        }
918    }
