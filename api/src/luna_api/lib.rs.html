1//! Luna's public, transport-agnostic API.
2//!
3//! Every feature the emulator exposes — load a ROM, step the CPU,
4//! peek memory, render a frame, drain audio — lives here as a method
5//! on [`Emulator`]. The CLI, GUI, MCP server and any future
6//! transport (HTTP, WebSocket, FFI) all call through this surface so
7//! the emulator stays drivable from outside the binary that built it.
8//!
9//! ## Stability
10//!
11//! From V1 onward this crate carries strict `SemVer` guarantees: new
12//! methods are additive, breaking changes bump the major version.
13//! Today (P3.3) we're still pre-V1 and the surface is allowed to
14//! churn freely.

45/// Errors surfaced from [`Emulator`] methods.
46#[derive(Debug, Error)]
47pub enum ApiError {
48    /// Caller asked for a ROM-dependent operation before
49    /// [`Emulator::load_rom`] succeeded.
50    #[error("no ROM loaded — call load_rom first")]
51    NoRom,
52    /// Cartridge parsing or layout detection failed.
53    #[error("cartridge: {0}")]
54    Cart(#[from] CartError),
55    /// Cartridge needs a coprocessor luna does not yet emulate
56    /// (S-DD1, SPC7110) — reachable via a forced mapper.
57    #[error("{0}")]
58    UnsupportedMapper(#[from] luna_core::UnsupportedMapper),
59    /// `step` / etc. panicked inside the core.
60    #[error("emulator panicked: {0}")]
61    Panic(String),
62    /// A debug call got an invalid argument (e.g. an unknown register name).
63    #[error("{0}")]
64    BadArg(String),
65    /// Save-state encode/decode failed, or a loaded state did not match
66    /// the running ROM / save-state format version.
67    #[error("save state: {0}")]
68    SaveState(String),
69    /// PNG encoding failed during `render_frame`.
70    #[error("image: {0}")]
71    Image(#[from] image::ImageError),
72    /// Generic I/O.
73    #[error("io: {0}")]
74    Io(#[from] std::io::Error),
75}

566    /// A save state records this so [`Emulator::load_state`] can refuse a
567    /// state produced against a different ROM.
568    rom_hash: u64,
569    /// Event Viewer category/DMA visibility config.
570    event_config: event_viewer::EventViewerConfig,
571    /// The just-completed frame's captured events (Mesen2 `_debugEvents`),
572    /// CPU IO + DMA merged and sorted by `(scanline, cycle)`.
573    last_frame_events: Vec<CapturedEvent>,

575    /// `show_previous_frame` is on.
576    prev_frame_events: Vec<CapturedEvent>,

579/// One raw captured Event Viewer event before category/filter decode — either
580/// a CPU bus access (mem-trace) or a DMA B-bus write (dma-trace). Kept as the
581/// raw source so [`Emulator::event_snapshot`] can re-decode under the live
582/// config (visibility / DMA-channel toggles) without re-running the frame.
583#[derive(Clone, Copy)]
584enum CapturedEvent {
585    /// A CPU bus access or NMI/IRQ marker.
586    Cpu(MemTraceEvent),
587    /// A DMA B-bus write, carrying its source channel.
588    Dma(DmaTraceEvent),
589}
590
591impl CapturedEvent {
592    /// Sort key `(scanline, cycle)` — Mesen2 orders the event list/overlay by
593    /// PPU position. Cycle is the exact H-clock (Mesen2 `GetHClock`).
594    const fn sort_key(&self) -> (u16, u16) {
595        match self {
596            Self::Cpu(e) => (e.line, e.hclock),
597            Self::Dma(e) => (e.line, e.hclock),
598        }
599    }
600
601    /// The position key Mesen2's `FilterEvents` compares against the cursor:
602    /// `(scanline << 16) + cycle`.
603    fn position_key(&self) -> u32 {
604        let (line, cycle) = self.sort_key();
605        (u32::from(line) << 16) | u32::from(cycle)
606    }
607
608    /// Decode under `cfg`, flagging prev-frame membership. `None` when the
609    /// access has no category or is hidden by a category / DMA-channel filter.
610    fn decode(
611        &self,
612        cfg: &event_viewer::EventViewerConfig,
613        is_prev_frame: bool,
614    ) -> Option<EventViewerEvent> {
615        match self {
616            Self::Cpu(e) => event_viewer::decode_event(e, cfg, is_prev_frame),
617            Self::Dma(e) => event_viewer::decode_dma_event(e, cfg, is_prev_frame),
618        }
619    }

668            }
669        }
670        self.load_cartridge(cart)
671    }

714    fn load_cartridge(&mut self, cart: Cartridge) -> Result<RomInfo, ApiError> {
715        // Capture a stable hash of the ROM bytes before the cartridge is
716        // consumed by `try_from_cartridge`; the save-state layer uses it to
717        // refuse states produced against a different ROM.
718        let rom_hash = {
719            use std::hash::{Hash, Hasher};
720            let mut h = std::collections::hash_map::DefaultHasher::new();
721            cart.rom.hash(&mut h);
722            h.finish()
723        };
724        let info = RomInfo {
725            title: cart.header.title.clone(),
726            mapper: format!("{:?}", cart.header.mapper_kind),
727            rom_bytes: cart.rom.len(),
728            header_rom_size_kb: cart.header.rom_size_kb,
729            sram_kb: cart.header.sram_size_kb,
730            region: format!("{:?}", cart.header.region),
731            fast_rom: cart.header.fast_rom,
732            version: cart.header.version,
733            checksum: cart.header.checksum,
734            checksum_complement: cart.header.checksum_complement,
735            checksum_valid: cart.header.checksum_valid(),
736            missing_firmware: if cart.needs_coprocessor_firmware() {
737                cart.required_firmware_filename().map(str::to_string)
738            } else {
739                None
740            },
741        };
742        // Unsupported coprocessor carts surface as a typed
743        // `UnsupportedMapper` error (no longer a panic). The
744        // `catch_unwind` stays as a backstop for any *other* panic
745        // during construction / `reset`, so a malformed ROM can't tear
746        // down the whole transport.
747        let result = std::panic::catch_unwind(std::panic::AssertUnwindSafe(|| {
748            let mut snes = Snes::try_from_cartridge(cart)?;
749            snes.reset();
750            Ok::<_, luna_core::UnsupportedMapper>(snes)
751        }));
752        match result {
753            Ok(Ok(snes)) => {
754                self.snes = Some(snes);
755                self.rom_info = Some(info.clone());
756                self.instructions_executed = 0;
757                self.rom_hash = rom_hash;
758                Ok(info)
759            }
760            Ok(Err(unsupported)) => Err(ApiError::UnsupportedMapper(unsupported)),
761            Err(payload) => Err(ApiError::Panic(panic_message(&payload))),
762        }
763    }

856    /// Step the CPU `count` instructions (or stop early if the CPU
857    /// halts or panics). Returns the number actually executed.
858    pub fn step(&mut self, count: u64) -> Result<u64, ApiError> {
859        let snes = self.snes.as_mut().ok_or(ApiError::NoRom)?;
860        let mut executed = 0u64;
861        let prev_hook = std::panic::take_hook();
862        std::panic::set_hook(Box::new(|_| {}));
863        let result = std::panic::catch_unwind(std::panic::AssertUnwindSafe(|| {
864            while executed < count {
865                if snes.cpu.stopped {
866                    break;
867                }
868                snes.step();
869                executed += 1;
870            }
871            executed
872        }));
873        std::panic::set_hook(prev_hook);
874        match result {
875            Ok(n) => {
876                self.instructions_executed += n;
877                Ok(n)
878            }
879            Err(payload) => {
880                self.instructions_executed += executed;
881                Err(ApiError::Panic(panic_message(&payload)))
882            }
883        }
884    }

886    /// Run instructions until the PPU completes one frame (i.e.
887    /// `frame_count` advances). Returns the number of instructions
888    /// executed. Bounded by `max_steps` as a safety belt.
889    pub fn step_until_frame(&mut self, max_steps: u64) -> Result<u64, ApiError> {
890        let snes = self.snes.as_mut().ok_or(ApiError::NoRom)?;
891        let start_frame = snes.frame_count;
892        let mut executed = 0u64;
893        let prev_hook = std::panic::take_hook();
894        std::panic::set_hook(Box::new(|_| {}));
895        let result = std::panic::catch_unwind(std::panic::AssertUnwindSafe(|| {
896            while executed < max_steps && !snes.cpu.stopped && snes.frame_count == start_frame {
897                snes.step();
898                executed += 1;
899            }
900            executed
901        }));
902        std::panic::set_hook(prev_hook);
903        match result {
904            Ok(n) => {
905                self.instructions_executed += n;
906                Ok(n)
907            }
908            Err(payload) => {
909                self.instructions_executed += executed;
910                Err(ApiError::Panic(panic_message(&payload)))
911            }
912        }
913    }

915    /// Probe CPU liveness from the current state: step up to `max_steps`
916    /// instructions (stopping early if the CPU halts), recording how many
917    /// **distinct** program addresses (`PB:PC`) were executed. A live game
918    /// touches hundreds–thousands of addresses; a hung game spins over a
919    /// handful — so a tiny `distinct_pcs` is the signal a frozen game is in
920    /// an infinite loop (vs. STP, which `cpu_state().stopped` already
921    /// reports). Mutates state (advances the CPU); a diagnostic, not part of
922    /// normal stepping. Panic-safe (a crashing ROM returns `Err`).
923    pub fn loop_probe(&mut self, max_steps: u64) -> Result<LoopProbe, ApiError> {
924        let snes = self.snes.as_mut().ok_or(ApiError::NoRom)?;
925        let mut seen = std::collections::HashSet::new();
926        let mut executed = 0u64;
927        let prev_hook = std::panic::take_hook();
928        std::panic::set_hook(Box::new(|_| {}));
929        let result = std::panic::catch_unwind(std::panic::AssertUnwindSafe(|| {
930            while executed < max_steps && !snes.cpu.stopped {
931                seen.insert((u32::from(snes.cpu.pb) << 16) | u32::from(snes.cpu.pc));
932                snes.step();
933                executed += 1;
934            }
935        }));
936        std::panic::set_hook(prev_hook);
937        self.instructions_executed += executed;
938        result.map_err(|p| ApiError::Panic(panic_message(&p)))?;
939        Ok(LoopProbe {
940            distinct_pcs: seen.len(),
941            executed,
942        })
943    }

1377    /// Disassemble `count` 65C816 instructions starting at the 24-bit address
1378    /// `start`, with effective accumulator/index widths `m8`/`x8`. Bytes are
1379    /// read side-effect-free through `peek_memory`; the line at the live
1380    /// `PB:PC` is flagged `is_pc`. Tracks `REP`/`SEP` forward so immediate
1381    /// widths stay correct across a width change inside the window. For a
1382    /// CPU disassembly panel.
1383    pub fn disassemble_cpu(
1384        &mut self,
1385        start: u32,
1386        count: u16,
1387        m8: bool,
1388        x8: bool,
1389    ) -> Result<Vec<DisasmLine>, ApiError> {
1390        let pc_full = {
1391            let snes = self.snes.as_ref().ok_or(ApiError::NoRom)?;
1392            (u32::from(snes.cpu.pb) << 16) | u32::from(snes.cpu.pc)
1393        };
1394        let bank = (start >> 16) as u8;
1395        let start_off = start as u16;
1396        // One side-effect-free peek of the whole window (instructions are
1397        // ≤4 bytes, so over-read a little to cover the last one).
1398        let span = u16::try_from(usize::from(count) * 4 + 3).unwrap_or(u16::MAX);
1399        let buf = self.peek_memory(bank, start_off, span)?;
1400        let read = |o: u16| {
1401            buf.get(usize::from(o.wrapping_sub(start_off)))
1402                .copied()
1403                .unwrap_or(0)
1404        };
1405
1406        let (mut m8, mut x8) = (m8, x8);
1407        let mut off = start_off;
1408        let mut out = Vec::with_capacity(usize::from(count));
1409        for _ in 0..count {
1410            let insn = luna_cpu_65c816::disassemble(read, off, m8, x8);
1411            let bytes = (0..u16::from(insn.length))
1412                .map(|i| read(off.wrapping_add(i)))
1413                .collect();
1414            let addr_full = (u32::from(bank) << 16) | u32::from(off);
1415            out.push(DisasmLine {
1416                addr: addr_full,
1417                bytes,
1418                text: insn.text,
1419                is_pc: addr_full == pc_full,
1420            });
1421            // Track REP/SEP so later lines use the right immediate widths.
1422            match read(off) {
1423                0xC2 => {
1424                    let v = read(off.wrapping_add(1));
1425                    if v & 0x20 != 0 {
1426                        m8 = false;
1427                    }
1428                    if v & 0x10 != 0 {
1429                        x8 = false;
1430                    }
1431                }
1432                0xE2 => {
1433                    let v = read(off.wrapping_add(1));
1434                    if v & 0x20 != 0 {
1435                        m8 = true;
1436                    }
1437                    if v & 0x10 != 0 {
1438                        x8 = true;
1439                    }
1440                }
1441                _ => {}
1442            }
1443            off = off.wrapping_add(u16::from(insn.length));
1444        }
1445        Ok(out)
1446    }

1537    /// Restore a blob produced by [`Emulator::save_state`] into the live
1538    /// emulator. The ROM must already be loaded and identical to the one
1539    /// the state was saved from (matched by hash) — this is a rewind of the
1540    /// running machine, not a ROM swap.
1541    ///
1542    /// Errors with [`ApiError::NoRom`] when no ROM is loaded, or
1543    /// [`ApiError::SaveState`] on a decode failure, a format-version
1544    /// mismatch, or a ROM-hash mismatch.
1545    pub fn load_state(&mut self, data: &[u8]) -> Result<(), ApiError> {
1546        if self.snes.is_none() {
1547            return Err(ApiError::NoRom);
1548        }
1549        let bundle: SaveStateBundle = bincode::deserialize(data)
1550            .map_err(|e| ApiError::SaveState(format!("bundle decode: {e}")))?;
1551        if bundle.version != SAVE_STATE_VERSION {
1552            return Err(ApiError::SaveState(format!(
1553                "format version mismatch: state is v{}, this build expects v{SAVE_STATE_VERSION}",
1554                bundle.version
1555            )));
1556        }
1557        if bundle.rom_hash != self.rom_hash {
1558            return Err(ApiError::SaveState(
1559                "ROM mismatch: this save state was produced against a different ROM".to_string(),
1560            ));
1561        }
1562        let mut restored: Snes = bincode::deserialize(&bundle.core)
1563            .map_err(|e| ApiError::SaveState(format!("core decode: {e}")))?;
1564        // The deserialized `Snes` has a placeholder mapper (the trait object
1565        // is `serde(skip)`). Move the LIVE mapper — which still owns the ROM
1566        // — into it, then replay the mapper's saved mutable state onto it.
1567        let snes = self.snes.as_mut().ok_or(ApiError::NoRom)?;
1568        restored.mapper = std::mem::replace(&mut snes.mapper, luna_core::null_mapper());
1569        restored.mapper.load_state(&bundle.mapper);
1570        *snes = restored;
1571        Ok(())
1572    }

1616    /// Memory search (L9): every 24-bit `$7E-$7F` WRAM address whose bytes
1617    /// match `pattern`. Empty `pattern` → no hits.

1633    /// Controlled execution (L10): step until the CPU PC reaches `pc`
1634    /// (`pb << 16 | pc`) or `max_steps` instructions elapse. Returns `true`
1635    /// if `pc` was reached (checked BEFORE each step, so a current match
1636    /// returns immediately).
1637    pub fn run_until_pc(&mut self, pc: u32, max_steps: u64) -> Result<bool, ApiError> {
1638        let snes = self.snes.as_mut().ok_or(ApiError::NoRom)?;
1639        for _ in 0..max_steps {
1640            let cur = (u32::from(snes.cpu.pb) << 16) | u32::from(snes.cpu.pc);
1641            if cur == pc {
1642                return Ok(true);
1643            }
1644            snes.step();
1645        }
1646        let cur = (u32::from(snes.cpu.pb) << 16) | u32::from(snes.cpu.pc);
1647        Ok(cur == pc)
1648    }

1690    fn run_until_mem_access(
1691        &mut self,
1692        addr_full: u32,
1693        want: MemEventKind,
1694        max_steps: u64,
1695    ) -> Result<Option<(u32, u8)>, ApiError> {
1696        let bank = (addr_full >> 16) as u8;
1697        {
1698            let snes = self.snes.as_mut().ok_or(ApiError::NoRom)?;
1699            // Capture this bank's accesses; drained every step so it never
1700            // overflows. (Re-enabling resets the buffer.)
1701            snes.enable_mem_trace(1 << 16, Some(bank), None);
1702        }
1703        for _ in 0..max_steps {
1704            {
1705                let snes = self.snes.as_mut().ok_or(ApiError::NoRom)?;
1706                snes.step();
1707            }
1708            let events = {
1709                let snes = self.snes.as_mut().ok_or(ApiError::NoRom)?;
1710                snes.take_mem_trace_log()
1711            };
1712            if let Some(ev) = events
1713                .into_iter()
1714                .find(|ev| ev.addr_full == addr_full && ev.kind == want)
1715            {
1716                return Ok(Some((ev.pc_full, ev.value)));
1717            }
1718        }
1719        Ok(None)
1720    }

1859    /// Diagnostic: per-page FNV-1a hashes of the 128 KiB WRAM (`$7E0000`-
1860    /// `$7FFFFF`), one `u64` per `page_size` bytes (default 4 KiB → 32
1861    /// pages). Frame-aligned (NMI-count) hashing of these is the
1862    /// confound-free way to bisect a CPU-state divergence vs a reference

1864    /// first diverging page pins the first real divergence — unlike scene-
1865    /// level windows which the boot-frame offset confounds. `page_size` must
1866    /// be a power of two that divides `0x20000`.

1869        let ps = if page_size == 0 { 0x1000 } else { page_size };
1870        assert!(
1871            ps.is_power_of_two() && 0x2_0000 % ps == 0,
1872            "page_size must be a power of two dividing 0x20000"
1873        );
1874        Ok(snes
1875            .wram
1876            .chunks_exact(ps)
1877            .map(|page| {
1878                // FNV-1a 64-bit
1879                let mut h: u64 = 0xcbf2_9ce4_8422_2325;
1880                for &b in page {
1881                    h ^= u64::from(b);
1882                    h = h.wrapping_mul(0x0000_0100_0000_01b3);
1883                }
1884                h
1885            })
1886            .collect())
1887    }

1923    /// Export the current APU state as a 66 048-byte `.spc` sound file
1924    /// (`SNES-SPC700 Sound File Data v0.30`): the live SPC700 registers,
1925    /// all 64 KB of ARAM, the 128 DSP registers, and the IPL ROM — the
1926    /// exact snapshot any SPC player needs to resume playback. Capture it
1927    /// once the game's music driver is running (step far enough in, and
1928    /// inject Start to pass title screens). A minimal ID666 text tag is
1929    /// filled from the cartridge title.
1930    pub fn export_spc(&self) -> Result<Vec<u8>, ApiError> {
1931        // Null-pad a UTF-8 string into a fixed-width ID666 text field.
1932        fn put_text(dst: &mut [u8], s: &str) {
1933            let n = s.len().min(dst.len());
1934            dst[..n].copy_from_slice(&s.as_bytes()[..n]);
1935        }
1936
1937        let snes = self.snes.as_ref().ok_or(ApiError::NoRom)?;
1938        let apu = &snes.apu_real;
1939        let mut buf = vec![0u8; 0x1_0200];
1940        // 0x00: 33-byte signature, then 0x1A 0x1A.
1941        buf[0x00..0x21].copy_from_slice(b"SNES-SPC700 Sound File Data v0.30");
1942        buf[0x21] = 0x1A;
1943        buf[0x22] = 0x1A;
1944        buf[0x23] = 0x1A; // ID666 tag present (text format)
1945        buf[0x24] = 30; // minor version
1946
1947        // 0x25: SPC700 register file at the dump moment.
1948        let c = &apu.cpu;
1949        buf[0x25..0x27].copy_from_slice(&c.pc.to_le_bytes());
1950        buf[0x27] = c.a;
1951        buf[0x28] = c.x;
1952        buf[0x29] = c.y;
1953        buf[0x2A] = c.psw.0;
1954        buf[0x2B] = c.sp;
1955        // 0x2C..0x2E reserved (zero).
1956
1957        // 0x2E: ID666 text tag.
1958        let title = self.rom_info.as_ref().map_or("", |r| r.title.trim_end());
1959        put_text(&mut buf[0x2E..0x4E], title); // song title
1960        put_text(&mut buf[0x4E..0x6E], title); // game title
1961        put_text(&mut buf[0x6E..0x7E], "luna"); // dumper
1962        put_text(&mut buf[0xA9..0xAC], "180"); // seconds before fade
1963        put_text(&mut buf[0xAC..0xB1], "10000"); // fade length (ms)
1964        // 0xD2 emulator used: 0 = unknown.
1965
1966        // 0x100: 64 KB ARAM. 0x10100: 128 DSP registers.
1967        buf[0x100..0x1_0100].copy_from_slice(&apu.aram[..]);
1968        buf[0x1_0100..0x1_0180].copy_from_slice(&apu.dsp.registers);
1969        // 0x10180..0x101C0 unused. 0x101C0: 64-byte IPL ROM.
1970        buf[0x1_01C0..0x1_0200].copy_from_slice(&luna_cpu_spc700::IPL_ROM);
1971        Ok(buf)
1972    }

2040    /// Roll the events captured since the last call into the completed-frame
2041    /// buffers (Mesen2's `ClearFrameEvents`: `_prevDebugEvents = _debugEvents`).
2042    /// The emu loop calls this once per emulated frame. A frame with no captured
2043    /// events keeps the previous set (avoids flicker on quiet frames).
2044    pub fn swap_frame_events(&mut self) {
2045        let Some(snes) = self.snes.as_mut() else {
2046            return;
2047        };
2048        // Drain BOTH traces: CPU IO accesses (mem-trace) and DMA B-bus
2049        // writes (dma-trace). The CPU is halted during a DMA burst, so DMA
2050        // register writes never appear in the mem-trace — merging the two is
2051        // what makes the overlay dense (OAM/VRAM DMA show up).
2052        let cpu = snes.take_mem_trace_log();
2053        let dma = snes.dma.take_dma_trace();
2054        if cpu.is_empty() && dma.is_empty() {
2055            // Quiet frame: keep the previous set (avoids overlay flicker).
2056            return;
2057        }
2058        let mut merged: Vec<CapturedEvent> = Vec::with_capacity(cpu.len() + dma.len());
2059        merged.extend(cpu.into_iter().map(CapturedEvent::Cpu));
2060        merged.extend(dma.into_iter().map(CapturedEvent::Dma));
2061        // Order by PPU position so the list/overlay are sequential.
2062        merged.sort_by_key(CapturedEvent::sort_key);
2063        std::mem::swap(&mut self.prev_frame_events, &mut self.last_frame_events);
2064        self.last_frame_events = merged;
2065    }

2067    /// The current Event Viewer snapshot: the just-finished frame's visible
2068    /// events, plus (when `show_previous_frame` is on) the previous frame's,
2069    /// each decoded + filtered by the category visibility mask.
2070    #[must_use]
2071    pub fn event_snapshot(&self) -> Vec<EventViewerEvent> {
2072        let mut out = Vec::with_capacity(self.last_frame_events.len());
2073        for ev in &self.last_frame_events {
2074            if let Some(e) = ev.decode(&self.event_config, false) {
2075                out.push(e);
2076            }
2077        }
2078        if self.event_config.show_previous_frame {
2079            // Mesen2 `BaseEventManager::FilterEvents` (`:9-22`): include a
2080            // previous-frame event only when its `(scanline<<16)+cycle` is
2081            // GREATER than the cursor, so the prev frame fills in the part of
2082            // the frame the current one has not reached yet (no duplication).
2083            // luna polls at the frame boundary (no pause), so the live cursor
2084            // is the last current-frame event's position — i.e. prev-frame
2085            // events later than that fill the tail of the frame.
2086            let cursor = self
2087                .last_frame_events
2088                .iter()
2089                .map(CapturedEvent::position_key)
2090                .max()
2091                .unwrap_or(0);
2092            for ev in &self.prev_frame_events {
2093                if ev.position_key() <= cursor {
2094                    continue;
2095                }
2096                if let Some(e) = ev.decode(&self.event_config, true) {
2097                    out.push(e);
2098                }
2099            }
2100        }
2101        out
2102    }

2429        rom[0x7FDD] = (complement >> 8) as u8;
2430        rom[0x7FDE] = checksum as u8;
2431        rom[0x7FDF] = (checksum >> 8) as u8;
