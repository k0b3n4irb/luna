1//! `luna bench` — run every ROM in a directory headless, detect anomalies,
2//! and write a compatibility report + one markdown "issue" per bug.
3//!
4//! Pure `luna-api` consumer (api-first). Panic-safe: `Emulator::step*` and
5//! `load_rom` catch unwinds and return errors, so a single crashing ROM
6//! becomes one ❌ row instead of aborting the whole run.
7
8use std::fmt::Write as _;
9use std::path::{Path, PathBuf};
10use std::process::ExitCode;
11
12use luna_api::Emulator;
13
14/// Instruction budget per frame for `step_until_frame` (matches `run_state`).
15const PER_FRAME_BUDGET: u64 = 30_000;
16/// Frame by which boot/intro should be underway; metrics measured after it.
17const WARMUP_FRAME: u64 = 120;
18
19/// Outcome bucket for a ROM, worst-first in `report.md`.
20#[derive(Clone, Copy, PartialEq, Eq)]
21enum Status {
22    Bug,
23    Suspect,
24    NeedsFirmware,
25    Ok,
26}
27
28impl Status {
29    const fn icon(self) -> &'static str {
30        match self {
31            Self::Bug => "❌ bug",
32            Self::Suspect => "⚠️ suspect",
33            Self::NeedsFirmware => "🔑 firmware",
34            Self::Ok => "✅ ok",
35        }
36    }
37}
38
39struct RomResult {
40    name: String,
41    slug: String,
42    title: String,
43    mapper: String,
44    status: Status,
45    frames_run: u64,
46    nmis: u64,
47    instrs: u64,
48    /// Peak absolute audio sample seen across the run + audio tail.
49    audio_peak: i32,
50    /// Whether the S-DSP ever keyed a voice with a non-zero envelope — the
51    /// sound driver was alive and trying to play, even if the captured PCM
52    /// peak stayed quiet. Distinguishes "engine never ran" from "just quiet".
53    audio_voice_seen: bool,
54    findings: Vec<String>,
55    repro: String,
56    screenshot: Option<String>,
57}
58
59/// A static/black screen with the CPU spinning over no more than this many
60/// distinct addresses is treated as a confirmed hang (vs. an alive
61/// render/forced-blank issue, which stays ⚠️ suspect).
62const HANG_LOOP_MAX: usize = 64;
63/// Instruction budget for the end-of-run liveness probe.
64const LIVENESS_PROBE_STEPS: u64 = 200_000;
65
66/// Peak `|sample|` above which audio counts as genuinely playing (not just
67/// DC/dither). Below it the main window heard nothing audible.
68const AUDIO_FLOOR: i32 = 256;
69/// If the main window was silent, keep stepping up to this many extra frames
70/// purely to sample audio. Many titles (esp. Konami) open with a silent
71/// publisher logo / intro and only start the music well after frame 600, so a
72/// 10 s window false-flags them as "silent". This is the audio twin of the
73/// coproc-testing rule's "a black smoke screenshot ≠ a bug": a silent boot
74/// window ≠ dead audio. ~30 s at 60 Hz is enough to reach their first track.
75const AUDIO_TAIL_FRAMES: u64 = 1800;
76
77/// Default scripted input: pulse Start a few times to clear title/menu screens
78/// (a static title is NOT a bug — `coproc-testing.md`). `frame:mask`, Start=0x1000.
79fn default_input() -> Vec<(u64, u16)> {
80    vec![
81        (90, 0x1000),
82        (95, 0x0000),
83        (240, 0x1000),
84        (245, 0x0000),
85        (420, 0x1000),
86        (425, 0x0000),
87    ]
88}

90/// `lowercase-kebab` slug of a ROM filename stem, for output paths.
91fn slug(name: &str) -> String {
92    let stem = Path::new(name)
93        .file_stem()
94        .and_then(|s| s.to_str())
95        .unwrap_or(name);
96    let mut out = String::with_capacity(stem.len());
97    let mut prev_dash = false;
98    for c in stem.chars() {
99        if c.is_ascii_alphanumeric() {
100            out.push(c.to_ascii_lowercase());
101            prev_dash = false;
102        } else if !prev_dash {
103            out.push('-');
104            prev_dash = true;
105        }
106    }
107    out.trim_matches('-').to_string()
108}

110/// Run one ROM and classify it.
111fn bench_one(path: &Path, frames: u64, input: &[(u64, u16)], screens_dir: &Path) -> RomResult {
112    let name = path
113        .file_name()
114        .and_then(|n| n.to_str())
115        .unwrap_or("?")
116        .to_string();
117    let slug = slug(&name);
118    let repro = format!(
119        "luna state -n 40000000 --screenshot /tmp/{slug}.png \"{}\"",
120        path.display()
121    );
122
123    let mut em = Emulator::new();
124    let info = match em.load_rom(path) {
125        Ok(info) => info,
126        Err(e) => {
127            return RomResult {
128                name,
129                slug,
130                title: "?".into(),
131                mapper: "?".into(),
132                status: Status::Bug,
133                frames_run: 0,
134                nmis: 0,
135                instrs: 0,
136                audio_peak: 0,
137                audio_voice_seen: false,
138                findings: vec![format!("load failed: {e}")],
139                repro,
140                screenshot: None,
141            };
142        }
143    };
144
145    // Drive frames, applying input checkpoints; sample metrics along the way.
146    let mut applied = 0usize;
147    let mut crash: Option<String> = None;
148    let mut nmis_at_warmup = 0u64;
149    let mut audio_peak = 0i32;
150    let mut audio_voice_seen = false;
151    let mut fb_first: Option<u64> = None;
152    let mut fb_changed = false;
153    let mut last_frame = 0u64;
154    for f in 0..frames {
155        while applied < input.len() && input[applied].0 <= f {
156            let _ = em.set_joypad(0, input[applied].1);
157            applied += 1;
158        }
159        if let Err(e) = em.step_until_frame(PER_FRAME_BUDGET) {
160            crash = Some(e.to_string());
161            break;
162        }
163        if em.cpu_state().is_ok_and(|c| c.stopped) {
164            break; // CPU halted (STP) — no point spinning
165        }
166        // Audio activity: drain this frame's samples, track peak amplitude.
167        if let Ok(samples) = em.drain_audio(usize::MAX) {
168            for (l, r) in samples {
169                audio_peak = audio_peak.max(i32::from(l).abs()).max(i32::from(r).abs());
170            }
171        }
172        let st = em.state();
173        // The driver is alive and trying to play if any voice keyed on with a
174        // live envelope — true even when the captured PCM happens to be quiet.
175        audio_voice_seen |= st.apu.voice_envelope.iter().any(|&e| e > 0);
176        last_frame = st.scheduler.frame_count;
177        if f == WARMUP_FRAME {
178            nmis_at_warmup = st.scheduler.nmis_serviced;
179        }
180        // Past warm-up, hash the framebuffer EVERY frame (exact, cheap — the
181        // API hashes the native RGB buffer, no RGBA re-render); flag static if
182        // it never changes. Per-frame beats sampling-every-Nth, which can
183        // stride right over a brief change and false-flag a live screen.
184        if f >= WARMUP_FRAME {
185            if let Ok(h) = em.framebuffer_hash() {
186                match fb_first {
187                    None => fb_first = Some(h),
188                    Some(h0) if h != h0 => fb_changed = true,
189                    Some(_) => {}
190                }
191            }
192        }
193    }
194
195    let st = em.state();
196    let nmis = st.scheduler.nmis_serviced;
197    let mut findings: Vec<String> = Vec::new();
198
199    // ❌ confirmed-bug signals.
200    if let Some(msg) = &crash {
201        findings.push(format!("emulator error during run: {msg}"));
202    }
203    if em.cpu_state().is_ok_and(|c| c.stopped) {
204        findings.push("65C816 halted (STP / runaway into data)".into());
205    }
206    if st.apu.spc_stopped {
207        findings.push("SPC700 stopped".into());
208    }
209    let mut confirmed = !findings.is_empty();
210
211    let static_fb = fb_first.is_some() && !fb_changed;
212    let blank = !em.frame_showed_content().unwrap_or(true);
213    let nmi_starved = nmis <= nmis_at_warmup;
214
215    // Dead screen (static or forced-blank) + a CPU spinning over a tiny set of
216    // addresses = a confirmed hang. If the CPU is alive (many distinct PCs),
217    // the dead screen is a render/forced-blank issue → ⚠️ suspect, not a hang.
218    // The liveness probe (`loop_probe`) is the new API signal that lets us tell
219    // the two apart — NMI-starvation alone never could (Super FX/IRQ games are
220    // alive without NMI).
221    if !confirmed && (static_fb || blank) {
222        let distinct = em
223            .loop_probe(LIVENESS_PROBE_STEPS)
224            .map_or(usize::MAX, |p| p.distinct_pcs);
225        let where_ = if blank { "black" } else { "static" };
226        let nmi = if nmi_starved {
227            ", no NMIs after warm-up"
228        } else {
229            ""
230        };
231        if distinct <= HANG_LOOP_MAX {
232            findings.push(format!(
233                "CPU stuck in a {distinct}-address loop while the screen is {where_}{nmi} — hang"
234            ));
235            confirmed = true;
236        } else {
237            findings.push(format!(
238                "screen {where_} for the whole run but CPU alive ({distinct} distinct PCs{nmi}) — render/forced-blank issue"
239            ));
240        }
241    }
242
243    // 🔑 firmware (informational alongside whatever else was found).
244    if let Some(fw) = &info.missing_firmware {
245        findings.push(format!("missing coprocessor firmware '{fw}' (inert)"));
246    }
247    // Audio peak is surfaced as its own report column (silence over a 10s
248    // window is common for intros), not a status-driving finding.
249
250    let status = if confirmed {
251        Status::Bug
252    } else if info.missing_firmware.is_some() {
253        Status::NeedsFirmware
254    } else if static_fb || blank {
255        Status::Suspect
256    } else {
257        Status::Ok
258    };
259
260    let screenshot = em.render_frame_png(false).ok().and_then(|png| {
261        std::fs::write(screens_dir.join(format!("{slug}.png")), png)
262            .ok()
263            .map(|()| format!("screenshots/{slug}.png"))
264    });
265
266    // Audio tail: if the main window heard nothing, keep stepping (pulsing
267    // Start) purely to sample audio. A silent boot window is NOT dead audio —
268    // it's usually a silent intro before the first track. Skipped for confirmed
269    // bugs / halted CPUs (pointless) so it only costs time on otherwise-✅ ROMs.
270    if !confirmed && crash.is_none() && audio_peak < AUDIO_FLOOR {
271        for f in 0..AUDIO_TAIL_FRAMES {
272            // Pulse Start every ~150 frames to nudge past attract/menu gates.
273            let mask = if f % 150 < 4 { 0x1000 } else { 0x0000 };
274            let _ = em.set_joypad(0, mask);
275            if em.step_until_frame(PER_FRAME_BUDGET).is_err() {
276                break;
277            }
278            if let Ok(samples) = em.drain_audio(usize::MAX) {
279                for (l, r) in samples {
280                    audio_peak = audio_peak.max(i32::from(l).abs()).max(i32::from(r).abs());
281                }
282            }
283            audio_voice_seen |= em.state().apu.voice_envelope.iter().any(|&e| e > 0);
284            if audio_peak >= AUDIO_FLOOR {
285                break; // found the music — no need to run the full tail
286            }
287        }
288    }
289
290    RomResult {
291        name,
292        slug,
293        title: info.title,
294        mapper: info.mapper,
295        status,
296        frames_run: last_frame,
297        nmis,
298        instrs: st.stats.instructions_executed,
299        audio_peak,
300        audio_voice_seen,
301        findings,
302        repro,
303        screenshot,
304    }
305}

307/// `luna bench` entry point.
308pub(crate) fn run_bench(
309    dir: &Path,
310    out: &Path,
311    frames: u64,
312    input: Option<Vec<(u64, u16)>>,
313) -> ExitCode {
314    let input = input.unwrap_or_else(default_input);
315
316    let mut roms: Vec<PathBuf> = match std::fs::read_dir(dir) {
317        Ok(rd) => rd
318            .filter_map(Result::ok)
319            .map(|e| e.path())
320            .filter(|p| matches!(p.extension().and_then(|e| e.to_str()), Some("sfc" | "smc")))
321            .collect(),
322        Err(e) => {
323            eprintln!("error: reading {}: {e}", dir.display());
324            return ExitCode::from(1);
325        }
326    };
327    roms.sort();
328    if roms.is_empty() {
329        eprintln!("error: no .sfc/.smc ROMs in {}", dir.display());
330        return ExitCode::from(1);
331    }
332
333    let screens_dir = out.join("screenshots");
334    let bugs_dir = out.join("bugs");
335    // Clear stale bug files so a now-fixed ROM doesn't keep a leftover report.
336    let _ = std::fs::remove_dir_all(&bugs_dir);
337    for d in [out, &screens_dir, &bugs_dir] {
338        if let Err(e) = std::fs::create_dir_all(d) {
339            eprintln!("error: creating {}: {e}", d.display());
340            return ExitCode::from(1);
341        }
342    }
343
344    let mut results = Vec::with_capacity(roms.len());
345    for (i, rom) in roms.iter().enumerate() {
346        let r = bench_one(rom, frames, &input, &screens_dir);
347        println!(
348            "[{}/{}] {:<11} {}",
349            i + 1,
350            roms.len(),
351            r.status.icon(),
352            r.name
353        );
354        results.push(r);
355    }
356
357    write_reports(out, &bugs_dir, dir, frames, &results);
358    let count = |s: Status| results.iter().filter(|r| r.status == s).count();
359    println!(
360        "\nReport: {}  ({} ok, {} bug, {} suspect, {} firmware)",
361        out.join("report.md").display(),
362        count(Status::Ok),
363        count(Status::Bug),
364        count(Status::Suspect),
365        count(Status::NeedsFirmware),
366    );
367    ExitCode::SUCCESS
368}

370/// Write `report.md` + per-bug/suspect markdown files.
371fn write_reports(out: &Path, bugs_dir: &Path, rom_dir: &Path, frames: u64, results: &[RomResult]) {
372    let count = |s: Status| results.iter().filter(|r| r.status == s).count();
373    let mut rep = String::new();
374    rep.push_str("# luna ROM benchmark\n\n");
375    let _ = writeln!(
376        rep,
377        "Corpus: `{}` · {} ROMs · {frames} frames/ROM · headless via `luna-api`.\n",
378        rom_dir.display(),
379        results.len()
380    );
381    let _ = writeln!(
382        rep,
383        "**Summary:** {} ✅ ok · {} ❌ bug · {} ⚠️ suspect · {} 🔑 firmware.\n",
384        count(Status::Ok),
385        count(Status::Bug),
386        count(Status::Suspect),
387        count(Status::NeedsFirmware),
388    );
389    rep.push_str("| ROM | Status | Mapper | Frames | NMIs | Audio | Notes |\n");
390    rep.push_str("|---|---|---|--:|--:|--:|---|\n");
391    for s in [
392        Status::Bug,
393        Status::Suspect,
394        Status::NeedsFirmware,
395        Status::Ok,
396    ] {
397        for r in results.iter().filter(|r| r.status == s) {
398            let note = r.findings.first().map_or("—", String::as_str);
399            let link = r
400                .screenshot
401                .as_ref()
402                .map_or_else(|| r.name.clone(), |p| format!("[{}]({p})", r.name));
403            // peak ≥ floor → real audio (show the number); below floor but a
404            // voice keyed on → engine alive, just quiet; otherwise no audio
405            // heard even after the tail.
406            let audio = if r.audio_peak >= AUDIO_FLOOR {
407                r.audio_peak.to_string()
408            } else if r.audio_voice_seen {
409                "quiet".to_string()
410            } else {
411                "silent".to_string()
412            };
413            let _ = writeln!(
414                rep,
415                "| {link} | {} | {} | {} | {} | {audio} | {note} |",
416                r.status.icon(),
417                r.mapper,
418                r.frames_run,
419                r.nmis,
420            );
421        }
422    }
423
424    rep.push_str(
425        "\n*Audio column: a number is the peak `|sample|` once real audio was \
426         heard; **quiet** = the sound engine keyed voices on but stayed below \
427         the audible floor; **silent** = nothing heard even after the audio \
428         tail. A silent *boot* window is not dead audio (intros are often \
429         silent) — the tail keeps sampling ~30 s past the main window to reach \
430         the first track before reporting.*\n",
431    );
432
433    rep.push_str("\n## API-coverage notes\n\n");
434    rep.push_str(
435        "Progress probing what `luna-api` lets the bench observe:\n\n\
436         - ✅ **CPU infinite-loop / liveness detector** — `Emulator::loop_probe` \
437           (distinct PB:PC count) separates a real hang from a wait-for-input \
438           loop. Promoted a ⚠️ suspect to a confirmed ❌ hang.\n\
439         - ✅ **Audio-activity metric** — peak PCM via `drain_audio` + a \
440           voice-keyed-on signal from `EmulatorState::apu`, sampled across the \
441           run and an audio tail. Caveat learned: a fixed boot window \
442           false-flags silent intros, so the tail is required for an honest call.\n\
443         - ⏳ **Per-frame `framebuffer_changed` flag** — cheaper + less \
444           false-prone than hashing RGBA each sample.\n\
445         - ⏳ **Reached-gameplay heuristic** — e.g. input-responsiveness or \
446           sprite activity, to separate \"sitting at a menu\" from \"actually \
447           playing\".\n",
448    );
449
450    if let Err(e) = std::fs::write(out.join("report.md"), &rep) {
451        eprintln!("error: writing report.md: {e}");
452    }
453
454    for r in results
455        .iter()
456        .filter(|r| matches!(r.status, Status::Bug | Status::Suspect))
457    {
458        let mut b = String::new();
459        let _ = writeln!(b, "# {} — {}\n", r.status.icon(), r.name);
460        let _ = writeln!(b, "- **Title (header):** {}", r.title);
461        let _ = writeln!(b, "- **Mapper:** {}", r.mapper);
462        let _ = writeln!(
463            b,
464            "- **Frames run:** {} · **NMIs:** {} · **Instrs:** {}\n",
465            r.frames_run, r.nmis, r.instrs
466        );
467        b.push_str("## Symptoms\n\n");
468        for f in &r.findings {
469            let _ = writeln!(b, "- {f}");
470        }
471        if let Some(p) = &r.screenshot {
472            let _ = writeln!(b, "\n## Screenshot\n\n![screenshot](../{p})");
473        }
474        let _ = writeln!(b, "\n## Repro\n\n```\n{}\n```", r.repro);
475        if r.status == Status::Suspect {
476            b.push_str(
477                "\n> ⚠️ Heuristic flag — **verify in the GUI** before treating as a real bug.\n",
478            );
479        }
480        let _ = std::fs::write(bugs_dir.join(format!("{}.md", r.slug)), b);
481    }
