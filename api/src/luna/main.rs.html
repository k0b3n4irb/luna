715/// Parse a `--input` script: comma-separated `frame:hex` entries.
716/// Returns the checkpoints sorted by ascending frame. The frame
717/// number is decimal; the hex mask accepts an optional `0x` prefix
718/// and is read as a 16-bit value.
719fn parse_input_script(script: &str) -> Result<Vec<(u64, u16)>, String> {
720    let mut out: Vec<(u64, u16)> = Vec::new();
721    for entry in script.split(',') {
722        let entry = entry.trim();
723        if entry.is_empty() {
724            continue;
725        }
726        let (frame_str, mask_str) = entry
727            .split_once(':')
728            .ok_or_else(|| format!("missing ':' in entry `{entry}`"))?;
729        let frame: u64 = frame_str
730            .trim()
731            .parse()
732            .map_err(|e| format!("bad frame `{frame_str}`: {e}"))?;
733        let mask_str = mask_str
734            .trim()
735            .trim_start_matches("0x")
736            .trim_start_matches("0X");
737        let mask: u16 = u16::from_str_radix(mask_str, 16)
738            .map_err(|e| format!("bad hex mask `{mask_str}`: {e}"))?;
739        out.push((frame, mask));
740    }
741    out.sort_by_key(|(f, _)| *f);
742    Ok(out)
743}

745/// A scripted mouse checkpoint: `(frame, (dx, dy, buttons))`.
746type MouseCheckpoint = (u64, (i32, i32, u8));
747
748/// Parse a `--mouse` script: `;`-separated `frame:dx,dy,buttons` entries
749/// (signed `dx`/`dy`; `buttons` bit0 = left, bit1 = right). Returns the
750/// checkpoints sorted by frame.
751fn parse_mouse_script(script: &str) -> Result<Vec<MouseCheckpoint>, String> {
752    let mut out = Vec::new();
753    for entry in script.split(';').map(str::trim).filter(|s| !s.is_empty()) {
754        let (frame, rest) = entry
755            .split_once(':')
756            .ok_or_else(|| format!("`{entry}`: expected `frame:dx,dy,buttons`"))?;
757        let frame: u64 = frame
758            .trim()
759            .parse()
760            .map_err(|_| format!("`{entry}`: bad frame"))?;
761        let p: Vec<&str> = rest.split(',').map(str::trim).collect();
762        if p.len() != 3 {
763            return Err(format!("`{entry}`: expected `dx,dy,buttons` (3 values)"));
764        }
765        let dx: i32 = p[0].parse().map_err(|_| format!("`{entry}`: bad dx"))?;
766        let dy: i32 = p[1].parse().map_err(|_| format!("`{entry}`: bad dy"))?;
767        let buttons: u8 = p[2]
768            .parse()
769            .map_err(|_| format!("`{entry}`: bad buttons"))?;
770        out.push((frame, (dx, dy, buttons)));
771    }
772    out.sort_by_key(|(f, _)| *f);
773    Ok(out)
774}

776/// Parse a `BANK:OFFSET:COUNT` peek spec (all hex, no `0x` prefix).
777fn parse_peek_spec(spec: &str) -> Result<(u8, u16, u16), String> {
778    let parts: Vec<&str> = spec.split(':').collect();
779    if parts.len() != 3 {
780        return Err(format!("expected BANK:OFFSET:COUNT, got `{spec}`"));
781    }
782    let bank = u8::from_str_radix(parts[0].trim(), 16)
783        .map_err(|e| format!("bad bank `{}`: {e}", parts[0]))?;
784    let offset = u16::from_str_radix(parts[1].trim(), 16)
785        .map_err(|e| format!("bad offset `{}`: {e}", parts[1]))?;
786    let count = u16::from_str_radix(parts[2].trim(), 16)
787        .map_err(|e| format!("bad count `{}`: {e}", parts[2]))?;
788    Ok((bank, offset, count))
789}

791/// Parse an even-length hex string (no `0x`) into bytes.
792fn parse_hex_bytes(s: &str) -> Result<Vec<u8>, String> {
793    let s = s.trim();
794    if s.is_empty() || s.len() % 2 != 0 {
795        return Err(format!("expected an even-length hex value, got `{s}`"));
796    }
797    (0..s.len())
798        .step_by(2)
799        .map(|i| {
800            u8::from_str_radix(&s[i..i + 2], 16)
801                .map_err(|e| format!("bad hex `{}`: {e}", &s[i..i + 2]))
802        })
803        .collect()
804}

806/// Parse a `BANK:OFFSET=HEXBYTES` assertion (all hex, no `0x`).
807fn parse_assert_spec(spec: &str) -> Result<(u8, u16, Vec<u8>), String> {
808    let (loc, val) = spec
809        .split_once('=')
810        .ok_or_else(|| format!("expected BANK:OFFSET=HEX, got `{spec}`"))?;
811    let (bank_s, off_s) = loc
812        .split_once(':')
813        .ok_or_else(|| format!("expected BANK:OFFSET=HEX, got `{spec}`"))?;
814    let bank =
815        u8::from_str_radix(bank_s.trim(), 16).map_err(|e| format!("bad bank `{bank_s}`: {e}"))?;
816    let offset =
817        u16::from_str_radix(off_s.trim(), 16).map_err(|e| format!("bad offset `{off_s}`: {e}"))?;
818    Ok((bank, offset, parse_hex_bytes(val)?))
819}

821/// Parse an `OFFSET=HEXBYTES` assertion (for ARAM / VRAM — no bank).
822fn parse_assert_spec_no_bank(spec: &str) -> Result<(u16, Vec<u8>), String> {
823    let (off_s, val) = spec
824        .split_once('=')
825        .ok_or_else(|| format!("expected OFFSET=HEX, got `{spec}`"))?;
826    let offset =
827        u16::from_str_radix(off_s.trim(), 16).map_err(|e| format!("bad offset `{off_s}`: {e}"))?;
828    Ok((offset, parse_hex_bytes(val)?))
829}

831/// Lower-case hex of a byte slice (for assert PASS/FAIL output).
832fn hex_str(bytes: &[u8]) -> String {
833    use std::fmt::Write as _;
834    bytes
835        .iter()
836        .fold(String::with_capacity(bytes.len() * 2), |mut s, b| {
837            let _ = write!(s, "{b:02x}");
838            s
839        })
840}

845/// Format a 24-bit program counter / bus address as `$BB:OOOO`
846/// (bank:offset) — the canonical PC column shared by the trace writers.
847fn fmt_pc(pc_full: u32) -> String {
848    format!("${:02X}:{:04X}", (pc_full >> 16) & 0xFF, pc_full & 0xFFFF)

851/// Shared skeleton for the trace CSV writers: create `path`, write the
852/// `header` line, then format each event via `row` (handed the writer,
853/// the event index, and the event). Centralises the
854/// File/BufWriter/header boilerplate every writer repeated.
855fn write_csv<T>(
856    path: &std::path::Path,
857    header: &str,
858    rows: &[T],
859    mut row: impl FnMut(&mut dyn std::io::Write, usize, &T) -> std::io::Result<()>,
860) -> std::io::Result<()> {
861    use std::io::Write;
862    let mut f = std::io::BufWriter::new(std::fs::File::create(path)?);
863    writeln!(f, "{header}")?;
864    for (i, ev) in rows.iter().enumerate() {
865        row(&mut f, i, ev)?;
866    }
867    f.flush()
868}

1134/// Print a 16-bytes-per-row hex dump to stderr.
1135fn print_hex_dump(bank: u8, base: u16, bytes: &[u8]) {
1136    for (row_idx, chunk) in bytes.chunks(16).enumerate() {
1137        let addr = (u32::from(bank) << 16) | (u32::from(base) + (row_idx as u32 * 16));
1138        let hex: String = chunk
1139            .iter()
1140            .map(|b| format!("{b:02X}"))
1141            .collect::<Vec<_>>()
1142            .join(" ");
1143        eprintln!("  ${addr:06X}  {hex}");
1144    }
1145}

1147/// Load `rom` into `em`, honouring an optional `--force-mapper` override.
1148/// Centralises the force-mapper parse + file read shared by the `state`
1149/// and `frames` subcommands. Returns a human-facing error string.
1150fn load_rom_into(
1151    em: &mut luna_api::Emulator,
1152    rom: &std::path::Path,
1153    force_mapper: Option<&str>,
1154    dsp1_rom: Option<&std::path::Path>,
1155) -> Result<(), String> {
1156    // `--dsp1-rom` installs the firmware into luna's firmware folder so it
1157    // is found now and on every future run.
1158    if let Some(fw) = dsp1_rom {
1159        match luna_api::Emulator::install_firmware(fw, "dsp1b.rom") {
1160            Ok(dest) => eprintln!("installed DSP firmware → {}", dest.display()),
1161            Err(e) => eprintln!("warning: could not install {}: {e}", fw.display()),
1162        }
1163    }
1164    let info = match force_mapper {
1165        Some(kind_str) => {

1169                std::fs::read(rom).map_err(|e| format!("reading {}: {e}", rom.display()))?;

1184            info.title.trim()
1185        );

1190/// `luna frames` — capture `count` exactly-consecutive PPU frames as
1191/// PNGs via the same `luna-api` render path the GUI uses, tagging each
1192/// with its frame number and forced-blank flag. Lets us reproduce the
1193/// temporal artefacts (flicker / page-flip desync) that a single
1194/// `state --screenshot` is structurally blind to.
1195fn run_frames(
1196    rom: &std::path::Path,
1197    steps: u64,
1198    count: u64,
1199    out_dir: &std::path::Path,
1200    force_mapper: Option<&str>,
1201    input_script: Option<&str>,
1202) -> ExitCode {
1203    const FRAME_BUDGET: u64 = 200_000;
1204    let mut em = luna_api::Emulator::new();
1205    if let Err(e) = load_rom_into(&mut em, rom, force_mapper, None) {
1206        eprintln!("error: {e}");
1207        return ExitCode::from(1);
1208    }
1209    if let Err(e) = std::fs::create_dir_all(out_dir) {
1210        eprintln!("error: creating {}: {e}", out_dir.display());
1211        return ExitCode::from(1);
1212    }
1213    // Scripted input during warm-up (same semantics as `state --input`),
1214    // so the capture can land in gameplay rather than at a title screen.
1215    let checkpoints: Vec<(u64, u16)> = match input_script {
1216        None => Vec::new(),
1217        Some(script) => match parse_input_script(script) {
1218            Ok(v) => v,
1219            Err(e) => {
1220                eprintln!("error: --input: {e}");
1221                return ExitCode::from(1);
1222            }
1223        },
1224    };
1225    for (frame, mask) in &checkpoints {
1226        while em.state().scheduler.frame_count < *frame {
1227            if em.step_until_frame(FRAME_BUDGET).unwrap_or(0) == 0 {
1228                break;
1229            }
1230        }
1231        if let Err(e) = em.set_joypad(0, *mask) {
1232            eprintln!("error: set_joypad: {e}");
1233            return ExitCode::from(1);
1234        }
1235    }
1236    if let Err(e) = em.step(steps) {
1237        eprintln!("step warning (warm-up): {e}");
1238    }
1239    // Capture loop: one PNG per consecutive frame, tagged frame# + blank.
1240    for i in 0..count {
1241        let executed = em.step_until_frame(FRAME_BUDGET).unwrap_or(0);
1242        let frame = em.frame_count().unwrap_or(0);
1243        let blanked = em.forced_blank().unwrap_or(false);
1244        let showed = em.frame_showed_content().unwrap_or(true);
1245        let png = match em.render_frame_png(false) {
1246            Ok(p) => p,
1247            Err(e) => {
1248                eprintln!("error: render_frame_png: {e}");
1249                return ExitCode::from(1);
1250            }
1251        };
1252        // Tag on the per-frame "showed visible content" latch (what the GUI
1253        // publishes), not the instantaneous forced-blank bit — the latter
1254        // mislabels Super FX frames that re-blank at VBlank as "blank".
1255        let tag = if showed { "live" } else { "blank" };
1256        let path = out_dir.join(format!("frame_{i:03}_f{frame}_{tag}.png"));
1257        if let Err(e) = std::fs::write(&path, &png) {
1258            eprintln!("error: writing {}: {e}", path.display());
1259            return ExitCode::from(1);
1260        }
1261        println!(
1262            "frame {i:>3}: ppu_frame={frame} showed_content={showed} forced_blank={blanked} (+{executed} instr) -> {}",
1263            path.display()
1264        );
1265        if executed == 0 {
1266            eprintln!("note: step_until_frame returned 0 (emulator halted?) — stopping early");
1267            break;
1268        }
1269    }
1270    ExitCode::SUCCESS
1271}

1338/// `luna assets-dump` — run to a scene, then write every loaded graphics
1339/// asset as PNGs (+ raw VRAM/CGRAM + OAM JSON). Warm-up + scripted-input

1353    let mut em = luna_api::Emulator::new();
1354    if let Err(e) = load_rom_into(&mut em, rom, force_mapper, dsp1_rom) {
1355        eprintln!("error: {e}");
1356        return ExitCode::from(1);
1357    }
1358    let checkpoints: Vec<(u64, u16)> = match input_script {
1359        None => Vec::new(),
1360        Some(script) => match parse_input_script(script) {
1361            Ok(v) => v,
1362            Err(e) => {
1363                eprintln!("error: --input: {e}");
1364                return ExitCode::from(1);
1365            }
1366        },
1367    };
1368    for (frame, mask) in &checkpoints {
1369        while em.state().scheduler.frame_count < *frame {
1370            if em.step_until_frame(FRAME_BUDGET).unwrap_or(0) == 0 {
1371                break;
1372            }
1373        }
1374        if let Err(e) = em.set_joypad(0, *mask) {
1375            eprintln!("error: set_joypad: {e}");
1376            return ExitCode::from(1);
1377        }
1378    }
1379    if let Err(e) = em.step(steps) {
1380        eprintln!("step warning (warm-up): {e}");
1381    }
1382    if let Err(e) = std::fs::create_dir_all(out) {
1383        eprintln!("error: creating {}: {e}", out.display());
1384        return ExitCode::from(1);
1385    }
1386
1387    // VRAM tile-sheet bpp: explicit, else auto from the current BG1 mode
1388    // (falling back to the common 4bpp).
1389    let bgmode = em.state().ppu.bgmode;
1390    let tile_bpp = bpp.unwrap_or_else(|| match em.bg_bpp(0) {
1391        Ok(b) if b != 0 => b,
1392        _ => 4,
1393    });
1394
1395    // Helper: write a PNG (Result<Vec<u8>>) to `out/<name>`, logging.
1396    let mut wrote = 0u32;
1397    let mut put = |name: &str, png: Result<Vec<u8>, luna_api::ApiError>| match png {
1398        Ok(bytes) => {
1399            let path = out.join(name);
1400            if let Err(e) = std::fs::write(&path, &bytes) {
1401                eprintln!("error: writing {}: {e}", path.display());
1402            } else {
1403                println!("  {} ({} bytes)", path.display(), bytes.len());
1404                wrote += 1;
1405            }
1406        }
1407        Err(e) => eprintln!("skip {name}: {e}"),
1408    };
1409
1410    println!(
1411        "assets @ frame {} (BGMODE ${bgmode:02X}, tile sheet {tile_bpp}bpp):",
1412        em.frame_count().unwrap_or(0)
1413    );
1414    put("screen.png", em.render_frame_png(true));
1415    put(
1416        "vram_tiles.png",
1417        em.render_vram_tiles_png(tile_bpp, palette),
1418    );
1419    put("palette.png", em.render_palette_png(16));
1420    put("sprites.png", em.render_sprite_sheet_png());
1421    // Mode 7 has a single 128×128 field (BG index is ignored); otherwise
1422    // dump each BG layer that is enabled in the current mode.
1423    if bgmode & 0x07 == 7 {
1424        put("bg1_tilemap_mode7.png", em.render_tilemap_png(0));
1425    } else {
1426        for bg in 0..4usize {
1427            if em.bg_bpp(bg).unwrap_or(0) != 0 {
1428                put(
1429                    &format!("bg{}_tilemap.png", bg + 1),
1430                    em.render_tilemap_png(bg),
1431                );
1432            }
1433        }
1434    }
1435
1436    // Raw dumps + OAM metadata for tools that want the bytes.
1437    if let Ok(v) = em.vram_bytes() {
1438        let _ = std::fs::write(out.join("vram.bin"), v);
1439    }
1440    if let Ok(cg) = em.peek_cgram() {
1441        let mut bytes = Vec::with_capacity(cg.len() * 2);
1442        for c in cg {
1443            bytes.extend_from_slice(&c.to_le_bytes());
1444        }
1445        let _ = std::fs::write(out.join("cgram.bin"), bytes);
1446    }
1447    if let Ok(sprites) = em.decode_sprites()
1448        && let Ok(json) = serde_json::to_string_pretty(&sprites)
1449    {
1450        let _ = std::fs::write(out.join("oam.json"), json);
1451    }
1452
1453    println!(
1454        "wrote {wrote} PNG(s) + raw VRAM/CGRAM + oam.json to {}",
1455        out.display()
1456    );
1457    ExitCode::SUCCESS
1458}

1460/// `luna wram-trace` — emit per-frame (vblank-aligned) WRAM page hashes
1461/// for a confound-free cross-emulator differential (see the subcommand
1462/// doc). One line per frame: `<ppu_frame> <h0> <h1> ...`.
1463#[allow(clippy::too_many_arguments)]
1464fn run_wram_trace(
1465    rom: &std::path::Path,
1466    steps: u64,
1467    count: u64,
1468    page_size: usize,
1469    out: &std::path::Path,
1470    dump_frame: Option<u64>,
1471    dump_out: &std::path::Path,
1472    force_mapper: Option<&str>,
1473    input_script: Option<&str>,
1474) -> ExitCode {
1475    use std::fmt::Write as _;
1476    const FRAME_BUDGET: u64 = 200_000;
1477    let mut em = luna_api::Emulator::new();
1478    if let Err(e) = load_rom_into(&mut em, rom, force_mapper, None) {
1479        eprintln!("error: {e}");
1480        return ExitCode::from(1);
1481    }
1482    let checkpoints: Vec<(u64, u16)> = match input_script {
1483        None => Vec::new(),
1484        Some(script) => match parse_input_script(script) {
1485            Ok(v) => v,
1486            Err(e) => {
1487                eprintln!("error: --input: {e}");
1488                return ExitCode::from(1);
1489            }
1490        },
1491    };
1492    if steps > 0 {
1493        if let Err(e) = em.step(steps) {
1494            eprintln!("step warning (warm-up): {e}");
1495        }
1496    }
1497    // Input checkpoints are applied DURING the capture loop, keyed by the
1498    // current PPU frame — so a scripted joypad pulse can span the frames
1499    // being hashed (front-loading them would consume the pulse before the
1500    // capture even starts).
1501    let mut ck_idx = 0usize;
1502    let mut buf = String::new();
1503    for _ in 0..count {
1504        let cur_frame = em.frame_count().unwrap_or(0);
1505        while ck_idx < checkpoints.len() && checkpoints[ck_idx].0 <= cur_frame {
1506            if let Err(e) = em.set_joypad(0, checkpoints[ck_idx].1) {
1507                eprintln!("error: set_joypad: {e}");
1508                return ExitCode::from(1);
1509            }
1510            ck_idx += 1;
1511        }
1512        let executed = em.step_until_frame(FRAME_BUDGET).unwrap_or(0);
1513        let frame = em.frame_count().unwrap_or(0);
1514        let hashes = match em.wram_page_hashes(page_size) {
1515            Ok(h) => h,
1516            Err(e) => {
1517                eprintln!("error: wram_page_hashes: {e}");
1518                return ExitCode::from(1);
1519            }
1520        };
1521        let _ = write!(buf, "{frame}");
1522        for h in &hashes {
1523            let _ = write!(buf, " {h:016x}");
1524        }
1525        buf.push('\n');
1526        if dump_frame == Some(frame) {
1527            match em.wram_snapshot() {
1528                Ok(bytes) => {
1529                    if let Err(e) = std::fs::write(dump_out, &bytes) {
1530                        eprintln!("error: writing {}: {e}", dump_out.display());
1531                        return ExitCode::from(1);
1532                    }
1533                    println!(
1534                        "dumped {} WRAM bytes at frame {frame} -> {}",
1535                        bytes.len(),
1536                        dump_out.display()
1537                    );
1538                }
1539                Err(e) => {
1540                    eprintln!("error: wram_snapshot: {e}");
1541                    return ExitCode::from(1);
1542                }
1543            }
1544        }
1545        if executed == 0 {
1546            eprintln!("note: step_until_frame returned 0 (emulator halted?) — stopping early");
1547            break;
1548        }
1549    }
1550    if let Err(e) = std::fs::write(out, &buf) {
1551        eprintln!("error: writing {}: {e}", out.display());
1552        return ExitCode::from(1);
1553    }
1554    println!(
1555        "wrote {count} frames x {} pages of {page_size}-byte WRAM hashes -> {}",
1556        0x2_0000 / page_size,
1557        out.display()
1558    );
1559    ExitCode::SUCCESS
1560}

1635                    "error: --port{} `{other}`: expected `pad`, `mouse`, or `superscope`",
1636                    port + 1
1637                );
1638                return ExitCode::from(1);
1639            }
1640        }
1641    }
1642    let mouse_checkpoints = match mouse_script.map(parse_mouse_script) {
1643        Some(Ok(v)) => v,
1644        Some(Err(e)) => {
1645            eprintln!("error: --mouse: {e}");
1646            return ExitCode::from(1);
1647        }
1648        None => Vec::new(),
1649    };
1650    // `--superscope` shares the `frame:a,b,c` triplet grammar with `--mouse`
1651    // (here a,b = absolute aim x,y; c = the button mask).
1652    let scope_checkpoints = match superscope_script.map(parse_mouse_script) {
1653        Some(Ok(v)) => v,
1654        Some(Err(e)) => {
1655            eprintln!("error: --superscope: {e}");
1656            return ExitCode::from(1);
1657        }
1658        None => Vec::new(),
1659    };
1660    // --srm-in: seed battery SRAM from a `.srm` file (the read half of a
1661    // cross-run power-cycle test), before the warm-up runs.
1662    if let Some(path) = srm_in {
1663        match std::fs::read(path) {
1664            Ok(data) => {
1665                if let Err(e) = em.load_sram(&data) {
1666                    eprintln!("error: --srm-in: {e}");
1667                    return ExitCode::from(1);
1668                }
1669                eprintln!(
1670                    "loaded {} bytes of SRAM from {}",
1671                    data.len(),
1672                    path.display()
1673                );
1674            }
1675            Err(e) => {
1676                eprintln!("error: --srm-in {}: {e}", path.display());
1677                return ExitCode::from(1);
1678            }
1679        }
1680    }
1681    // Optionally resume from a GUI-captured save state (api-first:
1682    // `Emulator::load_state`). Done right after `load_rom` so the `-n`
1683    // warm-up runs forward from the captured scene.
1684    if let Some(sp) = load_state_path {
1685        match std::fs::read(sp) {
1686            Ok(bytes) => {
1687                if let Err(e) = em.load_state(&bytes) {
1688                    eprintln!("error: --load-state {}: {e}", sp.display());
1689                    return ExitCode::from(1);
1690                }
1691                eprintln!("loaded save state from {}", sp.display());
1692            }
1693            Err(e) => {
1694                eprintln!("error: --load-state {}: {e}", sp.display());
1695                return ExitCode::from(1);
1696            }
1697        }
1698    }
1699    if apu_log_path.is_some() {
1700        if let Err(e) = em.enable_mailbox_log() {
1701            eprintln!("error: enable_mailbox_log: {e}");
1702            return ExitCode::from(1);
1703        }
1704    }
1705    if sa1_log_path.is_some() {
1706        if let Err(e) = em.enable_sa1_log() {
1707            eprintln!("error: enable_sa1_log: {e}");
1708            return ExitCode::from(1);
1709        }
1710    }
1711    if sa1_side_log_path.is_some() {
1712        if let Err(e) = em.enable_sa1_side_log() {
1713            eprintln!("error: enable_sa1_side_log: {e}");
1714            return ExitCode::from(1);
1715        }
1716    }
1717    if sa1_trace_path.is_some() {
1718        if let Err(e) = em.enable_sa1_trace(sa1_trace_max) {
1719            eprintln!("error: enable_sa1_trace: {e}");
1720            return ExitCode::from(1);
1721        }
1722    }
1723    if superfx_trace_path.is_some() {
1724        if let Err(e) = em.enable_superfx_trace(superfx_trace_max) {
1725            eprintln!("error: enable_superfx_trace: {e}");
1726            return ExitCode::from(1);
1727        }
1728    }
1729    if spc_trace_path.is_some() {
1730        if let Err(e) = em.enable_spc_trace(spc_trace_max) {
1731            eprintln!("error: enable_spc_trace: {e}");
1732            return ExitCode::from(1);
1733        }
1734    }
1735    // Parse `frame:hex` checkpoints into a sorted vector. We apply
1736    // them by stepping `step_until_frame` between checkpoints — a
1737    // ~30k-instruction budget per frame is enough for any real ROM,
1738    // including SA-1 carts.
1739    let checkpoints: Vec<(u64, u16)> = match input_script {
1740        None => Vec::new(),
1741        Some(script) => match parse_input_script(script) {
1742            Ok(v) => v,
1743            Err(e) => {
1744                eprintln!("error: --input: {e}");
1745                return ExitCode::from(1);
1746            }
1747        },
1748    };
1749    if !checkpoints.is_empty() || !mouse_checkpoints.is_empty() || !scope_checkpoints.is_empty() {
1750        const PER_FRAME_BUDGET: u64 = 60_000;
1751        // Merge gamepad (`--input`), mouse (`--mouse`) and super-scope
1752        // (`--superscope`) checkpoints into one frame-sorted event stream so
1753        // they all apply at the right moment.
1754        enum Ev {
1755            Pad(u16),
1756            Mouse(i32, i32, u8),
1757            Scope(i32, i32, u8),
1758        }
1759        let mut events: Vec<(u64, Ev)> = checkpoints
1760            .iter()
1761            .map(|&(f, m)| (f, Ev::Pad(m)))
1762            .chain(
1763                mouse_checkpoints
1764                    .iter()
1765                    .map(|&(f, (dx, dy, b))| (f, Ev::Mouse(dx, dy, b))),
1766            )
1767            .chain(
1768                scope_checkpoints
1769                    .iter()
1770                    .map(|&(f, (x, y, b))| (f, Ev::Scope(x, y, b))),
1771            )
1772            .collect();
1773        events.sort_by_key(|(f, _)| *f);
1774        for (frame, ev) in &events {
1775            // Step until we reach `frame` (no-op if we already crossed it).
1776            while em.state().scheduler.frame_count < *frame {
1777                if em.step_until_frame(PER_FRAME_BUDGET).unwrap_or(0) == 0 {
1778                    break;
1779                }
1780            }
1781            let applied = match ev {
1782                Ev::Pad(m) => em.set_joypad(0, *m),
1783                Ev::Mouse(dx, dy, b) => em.set_mouse(*dx, *dy, *b),
1784                Ev::Scope(x, y, b) => em.set_superscope(*x, *y, *b),
1785            };
1786            if let Err(e) = applied {
1787                eprintln!("error: applying scripted input: {e}");
1788                return ExitCode::from(1);
1789            }
1790        }
1791    }
1792    // Trace gating: bridge to the earliest of the two "trace-from"
1793    // targets, enable whichever crossed; bridge to the later one if
1794    // it's still ahead, enable that. Each trace caps itself at its
1795    // own --*-trace-max events regardless of remaining steps.
1796    let mut remaining = steps;
1797    let mut bridge_target = u64::MAX;
1798    if cpu_trace_path.is_some() {
1799        bridge_target = bridge_target.min(cpu_trace_from);
1800    }
1801    if mem_trace_path.is_some() {
1802        bridge_target = bridge_target.min(mem_trace_from);
1803    }
1804    let parsed_mem_bank: Option<u8> = match mem_trace_bank {
1805        None => None,
1806        Some(s) => match u8::from_str_radix(s.trim_start_matches("0x"), 16) {
1807            Ok(b) => Some(b),
1808            Err(e) => {
1809                eprintln!("error: --mem-trace-bank `{s}`: {e}");
1810                return ExitCode::from(1);
1811            }
1812        },
1813    };
1814    let parsed_mem_addr: Option<(u16, u16)> = match mem_trace_addr {
1815        None => None,
1816        Some(spec) => {
1817            let parse_hex16 = |h: &str| u16::from_str_radix(h.trim().trim_start_matches("0x"), 16);
1818            let Some((lo, hi)) = spec.split_once(':') else {
1819                eprintln!("error: --mem-trace-addr `{spec}`: expected `LO:HI` (e.g. 2100:21FF)");
1820                return ExitCode::from(1);
1821            };
1822            match (parse_hex16(lo), parse_hex16(hi)) {
1823                (Ok(lo), Ok(hi)) if lo <= hi => Some((lo, hi)),
1824                (Ok(lo), Ok(hi)) => {
1825                    eprintln!("error: --mem-trace-addr `{spec}`: LO ({lo:#06X}) > HI ({hi:#06X})");
1826                    return ExitCode::from(1);
1827                }
1828                _ => {
1829                    eprintln!("error: --mem-trace-addr `{spec}`: expected hex `LO:HI`");
1830                    return ExitCode::from(1);
1831                }
1832            }
1833        }
1834    };
1835    if bridge_target != u64::MAX {
1836        let current = em.instructions_executed();
1837        if bridge_target > current {
1838            let bridge = (bridge_target - current).min(remaining);
1839            if let Err(e) = em.step(bridge) {
1840                eprintln!("step warning (pre-trace bridge 1): {e}");
1841            }
1842            remaining = remaining.saturating_sub(bridge);
1843        }
1844    }
1845    // Enable whichever traces are at or past their target now.
1846    if cpu_trace_path.is_some() && em.instructions_executed() >= cpu_trace_from {
1847        if let Err(e) = em.enable_cpu_trace(cpu_trace_max) {
1848            eprintln!("error: enable_cpu_trace: {e}");
1849            return ExitCode::from(1);
1850        }
1851    }
1852    if mem_trace_path.is_some() && em.instructions_executed() >= mem_trace_from {
1853        if let Err(e) = em.enable_mem_trace(mem_trace_max, parsed_mem_bank, parsed_mem_addr) {
1854            eprintln!("error: enable_mem_trace: {e}");
1855            return ExitCode::from(1);
1856        }
1857    }
1858    // If the two targets differ, bridge to the second one and enable.
1859    if cpu_trace_path.is_some() && mem_trace_path.is_some() && cpu_trace_from != mem_trace_from {
1860        let second_target = cpu_trace_from.max(mem_trace_from);
1861        let current = em.instructions_executed();
1862        if second_target > current {
1863            let bridge = (second_target - current).min(remaining);
1864            if let Err(e) = em.step(bridge) {
1865                eprintln!("step warning (pre-trace bridge 2): {e}");
1866            }
1867            remaining = remaining.saturating_sub(bridge);
1868            // Re-check enables (whichever wasn't enabled yet).
1869            if cpu_trace_path.is_some() && em.instructions_executed() >= cpu_trace_from {
1870                let _ = em.enable_cpu_trace(cpu_trace_max);
1871            }
1872            if mem_trace_path.is_some() && em.instructions_executed() >= mem_trace_from {
1873                let _ = em.enable_mem_trace(mem_trace_max, parsed_mem_bank, parsed_mem_addr);
1874            }
1875        }
1876    }
1877    // DMA→VRAM trace: bridge to its start instruction independently of the
1878    // cpu/mem traces, then enable. Capturing from boot would drown the
1879    // window of interest, so `--dma-trace-from` skips the early uploads.
1880    if dma_trace_path.is_some() {
1881        let current = em.instructions_executed();
1882        if dma_trace_from > current {
1883            let bridge = (dma_trace_from - current).min(remaining);
1884            if let Err(e) = em.step(bridge) {
1885                eprintln!("step warning (pre-dma-trace bridge): {e}");
1886            }
1887            remaining = remaining.saturating_sub(bridge);
1888        }
1889        if let Err(e) = em.enable_dma_trace(dma_trace_max) {
1890            eprintln!("error: enable_dma_trace: {e}");
1891            return ExitCode::from(1);
1892        }
1893    }
1894    // When --audio-out is set, step in chunks and drain the APU's
1895    // ~16k-sample bounded queue after each chunk; otherwise we'd lose
1896    // ~99% of audio on any run longer than ~0.5 s of emulated time.
1897    // The accumulated Vec is written to disk after the run.
1898    let mut audio_accum: Vec<(i16, i16)> = Vec::new();
1899    if let Some(target_frame) = until_frame {
1900        // RFE-5: run to a specific PPU frame instead of the `-n` instruction
1901        // count, draining audio along the way so `--audio-out` still works.
1902        const FRAME_BUDGET: u64 = 200_000;
1903        while em.state().scheduler.frame_count < target_frame {
1904            if em.step_until_frame(FRAME_BUDGET).unwrap_or(0) == 0 {
1905                break;
1906            }
1907            if audio_out.is_some() {
1908                if let Ok(mut chunk) = em.drain_audio(usize::MAX) {
1909                    audio_accum.append(&mut chunk);
1910                }
1911            }
1912        }
1913    } else if audio_out.is_some() {
1914        // Chunk = 100k instructions ≈ ~10 ms of emulated SPC time
1915        // (well under the 512 ms queue capacity even at peak DSP
1916        // output rate). Drain the full queue after each chunk.
1917        const AUDIO_CHUNK: u64 = 100_000;
1918        let mut left = remaining;
1919        while left > 0 {
1920            let take = left.min(AUDIO_CHUNK);
1921            if let Err(e) = em.step(take) {
1922                eprintln!("step warning: {e}");
1923                break;
1924            }
1925            left -= take;
1926            match em.drain_audio(usize::MAX) {
1927                Ok(mut chunk) => audio_accum.append(&mut chunk),
1928                Err(e) => eprintln!("warning: drain_audio mid-run: {e}"),
1929            }
1930        }
1931    } else {
1932        match em.step(remaining) {
1933            Ok(_) => {}
1934            Err(e) => {
1935                // Step errors are informational — we still want a state
1936                // snapshot.
1937                eprintln!("step warning: {e}");
1938            }
1939        }
1940    }
1941
1942    for spec in peek_specs {
1943        match parse_peek_spec(spec) {
1944            Ok((bank, offset, count)) => match em.peek_memory(bank, offset, count) {
1945                Ok(bytes) => {
1946                    eprintln!("peek ${:02X}:{:04X} +{:04X}:", bank, offset, bytes.len());
1947                    print_hex_dump(bank, offset, &bytes);
1948                }
1949                Err(e) => eprintln!("error: peek_memory `{spec}`: {e}"),
1950            },
1951            Err(e) => eprintln!("error: --peek `{spec}`: {e}"),
1952        }
1953    }
1954
1955    // --assert / --assert-aram / --assert-vram: native pass/fail (exit 0/1) so
1956    // headless consumers don't have to parse the peek dump.
1957    let mut assert_failed = false;
1958    for spec in assert_specs {
1959        match parse_assert_spec(spec) {
1960            Ok((bank, offset, want)) => match em.peek_memory(bank, offset, want.len() as u16) {
1961                Ok(got) if got == want => {
1962                    println!("PASS ${bank:02X}:{offset:04X}={}", hex_str(&want));
1963                }
1964                Ok(got) => {
1965                    println!(
1966                        "FAIL ${bank:02X}:{offset:04X} expected {} got {}",
1967                        hex_str(&want),
1968                        hex_str(&got)
1969                    );
1970                    assert_failed = true;
1971                }
1972                Err(e) => {
1973                    eprintln!("error: --assert `{spec}`: {e}");
1974                    assert_failed = true;
1975                }
1976            },
1977            Err(e) => {
1978                eprintln!("error: --assert `{spec}`: {e}");
1979                assert_failed = true;
1980            }
1981        }
1982    }
1983    for (specs, label, kind) in [
1984        (assert_aram_specs, "aram", 0u8),
1985        (assert_vram_specs, "vram", 1u8),
1986    ] {
1987        for spec in specs {
1988            match parse_assert_spec_no_bank(spec) {
1989                Ok((offset, want)) => {
1990                    let res = if kind == 0 {
1991                        em.peek_aram(offset, want.len() as u16)
1992                    } else {
1993                        em.peek_vram(offset, want.len() as u16)
1994                    };
1995                    match res {
1996                        Ok(got) if got == want => {
1997                            println!("PASS {label}:{offset:04X}={}", hex_str(&want));
1998                        }
1999                        Ok(got) => {
2000                            println!(
2001                                "FAIL {label}:{offset:04X} expected {} got {}",
2002                                hex_str(&want),
2003                                hex_str(&got)
2004                            );
2005                            assert_failed = true;
2006                        }
2007                        Err(e) => {
2008                            eprintln!("error: --assert-{label} `{spec}`: {e}");
2009                            assert_failed = true;
2010                        }
2011                    }
2012                }
2013                Err(e) => {
2014                    eprintln!("error: --assert-{label} `{spec}`: {e}");
2015                    assert_failed = true;
2016                }
2017            }
2018        }
2019    }
2020    // CGRAM is 256 × 15-bit colours (512 bytes), low byte of each colour first.
2021    for spec in assert_cgram_specs {
2022        match parse_assert_spec_no_bank(spec) {
2023            Ok((offset, want)) => match em.peek_cgram() {
2024                Ok(colors) => {
2025                    let bytes: Vec<u8> = colors
2026                        .iter()
2027                        .flat_map(|c| [*c as u8, (*c >> 8) as u8])
2028                        .collect();
2029                    let start = offset as usize;
2030                    let end = start + want.len();

2032                        eprintln!(
2033                            "error: --assert-cgram `{spec}`: offset {offset:#X}+{} exceeds the 512-byte CGRAM",
2034                            want.len()
2035                        );
2036                        assert_failed = true;
2037                    } else if bytes[start..end] == want[..] {
2038                        println!("PASS cgram:{offset:04X}={}", hex_str(&want));
2039                    } else {
2040                        println!(
2041                            "FAIL cgram:{offset:04X} expected {} got {}",
2042                            hex_str(&want),
2043                            hex_str(&bytes[start..end])
2044                        );
2045                        assert_failed = true;
2046                    }
2047                }
2048                Err(e) => {
2049                    eprintln!("error: --assert-cgram `{spec}`: {e}");
2050                    assert_failed = true;
2051                }
2052            },
2053            Err(e) => {
2054                eprintln!("error: --assert-cgram `{spec}`: {e}");
2055                assert_failed = true;
2056            }
2057        }
2058    }

2183    if let Some(path) = mem_trace_path {
2184        match em.take_mem_trace_log() {
2185            Ok(events) => match write_mem_trace_csv(path, &events) {
2186                Ok(()) => eprintln!(
2187                    "Memory trace written to {} ({} events)",
2188                    path.display(),
2189                    events.len()
2190                ),
2191                Err(e) => eprintln!("error: writing memory trace: {e}"),
2192            },
2193            Err(e) => eprintln!("error: take_mem_trace_log: {e}"),
2194        }
2195    }
2196    if let Some(path) = dma_trace_path {
2197        match em.take_dma_trace() {
2198            Ok(events) => match write_dma_trace_csv(path, &events) {
2199                Ok(()) => eprintln!(
2200                    "DMA→VRAM trace written to {} ({} events)",
2201                    path.display(),
2202                    events.len()
2203                ),
2204                Err(e) => eprintln!("error: writing DMA trace: {e}"),
2205            },
2206            Err(e) => eprintln!("error: take_dma_trace: {e}"),
2207        }
2208    }
2209
2210    let state = em.state();
2211    let json = match serde_json::to_string_pretty(&state) {
2212        Ok(s) => s,
2213        Err(e) => {
2214            eprintln!("error: serialising state: {e}");
2215            return ExitCode::from(1);
2216        }
2217    };
2218    if out.as_os_str() == "-" {
2219        println!("{json}");
2220    } else if let Err(e) = std::fs::write(out, &json) {
2221        eprintln!("error: writing state JSON: {e}");
2222        return ExitCode::from(1);
2223    } else {
2224        eprintln!("State JSON written to {}", out.display());
2225    }
2226
2227    if let Some(p) = screenshot {
2228        match em.render_frame_png(false) {
2229            Ok(png) => {
2230                if let Err(e) = std::fs::write(p, &png) {
2231                    eprintln!("error: writing screenshot: {e}");
2232                    return ExitCode::from(1);
2233                }
2234                eprintln!("Screenshot written to {}", p.display());
2235            }
2236            Err(e) => {
2237                eprintln!("error: render_frame_png: {e}");
2238                return ExitCode::from(1);
2239            }
2240        }
2241    }
2242    if let Some(p) = audio_out {
2243        // Drain anything still queued at end-of-run, then write the
2244        // full accumulated stream.
2245        match em.drain_audio(usize::MAX) {
2246            Ok(mut tail) => audio_accum.append(&mut tail),
2247            Err(e) => eprintln!("warning: final drain_audio: {e}"),
2248        }
2249        if let Err(e) = write_wav(p, &audio_accum) {
2250            eprintln!("error: writing WAV: {e}");
2251            return ExitCode::from(1);
2252        }
2253        let secs = audio_accum.len() as f64 / 32_000.0;
2254        eprintln!(
2255            "Audio WAV written to {}  ({} samples = {:.2}s @ 32 kHz stereo)",
2256            p.display(),
2257            audio_accum.len(),
2258            secs
2259        );
2260    }
2261    // --srm-out: persist battery SRAM to a `.srm` file (the write half of a
2262    // cross-run power-cycle test).
2263    if let Some(path) = srm_out {
2264        let data = em.sram();
2265        match std::fs::write(path, &data) {
2266            Ok(()) => eprintln!("wrote {} bytes of SRAM to {}", data.len(), path.display()),
2267            Err(e) => eprintln!("error: --srm-out {}: {e}", path.display()),
2268        }
2269    }
2270    if assert_failed {
2271        ExitCode::from(1)
2272    } else {
2273        ExitCode::SUCCESS
2274    }
2275}

2277fn run(
2278    rom_path: &std::path::Path,
2279    steps: u64,
2280    screenshot: Option<&std::path::Path>,
2281    force_display: bool,
2282    bg: Option<u8>,
2283    audio_out: Option<&std::path::Path>,
2284    nocash_out: Option<&std::path::Path>,
2285    wdm_out: Option<&std::path::Path>,
2286    print_fbhash: bool,
2287) -> ExitCode {
2288    let mut em = luna_api::Emulator::new();
2289    let info = match em.load_rom(rom_path) {
2290        Ok(i) => i,
2291        Err(e) => {
2292            eprintln!("error: {e}");
2293            return ExitCode::from(1);
2294        }
2295    };
2296
2297    print_header(&info);
2298
2299    // Arm the Nocash ($21FC) capture before stepping so the program's
2300    // `SNES_NOCASH`/`SNES_ASSERT` output is recorded during the run.
2301    if nocash_out.is_some() {
2302        if let Err(e) = em.enable_nocash_log() {
2303            eprintln!("error: enable_nocash_log: {e}");
2304            return ExitCode::from(1);
2305        }
2306    }
2307    if wdm_out.is_some() {
2308        if let Err(e) = em.enable_wdm_log() {
2309            eprintln!("error: enable_wdm_log: {e}");
2310            return ExitCode::from(1);
2311        }
2312    }
2313
2314    // `load_rom` already runs the reset vector; report where it landed.
2315    {
2316        let cpu = em.state().cpu;
2317        println!("After reset: PC=${:02X}:{:04X}", cpu.pb, cpu.pc);
2318    }
2319    println!();
2320
2321    let mut panic_msg: Option<String> = None;
2322    let mut stopped = false;
2323    // If we're recording audio, drain the APU queue every batch of
2324    // steps so it doesn't overflow (capped at 16 384 samples).
2325    let mut audio_samples: Vec<(i16, i16)> = if audio_out.is_some() {
2326        Vec::with_capacity(1 << 20)
2327    } else {
2328        Vec::new()
2329    };
2330    if audio_out.is_some() {
2331        // Step in 4 096-instruction batches, draining the APU's bounded
2332        // queue after each so it never saturates over a long run. A
2333        // short batch (`ran < want`) means the CPU hit STP.
2334        const BATCH: u64 = 4_096;
2335        loop {
2336            let done = em.instructions_executed();
2337            if done >= steps {
2338                break;
2339            }
2340            let want = (steps - done).min(BATCH);
2341            match em.step(want) {
2342                Ok(ran) => {
2343                    match em.drain_audio(8_192) {
2344                        Ok(mut chunk) => audio_samples.append(&mut chunk),
2345                        Err(e) => eprintln!("warning: drain_audio mid-run: {e}"),
2346                    }
2347                    if ran < want {
2348                        stopped = true;
2349                        break;
2350                    }
2351                }
2352                Err(luna_api::ApiError::Panic(msg)) => {
2353                    panic_msg = Some(msg);
2354                    break;
2355                }
2356                Err(e) => {
2357                    eprintln!("error: step: {e}");
2358                    return ExitCode::from(1);
2359                }
2360            }
2361        }
2362        // Final drain.
2363        match em.drain_audio(usize::MAX) {
2364            Ok(mut chunk) => audio_samples.append(&mut chunk),
2365            Err(e) => eprintln!("warning: drain_audio final: {e}"),
2366        }
2367    } else {
2368        match em.step(steps) {
2369            Ok(ran) => stopped = ran < steps,
2370            Err(luna_api::ApiError::Panic(msg)) => panic_msg = Some(msg),
2371            Err(e) => {
2372                eprintln!("error: step: {e}");
2373                return ExitCode::from(1);
2374            }
2375        }
2376    }
2377    if stopped {
2378        println!(
2379            "CPU halted by STP after {} instructions.",
2380            em.instructions_executed()
2381        );
2382    }
2383
2384    println!("--- final state ---");
2385    print_cpu_state(&em.state().cpu);
2386    print_diag_state(&mut em);
2387    println!("Instructions executed: {}", em.instructions_executed());
2388    println!("Total master cycles:   {}", em.state().stats.total_mclk);
2389    if let Some(msg) = panic_msg {
2390        println!();
2391        println!("Stopped on CPU panic:");
2392        println!("  {msg}");
2393        // Returning success here: hitting an unimplemented opcode is the
2394        // expected state of P0.6, not a CLI failure.
2395    }
2396
2397    // Screenshot dump: render whatever the PPU has accumulated.
2398    if let Some(out_path) = screenshot {
2399        match save_screenshot(&em, out_path, force_display, bg) {
2400            Ok(()) => println!("\nScreenshot written to {}", out_path.display()),
2401            Err(e) => {
2402                eprintln!("\nerror: could not write screenshot: {e}");
2403                return ExitCode::from(1);
2404            }
2405        }
2406    }
2407    if let Some(out_path) = audio_out {
2408        match write_wav(out_path, &audio_samples) {
2409            Ok(()) => println!(
2410                "Audio WAV written to {}  ({} samples @ 32 kHz stereo, {} s)",
2411                out_path.display(),
2412                audio_samples.len(),
2413                audio_samples.len() as f64 / 32_000.0,
2414            ),
2415            Err(e) => {
2416                eprintln!("\nerror: could not write audio WAV: {e}");
2417                return ExitCode::from(1);
2418            }
2419        }
2420    }
2421    if let Some(path) = nocash_out {
2422        let bytes = em.take_nocash_log().unwrap_or_default();
2423        match std::fs::write(path, &bytes) {
2424            Ok(()) => println!(
2425                "Nocash ($21FC) log written to {}  ({} bytes)",
2426                path.display(),
2427                bytes.len(),
2428            ),
2429            Err(e) => {
2430                eprintln!("\nerror: could not write Nocash log: {e}");
2431                return ExitCode::from(1);
2432            }
2433        }
2434    }
2435    if let Some(path) = wdm_out {
2436        use std::fmt::Write as _;
2437        let hits = em.take_wdm_log().unwrap_or_default();
2438        let mut body = String::new();
2439        for (pc, op) in &hits {
2440            let _ = writeln!(body, "PC=${pc:06X} operand=${op:02X}");
2441        }
2442        match std::fs::write(path, &body) {
2443            Ok(()) => println!(
2444                "WDM log written to {}  ({} hit(s))",
2445                path.display(),
2446                hits.len(),
2447            ),
2448            Err(e) => {
2449                eprintln!("\nerror: could not write WDM log: {e}");
2450                return ExitCode::from(1);
2451            }
2452        }
2453    }
2454    // Stable, cross-arch visual-regression key (hashes the displayed RGBA,
2455    // pre-PNG). Printed last so a harness can `grep '^fbhash='`.
2456    if print_fbhash {
2457        match em.frame_hash(force_display) {
2458            Ok(h) => println!("fbhash={h:016x}"),
2459            Err(e) => {
2460                eprintln!("\nerror: could not hash frame: {e}");
2461                return ExitCode::from(1);
2462            }
2463        }
2464    }
2465    ExitCode::SUCCESS
2466}

2468/// Minimal RIFF/WAVE writer for 16-bit signed PCM stereo at 32 kHz.
2469/// We hand-roll instead of pulling a `hound` dependency just for
2470/// one diagnostic path.
2471fn write_wav(path: &std::path::Path, samples: &[(i16, i16)]) -> std::io::Result<()> {
2472    use std::io::Write;
2473    let mut f = std::fs::File::create(path)?;
2474    let sample_rate: u32 = 32_000;
2475    let channels: u16 = 2;
2476    let bits_per_sample: u16 = 16;
2477    let byte_rate = sample_rate * u32::from(channels) * u32::from(bits_per_sample) / 8;
2478    let block_align = channels * bits_per_sample / 8;
2479    let data_size =
2480        (samples.len() * usize::from(channels) * usize::from(bits_per_sample) / 8) as u32;
2481    let riff_size = 36 + data_size;
2482    f.write_all(b"RIFF")?;
2483    f.write_all(&riff_size.to_le_bytes())?;
2484    f.write_all(b"WAVE")?;
2485    f.write_all(b"fmt ")?;
2486    f.write_all(&16u32.to_le_bytes())?; // PCM chunk size
2487    f.write_all(&1u16.to_le_bytes())?; // PCM format
2488    f.write_all(&channels.to_le_bytes())?;
2489    f.write_all(&sample_rate.to_le_bytes())?;
2490    f.write_all(&byte_rate.to_le_bytes())?;
2491    f.write_all(&block_align.to_le_bytes())?;
2492    f.write_all(&bits_per_sample.to_le_bytes())?;
2493    f.write_all(b"data")?;
2494    f.write_all(&data_size.to_le_bytes())?;
2495    for (l, r) in samples {
2496        f.write_all(&l.to_le_bytes())?;
2497        f.write_all(&r.to_le_bytes())?;
2498    }
2499    Ok(())
2500}

2541    let p = &st.ppu;

2544        p.inidisp,

2551        "PPU:  INIDISP_writes={}  Backdrop=${:04X}",
2552        p.inidisp_write_count, p.backdrop
2553    );

2561            if vram.get(a).copied().unwrap_or(0) != 0 {
2562                nonzero += 1;
2563            }
2564        }

2566            "BG{}:  tile=${:04X} (byte ${:04X})  char=${:04X} (byte ${:04X})  hscroll={} vscroll={}  tilemap_nonzero={}/{}",
2567            i + 1,
2568            bg.tilemap_addr_words,
2569            base,
2570            bg.char_addr_words,
2571            (bg.char_addr_words as usize) << 1,
2572            bg.h_scroll,
2573            bg.v_scroll,
2574            nonzero,
2575            32 * 32 * 2,
2576        );

2579        "CPU regs:  NMITIMEN=${:02X}  HVBJOY=${:02X}  frames={}  NMIs_served={}  ppu_line={}",
2580        st.cpu_regs.nmitimen,
2581        st.cpu_regs.hvbjoy,
2582        st.scheduler.frame_count,

2600    let mvol_l = st.apu.mvol_l;
2601    let mvol_r = st.apu.mvol_r;
2602    let kon = st.apu.kon;

2672            sp.x,
2673            sp.y,
2674            sp.tile,
2675            sp.palette,
2676            sp.priority,
2677            sp.w,
2678            sp.h,
2679            if sp.h_flip { "H" } else { "-" },

2682    }

2688        "VRAM:  {}/65536 non-zero bytes  |  CGRAM: {}/256 non-zero colours",

2692    // @PC bytes need mutable bus access — run after all immutable
2693    // PPU diagnostics are done.
2694    let pc_bytes = em.peek_pc_bytes(8).unwrap_or_default();
2695    print!("@PC bytes:");
2696    for b in &pc_bytes {
2697        print!(" {b:02X}");
2698    }
2699    println!();
2700}

2702fn flag_string(p: u8, e: bool) -> String {
2703    let bit = |mask: u8, c: char, fallback: char| if p & mask != 0 { c } else { fallback };
2704    format!(
2705        "{}{}{}{}{}{}{}{} (e={})",
2706        bit(0b1000_0000, 'N', 'n'),
2707        bit(0b0100_0000, 'V', 'v'),
2708        bit(0b0010_0000, 'M', 'm'),
2709        bit(0b0001_0000, 'X', 'x'),
2710        bit(0b0000_1000, 'D', 'd'),
2711        bit(0b0000_0100, 'I', 'i'),
2712        bit(0b0000_0010, 'Z', 'z'),
2713        bit(0b0000_0001, 'C', 'c'),
2714        u8::from(e),
2715    )
2716}

2718fn save_screenshot(
2719    em: &luna_api::Emulator,
2720    path: &std::path::Path,
2721    force_display: bool,
2722    bg: Option<u8>,
2723) -> Result<(), luna_api::ApiError> {
2724    // Default path (no --bg, no --force-display) copies the persistent
2725    // framebuffer; debug paths (`--force-display` or single-BG render)
2726    // go through the one-shot renderer. All routed through luna-api so
2727    // the CLI and GUI render the exact same pixels.
2728    let png = match bg {
2729        Some(n) => {
2730            let idx = (n.saturating_sub(1).min(3)) as usize;
2731            em.render_frame_bg_png(idx, force_display)?
2732        }
2733        None => em.render_frame_png(force_display)?,
