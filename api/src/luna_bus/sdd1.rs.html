1//! S-DD1 graphics-decompression coprocessor (Star Ocean, Street Fighter
2//! Alpha 2).
3//!
4//! This module hosts the **decompressor** — a faithful port of the canonical
5//! ares / Mesen2 implementation (both byte-identical; see
6//! `docs/sdd1_reference.md`). It is a self-contained streaming state machine:
7//! the chip decompresses graphics on the fly while a DMA reads from it.
8//!
9//! Pipeline: Input Manager (bit reader) → Golomb-Code Decoder → 8 Bits
10//! Generators → Probability Estimation Module → Context Model (bitplane
11//! interleave) → Output Logic (bitplane → byte). The decompressor reads its
12//! compressed input only through the [`Sdd1RomBus`] trait, so it is fully
13//! testable in isolation from the SNES bus.
14//!
15//! [`Sdd1Mapper`] is the whole chip: the MMC (ROM banking + `$4800-$4807`
16//! control registers) plus — wired in a later stage — the DMA-triggered
17//! decompression.
18

117/// The S-DD1 streaming decompressor. One `init` per DMA stream, then one
118/// [`Self::decompress_byte`] per output byte. All six pipeline stages are
119/// flattened into this single struct (idiomatic Rust vs. ares' parent-ref
120/// object graph); the logic is a line-for-line port.
121#[derive(Clone, serde::Serialize, serde::Deserialize)]
122pub struct Sdd1Decompressor {
123    // Input Manager.
124    offset: u32,
125    bit_count: u32,
126    // Bits Generators (8, indexed by code_num).
127    mps_count: [u8; 8],
128    lps_index: [bool; 8],
129    // Probability Estimation Module (32 contexts).
130    context_status: [u8; 32],
131    context_mps: [u8; 32],
132    // Context Model.
133    bitplanes_info: u8,
134    context_bits_info: u8,
135    bit_number: u8,
136    curr_bitplane: u8,
137    prev_bitplane_bits: [u16; 8],
138    // Output Logic.
139    r0: u8,
140    r1: u8,
141    r2: u8,
142}
143
144impl Default for Sdd1Decompressor {
145    fn default() -> Self {
146        Self::new()
147    }

172    /// Begin a new decompression stream at MMC-decoded ROM `offset`. The byte
173    /// at `offset` is the header: bits 7:6 select the bitplane mode, 5:4 the
174    /// context-bit window; the compressed bitstream starts at bit 3 (the Input
175    /// Manager begins at `bit_count = 4`).
176    pub fn init(&mut self, rom: &impl Sdd1RomBus, offset: u32) {
177        let first_byte = rom.mmc_read(offset);
178        // Input Manager.
179        self.offset = offset;
180        self.bit_count = 4;
181        // Bits Generators.
182        self.mps_count = [0; 8];
183        self.lps_index = [false; 8];
184        // Probability Estimation Module.
185        self.context_status = [0; 32];
186        self.context_mps = [0; 32];
187        // Context Model.
188        self.bitplanes_info = first_byte & 0xc0;
189        self.context_bits_info = first_byte & 0x30;
190        self.bit_number = 0;
191        self.prev_bitplane_bits = [0; 8];
192        self.curr_bitplane = match self.bitplanes_info {
193            0x00 => 1,
194            0x40 => 7,
195            0x80 => 3,
196            // 0xc0: derived from `bit_number` per bit, init value unused.
197            _ => 0,
198        };
199        // Output Logic.
200        self.r0 = 1;
201        self.r1 = 0;
202        self.r2 = 0;
203    }
204
205    /// Produce the next decompressed byte (Output Logic). For the 2-bitplane
206    /// modes (`0x00/0x40/0x80`) one call reads 16 bits and returns the low
207    /// plane, the next returns the buffered high plane; for `0xc0` (8 bpp) one
208    /// call reads 8 bits and returns one byte.
209    pub fn decompress_byte(&mut self, rom: &impl Sdd1RomBus) -> u8 {
210        if self.bitplanes_info == 0xc0 {
211            self.r1 = 0;
212            self.r0 = 0x01;
213            while self.r0 != 0 {
214                if self.cm_get_bit(rom) != 0 {
215                    self.r1 |= self.r0;
216                }
217                self.r0 <<= 1;
218            }
219            return self.r1;
220        }
221        // Modes 0x00 / 0x40 / 0x80: two bitplanes, returned across two calls.
222        if self.r0 == 0 {
223            self.r0 = !self.r0;
224            return self.r2;
225        }
226        self.r0 = 0x80;
227        self.r1 = 0;
228        self.r2 = 0;
229        while self.r0 != 0 {
230            if self.cm_get_bit(rom) != 0 {
231                self.r1 |= self.r0;
232            }
233            if self.cm_get_bit(rom) != 0 {
234                self.r2 |= self.r0;
235            }
236            self.r0 >>= 1;
237        }
238        self.r1
239    }
240
241    /// Context Model: pick the bitplane, build the prediction context from the
242    /// plane's recent-bit history, decode one bit, and update the history.
243    fn cm_get_bit(&mut self, rom: &impl Sdd1RomBus) -> u8 {
244        match self.bitplanes_info {
245            0x00 => self.curr_bitplane ^= 0x01,
246            0x40 => {
247                self.curr_bitplane ^= 0x01;
248                if self.bit_number & 0x7f == 0 {
249                    self.curr_bitplane = (self.curr_bitplane + 2) & 0x07;
250                }
251            }
252            0x80 => {
253                self.curr_bitplane ^= 0x01;
254                if self.bit_number & 0x7f == 0 {
255                    self.curr_bitplane ^= 0x02;
256                }
257            }
258            // 0xc0
259            _ => self.curr_bitplane = self.bit_number & 0x07,
260        }
261
262        let idx = usize::from(self.curr_bitplane);
263        let context_bits = self.prev_bitplane_bits[idx];
264        let low = u8::try_from(context_bits & 0x0001).unwrap_or(0);
265        let low2 = u8::try_from(context_bits & 0x0003).unwrap_or(0);
266        let mut context = (self.curr_bitplane & 0x01) << 4;
267        context |= match self.context_bits_info {
268            0x00 => u8::try_from((context_bits & 0x01c0) >> 5).unwrap_or(0) | low,
269            0x10 => u8::try_from((context_bits & 0x0180) >> 5).unwrap_or(0) | low,
270            0x20 => u8::try_from((context_bits & 0x00c0) >> 5).unwrap_or(0) | low,
271            // 0x30
272            _ => u8::try_from((context_bits & 0x0180) >> 5).unwrap_or(0) | low2,
273        };
274
275        let bit = self.pem_get_bit(rom, context);
276        self.prev_bitplane_bits[idx] = (context_bits << 1) | u16::from(bit);
277        self.bit_number = self.bit_number.wrapping_add(1);
278        bit
279    }
280
281    /// Probability Estimation Module: decode one bit for `context`, evolve the
282    /// context's adaptive state on a run end, and apply the MPS inversion.
283    fn pem_get_bit(&mut self, rom: &impl Sdd1RomBus, context: u8) -> u8 {
284        let ctx = usize::from(context);
285        let status = self.context_status[ctx];
286        let mps = self.context_mps[ctx];
287        let state = EVOLUTION_TABLE[usize::from(status)];
288
289        let mut end_of_run = false;
290        let bit = self.bg_get_bit(rom, state.code_num, &mut end_of_run);
291
292        if end_of_run {
293            if bit != 0 {
294                if status & 0xfe == 0 {
295                    self.context_mps[ctx] ^= 0x01;
296                }
297                self.context_status[ctx] = state.next_lps;
298            } else {
299                self.context_status[ctx] = state.next_mps;
300            }
301        }
302
303        bit ^ mps
304    }
305
306    /// Bits Generator `code_num`: expand the current Golomb run into single
307    /// bits (MPS = 0 run, then one LPS = 1), fetching a fresh run when empty.
308    /// `end_of_run` is set when the run is exhausted.
309    fn bg_get_bit(&mut self, rom: &impl Sdd1RomBus, code_num: u8, end_of_run: &mut bool) -> u8 {
310        let i = usize::from(code_num);
311        if self.mps_count[i] == 0 && !self.lps_index[i] {
312            self.gcd_get_run_count(rom, code_num);
313        }
314        let bit = if self.mps_count[i] != 0 {
315            self.mps_count[i] -= 1;
316            0
317        } else {
318            self.lps_index[i] = false;
319            1
320        };
321        *end_of_run = self.mps_count[i] == 0 && !self.lps_index[i];
322        bit
323    }
324
325    /// Golomb-Code Decoder: fetch a codeword and resolve it into a
326    /// most-probable-symbol run count (and whether a least-probable symbol
327    /// terminates it).
328    fn gcd_get_run_count(&mut self, rom: &impl Sdd1RomBus, code_num: u8) {
329        let codeword = self.im_get_codeword(rom, code_num);
330        let i = usize::from(code_num);
331        if codeword & 0x80 != 0 {
332            self.lps_index[i] = true;
333            self.mps_count[i] = RUN_COUNT[usize::from(codeword >> (code_num ^ 0x07))];
334        } else {
335            self.mps_count[i] = 1u8 << code_num;
336        }
337    }
338
339    /// Input Manager: read a `code_len`-bit codeword from the compressed
340    /// bitstream, straddling byte boundaries. The `>> (9 - bit_count)` runs
341    /// over a `u16` to reproduce C++'s integer promotion: a shift of 8 yields
342    /// 0 (a `u8 >> 8` would panic in Rust and `wrapping_shr` would wrongly
343    /// shift by 0).
344    fn im_get_codeword(&mut self, rom: &impl Sdd1RomBus, code_len: u8) -> u8 {
345        let mut codeword = rom.mmc_read(self.offset).wrapping_shl(self.bit_count);
346        self.bit_count += 1;
347        if codeword & 0x80 != 0 {
348            let next = u16::from(rom.mmc_read(self.offset + 1));
349            codeword |= u8::try_from(next >> (9 - self.bit_count)).unwrap_or(0);
350            self.bit_count += u32::from(code_len);
351        }
352        if self.bit_count & 0x08 != 0 {
353            self.offset += 1;
354            self.bit_count &= 0x07;
355        }
356        codeword
357    }
358}

360/// The S-DD1 cartridge mapper: the MMC (ROM banking + `$4800-$4807` control
361/// registers) and SRAM, plus the DMA-triggered graphics decompression
362/// (streaming through [`Sdd1Decompressor`]).
363#[derive(Clone, serde::Serialize, serde::Deserialize)]
364pub struct Sdd1Mapper {
365    rom: Vec<u8>,
366    sram: Vec<u8>,
367    /// `$4800` hard-enable mask (per-channel S-DD1 DMA eligibility).
368    r4800: u8,
369    /// `$4801` soft-enable mask (per-channel arm; chip clears it at stream end).
370    r4801: u8,
371    /// `$4804-$4807` MMC bank selects for `$C0-CF/$D0-DF/$E0-EF/$F0-FF`
372    /// (bit 7 of `$4805`/`$4807` also flag the `$20-3F`/`$A0-BF` ROM mirror).
373    r4804: u8,
374    r4805: u8,
375    r4806: u8,
376    r4807: u8,
377    /// The streaming decompressor (one active stream at a time).
378    decompressor: Sdd1Decompressor,
379    /// Per-channel DMA source address / remaining length, captured by
380    /// observing `$43x2-6` writes — so a decompressing DMA read is recognised
381    /// (`addr == dma_address[ch]`, fixed-mode) and bounded.
382    dma_address: [u32; 8],
383    dma_size: [u16; 8],
384    /// `true` once the current stream's decompressor is initialised; reset
385    /// when a stream's length reaches 0.
386    dma_ready: bool,
387}
388
389/// Borrowed ROM view implementing the decompressor's input read (`mmcRead` —
390/// the `$C0-FF` four-bank MMC decode). Holds `&rom` plus *copies* of the bank
391/// selects, so it can be created while the decompressor field is borrowed
392/// mutably (disjoint from `rom`).
393struct Sdd1RomView<'a> {
394    rom: &'a [u8],
395    banks: [u8; 4],
396}
397
398impl Sdd1RomBus for Sdd1RomView<'_> {
399    fn mmc_read(&self, addr: u32) -> u8 {
400        if self.rom.is_empty() {
401            return 0;
402        }
403        let a = addr & 0x00FF_FFFF;
404        let sel = self.banks[((a >> 20) & 0x03) as usize];
405        let rom_addr = (u32::from(sel & 0x0F) << 20) | (a & 0x000F_FFFF);
406        self.rom[rom_mirror(rom_addr as usize, self.rom.len())]
407    }
408}

410impl Sdd1Mapper {

418            r4801: 0,

420            // `SDD1::power`): $C0-CF→megabit 0, $D0-DF→1, $E0-EF→2, $F0-FF→3.
421            // Games that only reconfigure the high banks ($4806/$4807) rely on
422            // this default for $C0-DF — all-zero made bank $D7 read megabit 0,
423            // so Star Ocean's logo palette (at $D7:EF60) came out cream not
424            // green.
425            r4804: 0,
426            r4805: 1,
427            r4806: 2,
428            r4807: 3,
429            decompressor: Sdd1Decompressor::new(),
430            dma_address: [0; 8],
431            dma_size: [0; 8],
432            dma_ready: false,
433        }
434    }

436    /// S-DD1 control-register read (`$4800-$480F`); `None` for the non-register
437    /// addresses in that window (games never read them → open bus).
438    const fn read_reg(&self, offset: u16) -> Option<u8> {
439        match offset & 0x000F {
440            0x00 => Some(self.r4800),
441            0x01 => Some(self.r4801),
442            0x04 => Some(self.r4804),
443            0x05 => Some(self.r4805),
444            0x06 => Some(self.r4806),
445            0x07 => Some(self.r4807),
446            _ => None,
447        }
448    }

450    /// MMC-decode a 24-bit address to a flat ROM offset — ares `mcuRead`'s
451    /// non-decompression path + `mmcRead`. `None` if the address is not ROM.
452    /// Two regions:
453    /// - `$00-3F/$80-BF : $8000-FFFF` — banked program ROM (`LoROM`-style), with
454    ///   the `$20-3F`/`$A0-BF` mirror controlled by `r4805`/`r4807` bit 7.
455    /// - bit 22 set (`$40-7F` / `$C0-FF`) — MMC-banked data ROM: address bits
456    ///   `[20:21]` pick `r4804-7`, whose low nibble forms ROM bits `[20:23]`.
457    fn rom_addr(&self, addr: Addr24) -> Option<u32> {
458        let a = addr & 0x00FF_FFFF;
459        if a & 0x0040_0000 == 0 {
460            if a & 0x0000_8000 == 0 {
461                return None; // low half of $00-3F/$80-BF is not ROM
462            }
463            let mut a = a;
464            if a & 0x0080_0000 == 0 && a & 0x0020_0000 != 0 && self.r4805 & 0x80 != 0 {
465                a &= !0x0020_0000; // $20-3F mirrors $00-1F
466            }
467            if a & 0x0080_0000 != 0 && a & 0x0020_0000 != 0 && self.r4807 & 0x80 != 0 {
468                a &= !0x0020_0000; // $A0-BF mirrors $80-9F
469            }
470            Some(((a & 0x003F_0000) >> 1) | (a & 0x0000_7FFF))
471        } else {
472            let sel = match (a >> 20) & 0x03 {
473                0 => self.r4804,
474                1 => self.r4805,
475                2 => self.r4806,
476                _ => self.r4807,
477            };
478            Some((u32::from(sel & 0x0F) << 20) | (a & 0x000F_FFFF))
479        }
480    }
481
482    /// SRAM offset (S-DD1 board: `$70-7D:$0000-7FFF`; wraps modulo the actual
483    /// SRAM size).
484    ///
485    /// IMPORTANT: only the `$70-7D` lower half is SRAM. Unlike a plain `LoROM`
486    /// cart, the S-DD1's `$C0-FF` (including the `$F0-FD` LoROM-mirror banks)
487    /// is the MMC-banked **ROM** region — never SRAM. Claiming `$F0-FD` as
488    /// SRAM made every `$F0-FD:0000-7FFF` read return zeroed save-RAM instead
489    /// of the MMC ROM, so Star Ocean's post-intro script engine (it walks a
490    /// linked list of script entries that live in `$FA:xxxx`) read null links
491    /// and dead-looped at `$C0:6C00`. The intro never hit it because it only
492    /// reads `$C0-EF`.
493    fn sram_offset(&self, bank: u8, offset: u16) -> Option<usize> {
494        if self.sram.is_empty() || !matches!(bank, 0x70..=0x7D) || offset >= 0x8000 {
495            return None;
496        }
497        let normalized_bank = (bank & 0x7F) - 0x70;
498        Some((usize::from(normalized_bank) * 0x8000 + usize::from(offset)) % self.sram.len())
499    }
500}
501
502impl Mapper for Sdd1Mapper {
503    fn kind(&self) -> MapperKind {
504        MapperKind::Sdd1
505    }
506
507    fn read(&mut self, addr: Addr24) -> Option<u8> {
508        let bank = bank_of(addr);
509        let offset = offset_of(addr);
510        if matches!(bank, 0x00..=0x3F | 0x80..=0xBF) && (0x4800..=0x480F).contains(&offset) {
511            if let Some(v) = self.read_reg(offset) {
512                return Some(v);
513            }
514        }
515        if let Some(o) = self.sram_offset(bank, offset) {
516            return Some(self.sram[o]);
517        }
518        // DMA-triggered decompression: in the bit-22 region (`$40-7F`/`$C0-FF`),
519        // if an armed channel's fixed source address matches this read, stream
520        // a decompressed byte (ares `mcuRead`). A normal (non-armed) read falls
521        // through to plain MMC ROM below.
522        let a = addr & 0x00FF_FFFF;
523        if a & 0x0040_0000 != 0 {
524            let active = self.r4800 & self.r4801;
525            if active != 0 {
526                for ch in 0..8 {
527                    if active & (1 << ch) != 0 && a == self.dma_address[ch] {
528                        // Build the ROM view inline (borrows `&self.rom`, copies
529                        // the bank regs) so it is disjoint from the
530                        // `&mut self.decompressor` borrow below.
531                        let view = Sdd1RomView {
532                            rom: &self.rom,
533                            banks: [self.r4804, self.r4805, self.r4806, self.r4807],
534                        };
535                        if !self.dma_ready {
536                            self.decompressor.init(&view, a);
537                            self.dma_ready = true;
538                        }
539                        let data = self.decompressor.decompress_byte(&view);
540                        self.dma_size[ch] = self.dma_size[ch].wrapping_sub(1);
541                        if self.dma_size[ch] == 0 {
542                            self.dma_ready = false;
543                            self.r4801 &= !(1 << ch);
544                        }
545                        return Some(data);
546                    }
547                }
548            }
549        }
550        if let Some(rom_addr) = self.rom_addr(addr) {
551            if self.rom.is_empty() {
552                return None;
553            }
554            return Some(self.rom[rom_mirror(rom_addr as usize, self.rom.len())]);
555        }
556        None
557    }
558
559    fn write(&mut self, addr: Addr24, value: u8) -> bool {
560        let bank = bank_of(addr);
561        let offset = offset_of(addr);
562        if matches!(bank, 0x00..=0x3F | 0x80..=0xBF) && (0x4800..=0x480F).contains(&offset) {
563            match offset & 0x000F {
564                0x00 => self.r4800 = value,
565                0x01 => self.r4801 = value,
566                0x04 => self.r4804 = value & 0x8F,
567                0x05 => self.r4805 = value & 0x8F,
568                0x06 => self.r4806 = value & 0x8F,
569                0x07 => self.r4807 = value & 0x8F,
570                _ => {}
571            }
572            return true;
573        }
574        // Observe the SNES DMA controller's `$43x2-6` writes to learn each
575        // channel's source address ($43x2-4) and length ($43x5-6) — needed to
576        // recognise + bound a decompressing DMA. We only observe; the real DMA
577        // controller (luna-core) owns these registers, so don't claim them.
578        if matches!(bank, 0x00..=0x3F | 0x80..=0xBF) && matches!(offset, 0x4300..=0x437F) {
579            let ch = usize::from((offset >> 4) & 0x07);
580            match offset & 0x000F {
581                0x02 => {
582                    self.dma_address[ch] = (self.dma_address[ch] & 0x00FF_FF00) | u32::from(value);
583                }
584                0x03 => {
585                    self.dma_address[ch] =
586                        (self.dma_address[ch] & 0x00FF_00FF) | (u32::from(value) << 8);
587                }
588                0x04 => {
589                    self.dma_address[ch] =
590                        (self.dma_address[ch] & 0x0000_FFFF) | (u32::from(value) << 16);
591                }
592                0x05 => self.dma_size[ch] = (self.dma_size[ch] & 0xFF00) | u16::from(value),
593                0x06 => self.dma_size[ch] = (self.dma_size[ch] & 0x00FF) | (u16::from(value) << 8),
594                _ => {}
595            }
596            return false;
597        }
598        if let Some(o) = self.sram_offset(bank, offset) {
599            self.sram[o] = value;
600            return true;
601        }
602        false
603    }
604
605    fn rom_size(&self) -> usize {
606        self.rom.len()
607    }
608
609    fn sram_size(&self) -> usize {
610        self.sram.len()
611    }
612
613    fn sram(&self) -> &[u8] {
614        &self.sram
615    }
616
617    fn load_sram(&mut self, data: &[u8]) {
618        let n = data.len().min(self.sram.len());
619        self.sram[..n].copy_from_slice(&data[..n]);
620    }
621
622    fn reset(&mut self) {
623        // Control registers + any in-flight decompression clear on reset;
624        // battery SRAM persists.
625        self.r4800 = 0;
626        self.r4801 = 0;
627        // Identity MMC mapping on reset (ares `SDD1::power`), not all-zero.
628        self.r4804 = 0;
629        self.r4805 = 1;
630        self.r4806 = 2;
631        self.r4807 = 3;
632        self.dma_address = [0; 8];
633        self.dma_size = [0; 8];
634        self.dma_ready = false;
635        self.decompressor = Sdd1Decompressor::new();
636    }
637
638    fn save_state(&self) -> Vec<u8> {
639        bincode::serialize(&(
640            &self.sram,
641            self.r4800,
642            self.r4801,
643            self.r4804,
644            self.r4805,
645            self.r4806,
646            self.r4807,
647            self.dma_address,
648            self.dma_size,
649            self.dma_ready,
650            &self.decompressor,
651        ))
652        .unwrap_or_default()
653    }
654
655    fn load_state(&mut self, data: &[u8]) {
656        type State = (
657            Vec<u8>,
658            u8,
659            u8,
660            u8,
661            u8,
662            u8,
663            u8,
664            [u32; 8],
665            [u16; 8],
666            bool,
667            Sdd1Decompressor,
668        );
669        if let Ok((
670            sram,
671            r4800,
672            r4801,
673            r4804,
674            r4805,
675            r4806,
676            r4807,
677            dma_addr,
678            dma_size,
679            ready,
680            dec,
681        )) = bincode::deserialize::<State>(data)
682        {
683            self.sram = sram;
684            self.r4800 = r4800;
685            self.r4801 = r4801;
686            self.r4804 = r4804;
687            self.r4805 = r4805;
688            self.r4806 = r4806;
689            self.r4807 = r4807;
690            self.dma_address = dma_addr;
691            self.dma_size = dma_size;
692            self.dma_ready = ready;
693            self.decompressor = dec;
694        }
695    }
696}

839    #[test]
840    fn mapper_dma_decompression_arms_streams_and_disarms() {
841        // 8 MB ramp ROM; put a compressed header + payload at a $C0 address.
842        let mut rom: Vec<u8> = (0..0x80_0000).map(|i| (i & 0xFF) as u8).collect();
843        // r4804 = 0 → $C0:xxxx decodes to ROM 0x00_xxxx. Header 0xC0 = 8 bpp.
844        rom[0x0000] = 0xC0;
845        let mut m = Sdd1Mapper::new(rom, 0x2000);
846
847        // Game protocol: set DMA channel-0 source ($C0:0000) + length (4),
848        // then arm S-DD1 on channel 0 (hard + soft enable).
849        m.write(make_addr(0x00, 0x4302), 0x00); // addr byte 0
850        m.write(make_addr(0x00, 0x4303), 0x00); // addr byte 1
851        m.write(make_addr(0x00, 0x4304), 0xC0); // addr byte 2 → $C0:0000
852        m.write(make_addr(0x00, 0x4305), 0x04); // length lo = 4
853        m.write(make_addr(0x00, 0x4306), 0x00); // length hi
854        m.write(make_addr(0x00, 0x4800), 0x01); // hard-enable ch0
855        m.write(make_addr(0x00, 0x4801), 0x01); // soft-enable ch0
856
857        // 4 armed reads at the fixed source → decompressed (not raw ROM=0xC0).
858        let out: Vec<u8> = (0..4)
859            .map(|_| m.read(make_addr(0xC0, 0x0000)).unwrap())
860            .collect();
861        // Stream exhausted → S-DD1 clears the soft-enable bit.
862        assert_eq!(
863            m.read(make_addr(0x00, 0x4801)),
864            Some(0x00),
865            "r4801 ch0 cleared"
866        );
867
868        // Deterministic: re-arm + re-read yields the same bytes.
869        m.write(make_addr(0x00, 0x4305), 0x04);
870        m.write(make_addr(0x00, 0x4801), 0x01);
871        let out2: Vec<u8> = (0..4)
872            .map(|_| m.read(make_addr(0xC0, 0x0000)).unwrap())
873            .collect();
874        assert_eq!(out, out2);
875
876        // A read NOT matching an armed DMA returns plain MMC ROM: after the
877        // second stream finished r4801 is clear, so $C0:0010 yields raw 0x10.
878        assert_eq!(m.read(make_addr(0xC0, 0x0010)), Some(0x10));
879    }

896    fn high_banks_f0_fd_read_mmc_rom_not_sram() {
897        // Regression: the S-DD1's `$C0-FF` (including the `$F0-FD` LoROM-mirror
898        // banks) is the MMC-banked ROM region — never SRAM. A `$F0-FD:0000-7FFF`
899        // read must return MMC ROM, not zeroed save-RAM. Star Ocean's post-intro
900        // script engine walks a linked list of script entries in `$FA:xxxx`;
901        // mapping those banks to (zeroed) SRAM returned null links and dead-looped
902        // the script interpreter at `$C0:6C00`. The intro never hit it (it only
903        // reads `$C0-EF`).
904        let mut m = mapper_with_ramp_rom(0x60_0000); // 6 MiB, rom[i] = i & 0xFF
905        // `$4807` selects the 1 MiB block mapped at `$F0-FF`; pick block 5.
906        m.write(make_addr(0x00, 0x4807), 0x05);
907        // Poison the real SRAM ($70) with a sentinel the bogus path would echo.
908        m.write(make_addr(0x70, 0x0003), 0xAB);
909        // `$FA:0003` → block 5 → ROM offset 0x5A0003 → ramp byte 0x03, i.e. the
910        // MMC ROM — NOT the SRAM sentinel and NOT zero.
911        assert_eq!(m.read(make_addr(0xFA, 0x0003)), Some(0x03));
912        // Writes to the ROM region are ignored (no SRAM aliasing).
913        m.write(make_addr(0xFA, 0x0003), 0x99);
914        assert_eq!(m.read(make_addr(0xFA, 0x0003)), Some(0x03));
915        // The genuine SRAM at `$70` is untouched and still reachable.
916        assert_eq!(m.read(make_addr(0x70, 0x0003)), Some(0xAB));
917    }

921        // ROM byte = its megabit index, so a read reveals which megabit the

932        assert_eq!(m.read(make_addr(0x00, 0x4807)), Some(3));
