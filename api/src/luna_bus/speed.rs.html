1//! SNES memory access speed lookup.
2//!
3//! Each region of the SNES memory map is accessed at one of three speeds.
4//! The cost is in master clock cycles per byte access, and is paid via
5//! [`crate::Bus::io_cycle`] at every read or write.
6//!
7//! References: <https://problemkaputt.de/fullsnes.htm> §"SNES Memory Map"
8//! and §"SNES Mem Speed (`FastROM` / `SlowROM`)".
9
10use crate::types::{Addr24, MCycles, bank_of, offset_of};
11
12/// SNES memory access speed for a single byte access.
13#[derive(Debug, Clone, Copy, PartialEq, Eq)]
14pub enum MemorySpeed {
15    /// 6 master cycles. ROM in banks `$80-$BF` / `$C0-$FF` at `$8000-$FFFF`
16    /// if the `FastROM` `MEMSEL` bit is set, and a few other fast regions.
17    Fast,
18    /// 8 master cycles. Default speed for most of the memory map (WRAM,
19    /// MMIO, SRAM, ROM in slow banks).
20    Slow,
21    /// 12 master cycles. Joypad-controller registers `$4016-$4017`.
22    XSlow,
23}
24
25impl MemorySpeed {
26    /// Master-cycle cost for one byte at this speed.
27    #[inline]
28    #[must_use]
29    pub const fn mcycles(self) -> MCycles {
30        match self {
31            Self::Fast => 6,
32            Self::Slow => 8,
33            Self::XSlow => 12,
34        }
35    }
36}
37
38/// Look up the access speed for a given 24-bit address.
39///
40/// `fast_rom` indicates whether the `FastROM` `MEMSEL` bit is currently set
41/// (register `$420D`): when true, ROM accesses in banks `$80-$FF` at
42/// `$8000-$FFFF` are FAST (6 mclk) instead of SLOW (8 mclk).
43///
44/// This function is timing-oriented only — it does **not** decide whether
45/// a region is RAM, ROM, MMIO, etc. The bus implementation handles that
46/// routing separately.
47#[must_use]
48pub const fn address_speed(addr: Addr24, fast_rom: bool) -> MemorySpeed {
49    let bank = bank_of(addr);
50    let offset = offset_of(addr);
51
52    match bank {
53        // System area in banks $00-$3F (and mirror $80-$BF). Sub-region
54        // costs are a faithful port of ares `CPU::wait` (sfc/cpu/memory.cpp):
55        // `$2000-$3FFF` and `$4200-$5FFF` are FAST (6); `$4000-$41FF` is
56        // XSLOW (12); the WRAM mirror and `$6000-$7FFF` are SLOW (8).
57        0x00..=0x3F | 0x80..=0xBF => match offset {
58            // LowRAM mirror.
59            0x0000..=0x1FFF => MemorySpeed::Slow,
60            // B-bus I/O: PPU ($21xx), APU ($2140-3), WRAM port ($2180-3).
61            0x2000..=0x3FFF => MemorySpeed::Fast,
62            // Joypad + auto-joypad strobe region (incl. $4016/$4017).
63            0x4000..=0x41FF => MemorySpeed::XSlow,
64            // CPU / DMA / IRQ / hardware-math registers ($42xx-$43xx).
65            0x4200..=0x5FFF => MemorySpeed::Fast,
66            // Open bus / expansion.
67            0x6000..=0x7FFF => MemorySpeed::Slow,
68            // ROM cartridge area.
69            0x8000..=0xFFFF => {
70                if fast_rom && bank >= 0x80 {
71                    MemorySpeed::Fast
72                } else {
73                    MemorySpeed::Slow
74                }
75            }
76        },
77
78        // HiROM / mid-banks: ROM (banks $40-$7D and mirror $C0-$FD).
79        0x40..=0x7D => MemorySpeed::Slow,
80        0xC0..=0xFD => {
81            if fast_rom {
82                MemorySpeed::Fast
83            } else {
84                MemorySpeed::Slow
85            }
86        }
87
88        // WRAM at banks $7E-$7F: always SLOW.
89        0x7E..=0x7F => MemorySpeed::Slow,
90
91        // Banks $FE-$FF: ROM, FastROM-eligible.
92        0xFE..=0xFF => {
93            if fast_rom {
94                MemorySpeed::Fast
95            } else {
96                MemorySpeed::Slow
97            }
98        }
99    }
100}

102#[cfg(test)]
103mod tests {
104    use super::*;
105    use crate::types::make_addr;
106
107    #[test]
108    fn fast_rom_only_kicks_in_above_0x80() {
109        assert_eq!(
110            address_speed(make_addr(0x00, 0x8000), true),
111            MemorySpeed::Slow
112        );
113        assert_eq!(
114            address_speed(make_addr(0x80, 0x8000), true),
115            MemorySpeed::Fast
116        );
117        assert_eq!(
118            address_speed(make_addr(0x80, 0x8000), false),
119            MemorySpeed::Slow
120        );
121    }
122
123    #[test]
124    fn joypad_registers_are_xslow() {
125        assert_eq!(
126            address_speed(make_addr(0x00, 0x4016), false),
127            MemorySpeed::XSlow
128        );
129        assert_eq!(
130            address_speed(make_addr(0x00, 0x4017), false),
131            MemorySpeed::XSlow
132        );
133        // Mirror in $80-$BF banks.
134        assert_eq!(
135            address_speed(make_addr(0x80, 0x4016), false),
136            MemorySpeed::XSlow
137        );
138        // FastROM bit shouldn't override the XSLOW joypad cost.
139        assert_eq!(
140            address_speed(make_addr(0x80, 0x4016), true),
141            MemorySpeed::XSlow
142        );
143    }
144
145    #[test]
146    fn wram_is_always_slow() {
147        assert_eq!(
148            address_speed(make_addr(0x7E, 0x0000), true),
149            MemorySpeed::Slow
150        );
151        assert_eq!(
152            address_speed(make_addr(0x7F, 0xFFFF), true),
153            MemorySpeed::Slow
154        );
155    }
156
157    #[test]
158    fn io_region_speeds_match_ares_wait() {
159        // $2000-$3FFF (B-bus I/O: PPU/APU/WRAM port) is FAST (6) — ares
160        // `CPU::wait` `address - 0x4000 & 0x7e00`. The $2118/$2119 VRAM
161        // ports live here; charging SLOW (8) slowed CPU VRAM uploads ~33%
162        // per access and drifted GSU launches by a frame.
163        assert_eq!(
164            address_speed(make_addr(0x00, 0x2100), false),
165            MemorySpeed::Fast
166        );
167        assert_eq!(
168            address_speed(make_addr(0x00, 0x2118), false),
169            MemorySpeed::Fast
170        );
171        assert_eq!(
172            address_speed(make_addr(0x00, 0x213F), false),
173            MemorySpeed::Fast
174        );
175        // $4000-$41FF (joypad / auto-joypad) is XSLOW (12).
176        assert_eq!(
177            address_speed(make_addr(0x00, 0x4000), false),
178            MemorySpeed::XSlow
179        );
180        assert_eq!(
181            address_speed(make_addr(0x00, 0x41FF), false),
182            MemorySpeed::XSlow
183        );
184        // $4200-$5FFF (CPU/DMA/IRQ/math regs) is FAST (6).
185        assert_eq!(
186            address_speed(make_addr(0x00, 0x4200), false),
187            MemorySpeed::Fast
188        );
189        assert_eq!(
190            address_speed(make_addr(0x00, 0x420B), false),
191            MemorySpeed::Fast
192        );
193        assert_eq!(
194            address_speed(make_addr(0x80, 0x4300), false),
195            MemorySpeed::Fast
196        );
197        // $0000-$1FFF and $6000-$7FFF stay SLOW (8).
198        assert_eq!(
199            address_speed(make_addr(0x00, 0x0000), false),
200            MemorySpeed::Slow
201        );
202        assert_eq!(
203            address_speed(make_addr(0x00, 0x6000), false),
204            MemorySpeed::Slow
205        );
206    }

208    #[test]
209    fn mcycles_lookup_matches_spec() {
210        assert_eq!(MemorySpeed::Fast.mcycles(), 6);
211        assert_eq!(MemorySpeed::Slow.mcycles(), 8);
212        assert_eq!(MemorySpeed::XSlow.mcycles(), 12);
213    }
