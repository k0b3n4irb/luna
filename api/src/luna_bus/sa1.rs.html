1//! SA-1 (Super Accelerator 1) cartridge mapping.
2//!
3//! The SA-1 is Nintendo's custom co-processor used by Super Mario RPG,
4//! DKC 2/3, Kirby Super Star, and ~25 other titles. Internally it's a
5//! 65C816 running at 10.74 MHz alongside the main CPU, plus banks of
6//! shared RAM, a DMA controller, character-conversion hardware, a
7//! hardware multiplier / divider / accumulator, and a complex ROM
8//! banking scheme.
9//!
10//! This module owns the shared cartridge memory + MMIO register file
11//! seen by both CPUs: ROM banking, I-RAM, BW-RAM, hardware
12//! multiplier / divider / accumulator, IRQ message latches, timer,
13//! normal-mode DMA, CC1 + CC2 character-conversion DMA, VLBP
14//! bit-stream reader, and the per-side I-RAM / BW-RAM
15//! write-protection masks. The SA-1's own 65C816 instance is
16//! layered on top in [`luna_coproc::Sa1Chip`].
17//!
18//! Reference: <https://problemkaputt.de/fullsnes.htm> §"SNES SA-1".
19//!
20//! # Mapping
21//!
22//! Four "super-bank" registers select which 1 MB of ROM is visible in
23//! each quarter of the CPU's 24-bit address space:
24//!
25//! - `$2220 CXB` — banks `$00-$1F` and `$80-$9F`'s upper half
26//! - `$2221 DXB` — banks `$20-$3F` and `$A0-$BF`'s upper half
27//! - `$2222 EXB` — banks `$40-$5F` (linear)
28//! - `$2223 FXB` — banks `$60-$7D` (linear)
29//!
30//! Within each LoROM-style super-bank region the byte mapping is the
31//! standard "32 KB at `$8000-$FFFF`, mirrored across 32 banks" used
32//! by the `LoROM` mapper, scaled by the super-bank offset (`bank << 20`
33//! into ROM).
34//!
35//! BW-RAM (up to 256 KB) appears as the cart's SRAM window at
36//! `$00-$3F:$6000-$7FFF` (an 8 KB sliding window selected by
37//! `$2224 BMAPS`) and linearly at `$40-$4F:$0000-$FFFF` (the
38//! contiguous 256 KB view). I-RAM (2 KB, shared with the SA-1 CPU)
39//! appears at `$00-$3F:$3000-$37FF`.

261/// Decode `bpp` source bytes (one bitmap pixel-row's worth) into SNES
262/// planar format inside `tile_buf` at the slot for `row`.
263///
264/// `tile_buf` is a 64-byte working buffer covering up to 8bpp; for
265/// shallower colour depths only the relevant prefix is touched.
266/// Plane interleaving matches the SNES tile layout — bp0/bp1 in
267/// bytes 0..16, bp2/bp3 in 16..32, bp4/bp5 in 32..48, bp6/bp7 in
268/// 48..64.
269///
270/// Shared by Type-1 ([`Sa1Mapper::run_cc1_dma`]) and Type-2
271/// ([`Sa1Mapper::cc2_consume_byte`]) character-conversion DMA.
272fn planarize_row(bpp: usize, src: &[u8], row: usize, tile_buf: &mut [u8; 64]) {
273    let mut pixels = [0u8; 8];
274    match bpp {
275        2 => {
276            for byte_idx in 0..2 {
277                let b = src[byte_idx];
278                pixels[byte_idx * 4] = (b >> 6) & 0x03;
279                pixels[byte_idx * 4 + 1] = (b >> 4) & 0x03;
280                pixels[byte_idx * 4 + 2] = (b >> 2) & 0x03;
281                pixels[byte_idx * 4 + 3] = b & 0x03;
282            }
283        }
284        4 => {
285            for byte_idx in 0..4 {
286                let b = src[byte_idx];
287                pixels[byte_idx * 2] = b >> 4;
288                pixels[byte_idx * 2 + 1] = b & 0x0F;
289            }
290        }
291        _ => {
292            for (i, b) in src.iter().enumerate().take(8) {
293                pixels[i] = *b;
294            }
295        }
296    }
297    for plane in 0..bpp {
298        let mut planar = 0u8;
299        for (col, p) in pixels.iter().enumerate() {
300            let bit = (p >> plane) & 1;
301            planar |= bit << (7 - col);
302        }
303        let plane_group = plane / 2;
304        let plane_inside = plane % 2;
305        let byte_off = plane_group * 16 + row * 2 + plane_inside;
306        tile_buf[byte_off] = planar;
307    }
308}

422    const fn iram_writable_for(&self, byte_off: usize, side: WriteSide) -> bool {
423        let mask = match side {
424            WriteSide::Main => self.siwp,
425            WriteSide::Sa1 => self.ciwp,
426        };
427        let page = (byte_off / 256) & 7;
428        (mask >> page) & 1 != 0
429    }

431    fn bwram_writable_for(&self, byte_off: usize, _side: WriteSide) -> bool {
432        // Per ares (`coprocessor/sa1/bwram.cpp:40-43, 73-84`) and
433        // Mesen2 (`CpuBwRamHandler.h:45-57`, `Sa1BwRamHandler.h:41-50`):
434        // BWRAM writes are gated by the **OR** of SBWE and CBWE bit 7,
435        // not each side independently. If EITHER enable is set, the
436        // write goes through from BOTH sides. Only when both are
437        // cleared does BWPA's protected first-region check kick in,
438        // and even then writes OUTSIDE the protected area still
439        // succeed. ares quotes Kirby's Dream Land 3 as the witness
440        // game: `BWPA=$02, SWEN=$80, CWEN=$00`, SA-1 writes to
441        // `$4001Ax`/`$40032x` must succeed.
442        //
443        // luna previously treated SBWE/CBWE as per-side hard gates;
444        // SMRPG (which never writes SBWE/CBWE during init, trusting
445        // the real-hardware default) had every main-CPU BWRAM write
446        // silently dropped, deadlocking the main↔SA-1 mailbox at
447        // `$40:3D00`. The reset-time backing-mmio default of `$00`
448        // also made the broken-gate case the default, so the bug
449        // affected most SA-1 carts.
450        //
451        // BWPA size formula `0x100 << min(bwp, 10)` matches Mesen2's
452        // clamp (max 256 KiB protection); ares uses the raw 4-bit
453        // value but no real cart sets `bwp > 10` so both behave the
454        // same in practice. `_side` is kept in the signature for
455        // call-site clarity but ignored — the gate is symmetric.
456        if (self.sbwe & 0x80) != 0 || (self.cbwe & 0x80) != 0 {
457            return true;
458        }
459        let bwp = (self.bwpa & 0x0F).min(0x0A);
460        let prot_bytes = 0x100usize << bwp;
461        byte_off >= prot_bytes
462    }

464    /// Effective bit-length for VLBP reads. VBD bits 0..3 carry the
465    /// length 1..15; an encoded value of 0 means 16.
466    fn vlbp_vlen(&self) -> u32 {
467        let n = u32::from(self.vbd & 0x0F);
468        if n == 0 { 16 } else { n }
469    }

471    /// Pull the current `vlen` bits from the source stream as a 16-bit
472    /// value. Reads three source bytes so a `vbit_offset` of up to 7
473    /// can still slide a 16-bit window cleanly.
474    fn vlbp_data(&mut self) -> u16 {
475        let byte_off = self.vbit_offset / 8;
476        let bit_off = self.vbit_offset & 7;
477        let base = self.vda_base;
478        let a0 = base.wrapping_add(byte_off) & 0x00FF_FFFF;
479        let a1 = base.wrapping_add(byte_off + 1) & 0x00FF_FFFF;
480        let a2 = base.wrapping_add(byte_off + 2) & 0x00FF_FFFF;
481        let b0 = u32::from(self.read(a0).unwrap_or(0));
482        let b1 = u32::from(self.read(a1).unwrap_or(0));
483        let b2 = u32::from(self.read(a2).unwrap_or(0));
484        let raw = b0 | (b1 << 8) | (b2 << 16);
485        let shifted = raw >> bit_off;
486        let vlen = self.vlbp_vlen();
487        let mask = if vlen >= 16 {
488            0xFFFFu32
489        } else {
490            (1u32 << vlen) - 1
491        };
492        (shifted & mask) as u16
493    }

495    /// Advance the VLBP bit cursor by `vlen` bits.
496    fn vlbp_advance(&mut self) {
497        self.vbit_offset = self.vbit_offset.wrapping_add(self.vlbp_vlen());

501    /// bytes from `sda` to `dda` through the regular bus dispatch so
502    /// ROM / BW-RAM / I-RAM source-destination combinations all work.
503    ///
504    /// At completion clears the DMA-enable bit + raises the DMA-IRQ
505    /// latch (gated by `CIE.4`).
506    fn run_normal_dma(&mut self) {
507        let n = self.dtc as usize;
508        for i in 0..n {
509            let s = (self.sda.wrapping_add(i as u32)) & 0x00FF_FFFF;
510            let d = (self.dda.wrapping_add(i as u32)) & 0x00FF_FFFF;
511            let byte = self.read(s).unwrap_or(0xFF);
512            // Use the raw byte path so MMIO doesn't try to interpret
513            // our DMA bursts as register writes.
514            self.write_raw_for_dma(d, byte);
515        }
516        self.dcnt &= 0x7F;
517        // Mirror into the memory-backed copy at $2230.
518        self.mmio[0x2230 - 0x2200] = self.dcnt;
519        self.dma_irq_to_sa1 = true;
520    }

522    /// Bypass the MMIO write path during DMA — we never want a DMA
523    /// stream into the `$2200-$23FF` window to start re-triggering DMA
524    /// or rebanking ROM mid-burst. The destination is restricted by
525    /// `DCNT.1-0` to BW-RAM / I-RAM, so a direct write is correct.
526    fn write_raw_for_dma(&mut self, addr: u32, value: u8) {
527        let bank = bank_of(addr);
528        let offset = offset_of(addr);
529        if let Some(o) = Self::iram_offset(bank, offset) {
530            self.iram[o] = value;
531            return;
532        }
533        if let Some(o) = self.bwram_offset(bank, offset, WriteSide::Main) {
534            self.bwram[o] = value;
535        }
536    }

538    /// Arm a Type-2 character-conversion stream. Captures the current
539    /// `cdma` colour mode and resets the staging buffer; subsequent
540    /// writes to BBF (`$223F`) feed pixel bytes in scanline order.

543            1 => 4,
544            2 => 2,

560    /// Consume one source pixel byte streamed by the SA-1 CPU into
561    /// BBF. When `bpp × 8` bytes have been staged we have a complete
562    /// 8×8 tile worth of source data — planarize it, write it to
563    /// `DDA`, auto-increment DDA, raise the CC IRQ latch, and reset
564    /// the staging buffer for the next tile.
565    fn cc2_consume_byte(&mut self, byte: u8) {
566        if !self.cc2_active {
567            return;
568        }
569        let bytes_per_tile = self.cc2_bpp * 8;
570        self.cc2_buf[self.cc2_byte_idx] = byte;
571        self.cc2_byte_idx += 1;
572        if self.cc2_byte_idx < bytes_per_tile {
573            return;
574        }
575        // Tile complete — planarize row-by-row.
576        let bpp = self.cc2_bpp;
577        let buf = self.cc2_buf;
578        let mut planar = [0u8; 64];
579        for row in 0..8 {
580            let src = &buf[row * bpp..(row + 1) * bpp];
581            planarize_row(bpp, src, row, &mut planar);
582        }
583        // Write planar tile to DDA, advance DDA.
584        let dst_base = self.dda & 0x00FF_FFFF;
585        for (k, b) in planar.iter().enumerate().take(bytes_per_tile) {
586            self.write_raw_for_dma(dst_base.wrapping_add(k as u32) & 0x00FF_FFFF, *b);
587        }
588        self.dda = self.dda.wrapping_add(bytes_per_tile as u32) & 0x00FF_FFFF;
589        self.cc2_byte_idx = 0;
590        self.cc1_irq_to_main = true;
591    }

593    /// Run a Type-1 Character-Conversion DMA — convert a linear
594    /// `bpp`-packed bitmap at SDA into SNES planar tile data at DDA.
595    /// Tile data is laid out in raster (left-to-right, top-to-bottom)
596    /// order so destination consumers can stream it straight into a
597    /// VRAM-bound DMA channel.
598    ///
599    /// CDMA layout:
600    ///   * bits 4..2 = colour mode (`0`=8bpp, `1`=4bpp, `2`=2bpp)
601    ///   * bits 1..0 = virtual bitmap width
602    ///     (`0`=8 / `1`=16 / `2`=32 / `3`=64 tiles wide)
603    ///
604    /// Source pixel-byte packing (per Anomie's SA-1 reference):
605    ///   * 8bpp — 1 byte/pixel
606    ///   * 4bpp — 2 pixels/byte, leftmost pixel in high nibble
607    ///   * 2bpp — 4 pixels/byte, leftmost pixel in bits 7..6
608    ///
609    /// DTC defines the number of *output* bytes to produce; we floor
610    /// that to a whole-tile multiple.
611    ///
612    /// On completion: clears DCNT.7, raises `cc1_irq_to_main`.
613    fn run_cc1_dma(&mut self) {
614        let bpp = match (self.cdma >> 2) & 0x07 {
615            1 => 4,
616            2 => 2,
617            _ => 8, // 0 + reserved (3..=7) → 8bpp rather than panic
618        };
619        let tile_width_tiles: u32 = 8u32 << (self.cdma & 0x03);
620        let bytes_per_tile: u32 = (bpp as u32) * 8;
621        let total_out = u32::from(self.dtc);
622        let num_tiles = total_out / bytes_per_tile;
623        // Pixel-row stride within the source bitmap (bytes per row).
624        let row_stride = tile_width_tiles * (bpp as u32);
625
626        for tile_idx in 0..num_tiles {
627            let tile_col = tile_idx % tile_width_tiles;
628            let tile_row = tile_idx / tile_width_tiles;
629            let mut tile_buf = [0u8; 64];
630
631            for row in 0..8 {
632                let pixel_row = (tile_row as usize) * 8 + row;
633                let src_row_off = pixel_row * (row_stride as usize) + (tile_col as usize) * bpp;
634
635                // Pull `bpp` source bytes for one pixel-row of one tile.
636                let mut src_bytes = [0u8; 8];
637                for (k, slot) in src_bytes.iter_mut().enumerate().take(bpp) {
638                    let a = self
639                        .sda
640                        .wrapping_add(u32::try_from(src_row_off + k).unwrap_or(0))
641                        & 0x00FF_FFFF;
642                    *slot = self.read(a).unwrap_or(0);
643                }
644                planarize_row(bpp, &src_bytes[..bpp], row, &mut tile_buf);
645            }
646
647            // Spill the converted tile to the destination buffer.
648            let tile_dst_base =
649                self.dda.wrapping_add(tile_idx.wrapping_mul(bytes_per_tile)) & 0x00FF_FFFF;
650            for (k, byte) in tile_buf.iter().enumerate().take(bytes_per_tile as usize) {
651                self.write_raw_for_dma(tile_dst_base.wrapping_add(k as u32) & 0x00FF_FFFF, *byte);
652            }
653        }
654
655        self.dcnt &= 0x7F;
656        self.mmio[0x2230 - 0x2200] = self.dcnt;
657        self.cc1_irq_to_main = true;
658    }

660    /// HCNT/VCNT compare values (9-bit, in dots) from their lo/hi pairs.
661    fn timer_compare(&self) -> (u16, u16) {
662        let hcnt = ((u16::from(self.hcnt_hi) << 8) | u16::from(self.hcnt_lo)) & 0x01FF;
663        let vcnt = ((u16::from(self.vcnt_hi) << 8) | u16::from(self.vcnt_lo)) & 0x01FF;
664        (hcnt, vcnt)
665    }

667    /// Advance the SA-1 timer by `ticks` master clocks, in ares' exact
668    /// 2-clock steps (`SA1::step`). Both HV and linear modes keep their own
669    /// H/V counters; only the advance differs, the IRQ compare is the same
670    /// `hen | ven<<1` switch in either mode. `timer_rem` carries any odd
671    /// clock between calls so the cadence is independent of batch size.
672    pub fn tick_timer(&mut self, ticks: u32) {
673        self.timer_rem += ticks;
674        while self.timer_rem >= 2 {
675            self.timer_rem -= 2;
676            self.timer_step2();
677        }
678    }

680    /// One ares timer step (+2 master clocks): advance the counters per the
681    /// selected mode, then test the H/V compare for the timer IRQ.
682    fn timer_step2(&mut self) {
683        if self.tmc & 0x80 == 0 {
684            // HV timer — H wraps each scanline (1364 clocks), V each frame.
685            self.hcounter += 2;
686            if self.hcounter >= 1364 {
687                self.hcounter = 0;
688                self.vcounter += 1;
689                if self.vcounter >= self.scanlines {
690                    self.vcounter = 0;
691                }
692            }
693        } else {
694            // Linear timer — an 11-bit H feeding a 9-bit V.
695            self.hcounter += 2;
696            self.vcounter = self.vcounter.wrapping_add(self.hcounter >> 11);
697            self.hcounter &= 0x07FF;
698            self.vcounter &= 0x01FF;
699        }
700        // Timer IRQ compare. Mode = hen | ven<<1 = TMC[1:0]. The flag stays
701        // set (level) until the SA-1 clears it via $220B; the bare `==` fires
702        // once per H period (line) / V period (frame), like ares.
703        let (hcnt, vcnt) = self.timer_compare();
704        match self.tmc & 0x03 {
705            1 if self.hcounter == hcnt << 2 => self.timer_irq_to_sa1 = true,
706            2 if self.vcounter == vcnt && self.hcounter == 0 => self.timer_irq_to_sa1 = true,
707            3 if self.vcounter == vcnt && self.hcounter == hcnt << 2 => {
708                self.timer_irq_to_sa1 = true;
709            }
710            _ => {}
711        }
712    }

837    /// Translate a CPU-side ROM access through the four super-bank
838    /// registers into a linear byte offset into the ROM vector.
839    /// Returns `None` if the address doesn't fall in a ROM region.
840    ///
841    /// SA-1 banking (per ares' `rom.cpp` + Mesen2's `UpdatePrgRomMappings`):
842    ///
843    /// * `$00-1F:8000-FFFF` + `$C0-CF:0000-FFFF` → bank C (CXB / `Banks[0]`)
844    /// * `$20-3F:8000-FFFF` + `$D0-DF:0000-FFFF` → bank D (DXB / `Banks[1]`)
845    /// * `$80-9F:8000-FFFF` + `$E0-EF:0000-FFFF` → bank E (EXB / `Banks[2]`)
846    /// * `$A0-BF:8000-FFFF` + `$F0-FF:0000-FFFF` → bank F (FXB / `Banks[3]`)
847    ///
848    /// Each super-bank register has bit 7 = "remap mode" flag:
849    /// * Cleared → `LoROM` half-bank windows use the *default* MB
850    ///   slot for that bank (Banks[0]→MB0, [1]→MB1, [2]→MB2, [3]→MB3).
851    /// * Set → `LoROM` half-bank windows use the MB selected by bits
852    ///   0..2 of the bank register.
853    ///
854    /// The `HiROM` full-bank windows (`$C0+`) always use the banked MB
855    /// regardless of the mode flag.
856    fn rom_offset(&self, bank: u8, offset: u16) -> Option<usize> {
857        const MB: usize = 0x10_0000;
858        // Identify which super-bank region this address belongs to, the
859        // local bank index within that region (0..31 for LoROM windows,
860        // 0..15 for HiROM windows), and whether it's the LoROM half-bank
861        // ($8000-$FFFF) or the HiROM full-bank view.
862        let (reg, default_mb, local_bank, is_lo) = match bank {
863            0x00..=0x1F if offset >= 0x8000 => (self.cxb, 0u8, bank, true),
864            0x20..=0x3F if offset >= 0x8000 => (self.dxb, 1u8, bank - 0x20, true),
865            0x80..=0x9F if offset >= 0x8000 => (self.exb, 2u8, bank - 0x80, true),
866            0xA0..=0xBF if offset >= 0x8000 => (self.fxb, 3u8, bank - 0xA0, true),
867            0xC0..=0xCF => (self.cxb, 0u8, bank - 0xC0, false),
868            0xD0..=0xDF => (self.dxb, 1u8, bank - 0xD0, false),
869            0xE0..=0xEF => (self.exb, 2u8, bank - 0xE0, false),
870            0xF0..=0xFF => (self.fxb, 3u8, bank - 0xF0, false),
871            _ => return None,
872        };
873        // Pick the MB slot. LoROM half-banks defer to `default_mb`
874        // unless the mode flag is set; HiROM full-banks always use
875        // the banked MB.
876        let mb = if !is_lo || (reg & 0x80) != 0 {
877            usize::from(reg & 0x07)
878        } else {
879            usize::from(default_mb)
880        };
881        let base = mb * MB;
882        let within_mb = if is_lo {
883            usize::from(local_bank) * 0x8000 + (usize::from(offset) - 0x8000)
884        } else {
885            usize::from(local_bank) * 0x1_0000 + usize::from(offset)
886        };
887        let off = base + within_mb;
888        if off < self.rom.len() {
889            Some(off)
890        } else {
891            None
892        }
893    }

927    /// BW-RAM access: two views, both gated by the cart having
928    /// declared SRAM:
929    ///   * 8 KB sliding window at `$00-$3F:$6000-$7FFF`, offset by
930    ///     `BMAPS << 13` within BW-RAM.
931    ///   * Linear 256 KB at `$40-$4F:$0000-$FFFF` for the SA-1's own
932    ///     full-bandwidth view.
933    fn bwram_offset(&self, bank: u8, offset: u16, side: WriteSide) -> Option<usize> {
934        if self.bwram.is_empty() {
935            return None;
936        }
937        if matches!(bank, 0x00..=0x3F | 0x80..=0xBF) && (0x6000..=0x7FFF).contains(&offset) {
938            // The `$6000-$7FFF` 8 KB window is bank-selected PER SIDE: the main
939            // CPU uses SBM ($2224), the SA-1 uses CBM ($2225). (CBM bit 7 =
940            // bitmap source mode — not yet modelled; SMRPG's attract uses
941            // normal banking, bit 7 clear.)
942            let bank_reg = match side {
943                WriteSide::Main => self.bmaps,
944                WriteSide::Sa1 => self.cbm,
945            };
946            let window = usize::from(bank_reg & 0x1F) * 0x2000;
947            let off = window + usize::from(offset - 0x6000);
948            return Some(off % self.bwram.len());
949        }
950        if matches!(bank, 0x40..=0x4F) {
951            let off = usize::from(bank - 0x40) * 0x1_0000 + usize::from(offset);
952            return Some(off % self.bwram.len());
953        }
954        None
955    }

968    /// Re-run the multiplier / divider per `mcnt`. Triggered by a
969    /// write to `$2254` MB-high.
970    /// Run the arithmetic op latched by the `$2254` (MBH) write — a
971    /// direct port of ares `io.cpp:398-423`. The accumulate flag
972    /// (`acm`, MCNT bit 1) takes precedence over the multiply/divide
973    /// select (`md`, bit 0).
974    fn update_arith(&mut self) {
975        let acm = self.mcnt & 0x02 != 0;
976        let md = self.mcnt & 0x01 != 0;
977        if acm {
978            // Sigma: cumulative MAC into a 40-bit accumulator with an
979            // overflow flag (ares io.cpp:419-422).
980            let product = i64::from(self.ma) * i64::from(self.mb);
981            let acc = (self.mr as u64).wrapping_add(product as u64);
982            self.overflow = (acc >> 40) & 1 != 0;
983            self.mr = (acc & ((1u64 << 40) - 1)) as i64;
984        } else if md {
985            // Division: SIGNED dividend ÷ UNSIGNED divisor, floored
986            // (non-negative remainder). MA is reset (MB below).
987            if self.mb == 0 {
988                self.mr = 0;
989            } else {
990                let dividend = i32::from(self.ma);
991                let divisor = i32::from(self.mb as u16);
992                let remainder = if dividend >= 0 {
993                    dividend % divisor
994                } else {
995                    (dividend % divisor + divisor) % divisor
996                };
997                let quotient = (dividend - remainder) / divisor;
998                self.mr = (i64::from(remainder as u16) << 16) | i64::from(quotient as u16);
999            }
1000            self.ma = 0;
1001        } else {
1002            // Signed 16×16 → unsigned-32-bit multiply.
1003            let product = i32::from(self.ma) * i32::from(self.mb);
1004            self.mr = i64::from(product as u32);
1005        }
1006        // Every op resets MB (ares io.cpp:402,415,422).
1007        self.mb = 0;
1008    }
1009}

1011impl Mapper for Sa1Mapper {
1012    fn kind(&self) -> MapperKind {
1013        MapperKind::Sa1
1014    }

1016    fn read(&mut self, addr: Addr24) -> Option<u8> {
1017        let bank = bank_of(addr);
1018        let offset = offset_of(addr);
1019        // I/O reads — multiplier result is the only "live" path; the
1020        // rest of the window is open-bus / memory-backed stub.
1021        if let Some(idx) = Self::mmio_offset(addr) {
1022            let mr_addr = 0x2200 + idx as u16;
1023            // VDP reads are side-effecting (auto-advance the bit
1024            // cursor on the matching trigger half), so they're not in
1025            // the pure-`match` arm below.
1026            match mr_addr {
1027                0x230C => {
1028                    let data = self.vlbp_data();
1029                    let lo = data as u8;
1030                    if (self.vbd & 0x80) == 0 {
1031                        self.vlbp_advance();
1032                    }
1033                    return Some(lo);
1034                }
1035                0x230D => {
1036                    let data = self.vlbp_data();
1037                    let hi = (data >> 8) as u8;
1038                    if (self.vbd & 0x80) != 0 {
1039                        self.vlbp_advance();
1040                    }
1041                    return Some(hi);
1042                }
1043                _ => {}
1044            }
1045            // $2306-$230A → 40-bit MR result. We expose 5 bytes
1046            // little-endian.
1047            return Some(match mr_addr {
1048                0x2300 => self.read_sfr(),
1049                0x2301 => self.read_cfr(),
1050                // HCR / VCR — the live H/V timer counters, read back in DOTS
1051                // (4 clocks = 1 dot), 9 bits each. H = hcounter >> 2.
1052                0x2302 => (self.hcounter >> 2) as u8,
1053                0x2303 => ((self.hcounter >> 2) >> 8) as u8 & 0x01,
1054                0x2304 => self.vcounter as u8,
1055                0x2305 => (self.vcounter >> 8) as u8 & 0x01,
1056                0x2306 => self.mr as u8,
1057                0x2307 => (self.mr >> 8) as u8,
1058                0x2308 => (self.mr >> 16) as u8,
1059                0x2309 => (self.mr >> 24) as u8,
1060                0x230A => (self.mr >> 32) as u8,
1061                // (OF) sigma-mode overflow flag in bit 7.
1062                0x230B => u8::from(self.overflow) << 7,
1063                _ => self.mmio[idx],
1064            });
1065        }
1066        // Main-CPU vector override — when the SA-1 is currently
1067        // asserting an IRQ/NMI to the S-CPU and the matching IVSW /
1068        // NMIVW bit is set, the bank-0 vector slots read back as the
1069        // SA-1's SIV / SNV instead of the real ROM bytes.
1070        if let Some(v) = self.main_vector_override(bank, offset) {
1071            return Some(v);
1072        }
1073        if let Some(o) = Self::iram_offset(bank, offset) {
1074            return Some(self.iram[o]);
1075        }
1076        if let Some(o) = self.bwram_offset(bank, offset, WriteSide::Main) {
1077            return Some(self.bwram[o]);
1078        }
1079        if let Some(o) = self.rom_offset(bank, offset) {
1080            return Some(self.rom[o]);
1081        }
1082        None
1083    }
1084
1085    fn write(&mut self, addr: Addr24, value: u8) -> bool {
1086        // The trait entry point is the main-CPU view. `Sa1Bus` calls
1087        // `write_from_sa1` directly so I-RAM / BW-RAM write protection
1088        // can distinguish the two sides.
1089        self.write_with_side(addr, value, WriteSide::Main)
1090    }

1092    fn rom_size(&self) -> usize {
1093        self.rom.len()
1094    }
1095
1096    fn sram_size(&self) -> usize {
1097        self.bwram.len()
1098    }
1099
1100    fn save_state(&self) -> Vec<u8> {
1101        bincode::serialize(self).unwrap_or_default()
1102    }
1103
1104    fn load_state(&mut self, data: &[u8]) {

1110        }
1111    }

1115    /// Side-aware read entry for the SA-1's own bus. Uses
1116    /// [`Sa1Mapper::iram_offset_sa1`] which also exposes the I-RAM
1117    /// mirror at `$0000-$07FF`, and skips the main-CPU vector
1118    /// override (the SA-1 has its own override via
1119    /// [`Sa1Mapper::sa1_vector_override`], applied by [`super::Sa1Bus`]).
1120    /// MMIO reads route identically to the main-CPU path.
1121    pub fn read_from_sa1(&mut self, addr: Addr24) -> Option<u8> {
1122        let bank = bank_of(addr);
1123        let offset = offset_of(addr);
1124        if let Some(idx) = Self::mmio_offset(addr) {
1125            // Same MMIO handling as the trait's `read`. We re-dispatch
1126            // through it but mask out the main-CPU `main_vector_override`
1127            // by intercepting it ourselves first (MMIO addresses live
1128            // outside that range so this is a no-op early-out).
1129            let _ = idx;
1130            return self.read(addr);
1131        }
1132        if let Some(o) = Self::iram_offset_sa1(bank, offset) {
1133            return Some(self.iram[o]);
1134        }
1135        if let Some(o) = self.bwram_offset(bank, offset, WriteSide::Sa1) {
1136            return Some(self.bwram[o]);
1137        }
1138        if let Some(o) = self.rom_offset(bank, offset) {
1139            return Some(self.rom[o]);
1140        }
1141        None
1142    }

1144    /// Base SA-1 access cost in **SA-1 steps** (1 step = 2 master cycles)
1145    /// for `addr`, faithful to ares `coprocessor/sa1/memory.cpp`: IO = 1,
1146    /// ROM = 1, IRAM = 1, BWRAM = 2 (+ `conflict()` contention steps, which
1147    /// are Increment B). The region order mirrors [`Self::read_from_sa1`]
1148    /// exactly so the cycle decode can never drift from the data path.
1149    #[must_use]
1150    pub fn sa1_region_steps(&self, addr: Addr24) -> u8 {
1151        let bank = bank_of(addr);
1152        let offset = offset_of(addr);
1153        // BWRAM is the only 2-step region; MMIO/IRAM/ROM/open-bus are all
1154        // 1 step. Honour the same priority as `read_from_sa1` (MMIO and
1155        // IRAM win over BWRAM), so a BWRAM cost is charged only when the
1156        // address actually resolves to BWRAM.
1157        let is_bwram = Self::mmio_offset(addr).is_none()
1158            && Self::iram_offset_sa1(bank, offset).is_none()
1159            && self.bwram_offset(bank, offset, WriteSide::Sa1).is_some();
1160        if is_bwram { 2 } else { 1 }
1161    }

1163    /// Extra SA-1 bus-**contention** steps for an access at `sa1_addr` while
1164    /// the S-CPU holds `scpu_mar` on the shared bus — ares' `conflict()`
1165    /// model (`coprocessor/sa1/{rom,bwram,iram}.cpp` + the conditional
1166    /// `step()`s in `memory.cpp` read/write). The SA-1's own access region
1167    /// (ROM / BW-RAM / I-RAM, classified by ares' raw address masks, MMIO
1168    /// and ROM winning over BW-RAM/I-RAM) selects which resource is in play;
1169    /// the penalty fires only when the **S-CPU**'s last access address
1170    /// (`scpu_mar`, ares `cpu.r.mar`) targets that same resource:
1171    /// ROM `+1`, BW-RAM `+2`, I-RAM `+2`. MMIO / open-bus never contend.
1172    ///
1173    /// Charged on top of [`Self::sa1_region_steps`] (the base cost). This is
1174    /// the "Increment B" the Phase-5b doc deferred.
1175    ///
1176    /// Two faithful approximations vs ares, both bounded: (1) luna evaluates
1177    /// `scpu_mar` once per SA-1 batch (the deficit model runs ~1-2 SA-1
1178    /// instructions per S-CPU access, so the S-CPU address is effectively
1179    /// fixed across the batch); (2) the I-RAM exemption during S-CPU DRAM
1180    /// refresh (`iram.conflict()` returns `cpu.refresh()==0`) is not modelled
1181    /// — a ≤2-step over-charge confined to the ~40-mclk refresh window.
1182    #[must_use]
1183    pub fn sa1_conflict_steps(&self, sa1_addr: Addr24, scpu_mar: u32) -> u8 {
1184        let a = sa1_addr & 0xFF_FFFF;
1185        // MMIO ($2200-23ff): no shared-bus contention.
1186        if a & 0x40_FE00 == 0x00_2200 {
1187            return 0;
1188        }
1189        // SA-1 ROM region → conflict iff the S-CPU is also in ROM (+1).
1190        if a & 0x40_8000 == 0x00_8000 || a & 0xC0_0000 == 0xC0_0000 {
1191            return u8::from(Self::scpu_rom_conflict(scpu_mar));
1192        }
1193        // SA-1 BW-RAM region → conflict iff the S-CPU is also in BW-RAM (+2).
1194        if a & 0x40_E000 == 0x00_6000 || a & 0xE0_0000 == 0x40_0000 || a & 0xF0_0000 == 0x60_0000 {
1195            return if Self::scpu_bwram_conflict(scpu_mar) {
1196                2
1197            } else {
1198                0
1199            };
1200        }
1201        // SA-1 I-RAM region ($0000-07ff mirror or $3000-37ff) → conflict iff
1202        // the S-CPU is also in I-RAM (+2).
1203        if a & 0x40_F800 == 0x00_0000 || a & 0x40_F800 == 0x00_3000 {
1204            return if Self::scpu_iram_conflict(scpu_mar) {
1205                2
1206            } else {
1207                0
1208            };
1209        }
1210        // Open-bus fall-through: 1 base step, no contention.
1211        0
1212    }

1215    /// `c0-ff:0000-ffff`.
1216    const fn scpu_rom_conflict(mar: u32) -> bool {
1217        mar & 0x40_8000 == 0x00_8000 || mar & 0xC0_0000 == 0xC0_0000
1218    }

1220    /// ares `SA1::BWRAM::conflict()` — S-CPU in `00-3f/80-bf:6000-7fff` or

1233    /// Side-aware write entry for the SA-1's own bus. Drives I-RAM /
1234    /// BW-RAM through the CIWP / CBWE protection masks instead of
1235    /// the S-CPU's SIWP / SBWE / BWPA. MMIO writes are routed
1236    /// identically.

1239    }

1241    fn write_with_side(&mut self, addr: Addr24, value: u8, side: WriteSide) -> bool {
1242        let bank = bank_of(addr);
1243        let offset = offset_of(addr);
1244        if let Some(idx) = Self::mmio_offset(addr) {
1245            let absolute = 0x2200 + idx as u16;
1246            let prev = self.mmio[idx];
1247            self.mmio[idx] = value;
1248            match absolute {
1249                // -------- S-CPU → SA-1 control --------
1250                0x2200 => {
1251                    // CCNT bit layout (per ares `io.cpp` $2200 +
1252                    // Mesen2 `Sa1.cpp:240-258`):
1253                    //   bit 7 = SA-1 IRQ request (level-driven)
1254                    //   bit 6 = SA-1 wait (not modelled)
1255                    //   bit 5 = SA-1 reset (handled in Sa1Chip::write)
1256                    //   bit 4 = SA-1 NMI request (level-driven)
1257                    //   bits 3..0 = message to SA-1
1258                    // Both refs latch the IRQ/NMI flag on every write
1259                    // whose corresponding bit is set, regardless of the
1260                    // previous value. luna previously edge-detected
1261                    // (0→1) which silently dropped re-trigger requests
1262                    // — games that re-write CCNT=$80 after a single ack
1263                    // never got the second IRQ. Acks are explicit via
1264                    // CIC ($220B), not implicit on a CCNT clear.
1265                    let _ = prev;
1266                    self.ccnt_msg = value & 0x0F;
1267                    if (value & 0x80) != 0 {
1268                        self.main_irq_to_sa1 = true;
1269                    }
1270                    if (value & 0x10) != 0 {
1271                        self.main_nmi_to_sa1 = true;
1272                    }
1273                }
1274                0x2201 => self.sie = value,
1275                0x2202 => {
1276                    if (value & 0x80) != 0 {
1277                        self.s_irq_to_main = false;
1278                    }
1279                    if (value & 0x40) != 0 {
1280                        self.s_nmi_to_main = false;
1281                    }
1282                    if (value & 0x20) != 0 {
1283                        self.cc1_irq_to_main = false;
1284                    }
1285                }
1286                0x2203 | 0x2204 => {}
1287                0x2205 => self.cnv_lo = value,
1288                0x2206 => self.cnv_hi = value,
1289                0x2207 => self.civ_lo = value,
1290                0x2208 => self.civ_hi = value,
1291
1292                // -------- SA-1 → S-CPU control --------
1293                0x2209 => {
1294                    // SCNT bit layout (per ares `io.cpp` $2209 lines
1295                    // ~233-246 + Mesen2 `Sa1.cpp:84-110`):
1296                    //   bit 7 = SA-1 → S-CPU IRQ request (level-driven)
1297                    //   bit 6 = IVSW (vector override for S-CPU IRQ)
1298                    //   bit 4 = NMIVW (vector override for S-CPU NMI)
1299                    //   bits 3..0 = message to S-CPU
1300                    // The IRQ latch is set on every write whose bit 7
1301                    // is set, regardless of prior value. luna's old
1302                    // edge-detect dropped SMRPG's repeated SCNT=$87
1303                    // pulses — after the first ack via SIC, the second
1304                    // and subsequent SCNT writes never re-latched, so
1305                    // the main↔SA-1 mailbox deadlocked on the second
1306                    // handshake. Acks are explicit through SIC ($2202).
1307                    let _ = prev;
1308                    self.scnt = value;
1309                    if (value & 0x80) != 0 {
1310                        self.s_irq_to_main = true;
1311                    }
1312                }
1313                // CIE bit layout (per ares + Mesen2):
1314                //   bit 7 = S-CPU → SA-1 IRQ enable
1315                //   bit 6 = timer IRQ enable
1316                //   bit 5 = DMA IRQ enable
1317                //   bit 4 = S-CPU → SA-1 NMI enable
1318                0x220A => self.cie = value,
1319                // CIC mirror: each bit clears its CFR latch.
1320                0x220B => {
1321                    if (value & 0x80) != 0 {
1322                        self.main_irq_to_sa1 = false;
1323                    }
1324                    if (value & 0x40) != 0 {
1325                        self.timer_irq_to_sa1 = false;
1326                    }
1327                    if (value & 0x20) != 0 {
1328                        self.dma_irq_to_sa1 = false;
1329                    }
1330                    if (value & 0x10) != 0 {
1331                        self.main_nmi_to_sa1 = false;
1332                    }
1333                }
1334                0x220C => self.snv_lo = value,
1335                0x220D => self.snv_hi = value,
1336                0x220E => self.siv_lo = value,
1337                0x220F => self.siv_hi = value,
1338
1339                // -------- SA-1 timer --------
1340                0x2210 => self.tmc = value,
1341                0x2211 => {
1342                    // CTR — restart the timer (ares io.cpp $2211).
1343                    self.hcounter = 0;
1344                    self.vcounter = 0;
1345                    self.timer_rem = 0;
1346                }
1347                0x2212 => self.hcnt_lo = value,
1348                0x2213 => self.hcnt_hi = value,
1349                0x2214 => self.vcnt_lo = value,
1350                0x2215 => self.vcnt_hi = value,
1351
1352                // -------- SA-1 DMA --------
1353                //
1354                // DCNT only *configures* the DMA. Per ares' `io.cpp`
1355                // and Mesen2's `WriteSharedRegisters`, the actual
1356                // trigger is on the final DDA byte for normal /
1357                // CC1 DMA, and on BRF[7] / BRF[15] for CC2.
1358                0x2230 => {
1359                    self.dcnt = value;
1360                    let prev_en = self.dma_en;
1361                    self.dma_en = (value & 0x80) != 0;
1362                    self.dma_cden = (value & 0x20) != 0;
1363                    self.dma_cdsel = (value & 0x10) != 0;
1364                    self.dma_dd = (value & 0x04) != 0;
1365                    // Clearing DMA enable also terminates any in-flight
1366                    // CC2 stream.
1367                    if prev_en && !self.dma_en && self.cc2_active {
1368                        self.cc2_end();
1369                    }
1370                    // If the game arms CC2 (cden=1, cdsel=0), prep the
1371                    // streaming buffer. Real hardware doesn't *need*
1372                    // this — CC2 also auto-arms on the first BRF write
1373                    // — but priming here matches our existing CC2
1374                    // model and lets the streaming buffer track bpp
1375                    // captured from CDMA.
1376                    if self.dma_en && self.dma_cden && !self.dma_cdsel && !self.cc2_active {
1377                        self.cc2_arm();
1378                    }
1379                }
1380                0x2231 => {
1381                    self.cdma = value;
1382                    // CDMA bit 7 = CDEND. Writing 1 terminates a CC2
1383                    // stream immediately.
1384                    if (value & 0x80) != 0 && self.cc2_active {
1385                        self.cc2_end();
1386                    }
1387                }
1388                0x2232 => self.sda = (self.sda & !0x00_00FF) | u32::from(value),
1389                0x2233 => self.sda = (self.sda & !0x00_FF00) | (u32::from(value) << 8),
1390                0x2234 => self.sda = (self.sda & !0xFF_0000) | (u32::from(value) << 16),
1391                0x2235 => self.dda = (self.dda & !0x00_00FF) | u32::from(value),
1392                0x2236 => {
1393                    self.dda = (self.dda & !0x00_FF00) | (u32::from(value) << 8);
1394                    // Trigger: normal DMA → I-RAM, or CC1.
1395                    if self.dma_en {
1396                        if !self.dma_cden && !self.dma_dd {
1397                            self.run_normal_dma();
1398                        } else if self.dma_cden && self.dma_cdsel {
1399                            self.run_cc1_dma();
1400                        }
1401                    }
1402                }
1403                0x2237 => {
1404                    self.dda = (self.dda & !0xFF_0000) | (u32::from(value) << 16);
1405                    // Trigger: normal DMA → BW-RAM.
1406                    if self.dma_en && !self.dma_cden && self.dma_dd {
1407                        self.run_normal_dma();
1408                    }
1409                }
1410                0x2238 => self.dtc = (self.dtc & 0xFF00) | u16::from(value),
1411                0x2239 => self.dtc = (self.dtc & 0x00FF) | (u16::from(value) << 8),
1412                // $223F BBF is the BW-RAM bitmap-format selector on
1413                // real hardware. We also accept it as a luna-internal
1414                // CC2 byte feed for back-compat with existing tests.
1415                // BRF[7] / BRF[15] ($2247 / $224F) are the real-HW
1416                // CC2 row triggers — same per-byte staging path.
1417                0x223F | 0x2247 | 0x224F if self.cc2_active => {
1418                    self.cc2_consume_byte(value);
1419                }
1420
1421                0x2220 => self.cxb = value,
1422                0x2221 => self.dxb = value,
1423                0x2222 => self.exb = value,
1424                0x2223 => self.fxb = value,
1425                0x2224 => self.bmaps = value,
1426                // $2225 BMAP (CBM) — the SA-1's BW-RAM $6000-$7FFF bank.
1427                0x2225 => self.cbm = value,
1428                0x2226 => self.sbwe = value,
1429                0x2227 => self.cbwe = value,
1430                0x2228 => self.bwpa = value,
1431                // Per ares + Mesen2: SIWP at $2229, CIWP at $222A.
1432                // (Older docs had these one address higher.)
1433                0x2229 => self.siwp = value,
1434                0x222A => self.ciwp = value,
1435                0x2250 => {
1436                    self.mcnt = value;
1437                    // ares io.cpp:374 — entering sigma mode (acm, bit 1)
1438                    // clears the accumulator, regardless of the other bits.
1439                    if value & 0x02 != 0 {
1440                        self.mr = 0;
1441                    }
1442                }
1443                0x2251 => self.ma = (self.ma & !0xFF) | i16::from(value),
1444                0x2252 => self.ma = (self.ma & 0xFF) | (i16::from(value as i8) << 8),
1445                0x2253 => self.mb = (self.mb & !0xFF) | i16::from(value),
1446                0x2254 => {
1447                    self.mb = (self.mb & 0xFF) | (i16::from(value as i8) << 8);
1448                    self.update_arith();
1449                }
1450                // -------- VLBP (Variable-Length Bit Processor) --------
1451                0x2258 => self.vbd = value,
1452                0x2259 => {
1453                    self.vda_base = (self.vda_base & !0x00_00FF) | u32::from(value);
1454                }
1455                0x225A => {
1456                    self.vda_base = (self.vda_base & !0x00_FF00) | (u32::from(value) << 8);
1457                }
1458                0x225B => {
1459                    self.vda_base = (self.vda_base & !0xFF_0000) | (u32::from(value) << 16);
1460                    self.vbit_offset = 0;
1461                }
1462                _ => {}
1463            }
1464            return true;
1465        }
1466        // SA-1 writes can also reach the I-RAM mirror at $0000-$07FF;
1467        // the main CPU only sees I-RAM at $3000-$37FF (so its writes
1468        // into $0000-$07FF should fall through to the bus's WRAM).
1469        let iram = match side {
1470            WriteSide::Main => Self::iram_offset(bank, offset),
1471            WriteSide::Sa1 => Self::iram_offset_sa1(bank, offset),
1472        };
1473        if let Some(o) = iram {
1474            if self.iram_writable_for(o, side) {
1475                self.iram[o] = value;
1476            }
1477            // Always claim the access — the bus would otherwise fall
1478            // through to WRAM, which isn't what protection means.
1479            return true;
1480        }
1481        if let Some(o) = self.bwram_offset(bank, offset, side) {

2419    fn sa1_conflict_steps_charge_only_on_same_shared_resource() {
2420        // Address-only contention model (ares `conflict()`); mapper config
2421        // is irrelevant. Use a mapper with BW-RAM so the regions resolve.
2422        let m = Sa1Mapper::new(ramp_rom(0x20_0000), 0x10000);
2423
2424        // S-CPU addresses representative of each shared resource + WRAM.
2425        let scpu_rom = make_addr(0x01, 0x9000); // 01:8000-ffff → ROM
2426        let scpu_rom_hi = make_addr(0xC5, 0x0000); // c0-ff → ROM
2427        let scpu_bwram = make_addr(0x40, 0x0000); // 40-4f:0000 → BW-RAM
2428        let scpu_iram = make_addr(0x00, 0x3100); // 00:3000-37ff → I-RAM
2429        let scpu_wram = make_addr(0x00, 0x0100); // low RAM → no resource
2430
2431        // SA-1 ROM access: +1 iff the S-CPU also holds ROM.
2432        let sa1_rom = make_addr(0x00, 0x8000);
2433        assert_eq!(m.sa1_conflict_steps(sa1_rom, scpu_rom), 1);
2434        assert_eq!(m.sa1_conflict_steps(sa1_rom, scpu_rom_hi), 1);
2435        assert_eq!(m.sa1_conflict_steps(sa1_rom, scpu_bwram), 0);
2436        assert_eq!(m.sa1_conflict_steps(sa1_rom, scpu_wram), 0);

2439        let sa1_bwram = make_addr(0x00, 0x6000);
2440        assert_eq!(m.sa1_conflict_steps(sa1_bwram, scpu_bwram), 2);
2441        assert_eq!(m.sa1_conflict_steps(sa1_bwram, scpu_rom), 0);
2442        assert_eq!(m.sa1_conflict_steps(sa1_bwram, scpu_iram), 0);

2444        // SA-1 I-RAM access: +2 iff the S-CPU also holds I-RAM (the
2445        // $3000-37ff window only — the $0000-07ff mirror does not count,
