1//! SNES ROM file parsing.
2//!
3//! Detects an optional 512-byte SMC "copier" header, infers the internal
4//! header location (`$7FC0` for `LoROM`, `$FFC0` for `HiROM`), parses the
5//! title / mapper / sizes / region, and builds a [`Cartridge`] ready to
6//! be wrapped in a `luna-bus` mapper.
7//!
8//! See `ARCHITECTURE.md` §5.
9
10use luna_bus::MapperKind;
11use std::fs;
12use std::path::Path;
13use thiserror::Error;

17// =============================================================================

19/// Errors that may surface while loading or parsing a SNES ROM.
20#[derive(Debug, Error)]
21pub enum CartError {
22    /// Underlying filesystem error.
23    #[error("I/O error reading ROM: {0}")]
24    Io(#[from] std::io::Error),
25    /// File is smaller than the minimum SNES ROM page (32 KB).
26    #[error("ROM is too small ({0} bytes); minimum is 32 KB")]
27    TooSmall(usize),
28    /// No internal header at the expected offsets passed the
29    /// checksum-complement validation.
30    #[error(
31        "could not detect cartridge layout (LoROM / HiROM): both internal headers fail the checksum complement check"
32    )]
33    LayoutUnknown,
34}

40/// Cartridge region / video standard derived from the country byte.
41#[derive(Debug, Clone, Copy, PartialEq, Eq, serde::Serialize, serde::Deserialize)]
42pub enum Region {
43    /// NTSC (Japan / North America).
44    Ntsc,
45    /// PAL (Europe / Australia).
46    Pal,
47    /// Unknown country code.
48    Unknown,
49}
50
51impl Region {
52    /// Decode region from the SNES country byte at offset `$xxD9`.
53    #[must_use]
54    pub const fn from_country(byte: u8) -> Self {
55        match byte {
56            // NTSC: Japan, USA, Canada, South Korea, Brazil
57            0x00 | 0x01 | 0x0D | 0x0F | 0x10 => Self::Ntsc,
58            // PAL: Europe and friends
59            0x02..=0x0C | 0x11 => Self::Pal,
60            _ => Self::Unknown,
61        }
62    }
63}

65/// Decoded SNES internal header.
66#[derive(Debug, Clone)]
67pub struct Header {
68    /// ROM title, ASCII-decoded (Japanese ROMs use Shift-JIS — best-effort).
69    pub title: String,
70    /// Cartridge mapping mode.
71    pub mapper_kind: MapperKind,
72    /// For a `Dsp1` cartridge, whether the base ROM layout is `HiROM`
73    /// (`true`, e.g. Super Mario Kart) or `LoROM` (`false`). Ignored for
74    /// non-DSP mappers.
75    pub dsp_hirom: bool,
76    /// `true` if the `FastROM` bit is set in the mapping byte.
77    pub fast_rom: bool,
78    /// ROM size in kilobytes (advertised by the cartridge, may exceed the
79    /// actual file size for over-dumped or padded ROMs).
80    pub rom_size_kb: u32,
81    /// SRAM size in kilobytes (0 = no SRAM).
82    pub sram_size_kb: u32,
83    /// Expansion (coprocessor work) RAM size in kilobytes, from the
84    /// extended-header `$FFBD` byte (`1024 << n`). `0` if the byte is not a
85    /// valid `1..=7` exponent. This is the Super FX Game Pak work RAM size —
86    /// distinct from `sram_size_kb` (battery save RAM, `$FFD8`). See Mesen2
87    /// `BaseCartridge.cpp` (`ExpansionRamSize`).
88    pub expansion_ram_kb: u32,
89    /// Region / video standard.
90    pub region: Region,
91    /// Maker code (old-style single byte).
92    pub maker: u8,
93    /// Mask ROM revision.
94    pub version: u8,
95    /// 16-bit checksum claimed by the header.
96    pub checksum: u16,
97    /// 16-bit checksum complement (should be `!checksum`).
98    pub checksum_complement: u16,
99}
100
101impl Header {
102    /// `true` iff `checksum ^ complement == 0xFFFF`. Used as the primary
103    /// signal to disambiguate `LoROM` vs `HiROM`.
104    #[must_use]
105    pub const fn checksum_valid(&self) -> bool {
106        self.checksum ^ self.checksum_complement == 0xFFFF
107    }
108}

114/// A parsed SNES cartridge.
115#[derive(Debug, Clone)]
116pub struct Cartridge {
117    /// Pure ROM bytes (SMC copier header stripped if present).
118    pub rom: Vec<u8>,
119    /// Decoded header.
120    pub header: Header,
121    /// Coprocessor microcode (e.g. an 8 KB DSP-1 `dsp1b.rom`), if the cart
122    /// needs one and it was found embedded in the dump or supplied
123    /// externally. `None` for non-coprocessor carts or until resolved.
124    coprocessor_firmware: Option<Vec<u8>>,
125}
126
127/// Combined DSP-1 firmware size (program `0x1800` + data `0x800`).
128const DSP1_FIRMWARE_LEN: usize = 0x2000;
129
130impl Cartridge {
131    /// Load and parse a ROM file from disk. For a DSP game with no firmware
132    /// embedded in the dump, auto-discovers `dsp1b.rom` next to the ROM file.
133    pub fn load(path: impl AsRef<Path>) -> Result<Self, CartError> {
134        let path = path.as_ref();
135        let mut cart = Self::from_bytes(fs::read(path)?)?;
136        if cart.needs_coprocessor_firmware() {
137            if let Some(dir) = path.parent() {
138                if let Ok(bytes) = fs::read(dir.join("dsp1b.rom")) {
139                    cart.set_coprocessor_firmware(bytes);
140                }
141            }
142        }
143        Ok(cart)
144    }
145
146    /// Parse a ROM image from bytes already in memory. Strips a 512-byte
147    /// SMC copier header if present, and (DSP games) extracts firmware
148    /// appended to the dump.
149    pub fn from_bytes(mut rom: Vec<u8>) -> Result<Self, CartError> {
150        // SMC copier prepends 512 bytes if `(rom.len() % 1024) == 512`.
151        if rom.len() % 1024 == 512 {
152            rom.drain(..512);
153        }
154        if rom.len() < 0x8000 {
155            return Err(CartError::TooSmall(rom.len()));
156        }
157        let header = detect_and_parse(&rom).ok_or(CartError::LayoutUnknown)?;
158        // Some DSP-1 dumps append the chip's 8 KB firmware (Mesen2
159        // `BaseCartridge.cpp`: ROM length is `32KB·n + 0x2000`). Strip it
160        // off the ROM and keep it as the coprocessor firmware.
161        let coprocessor_firmware = if matches!(header.mapper_kind, MapperKind::Dsp1)
162            && rom.len() & 0x7FFF == DSP1_FIRMWARE_LEN
163        {
164            Some(rom.split_off(rom.len() - DSP1_FIRMWARE_LEN))
165        } else {
166            None
167        };
168        Ok(Self {
169            rom,
170            header,
171            coprocessor_firmware,
172        })
173    }
174
175    /// Parse a ROM image but **force** the mapper layout and skip the
176    /// checksum-complement validation that [`Self::from_bytes`] requires.
177    ///
178    /// For headerless / homebrew / hardware-test ROMs (e.g. the Peter
179    /// Lemon SNES suite) whose internal checksum is blank or wrong, where
180    /// layout auto-detection would reject them. The header fields are
181    /// still parsed at the forced mapper's offset (best effort), but
182    /// `mapper_kind` is overridden to `mapper`.
183    pub fn from_bytes_forced(mut rom: Vec<u8>, mapper: MapperKind) -> Result<Self, CartError> {
184        if rom.len() % 1024 == 512 {
185            rom.drain(..512);
186        }
187        if rom.len() < 0x8000 {
188            return Err(CartError::TooSmall(rom.len()));
189        }
190        let off = match mapper {
191            // LoROM-region layouts (header at $7FC0).
192            MapperKind::LoRom | MapperKind::Sa1 | MapperKind::SuperFx => HEADER_OFFSET_LOROM,
193            // HiROM-region layouts (header at $FFC0). Forced DSP-1 assumes
194            // the HiROM board (Super Mario Kart); LoROM DSP-1 isn't forced.
195            MapperKind::HiRom | MapperKind::Sdd1 | MapperKind::Spc7110 | MapperKind::Dsp1 => {
196                HEADER_OFFSET_HIROM
197            }
198            MapperKind::ExHiRom => HEADER_OFFSET_EXHIROM,
199        };
200        if off + 0x20 > rom.len() {
201            return Err(CartError::LayoutUnknown);
202        }
203        let mut header = parse_at(&rom, off);
204        header.mapper_kind = mapper;
205        Ok(Self {
206            rom,
207            header,
208            coprocessor_firmware: None,
209        })
210    }
211
212    /// `true` when this cartridge needs an external coprocessor firmware
213    /// image that hasn't been supplied yet (a DSP game with no `dsp1b.rom`
214    /// embedded in the dump or loaded beside it).
215    #[must_use]
216    pub const fn needs_coprocessor_firmware(&self) -> bool {
217        matches!(self.header.mapper_kind, MapperKind::Dsp1) && self.coprocessor_firmware.is_none()
218    }
219
220    /// Supply the coprocessor firmware (e.g. an 8 KB `dsp1b.rom`). Used by
221    /// front-ends that resolve the file via a CLI flag / firmware folder.
222    pub fn set_coprocessor_firmware(&mut self, bytes: Vec<u8>) {
223        self.coprocessor_firmware = Some(bytes);
224    }
225
226    /// The loaded coprocessor firmware, if any.
227    #[must_use]
228    pub fn coprocessor_firmware(&self) -> Option<&[u8]> {
229        self.coprocessor_firmware.as_deref()
230    }
231
232    /// The external firmware file this cartridge needs (e.g. `dsp1b.rom`
233    /// for a DSP-1 game), or `None` if it needs none. Front-ends use this
234    /// to name the file in a prompt / error and to look it up by name.
235    #[must_use]
236    pub const fn required_firmware_filename(&self) -> Option<&'static str> {
237        match self.header.mapper_kind {
238            MapperKind::Dsp1 => Some("dsp1b.rom"),
239            _ => None,
240        }
241    }
242}

248const HEADER_OFFSET_LOROM: usize = 0x7FC0;
249const HEADER_OFFSET_HIROM: usize = 0xFFC0;
250const HEADER_OFFSET_EXHIROM: usize = 0x40_FFC0;
251
252/// Detect the cartridge layout by examining each plausible internal-header
253/// offset.
254///
255/// The checksum-complement test (`!ck == ckcomp`) is kept as the strict
256/// **acceptance gate** — it reliably rejects all-zero / non-ROM input,
257/// which is luna's deliberate trade-off (forced-mapper loading handles
258/// unlicensed dumps with bogus checksums). Among the offsets that pass
259/// the gate, we no longer take the *first* one: we pick the
260/// highest-[`score_header`] candidate (ties → the earlier offset, i.e.
261/// `LoROM`), a port of ares' `SuperFamicom::scoreHeader`
262/// (mia/medium/super-famicom.cpp). This disambiguates the rare case where
263/// more than one offset's checksum coincidentally validates.
264fn detect_and_parse(rom: &[u8]) -> Option<Header> {
265    let mut best: Option<(i32, usize)> = None;
266    for off in [
267        HEADER_OFFSET_LOROM,
268        HEADER_OFFSET_HIROM,
269        HEADER_OFFSET_EXHIROM,
270    ] {
271        if off + 0x20 > rom.len() {
272            continue;
273        }
274        if !parse_at(rom, off).checksum_valid() {
275            continue;
276        }
277        let score = score_header(rom, off);
278        if best.is_none_or(|(bs, _)| score > bs) {
279            best = Some((score, off));
280        }
281    }
282    best.map(|(_, off)| parse_at(rom, off))
283}
284
285/// Heuristic confidence that the internal header at `off` is the real one,
286/// a faithful port of ares' `SuperFamicom::scoreHeader`
287/// (mia/medium/super-famicom.cpp). ares' header base is `off - 0x10`, so
288/// its `address + 0x25` map-mode byte is luna's `off + 0x15`, etc. Scores
289/// the reset-vector validity, the plausibility of the first opcode the CPU
290/// would execute, the checksum complement, and the map-mode/offset match.
291fn score_header(rom: &[u8], off: usize) -> i32 {
292    // ares requires `address + 0x50` bytes (= luna `off + 0x40`): the
293    // header plus the native reset vector at `off + 0x3C`.
294    if off + 0x40 > rom.len() {
295        return 0;
296    }
297    let map_mode = rom[off + 0x15] & !0x10; // ignore the FastROM bit
298    let complement = u16::from_le_bytes([rom[off + 0x1C], rom[off + 0x1D]]);
299    let checksum = u16::from_le_bytes([rom[off + 0x1E], rom[off + 0x1F]]);
300    let reset_vector = u16::from_le_bytes([rom[off + 0x3C], rom[off + 0x3D]]);
301    if reset_vector < 0x8000 {
302        // $00:0000-7FFF is never ROM data — this offset can't be a header.
303        return 0;
304    }
305
306    // The first instruction the CPU would execute at the reset vector.
307    let ares_base = off.wrapping_sub(0x10);
308    let opcode_off = (ares_base & !0x7FFF) | (reset_vector as usize & 0x7FFF);
309    let opcode = rom.get(opcode_off).copied().unwrap_or(0);
310
311    let mut score: i32 = 0;
312    match opcode {
313        // most likely: sei / clc / sec / stz $nnnn / jmp / jml
314        0x78 | 0x18 | 0x38 | 0x9C | 0x4C | 0x5C => score += 8,
315        // plausible: rep/sep/lda/ldx/ldy/jsr/jsl
316        0xC2 | 0xE2 | 0xAD | 0xAE | 0xAC | 0xAF | 0xA9 | 0xA2 | 0xA0 | 0x20 | 0x22 => score += 4,
317        // implausible: rti/rts/rtl/cmp/cpx/cpy
318        0x40 | 0x60 | 0x6B | 0xCD | 0xEC | 0xCC => score -= 4,
319        // least likely: brk/cop/stp/wdm/sbc $nnnnnn,x
320        0x00 | 0x02 | 0xDB | 0x42 | 0xFF => score -= 8,
321        _ => {}
322    }
323    if checksum.wrapping_add(complement) == 0xFFFF {
324        score += 4;
325    }
326    if off == HEADER_OFFSET_LOROM && map_mode == 0x20 {
327        score += 2;
328    }
329    if off == HEADER_OFFSET_HIROM && map_mode == 0x21 {
330        score += 2;
331    }
332    score.max(0)
333}

335fn parse_at(rom: &[u8], off: usize) -> Header {
336    let mut title_bytes = [0u8; 21];
337    title_bytes.copy_from_slice(&rom[off..off + 21]);
338    let title = decode_title(&title_bytes);
339
340    let map_byte = rom[off + 0x15];
341    let chipset = rom[off + 0x16];
342    // Coprocessor override from the chipset byte ($FFD6): when the low
343    // nibble flags a coprocessor (>= 3) the high nibble selects which.
344    // Super FX games (Star Fox = $13, Yoshi's Island = $15) carry a LoROM
345    // map mode ($20), so the GSU is only visible via this byte — high
346    // nibble 1 = GSU. (Empirically verified against both ROMs' headers.)
347    // Coprocessor overrides keyed on the chipset byte: low nibble >= 3 flags
348    // a coprocessor, high nibble selects which (1 = Super FX, 0 = NEC DSP).
349    let is_superfx = (chipset & 0x0F) >= 0x03 && (chipset & 0xF0) == 0x10;
350    let is_dsp = (chipset & 0x0F) >= 0x03 && (chipset & 0xF0) == 0x00;
351    // SA-1's canonical signal is the chipset/RomType byte ($FFD6): low
352    // nibble >= 3 (coprocessor present) + high nibble 3 (= SA-1) — e.g.
353    // SMRPG / Kirby Super Star carry $34/$35. The MapMode byte's low
354    // nibble 3 is a weaker secondary signal (kept via `mapper_from_byte`),
355    // but RomType is what hardware/ares key on.
356    let is_sa1 = (chipset & 0x0F) >= 0x03 && (chipset & 0xF0) == 0x30;
357    // S-DD1 (graphics decompression — Star Ocean, Street Fighter Alpha 2):
358    // chipset high nibble 4. LoROM-based; the chip is a `Sdd1Mapper` shim.
359    let is_sdd1 = (chipset & 0x0F) >= 0x03 && (chipset & 0xF0) == 0x40;
360    let base_kind = mapper_from_byte(map_byte).unwrap_or(MapperKind::LoRom);
361    // DSP-1 boards exist in both LoROM (DR/SR at $8000) and HiROM (DR/SR at
362    // $6000) flavours — the base layout follows the map byte.
363    let dsp_hirom = matches!(base_kind, MapperKind::HiRom | MapperKind::ExHiRom);
364    let mapper_kind = if is_superfx {
365        MapperKind::SuperFx
366    } else if is_dsp {
367        MapperKind::Dsp1
368    } else if is_sa1 {
369        MapperKind::Sa1
370    } else if is_sdd1 {
371        MapperKind::Sdd1
372    } else {
373        base_kind
374    };
375    let fast_rom = (map_byte & 0x10) != 0;
376    // The size bytes are exponents (KB = 1 << byte). Garbage cartridges
377    // (or our wrong-offset probing) can produce arbitrary byte values
378    // which would propagate downstream into multi-terabyte allocation
379    // requests. Clamp the exponents to ranges that span the real SNES
380    // catalogue: ROM up to 64 MB (1 << 16 KB) and SRAM up to 128 KB
381    // (1 << 7 KB). Larger advertised values are saturated, not trusted.
382    let rom_size_kb = 1u32 << u32::from(rom[off + 0x17]).min(16);
383    let sram_byte = rom[off + 0x18];
384    let sram_size_kb = if sram_byte == 0 {
385        0
386    } else {
387        1u32 << u32::from(sram_byte).min(7)
388    };
389    // Expansion (coprocessor) RAM size: extended-header byte $FFBD, three
390    // bytes before the standard header (`off - 3`). Only a `1..=7` exponent
391    // is a valid size (`1024 << n`); anything else (incl. the 0xFF that
392    // non-extended-header carts like Star Fox carry there) reads as "absent".
393    let expansion_ram_kb = off
394        .checked_sub(3)
395        .and_then(|i| rom.get(i))
396        .copied()
397        .filter(|&b| (1..=7).contains(&b))
398        .map_or(0, |b| 1u32 << b);
399
400    Header {
401        title,
402        mapper_kind,
403        dsp_hirom,
404        fast_rom,
405        rom_size_kb,
406        sram_size_kb,
407        expansion_ram_kb,
408        region: Region::from_country(rom[off + 0x19]),
409        maker: rom[off + 0x1A],
410        version: rom[off + 0x1B],
411        checksum_complement: u16::from_le_bytes([rom[off + 0x1C], rom[off + 0x1D]]),
412        checksum: u16::from_le_bytes([rom[off + 0x1E], rom[off + 0x1F]]),
413    }
414}

416const fn mapper_from_byte(byte: u8) -> Option<MapperKind> {
417    match byte & 0x0F {
418        0 => Some(MapperKind::LoRom),
419        1 => Some(MapperKind::HiRom),
420        3 => Some(MapperKind::Sa1),
421        5 => Some(MapperKind::ExHiRom),
422        _ => None,
423    }
424}

426fn decode_title(bytes: &[u8; 21]) -> String {
427    bytes
428        .iter()
429        .map(|&b| {
430            if (0x20..=0x7E).contains(&b) {
431                b as char
432            } else {
433                ' '
434            }
435        })
436        .collect::<String>()
437        .trim_end()
438        .to_string()
439}

449    /// Build a 32 KB synthetic `LoROM` with a valid header.
450    fn synth_lorom(title: &str, sram_kb_log2: u8) -> Vec<u8> {
451        let mut rom = vec![0xEA; 32 * 1024]; // NOP-padded
452        let header_off = HEADER_OFFSET_LOROM;
453        // Title (21 bytes, space-padded).
454        let title_bytes: Vec<u8> = title
455            .bytes()
456            .chain(std::iter::repeat(b' '))
457            .take(21)
458            .collect();
459        rom[header_off..header_off + 21].copy_from_slice(&title_bytes);
460        rom[header_off + 0x15] = 0x20; // LoROM, slow
461        rom[header_off + 0x16] = 0x00; // ROM only
462        rom[header_off + 0x17] = 0x05; // 32 KB
463        rom[header_off + 0x18] = sram_kb_log2;
464        rom[header_off + 0x19] = 0x01; // USA (NTSC)
465        rom[header_off + 0x1A] = 0x33;
466        rom[header_off + 0x1B] = 0x00;
467        // Checksum complement = 0x1234, checksum = !0x1234 = 0xEDCB
468        rom[header_off + 0x1C] = 0x34;
469        rom[header_off + 0x1D] = 0x12;
470        rom[header_off + 0x1E] = 0xCB;
471        rom[header_off + 0x1F] = 0xED;
472        rom
473    }

501    #[test]
502    fn score_header_prefers_plausible_reset_opcode() {
503        // Identical headers; only the byte at the reset-vector target
504        // differs. A plausible first opcode (sei) must outscore an
505        // implausible one (brk).
506        let mut good = synth_lorom("SCORE", 0);
507        // Point the reset vector at $8100 → file offset $0100.
508        good[HEADER_OFFSET_LOROM + 0x3C] = 0x00;
509        good[HEADER_OFFSET_LOROM + 0x3D] = 0x81;
510        let mut bad = good.clone();
511        good[0x0100] = 0x78; // sei  → +8
512        bad[0x0100] = 0x00; // brk  → -8
513        assert!(score_header(&good, HEADER_OFFSET_LOROM) > score_header(&bad, HEADER_OFFSET_LOROM));
514    }

516    #[test]
517    fn expansion_ram_byte_sizes_superfx_work_ram() {
518        // Extended-header $FFBD (off-3) = exponent n → 1024<<n KB. Yoshi's
519        // Island carries 5 (= 32 KB); Doom/Stunt Race carry 6 (= 64 KB).
520        let mut rom = synth_lorom("EXP RAM", 0);
521        rom[HEADER_OFFSET_LOROM - 3] = 5;
522        assert_eq!(
523            Cartridge::from_bytes(rom).unwrap().header.expansion_ram_kb,

527        // An out-of-range byte (e.g. the 0xFF a non-extended-header cart
528        // like Star Fox carries there) reads as "absent" → 0; the Super FX
529        // builder then defaults GSU work RAM to 64 KB.
530        let mut rom = synth_lorom("NO EXP", 0);
531        rom[HEADER_OFFSET_LOROM - 3] = 0xFF;
532        assert_eq!(
533            Cartridge::from_bytes(rom).unwrap().header.expansion_ram_kb,
534            0
535        );
536    }
