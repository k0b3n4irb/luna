1//! Standalone SPC700 disassembler.
2//!
3//! No SNES glue: [`disassemble`] takes a `read(addr) -> u8` closure and a
4//! program counter, decodes one instruction, and returns its canonical
5//! text + byte length. The opcode → (mnemonic, addressing-mode) table is
6//! authored from the ares-cross-checked per-opcode reference in
7//! [`crate::cycles`]; instruction length is fully determined by the mode.
8//!
9//! Operand order matches the execution handlers in [`crate::opcodes`]:
10//! `dd,ds` is `[op][src][dst]`, `d,#i` is `[op][imm][dp]`, and a membit
11//! operand is a little-endian 16-bit word split `addr = w & 0x1FFF`,
12//! `bit = w >> 13` (see `Spc700::fetch_mem_bit`).
13
14/// Addressing mode — selects how many operand bytes follow the opcode and
15/// how they are formatted.
16#[derive(Copy, Clone, Eq, PartialEq, Debug)]
17enum Mode {
18    /// No operand bytes (the template is the whole instruction).
19    Imp,
20    /// `#$ii` immediate.
21    Imm,
22    /// `$dd` direct page.
23    Dp,
24    /// `$dd+X`.
25    DpX,
26    /// `$dd+Y`.
27    DpY,
28    /// `[$dd+X]` indexed-indirect.
29    IndIdxX,
30    /// `[$dd]+Y` indirect-indexed.
31    IdxIndY,
32    /// `$aaaa` absolute.
33    Abs,
34    /// `$aaaa+X`.
35    AbsX,
36    /// `$aaaa+Y`.
37    AbsY,
38    /// `[$aaaa+X]` absolute indexed-indirect (JMP).
39    AbsIndX,
40    /// `$dst, $src` — two direct-page bytes, encoded `[op][src][dst]`.
41    DpDp,
42    /// `$dp, #$ii` — encoded `[op][imm][dp]`.
43    DpImm,
44    /// Relative branch; operand is the resolved target `$aaaa`.
45    Rel,
46    /// `$dp, $target` — direct byte then relative (CBNE/DBNZ d).
47    DpRel,
48    /// `$dp+X, $target` (CBNE d+X).
49    DpXRel,
50    /// `$dp, $target` — direct then relative (BBS/BBC; the bit is in the
51    /// mnemonic).
52    DpBitRel,
53    /// `$dp.b` — direct byte, bit number from the opcode (SET1/CLR1).
54    DpBit,
55    /// `$aaaa.b` membit (13-bit address + 3-bit bit).
56    MemBit,
57    /// `$FFuu` PCALL target.
58    PCall,
59    /// `n` TCALL vector, from the opcode.
60    TCall,
61}
62
63impl Mode {
64    const fn operand_bytes(self) -> u8 {
65        match self {
66            Self::Imp | Self::TCall => 0,
67            Self::Imm
68            | Self::Dp
69            | Self::DpX
70            | Self::DpY
71            | Self::IndIdxX
72            | Self::IdxIndY
73            | Self::Rel
74            | Self::DpBit
75            | Self::PCall => 1,
76            Self::Abs
77            | Self::AbsX
78            | Self::AbsY
79            | Self::AbsIndX
80            | Self::DpDp
81            | Self::DpImm
82            | Self::DpRel
83            | Self::DpXRel
84            | Self::DpBitRel
85            | Self::MemBit => 2,
86        }
87    }
88}

90/// A decoded instruction: canonical assembly text and its byte length.
91#[derive(Clone, Eq, PartialEq, Debug)]
92pub struct Instruction {
93    /// Canonical mnemonic + operands, e.g. `"MOV A, #$12"`.
94    pub text: String,
95    /// Total instruction length in bytes (1..=3).
96    pub length: u8,
97}
98
99/// Disassemble the single instruction at `pc`, reading bytes through
100/// `read`. Never panics; unknown encodings cannot occur (all 256 opcodes
101/// are tabulated).
102pub fn disassemble(read: impl Fn(u16) -> u8, pc: u16) -> Instruction {
103    let op = read(pc);
104    let (tmpl, mode) = OPCODES[op as usize];
105    let length = 1 + mode.operand_bytes();
106    let b1 = read(pc.wrapping_add(1));
107    let b2 = read(pc.wrapping_add(2));
108    let word = u16::from(b1) | (u16::from(b2) << 8);
109    let target = |rel: u8| {
110        pc.wrapping_add(u16::from(length))
111            .wrapping_add(rel as i8 as u16)
112    };
113
114    let operand = match mode {
115        Mode::Imp => String::new(),
116        Mode::Imm => format!("#${b1:02X}"),
117        Mode::Dp => format!("${b1:02X}"),
118        Mode::DpX => format!("${b1:02X}+X"),
119        Mode::DpY => format!("${b1:02X}+Y"),
120        Mode::IndIdxX => format!("[${b1:02X}+X]"),
121        Mode::IdxIndY => format!("[${b1:02X}]+Y"),
122        Mode::Abs => format!("${word:04X}"),
123        Mode::AbsX => format!("${word:04X}+X"),
124        Mode::AbsY => format!("${word:04X}+Y"),
125        Mode::AbsIndX => format!("[${word:04X}+X]"),
126        Mode::DpDp => format!("${b2:02X}, ${b1:02X}"),
127        Mode::DpImm => format!("${b2:02X}, #${b1:02X}"),
128        Mode::Rel => format!("${:04X}", target(b1)),
129        Mode::DpRel => format!("${b1:02X}, ${:04X}", target(b2)),
130        Mode::DpXRel => format!("${b1:02X}+X, ${:04X}", target(b2)),
131        Mode::DpBitRel => format!("${b1:02X}, ${:04X}", target(b2)),
132        Mode::DpBit => format!("${b1:02X}.{}", (op >> 5) & 7),
133        Mode::MemBit => format!("${:04X}.{}", word & 0x1FFF, word >> 13),
134        Mode::PCall => format!("${:04X}", 0xFF00 | u16::from(b1)),
135        Mode::TCall => format!("{}", op >> 4),
136    };
137
138    let text = if tmpl.contains("{}") {
139        tmpl.replace("{}", &operand)
140    } else {
141        tmpl.to_string()
142    };
143    Instruction { text, length }
144}

241    /// Disassemble a fixed byte slice starting at `pc`.
242    fn disasm(bytes: &[u8], pc: u16) -> Instruction {
243        disassemble(
244            |a| {
245                bytes
246                    .get(usize::from(a.wrapping_sub(pc)))
247                    .copied()
248                    .unwrap_or(0)
249            },
250            pc,
251        )
252    }

254    #[test]
255    fn one_per_addressing_mode() {
256        let cases: &[(&[u8], u16, &str, u8)] = &[
257            (&[0x00], 0, "NOP", 1),
258            (&[0xE8, 0x12], 0, "MOV A, #$12", 2),
259            (&[0xE4, 0x10], 0, "MOV A, $10", 2),
260            (&[0xF5, 0x34, 0x12], 0, "MOV A, $1234+X", 3),
261            (&[0x5F, 0x34, 0x12], 0, "JMP $1234", 3),
262            (&[0x1F, 0x34, 0x12], 0, "JMP [$1234+X]", 3),
263            (&[0xE6], 0, "MOV A, (X)", 1),
264            (&[0xBF], 0, "MOV A, (X)+", 1),
265            (&[0x07, 0x10], 0, "OR A, [$10+X]", 2),
266            (&[0x17, 0x10], 0, "OR A, [$10]+Y", 2),
267            (&[0x09, 0x10, 0x20], 0, "OR $20, $10", 3), // dst=$20, src=$10
268            (&[0x18, 0x05, 0x10], 0, "OR $10, #$05", 3), // dp=$10, imm=$05
269            (&[0x0A, 0x00, 0x28], 0, "OR1 C, $0800.1", 3), // word 0x2800
270            (&[0x2A, 0x00, 0x28], 0, "OR1 C, /$0800.1", 3),
271            (&[0x02, 0x10], 0, "SET1 $10.0", 2),
272            (&[0x22, 0x10], 0, "SET1 $10.1", 2),
273            (&[0x12, 0x10], 0, "CLR1 $10.0", 2),
274            (&[0x01], 0, "TCALL 0", 1),
275            (&[0xF1], 0, "TCALL 15", 1),
276            (&[0x4F, 0x30], 0, "PCALL $FF30", 2),
277            (&[0xCF], 0, "MUL YA", 1),
278            (&[0x9E], 0, "DIV YA, X", 1),
279            (&[0xC4, 0x10], 0, "MOV $10, A", 2),
280        ];
281        for (bytes, pc, want, len) in cases {
282            let got = disasm(bytes, *pc);
283            assert_eq!(got.text, *want, "opcode {:02X}", bytes[0]);
284            assert_eq!(got.length, *len, "len for {want}");
285        }
286    }

288    #[test]
289    fn branch_targets_are_resolved() {
290        // BNE $rel at PC=0x0100, rel = -2 → target = 0x0100+2-2 = 0x0100.
291        let i = disasm(&[0xD0, 0xFE], 0x0100);
292        assert_eq!(i.text, "BNE $0100");
293        assert_eq!(i.length, 2);
294        // Forward: BRA at 0x0200, rel = +0x10 → 0x0200+2+0x10 = 0x0212.
295        let i = disasm(&[0x2F, 0x10], 0x0200);
296        assert_eq!(i.text, "BRA $0212");
297        // BBS0 $10, rel at 0x0100, rel=-3 → 0x0100+3-3 = 0x0100.
298        let i = disasm(&[0x03, 0x10, 0xFD], 0x0100);
299        assert_eq!(i.text, "BBS0 $10, $0100");
300        assert_eq!(i.length, 3);
301        // CBNE d+X (0xDE).
302        let i = disasm(&[0xDE, 0x10, 0x00], 0x0100);
303        assert_eq!(i.text, "CBNE $10+X, $0103");
304        assert_eq!(i.length, 3);
305        // DBNZ Y, rel (0xFE) — 2 bytes.
306        let i = disasm(&[0xFE, 0x00], 0x0100);
307        assert_eq!(i.text, "DBNZ Y, $0102");
308        assert_eq!(i.length, 2);
309    }

312    fn every_opcode_decodes_without_panic_and_has_sane_length() {
313        for op in 0u16..=255 {
314            let bytes = [op as u8, 0x55, 0xAA];
315            let i = disassemble(|a| bytes[usize::from(a) % 3], a_pc(op));
316            assert!((1..=3).contains(&i.length));
317            assert!(!i.text.is_empty());
318        }
