1//! The canonical 64-byte SNES SPC700 IPL ROM.
2//!
3//! This is a literal byte-for-byte copy of the SPC700 boot ROM that
4//! sits at `$FFC0..=$FFFF` of every real SPC700's address space. It
5//! is part of the SNES hardware — the same bytes appear on every
6//! console ever made. Cross-checked against fullsnes (Nocash) and
7//! ares' `sfc/smp/iplrom.cpp`.
8//!
9//! On reset the SPC700 reads its reset vector from `$FFFE/$FFFF`
10//! (which the IPL ROM populates as `$FFC0`) and starts executing the
11//! IPL ROM, which:
12//!
13//! 1. Initialises the stack pointer.
14//! 2. Clears direct page RAM (`$0000..=$00EF`).
15//! 3. Writes `$AA` to mailbox 0 (`$F4`) and `$BB` to mailbox 1 (`$F5`).
16//! 4. Spins until the main CPU writes `$CC` to mailbox 0 — the
17//!    handshake "kick".
18//! 5. Enters a byte-transfer loop driven by counters on `$F4`/`$F5`
19//!    and the target address on `$F6`/`$F7`.
20//! 6. Finishes by `JMP [$0000+X]` into whatever code was uploaded.
21//!
22//! The ROM is read-only on real hardware (the SPC700 has a bit in
23//! its control register `$F1` to expose it or hide it), but we only
24//! model the "exposed" path for now — it's how every commercial
25//! SNES game boots.
26//!
27//! See also: <https://problemkaputt.de/fullsnes.htm#snesapu>.
28
29/// The 64 bytes of the SPC700 boot ROM, mapped at `$FFC0..=$FFFF`.
30/// Same on every SNES — these are documented hardware bytes, not
31/// reverse-engineered firmware.
32pub const IPL_ROM: [u8; 64] = [
33    // $FFC0: MOV X,#$EF        ; init top-of-stack value
34    0xCD, 0xEF, //
35    // $FFC2: MOV SP,X           ; SP = $EF
36    0xBD, //
37    // $FFC3: MOV A,#$00         ; A = 0
38    0xE8, 0x00, //
39    // $FFC5: MOV (X),A          ; *(direct[X]) = A; clear dp byte
40    0xC6, //
41    // $FFC6: DEC X
42    0x1D, //
43    // $FFC7: BNE $FFC5          ; loop while X != 0
44    0xD0, 0xFC, //
45    // $FFC9: MOV $F4,#$AA       ; mailbox 0 ← $AA
46    0x8F, 0xAA, 0xF4, //
47    // $FFCC: MOV $F5,#$BB       ; mailbox 1 ← $BB
48    0x8F, 0xBB, 0xF5, //
49    // $FFCF: CMP $F4,#$CC       ; spin until CPU writes $CC kick
50    0x78, 0xCC, 0xF4, //
51    // $FFD2: BNE $FFCF
52    0xD0, 0xFB, //
53    // $FFD4: BRA $FFEF          ; jump to per-byte transfer loop
54    0x2F, 0x19, //
55    // $FFD6: MOV Y,$F4
56    0xEB, 0xF4, //
57    // $FFD8: BNE $FFD6          ; wait for CPU's first counter byte
58    0xD0, 0xFC, //
59    // $FFDA: CMP Y,$F4          ; CPU just wrote a new counter
60    0x7E, 0xF4, //
61    // $FFDC: BNE $FFE9
62    0xD0, 0x0B, //
63    // $FFDE: MOV A,$F5          ; A = data byte
64    0xE4, 0xF5, //
65    // $FFE0: MOV $F4,Y          ; ack: $F4 ← counter
66    0xCB, 0xF4, //
67    // $FFE2: MOV ($00)+Y,A      ; store byte at *target+Y
68    0xD7, 0x00, //
69    // $FFE4: INC Y
70    0xFC, //
71    // $FFE5: BNE $FFDA          ; loop until Y wraps
72    0xD0, 0xF3, //
73    // $FFE7: INC $01            ; high byte of target++
74    0xAB, 0x01, //
75    // $FFE9: BPL $FFDA          ; non-counter write → loop
76    0x10, 0xEF, //
77    // $FFEB: CMP Y,$F4
78    0x7E, 0xF4, //
79    // $FFED: BPL $FFDA          ; index still ahead → keep receiving;
80    //                            falls through to the new-block / execute
81    //                            dispatch only when the CPU signals it.
82    //                            (was $FB = BPL $FFEA, branching into the
83    //                            middle of the prior instruction — broke
84    //                            every multi-block IPL upload.)
85    0x10, 0xEB, //
86    // $FFEF: BA $F6             ; MOVW YA,$F6 = read target address
87    0xBA, 0xF6, //
88    // $FFF1: DA $00             ; MOVW $00,YA = store as transfer target
89    0xDA, 0x00, //
90    // $FFF3: BA $F4             ; MOVW YA,$F4 = read entry address
91    0xBA, 0xF4, //
92    // $FFF5: MOV $F4,A          ; ack low byte
93    0xC4, 0xF4, //
94    // $FFF7: MOV A,Y            ; A = entry high byte
95    0xDD, //
96    // $FFF8: MOV X,A
97    0x5D, //
98    // $FFF9: BNE $FFD6          ; if entry != 0, do byte transfer
99    0xD0, 0xDB, //
100    // $FFFB: JMP [$0000+X]      ; jump to uploaded code
101    0x1F, 0x00, 0x00, //
102    // $FFFE: <reset vector — $C0 $FF (little-endian = $FFC0)>
103    0xC0, 0xFF,
104];

106/// Where the IPL ROM lives in the SPC700's address space.
107pub const IPL_ROM_BASE: u16 = 0xFFC0;
