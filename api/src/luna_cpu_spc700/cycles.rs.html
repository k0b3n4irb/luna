1//! SPC700 per-opcode cycle counts (canonical Sony / Anomie / ares
2//! reference).
3//!
4//! Each entry is the total bus-cycle cost of the opcode INCLUDING the
5//! initial opcode fetch. Branches list the not-taken cost; the
6//! dispatcher should add the per-instruction "taken" penalty
7//! separately if it cares (BRA family = +2 when taken, BBC/BBS = +2,
8//! CBNE = +2, DBNZ = +2).
9//!
10//! Cross-checked against ares' `processor/spc700/instructions.cpp`
11//! addressing-mode helpers (per the spec in
12//! `docs/apu_dsp_reference.md` §2): each `fetch()` / `read()` /
13//! `write()` / `idle()` / `push()` / `pull()` / `load()` / `store()`
14//! call inside ares costs exactly one bus cycle, and the table here
15//! is the sum for each opcode.
16//!
17//! Replaces the previous "flat 4 cycles per opcode" approximation
18//! that desynchronised music tempo (T0/T1/T2 timers) AND playback
19//! pitch (DSP sample tick) by the same factor of ~10-50% (gap A1 in
20//! `docs/luna_apu_gaps.md`).
21

98/// Extra SPC cycles a branch-family opcode costs when the branch is
99/// **taken** — two pipeline idles on real hardware (ares
100/// `instructionBranch`: two `idle()` calls past the condition). Added
101/// by [`crate::Spc700::step`] on top of the not-taken base in
102/// [`SPC700_CYCLES`] for BRA / Bcc / CBNE / DBNZ / BBS / BBC.
103pub const SPC700_BRANCH_TAKEN_PENALTY: u8 = 2;

105/// Quick sanity check: every entry must be in [2, 12]. Used by the
106/// build-time test below.
107#[cfg(test)]
108const _: () = {
109    let mut i = 0;
110    while i < 256 {
111        let c = SPC700_CYCLES[i];
112        assert!(c >= 2 && c <= 12, "cycle table entry out of range");
113        i += 1;
114    }
115};

121    #[test]
122    fn well_known_opcodes_have_canonical_cycle_counts() {
123        // Spot-check against the agent's ares research notes.
124        assert_eq!(SPC700_CYCLES[0x00], 2, "NOP");
125        assert_eq!(SPC700_CYCLES[0x01], 8, "TCALL 0");
126        assert_eq!(SPC700_CYCLES[0x08], 2, "OR A,#i (immediate)");
127        assert_eq!(SPC700_CYCLES[0x05], 4, "OR A,!a (absolute)");
128        assert_eq!(SPC700_CYCLES[0x04], 3, "OR A,d (direct page)");
129        assert_eq!(SPC700_CYCLES[0x3F], 8, "CALL !a");
130        assert_eq!(SPC700_CYCLES[0x6F], 5, "RET");
131        assert_eq!(SPC700_CYCLES[0x7F], 6, "RETI");
132        assert_eq!(SPC700_CYCLES[0x0F], 8, "BRK");
133        assert_eq!(SPC700_CYCLES[0x10], 2, "BPL not-taken");
134        assert_eq!(
135            SPC700_CYCLES[0x2F], 2,
136            "BRA not-taken base; step() adds +2 taken = 4"
137        );
138        assert_eq!(SPC700_CYCLES[0xCF], 9, "MUL YA");
139        assert_eq!(SPC700_CYCLES[0x9E], 12, "DIV YA,X");
140        assert_eq!(SPC700_CYCLES[0xDF], 3, "DAA");
141        assert_eq!(SPC700_CYCLES[0xBE], 3, "DAS");
142        assert_eq!(SPC700_CYCLES[0x9F], 5, "XCN");
143        assert_eq!(SPC700_CYCLES[0xED], 3, "NOTC");
144        // SLEEP/STOP halt the core; luna charges the fetch+execute (2) and
145        // then returns 2 per step while halted. The Tom Harte trace length
146        // (7) is a fixed halt-window artifact, not a completing instruction
147        // cost, so the cycle backstop excludes these two opcodes.
148        assert_eq!(SPC700_CYCLES[0xEF], 2, "SLEEP");
149        assert_eq!(SPC700_CYCLES[0xFF], 2, "STOP");
150        assert_eq!(SPC700_CYCLES[0x8F], 5, "MOV d,#i (direct-immediate write)");
151        assert_eq!(SPC700_CYCLES[0xFA], 5, "MOV dd,ds");
152
153        // The carry/bit group + MOV A,(X)+ — corrected after the Tom Harte
154        // cycle backstop caught luna charging each one cycle too many
155        // (tests/tom_harte.rs validates step() == bus-cycle trace length).
156        assert_eq!(SPC700_CYCLES[0x0A], 5, "OR1 C,m.b");
157        assert_eq!(SPC700_CYCLES[0x2A], 5, "OR1 C,/m.b");
158        assert_eq!(SPC700_CYCLES[0x4A], 4, "AND1 C,m.b");
159        assert_eq!(SPC700_CYCLES[0x6A], 4, "AND1 C,/m.b");
160        assert_eq!(SPC700_CYCLES[0x8A], 5, "EOR1 C,m.b");
161        assert_eq!(SPC700_CYCLES[0xAA], 4, "MOV1 C,m.b");
162        assert_eq!(SPC700_CYCLES[0xCA], 6, "MOV1 m.b,C");
163        assert_eq!(SPC700_CYCLES[0xEA], 5, "NOT1 m.b");
164        assert_eq!(SPC700_CYCLES[0xBF], 4, "MOV A,(X)+");
165    }

167    #[test]
168    fn cycle_table_average_is_in_realistic_range() {
169        // Sanity: the avg cycle count across all 256 opcodes should
170        // land between 3.5 and 5.5 — anything outside indicates I
171        // mistyped a row.
172        let sum: u32 = SPC700_CYCLES.iter().map(|&c| c as u32).sum();
173        let avg = sum as f64 / 256.0;
174        assert!(
175            (3.5..=5.5).contains(&avg),
176            "avg cycles per opcode out of plausible range: {avg}"
177        );
178    }
